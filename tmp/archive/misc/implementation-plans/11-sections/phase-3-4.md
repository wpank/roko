## 3.0 Enhanced template schema

Extend `AgentTemplate` in `roko-serve/src/templates.rs`:

```rust
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct AgentTemplate {
    // Existing fields
    pub name: String,
    pub description: String,
    #[serde(default = "default_model")]
    pub model: String,
    pub system_prompt: String,
    #[serde(default = "default_max_turns")]
    pub max_turns: u32,
    #[serde(default)]
    pub parameters: HashMap<String, ParameterSpec>,

    // New fields
    /// Role for RoleSystemPromptSpec (e.g., "reviewer", "planner", "operator").
    #[serde(default)]
    pub role: Option<String>,

    /// Signal kinds this agent responds to (documentation + validation).
    #[serde(default)]
    pub triggers: Vec<String>,

    /// MCP server names this agent needs (looked up in mcp-servers.toml).
    #[serde(default)]
    pub mcp_servers: Vec<String>,

    /// Gate specs for validating agent output.
    #[serde(default)]
    pub gates: Vec<GateSpec>,

    /// A/B experiment configuration.
    #[serde(default)]
    pub experiment: Option<ExperimentSpec>,

    /// Working directory override.
    #[serde(default)]
    pub repo_context: Option<PathBuf>,

    /// Max concurrent instances of this agent.
    #[serde(default = "default_max_concurrent")]
    pub max_concurrent: u32,

    /// Min seconds between triggers (rate limiting).
    #[serde(default)]
    pub cooldown_secs: Option<u64>,

    /// How to extract template params from the trigger signal body.
    /// Maps param name → JSONPath expression on the signal body.
    #[serde(default)]
    pub input_transform: HashMap<String, String>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct GateSpec {
    pub gate_type: String,  // "compile", "test", "clippy", "diff", "custom"
    #[serde(default = "default_true")]
    pub required: bool,
    /// For "custom" gates: command to run.
    #[serde(default)]
    pub command: Option<String>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ExperimentSpec {
    pub name: String,
    pub variants: Vec<String>,
    pub metric: String,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ParameterSpec {
    pub description: String,
    #[serde(default)]
    pub required: bool,
    #[serde(default)]
    pub default: Option<String>,
}

fn default_model() -> String { "claude-sonnet-4-20250514".into() }
fn default_max_turns() -> u32 { 10 }
fn default_max_concurrent() -> u32 { 3 }
fn default_true() -> bool { true }
```

```toml
required = false

name = "doc-review-strictness"
```

```rust
fn validate_template(template: &AgentTemplate) -> Result<(), Vec<String>> {
    let mut errors = Vec::new();

    if template.name.is_empty() {
        errors.push("name is required".into());
    }
    if template.system_prompt.is_empty() {
        errors.push("system_prompt is required".into());
    }
    if template.max_turns == 0 {
        errors.push("max_turns must be > 0".into());
    }
    for trigger in &template.triggers {
        if !trigger.contains('.') {
            errors.push(format!("trigger '{trigger}' should be dot-separated"));
        }
    }
    if let Some(exp) = &template.experiment {
        if exp.variants.len() < 2 {
            errors.push("experiment needs at least 2 variants".into());
        }
    }

    if errors.is_empty() { Ok(()) } else { Err(errors) }
}
```

```rust
//! Cron scheduler — emits scheduler.cron signals at configured intervals.

use std::sync::Arc;

use anyhow::Result;
use tokio_cron_scheduler::{Job, JobScheduler};
use tracing::{info, error};

use roko_core::kind::Kind;
use roko_core::signal::{Signal, Body};

use crate::events::ServerEvent;
use crate::state::{AppState, Subscription};

/// Start cron jobs for all subscriptions that have a `schedule` field.
pub async fn start_scheduler(state: Arc<AppState>) -> Result<JobScheduler> {
    let scheduler = JobScheduler::new().await?;

    let subs = state.subscriptions.read().await;
    for sub in subs.iter() {
        if !sub.enabled { continue; }
        let Some(ref schedule) = sub.schedule else { continue; };

        let state_clone = Arc::clone(&state);
        let template_name = sub.agent_template.clone();
        let pattern = sub.pattern.clone();

        match Job::new_async(schedule.as_str(), move |_uuid, _lock| {
            let state = Arc::clone(&state_clone);
            let template = template_name.clone();
            let pat = pattern.clone();
            Box::pin(async move {
                info!(template = %template, schedule = %pat, "cron tick");

                let signal = Signal::new(
                    Kind::Custom("scheduler.cron".into()),
                    Body::Json(serde_json::json!({
                        "template": template,
                        "pattern": pat,
                        "timestamp": chrono::Utc::now().to_rfc3339(),
                    })),
                );

                state.event_bus.emit(ServerEvent::WebhookReceived {
                    kind: "scheduler.cron".into(),
                    signal_hash: signal.hash.clone(),
                    source: "scheduler".into(),
                });
            })
        }) {
            Ok(job) => {
                scheduler.add(job).await?;
                info!(template = %sub.agent_template, schedule = %schedule, "scheduled cron job");
            }
            Err(e) => {
                error!(template = %sub.agent_template, schedule = %schedule, error = %e, "invalid cron expression");
            }
        }
    }

    scheduler.start().await?;
    info!("cron scheduler started");

    Ok(scheduler)
}
```

**File:** `crates/roko-serve/src/fswatcher.rs`

### Dependencies

### Implementation

```rust
//! File system watcher — emits watcher.fs_change signals on file modifications.

use std::collections::HashSet;
use std::path::PathBuf;
use std::sync::Arc;
use std::time::Duration;

use anyhow::Result;
use notify::{Config, RecommendedWatcher, RecursiveMode, Watcher};
use notify_debouncer_mini::{new_debouncer, DebouncedEventKind};
use tracing::{info, error, warn};

use roko_core::kind::Kind;
use roko_core::signal::{Signal, Body};

use crate::events::ServerEvent;
use crate::state::{AppState, Subscription};

/// Start file watchers for all subscriptions with `watch_path`.
pub async fn start_watchers(state: Arc<AppState>) -> Result<Vec<notify_debouncer_mini::Debouncer<RecommendedWatcher>>> {
    let mut watchers = Vec::new();

    let subs = state.subscriptions.read().await;
    for sub in subs.iter() {
        if !sub.enabled { continue; }
        let Some(ref watch_path) = sub.watch_path else { continue; };

        if !watch_path.exists() {
            warn!(path = %watch_path.display(), "watch path does not exist, skipping");
            continue;
        }

        let state_clone = Arc::clone(&state);
        let template = sub.agent_template.clone();
        let glob_pattern = sub.watch_glob.clone();
        let watch_path_clone = watch_path.clone();

        let debounce_ms = sub.cooldown_secs.unwrap_or(5) * 1000;

        let (tx, rx) = std::sync::mpsc::channel();

        let mut debouncer = new_debouncer(
            Duration::from_millis(debounce_ms),
            tx,
        )?;

        debouncer.watcher().watch(watch_path, RecursiveMode::Recursive)?;

        // Spawn a task to process debounced events
        tokio::spawn(async move {
            loop {
                match rx.recv() {
                    Ok(Ok(events)) => {
                        let mut changed: HashSet<PathBuf> = HashSet::new();
                        for event in events {
                            if event.kind == DebouncedEventKind::Any {
                                // Apply glob filter
                                if let Some(ref pattern) = glob_pattern {
                                    if let Ok(glob) = glob::Pattern::new(pattern) {
                                        let relative = event.path.strip_prefix(&watch_path_clone).unwrap_or(&event.path);
                                        if !glob.matches_path(relative) {
                                            continue;
                                        }
                                    }
                                }
                                changed.insert(event.path);
                            }
                        }

                        if changed.is_empty() { continue; }

                        let paths: Vec<String> = changed.iter()
                            .map(|p| p.display().to_string())
                            .collect();

                        info!(template = %template, paths = ?paths, "fs change detected");

                        let signal = Signal::new(
                            Kind::Custom("watcher.fs_change".into()),
                            Body::Json(serde_json::json!({
                                "paths": paths,
                                "watch_path": watch_path_clone.display().to_string(),
                                "template": template,
                            })),
                        );

                        state_clone.event_bus.emit(ServerEvent::WebhookReceived {
                            kind: "watcher.fs_change".into(),
                            signal_hash: signal.hash.clone(),
                            source: "fswatcher".into(),
                        });
                    }
                    Ok(Err(e)) => {
                        error!(error = %e, "fs watcher error");
                    }
                    Err(_) => break, // Channel closed
                }
            }
        });

        info!(path = %watch_path.display(), template = %sub.agent_template, "started file watcher");
        watchers.push(debouncer);
    }

    Ok(watchers)
}
```
