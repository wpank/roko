## Task

```rust
#[tokio::test]
async fn ingestor_called_on_successful_task() {
    use std::sync::atomic::{AtomicU32, Ordering};
    use std::sync::Arc;

    #[derive(Debug)]
    struct CountingIngestor(AtomicU32);

    #[async_trait]
    impl KnowledgeIngestor for CountingIngestor {
        async fn ingest(&self, _: &KnowledgeCandidate) -> Result<(), anyhow::Error> {
            self.0.fetch_add(1, Ordering::SeqCst);
            Ok(())
        }
    }

    let dir = tempdir().unwrap();
    let path = dir.path().join("kc.jsonl");
    let counter = Arc::new(CountingIngestor(AtomicU32::new(0)));
    let sink = KnowledgeIngestionSink::at(&path).with_ingestor(counter.clone());

    sink.on_event(&FeedbackEvent::TaskCompleted {
        plan_id: "p".into(),
        task_id: "t".into(),
        outcome: outcome(),
        model_source: ModelChoiceSource::Router,
        succeeded: true,
    }).await.unwrap();

    assert_eq!(counter.0.load(Ordering::SeqCst), 1);
    // JSONL still written (durable record):
    let txt = tokio::fs::read_to_string(&path).await.unwrap();
    assert!(txt.contains("\"kind\":\"success\""));
}
```

## Do NOT

- Do NOT remove the JSONL write. The disk record is the durable source of truth for replay/audit.
- Do NOT make `.ingest` synchronous if it blocks dispatch. The trait is `async fn`; respect it. If admission is slow, fix `roko-neuro`, not this sink.
- Do NOT open a new store handle per event. Construct once at startup, share via `Arc`.
- Do NOT delete `KnowledgeIngestionSink::at()` (no-ingestor constructor) — it's used in tests and deferred-ingest scenarios.
- Do NOT touch `EpisodeSink` or `RoutingObservationSink` here (separate concerns).
- Do NOT invent fields on `KnowledgeAdmissionInput`. If a required field can't be derived from `KnowledgeCandidate`, that's a neuro API gap; log and stop.
- Do NOT add `tracing::error!` on ingestion failure that aborts the runner. Plan 25 § Phase B (S-learn-B) adds a proper failure budget; today, just log via `tracing::warn!`.
