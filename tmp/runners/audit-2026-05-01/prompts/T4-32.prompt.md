### 2. Add `playbooks` layer to `SystemPromptBuilder`

`crates/roko-prompt/src/lib.rs` (or wherever `SystemPromptBuilder` lives):

```rust
use roko_learn::playbook::PlaybookEntry;

impl<'a> SystemPromptBuilder<'a> {
    /// Inject up to N relevant playbooks into the prompt.
    /// When `entries` is empty, the section is omitted entirely.
    pub fn playbooks(mut self, entries: &[PlaybookEntry]) -> Self {
        self.playbooks_section = if entries.is_empty() {
            None
        } else {
            let mut s = String::from("## Relevant Playbooks\n");
            for (i, e) in entries.iter().enumerate() {
                s.push_str(&format!(
                    "{}. [success_rate: {:.2}, used: {} times] {}\n   {}\n",
                    i + 1, e.success_rate, e.used_count, e.title, e.body,
                ));
            }
            Some(s)
        };
        self
    }
}
```

In the `build()` method, render `playbooks_section` between the existing `context` (HDC) and `scratch` layers per the 9-layer order.

If `SystemPromptBuilder` already has a `.playbooks(...)` method but it's a no-op or hardcoded, replace the body with the implementation above.

```rust
fn playbook_layer_omitted_when_empty() {
    let prompt = SystemPromptBuilder::new()
        .task("hello")

        .build();
    assert!(!prompt.contains("Playbook"), "no playbook section when empty: got {prompt:?}");

#[test]
fn playbook_entries_inserted_in_section() {
    let entries = vec![
        PlaybookEntry {
            title: "Auth refactor".into(),
            body: "Use AsyncRead/AsyncWrite at the boundary.".into(),
            success_rate: 0.92,
            used_count: 12,
            task_kind: "refactor".into(),
        },
    ];
    let prompt = SystemPromptBuilder::new()
        .task("refactor auth")
        .playbooks(&entries)
        .build();
    assert!(prompt.contains("AsyncRead/AsyncWrite at the boundary"));
    assert!(prompt.contains("0.92") || prompt.contains("92%"));
    assert!(prompt.contains("Relevant Playbooks"));
}
```
