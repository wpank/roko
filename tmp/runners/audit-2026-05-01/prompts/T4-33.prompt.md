### 1. Add `crates/roko-cli/src/runtime_feedback/rotator.rs`

```rust
//! JSONL rotation helper. Triggered before each append; rotates when the
//! file size exceeds the threshold.
//!
//! Convention: source `episodes.jsonl` rotates to `episodes.jsonl.1`,
//! pushing `episodes.jsonl.N` to `episodes.jsonl.N+1`. Oldest beyond
//! `keep` is deleted.

use std::path::Path;

const ROTATION_THRESHOLD_BYTES: u64 = 10 * 1024 * 1024; // 10 MiB
const KEEP_ROTATED: usize = 5;

#[must_use]
pub fn default_rotation() -> (u64, usize) {
    (ROTATION_THRESHOLD_BYTES, KEEP_ROTATED)
}

/// Rotate `path` if it exceeds `threshold`. Keeps the last `keep` archives.
/// Returns `Ok(true)` if rotation happened, `Ok(false)` if not, `Err(_)` on IO error.
pub async fn maybe_rotate(
    path: &Path,
    threshold: u64,
    keep: usize,
) -> Result<bool, std::io::Error> {
    let metadata = match tokio::fs::metadata(path).await {
        Ok(m) => m,
        Err(e) if e.kind() == std::io::ErrorKind::NotFound => return Ok(false),
        Err(e) => return Err(e),
    };
    if metadata.len() < threshold {
        return Ok(false);
    }

    // Shift archives N..1 down: path.{n} -> path.{n+1}, oldest deleted.
    for i in (1..keep).rev() {
        let from = path.with_extension(format!("jsonl.{i}"));
        let to = path.with_extension(format!("jsonl.{}", i + 1));
        match tokio::fs::rename(&from, &to).await {
            Ok(()) => {}
            Err(e) if e.kind() == std::io::ErrorKind::NotFound => {} // ok, that slot empty
            Err(e) => return Err(e),
        }
    }

    let oldest = path.with_extension(format!("jsonl.{keep}"));
    match tokio::fs::remove_file(&oldest).await {
        Ok(()) => {}
        Err(e) if e.kind() == std::io::ErrorKind::NotFound => {}
        Err(e) => return Err(e),
    }

    let first_archive = path.with_extension("jsonl.1");
    tokio::fs::rename(path, &first_archive).await?;
    Ok(true)
}
```

With:

```rust
async fn write(&self, candidate: &KnowledgeCandidate) -> Result<(), anyhow::Error> {
    if let Some(parent) = self.candidates_path.parent() {
        tokio::fs::create_dir_all(parent).await.ok();
    }
    let mut guard = self.file.lock().await;

    // Check rotation before append. If rotated, drop our cached file handle.
    let (threshold, keep) = crate::runtime_feedback::rotator::default_rotation();
    match crate::runtime_feedback::rotator::maybe_rotate(
        &self.candidates_path, threshold, keep,
    ).await {
        Ok(true) => { *guard = None; }   // rotated; force re-open
        Ok(false) => {}
        Err(e) => {
            tracing::warn!(error = %e, "knowledge candidate rotation failed; appending to current");
        }
    }

    if guard.is_none() {
        let file = tokio::fs::OpenOptions::new()
            .create(true)
            .append(true)
            .open(&self.candidates_path)
            .await?;
        *guard = Some(file);
    }
    let line = serde_json::to_string(candidate)?;
    let bytes = format!("{line}\n");
    let file = guard.as_mut().unwrap();
    file.write_all(bytes.as_bytes()).await?;
    file.flush().await?;
    Ok(())
}
```

#[tokio::test]
async fn rotates_when_threshold_exceeded() {
    use tokio::io::AsyncWriteExt;
    let dir = tempfile::tempdir().unwrap();
    let path = dir.path().join("test.jsonl");
    let big = b"x".repeat(1_100_000);  // ~1.1 MB per write
    for _ in 0..15 {
        let mut f = tokio::fs::OpenOptions::new()
            .create(true).append(true).open(&path).await.unwrap();
        f.write_all(&big).await.unwrap();
    }
    assert!(tokio::fs::metadata(&path).await.unwrap().len() > 10_000_000);

    let rotated = maybe_rotate(&path, 10_000_000, 3).await.unwrap();
    assert!(rotated);
    assert!(path.with_extension("jsonl.1").exists());
    // Original is now gone (or empty); a new write will recreate.
}

#[tokio::test]
async fn keeps_only_n_archives() {
    let dir = tempfile::tempdir().unwrap();
    let path = dir.path().join("test.jsonl");
    let big = b"x".repeat(11_000_000);

    for round in 1..=5 {
        let mut f = tokio::fs::OpenOptions::new()
            .create(true).append(true).open(&path).await.unwrap();
        f.write_all(&big).await.unwrap();
        let _ = maybe_rotate(&path, 10_000_000, 3).await.unwrap();
        let _ = round;
    }
    assert!(path.with_extension("jsonl.1").exists());
    assert!(path.with_extension("jsonl.2").exists());
    assert!(path.with_extension("jsonl.3").exists());
    assert!(!path.with_extension("jsonl.4").exists(), "keep=3 means no jsonl.4");
}

#[tokio::test]
async fn no_rotation_when_under_threshold() {
    let dir = tempfile::tempdir().unwrap();
    let path = dir.path().join("test.jsonl");
    tokio::fs::write(&path, b"small content").await.unwrap();
    assert_eq!(maybe_rotate(&path, 10_000_000, 3).await.unwrap(), false);
}

#[tokio::test]
async fn missing_file_is_no_op() {
    let dir = tempfile::tempdir().unwrap();
    let path = dir.path().join("missing.jsonl");
    assert_eq!(maybe_rotate(&path, 10_000_000, 3).await.unwrap(), false);
}

## Acceptance Criteria

- `rotator.rs` (or `roko-fs/rotation.rs`) implemented with `maybe_rotate`.
- All three JSONL writers (episodes, knowledge candidates, efficiency) call `maybe_rotate` before each append.
- File handle is dropped and re-opened after rotation.
- 4 unit tests covering rotate / keep-N / under-threshold / missing-file.

## Do NOT

- Do NOT compress rotated files (separate task).
- Do NOT use age-based rotation (size-based is simpler, adequate).
- Do NOT use `std::fs` blocking calls in the async sinks. Use `tokio::fs`.
- Do NOT swallow rotation errors silently. Log via `tracing::warn!` and append to the existing file.
- Do NOT introduce a separate rotation thread / timer. The check is per-append.
- Do NOT change the JSONL line format (separate task: S-learn-C adds `schema_version`).
- Do NOT delete the `.jsonl.{N}` files at startup; let the rotation chain manage them.
