### 1. Extend `RunLedgerEntry`

`crates/roko-runtime/src/run_ledger.rs`:

```rust
use roko_gate::{Rung, GateStatus};

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(tag = "kind", rename_all = "snake_case")]
pub enum RunLedgerEntry {
    // ... existing variants
    Gate {
        rung: Rung,
        status: GateStatus,
        duration_ms: u64,
        attempt: u32,
        #[serde(skip_serializing_if = "Option::is_none")]
        rationale: Option<String>,
    },
}

impl RunLedger {
    pub fn record_gate(
        &mut self,
        rung: Rung,
        status: GateStatus,
        duration: std::time::Duration,
        attempt: u32,
        rationale: Option<String>,
    ) -> Result<(), LedgerWriteError> {
        self.append(RunLedgerEntry::Gate {
            rung,
            status,
            duration_ms: duration.as_millis() as u64,
            attempt,
            rationale,
        })
    }

    /// Latest verdict per rung. Used by the report builder (slice S-ledger-2).
    pub fn gate_verdicts_by_rung(&self) -> std::collections::HashMap<Rung, GateStatus> {
        let mut latest = std::collections::HashMap::new();
        for entry in self.entries() {
            if let RunLedgerEntry::Gate { rung, status, .. } = entry {
                latest.insert(*rung, *status);
            }
        }
        latest
    }
}
```

If `LedgerWriteError` doesn't exist yet, return `anyhow::Result<()>` for now; S-ledger-1 introduces typed errors later.

### 3. Tests

```rust
#[test]
fn gate_ledger_round_trips_status() {
    let mut ledger = RunLedger::new();
    ledger.record_gate(Rung::Compile, GateStatus::Passed, std::time::Duration::from_secs(2), 1, None).unwrap();
    ledger.record_gate(
        Rung::Test,
        GateStatus::Failed { code: Some(1) },
        std::time::Duration::from_secs(5),
        1,
        Some("test_foo: assertion failed".into()),
    ).unwrap();
    let map = ledger.gate_verdicts_by_rung();
    assert_eq!(map[&Rung::Compile], GateStatus::Passed);
    assert!(matches!(map[&Rung::Test], GateStatus::Failed { code: Some(1) }));
}

#[test]
fn gate_ledger_keeps_latest_per_rung() {
    let mut ledger = RunLedger::new();
    ledger.record_gate(Rung::Compile, GateStatus::Failed { code: Some(1) }, ..., 1, None).unwrap();
    ledger.record_gate(Rung::Compile, GateStatus::Passed, ..., 2, None).unwrap();
    let map = ledger.gate_verdicts_by_rung();
    assert_eq!(map[&Rung::Compile], GateStatus::Passed);
}
```
