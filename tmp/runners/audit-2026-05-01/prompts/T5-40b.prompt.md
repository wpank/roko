### 2. Workflow finalizer reads ledger

`crates/roko-runtime/src/workflow_engine.rs`:

```rust
impl WorkflowEngine {
    fn finalize_outcome(&self, ledger: &RunLedger) -> WorkflowOutcome {
        let invalid_required: Vec<_> = ledger.artifacts()
            .filter(|a| a.required &&
                matches!(a.outcome, ArtifactOutcome::Invalid | ArtifactOutcome::Missing))
            .collect();
        if !invalid_required.is_empty() {
            return WorkflowOutcome::Failed {
                reason: format!(
                    "{} required artifacts invalid/missing: {}",
                    invalid_required.len(),
                    invalid_required.iter().map(|a| a.id.as_str()).collect::<Vec<_>>().join(", ")
                ),
            };
        }
        // ... existing logic
    }
}
```
