## Exact changes

### 1. Add TypeScript `CommandEvent` types

`demo/demo-app/src/lib/terminal-events.ts` (new):

```typescript
// Mirror exactly the serde JSON shape produced by Rust's CommandEvent
// (crates/roko-serve/src/command_events.rs).

export type CommandEvent =
  | { type: 'started'; session_id: string; pid: number; command: string }
  | { type: 'output'; session_id: string; bytes: string }
  | { type: 'exited'; session_id: string; code: number | null; duration_ms: number }
  | { type: 'spawn_failed'; session_id: string; reason: string }
  | { type: 'cancelled'; session_id: string };

export interface CommandRun {
  code: number | null;
  output: string;
  duration_ms: number;
  failed: boolean;
  reason?: string;
}

export async function runCommand(
  sessionId: string,
  command: string,
  apiBase: string,
): Promise<CommandRun> {
  return new Promise((resolve, reject) => {
    const ws = new WebSocket(`${apiBase}/api/terminal/sessions/${sessionId}/events`);
    let buffer = '';

    ws.onmessage = (msg) => {
      const event: CommandEvent = JSON.parse(msg.data);
      switch (event.type) {
        case 'output':
          buffer += event.bytes;
          break;
        case 'exited':
          ws.close();
          resolve({
            code: event.code,
            output: buffer,
            duration_ms: event.duration_ms,
            failed: event.code !== 0,
          });
          break;
        case 'spawn_failed':
          ws.close();
          resolve({ code: null, output: buffer, duration_ms: 0, failed: true, reason: event.reason });
          break;
        case 'cancelled':
          ws.close();
          resolve({ code: null, output: buffer, duration_ms: 0, failed: true, reason: 'cancelled' });
          break;
        case 'started':
          break;
      }
    };

    ws.onerror = () => reject(new Error('events ws error'));
    ws.onopen = () => sendCommand(sessionId, command, apiBase);
  });
}

async function sendCommand(sessionId: string, cmd: string, apiBase: string) {
  // Use the existing /api/terminal/sessions/{id}/io WS or a POST endpoint
  // to deliver `cmd` to the PTY. Reuse whatever the demo app already uses.
}
```

The exact `sendCommand` implementation depends on the app's terminal-session helper; reuse from `demo/demo-app/src/lib/terminal-session.ts`.

### 2. Migrate `prd-pipeline.ts`

```bash
cat demo/demo-app/src/lib/scenario-runners/prd-pipeline.ts
```

Find the regex patterns. Common forms:

```typescript
async function runStep(prompt: string) {
    await terminal.sendText(prompt);
    while (true) {
        const output = await terminal.readUntilTimeout(500);
        if (output.match(/\$ $/)) break;
        if (output.match(/error/i)) throw new Error('failed');
    }
}
```

Replace with:

```typescript
import { runCommand, CommandRun } from '../terminal-events';

async function runStep(sessionId: string, command: string, apiBase: string): Promise<CommandRun> {
    const result = await runCommand(sessionId, command, apiBase);
    if (result.failed) {
        throw new Error(
            `step failed: ${result.reason ?? `exit ${result.code}`}\n${result.output.slice(-200)}`,
        );
    }
    return result;
}
```

Update each step in the scenario to use `runStep(sessionId, "roko prd plan auth-redesign", apiBase)` and check the returned `code`.
