# BI_17: /gate <name> on|off toggles runtime config

## Task

## Problem

### Step 2: Replace the stub `/gate <args>` arm

Replace lines 2980-2986 (the `else` branch printing `gate toggle: {arg}`) with a real parser. Keep the empty-arg branch (lines 2952-2978) intact, but change its hard-coded values to read from `session.runtime_gates`:

```rust
        _ if cmd.starts_with("/gate") => {
            let arg = cmd.strip_prefix("/gate").unwrap().trim();
            if arg.is_empty() {
                let gates = &session.runtime_gates;
                let tests_state = if gates.skip_tests { "off" } else { "on" };
                let clippy_state = if gates.clippy_enabled { "on" } else { "off" };
                term.push_lines(&[
                    styled::section_start(theme, "gates", "(runtime)", None),
                    styled::continuation(theme, "compile", "on (always)", None),
                    styled::continuation(theme, "test", tests_state, None),
                    styled::continuation(theme, "clippy", clippy_state, None),
                    styled::section_end(
                        theme,
                        "max_iter",
                        &gates.max_iterations.to_string(),
                    ),
                ])?;
            } else {
                let parts: Vec<&str> = arg.split_whitespace().collect();
                if parts.len() != 2 {
                    term.push_lines(&[styled::continuation(
                        theme,
                        "gate",
                        "usage: /gate <name> on|off  OR  /gate max_iter <n>",
                        Some("names: clippy, tests, max_iter"),
                    )])?;
                } else {
                    let (name, value) = (parts[0].to_ascii_lowercase(), parts[1]);
                    let result: Result<(String, String), &'static str> = match name.as_str() {
                        "clippy" => match value {
                            "on" | "true" | "1" => {
                                session.runtime_gates.clippy_enabled = true;
                                Ok(("clippy".into(), "on".into()))
                            }
                            "off" | "false" | "0" => {
                                session.runtime_gates.clippy_enabled = false;
                                Ok(("clippy".into(), "off".into()))
                            }
                            _ => Err("clippy expects on|off"),
                        },
                        "tests" | "test" => match value {
                            // `tests on` means tests enabled, i.e. skip_tests = false
                            "on" | "true" | "1" => {
                                session.runtime_gates.skip_tests = false;
                                Ok(("tests".into(), "on".into()))
                            }
                            "off" | "false" | "0" => {
                                session.runtime_gates.skip_tests = true;
                                Ok(("tests".into(), "off".into()))
                            }
                            _ => Err("tests expects on|off"),
                        },
                        "max_iter" | "max_iterations" => match value.parse::<u32>() {
                            Ok(n) => {
                                session.runtime_gates.max_iterations = n;
                                Ok(("max_iter".into(), n.to_string()))
                            }
                            Err(_) => Err("max_iter expects a non-negative integer"),
                        },
                        _ => Err("unknown gate name; expected: clippy, tests, max_iter"),
                    };
                    match result {
                        Ok((field, new_value)) => {
                            term.push_lines(&[styled::continuation(
                                theme,
                                "gate",
                                &format!("{field} = {new_value}"),
                                Some("runtime overlay; not written to roko.toml"),
                            )])?;
                        }
                        Err(msg) => {
                            term.push_lines(&[styled::continuation(
                                theme,
                                "gate",
                                msg,
                                Some("usage: /gate <name> on|off"),
                            )])?;
                        }
                    }
                }
            }
        }
```
