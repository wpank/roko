### Step 2: Add a small dotted-key applier near the bottom of the file

Near the other helpers around `read_roko_toml` (`chat_inline.rs:2317-2323`), add:

```rust
/// Result of applying a `/config set` mutation.
struct ConfigSetOutcome {
    /// Human-readable echo of the new effective value.
    echoed: String,
    /// `Some(path)` when the change was persisted to disk; `None` otherwise.
    persisted_to: Option<std::path::PathBuf>,
}

/// Apply `<key> = <value>` to a runtime `RokoConfig` overlay.
///
/// `key` is a dotted path; only the keys listed in BI_18 are supported.
/// Returns a `&'static str` error message on parse / unknown-key failures
/// so the caller can render it without losing typed context.
fn apply_runtime_config_set(
    cfg: &mut roko_core::config::schema::RokoConfig,
    key: &str,
    value: &str,
) -> Result<String, &'static str> {
    let trimmed = value.trim();
    let parsed_key = key.trim().to_ascii_lowercase();
    match parsed_key.as_str() {
        "agent.default_model" => {
            cfg.agent.default_model = trimmed.to_string();
            Ok(format!("agent.default_model = {trimmed}"))
        }
        "agent.default_effort" => {
            cfg.agent.default_effort = trimmed.to_string();
            Ok(format!("agent.default_effort = {trimmed}"))
        }
        "agent.fallback_model" => {
            if trimmed.is_empty() || trimmed == "none" || trimmed == "-" {
                cfg.agent.fallback_model = None;
                Ok("agent.fallback_model = (cleared)".to_string())
            } else {
                cfg.agent.fallback_model = Some(trimmed.to_string());
                Ok(format!("agent.fallback_model = {trimmed}"))
            }
        }
        "gates.clippy_enabled" => {
            cfg.gates.clippy_enabled = parse_bool_arg(trimmed)
                .ok_or("gates.clippy_enabled expects true|false|on|off")?;
            Ok(format!("gates.clippy_enabled = {}", cfg.gates.clippy_enabled))
        }
        "gates.skip_tests" => {
            cfg.gates.skip_tests = parse_bool_arg(trimmed)
                .ok_or("gates.skip_tests expects true|false|on|off")?;
            Ok(format!("gates.skip_tests = {}", cfg.gates.skip_tests))
        }
        "gates.max_iterations" => {
            cfg.gates.max_iterations = trimmed
                .parse::<u32>()
                .map_err(|_| "gates.max_iterations expects a non-negative integer")?;
            Ok(format!("gates.max_iterations = {}", cfg.gates.max_iterations))
        }
        "budget.max_plan_usd" => {
            cfg.budget.max_plan_usd = trimmed
                .parse::<f32>()
                .map_err(|_| "budget.max_plan_usd expects a decimal number")?;
            Ok(format!("budget.max_plan_usd = {}", cfg.budget.max_plan_usd))
        }
        "budget.max_turn_usd" => {
            cfg.budget.max_turn_usd = trimmed
                .parse::<f32>()
                .map_err(|_| "budget.max_turn_usd expects a decimal number")?;
            Ok(format!("budget.max_turn_usd = {}", cfg.budget.max_turn_usd))
        }
        _ => Err(
            "unsupported key. Supported: agent.default_model, agent.default_effort, \
             agent.fallback_model, gates.{clippy_enabled,skip_tests,max_iterations}, \
             budget.{max_plan_usd,max_turn_usd}",
        ),
    }
}

fn parse_bool_arg(value: &str) -> Option<bool> {
    match value.to_ascii_lowercase().as_str() {
        "true" | "on" | "yes" | "1" => Some(true),
        "false" | "off" | "no" | "0" => Some(false),
        _ => None,
    }
}

/// Persist the given runtime config to `<workdir>/roko.toml` via the
/// schema's TOML serializer. Returns the file path on success.
fn persist_runtime_config(
    cfg: &roko_core::config::schema::RokoConfig,
) -> anyhow::Result<std::path::PathBuf> {
    let path = std::env::current_dir()
        .unwrap_or_else(|_| std::path::PathBuf::from("."))
        .join("roko.toml");
    let toml = roko_core::config::schema::RokoConfig::to_toml(cfg);
    std::fs::write(&path, toml)?;
    Ok(path)
}
```

If the `to_toml` associated function does not exist on `RokoConfig` (verify with `rg -n 'fn to_toml|impl RokoConfig' crates/roko-core/src/config/schema.rs`), fall back to `toml::to_string_pretty(cfg)?` (the `toml` crate is already a workspace dep — confirm with `rg -n '^toml ' crates/roko-cli/Cargo.toml`). Pick whichever is available; do not invent a new method on `RokoConfig` from this batch.

### Step 3: Replace the `/config set` arm with the real handler

At `chat_inline.rs:3124-3143`, replace the `else` block (the `set {key} = {value}` echo) with:

```rust
        _ if cmd.starts_with("/config set ") => {
            let rest = cmd.strip_prefix("/config set ").unwrap().trim();
            // `:persist` (suffix) requests writing the change back to roko.toml.
            let (body, persist) = match rest.strip_suffix(":persist") {
                Some(stripped) => (stripped.trim(), true),
                None => (rest, false),
            };
            let parts: Vec<&str> = body.splitn(2, ' ').collect();
            if parts.len() < 2 {
                term.push_lines(&[styled::continuation(
                    theme,
                    "config",
                    "usage: /config set <key> <value> [:persist]",
                    Some("e.g. /config set agent.default_model gpt-4o :persist"),
                )])?;
            } else {
                let (key, value) = (parts[0], parts[1]);
                match apply_runtime_config_set(&mut session.runtime_config, key, value) {
                    Ok(echo) => {
                        let hint = if persist {
                            match persist_runtime_config(&session.runtime_config) {
                                Ok(path) => Some(format!("persisted to {}", path.display())),
                                Err(err) => Some(format!("persist failed: {err}")),
                            }
                        } else {
                            Some("runtime overlay; add ` :persist` to write to roko.toml".into())
                        };
                        term.push_lines(&[styled::continuation(
                            theme,
                            "config",
                            &echo,
                            hint.as_deref(),
                        )])?;
                    }
                    Err(msg) => {
                        term.push_lines(&[styled::continuation(
                            theme,
                            "config",
                            msg,
                            Some("usage: /config set <key> <value> [:persist]"),
                        )])?;
                    }
                }
            }
        }
```
