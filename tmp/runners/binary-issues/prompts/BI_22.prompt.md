```304:345:crates/roko-std/src/tool/builtin/web_search.rs
    async fn execute(&self, call: ToolCall, ctx: &ToolContext) -> ToolResult {
        if !ctx.capabilities.network {
            return ToolResult::Err(ToolError::PermissionDenied(
                "web_search requires network capability".into(),
            ));

        let query = match require_string(&call.arguments, "query") {
            Ok(q) => q,
            Err(e) => return ToolResult::Err(e),

        if query.trim().is_empty() {
            return ToolResult::Err(ToolError::SchemaInvalid(
                "web_search: `query` must be non-empty".into(),
            ));

        if let Some(model_caller) = self.model_caller.clone() {

                ));
```

```rust
/// Run a single `web_search` tool invocation inline. Returns the
/// formatted text the tool would have returned to an agent. The chat
/// handler renders this text directly; the chat session's model is
/// not invoked.
async fn run_research_inline(query: &str) -> anyhow::Result<String> {
    use roko_core::tool::{ToolCall, ToolContext};
    use roko_std::tool::builtin::web_search;
    let mut ctx = ToolContext::testing(
        std::env::current_dir()
            .unwrap_or_else(|_| std::path::PathBuf::from("."))
            .to_string_lossy()
            .as_ref(),
    );
    ctx.capabilities.network = true;
    let call = ToolCall::new(
        "chat-research",
        web_search::NAME,
        serde_json::json!({ "query": query }),
    );
    let handler = web_search::Handler::default();
    use roko_core::tool::ToolHandler as _;
    match handler.execute(call, &ctx).await {
        roko_core::tool::ToolResult::Ok(payload) => {
            // ToolResult::text(...) wraps the response as a structured
            // payload. Pull the text body out for the TUI.
            payload
                .as_text()
                .map(|s| s.to_string())
                .ok_or_else(|| anyhow::anyhow!("web_search returned non-text payload"))
        }
        roko_core::tool::ToolResult::Err(err) => Err(anyhow::anyhow!("{err}")),
    }
}
```
