```rust
/// Configuration for the pipeline. Determines which phases are active.
#[derive(Debug, Clone)]
pub struct WorkflowConfig {
    /// Include a strategist phase before implementation.
    pub has_strategy: bool,
    /// Include a review phase after gates pass.
    pub has_review: bool,
    /// Maximum implement -> gate -> review iterations.
    pub max_iterations: u32,
    /// Maximum autofix attempts per gate failure.
    pub max_autofix_attempts: u32,
}

impl WorkflowConfig {
    /// Express: implement -> gate -> commit.
    pub fn express() -> Self {
        Self { has_strategy: false, has_review: false, max_iterations: 1, max_autofix_attempts: 1 }
    }

    /// Standard: implement -> gate -> review -> commit.
    pub fn standard() -> Self {
        Self { has_strategy: false, has_review: true, max_iterations: 2, max_autofix_attempts: 2 }
    }

    /// Full: strategy -> implement -> gate -> review -> commit.
    pub fn full() -> Self {
        Self { has_strategy: true, has_review: true, max_iterations: 3, max_autofix_attempts: 2 }
    }
}
```

The TOML format the loader must parse. This is the canonical shape used in `.roko/workflow.toml`
and inline task configs:
```toml
[workflow]
template = "standard"       # optional: "express" | "standard" | "full" (overrides individual fields)
has_strategy = false        # optional: override template default
has_review = true           # optional: override template default
max_iterations = 3          # optional: override template default
max_autofix_attempts = 2    # optional: override template default

[[workflow.steps]]
name = "strategy"
role = "strategist"
on_success = "implement"
on_failure = "halt"

[[workflow.steps]]
name = "implement"
role = "implementer"
on_success = "gate"
on_failure = "halt"
```

The `steps` array is parsed for forward-compatibility but not yet consumed by the state machine
(it drives `has_strategy` / `has_review` inference instead — see Task below).

```rust
    has_strategy: Option<bool>,
```

Parse logic:

1. Try to deserialize as `{ workflow: WorkflowConfigToml }` (the table form). If that fails, try

3. Apply overrides: any `Some(v)` in the intermediate struct replaces the base field.
4. Return the merged `WorkflowConfig`.
