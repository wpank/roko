### Task

#### 1. Ensure `RunReport` and `GateResult` derive `Serialize`

If not already present, add `#[derive(serde::Serialize)]`. For `Duration` fields
add a serialization helper converting to `f64` seconds.

#### 2. Add `generate_share_token` to `crates/roko-core/src/lib.rs`

```rust
/// Generate a short share token: `<unix_millis_hex>-<rand_hex>`.
pub fn generate_share_token() -> String {
    use std::time::{SystemTime, UNIX_EPOCH};
    let millis = SystemTime::now().duration_since(UNIX_EPOCH)
        .unwrap_or_default().as_millis();
    let rand_part: u16 = (millis as u16) ^ (std::process::id() as u16);
    format!("{millis:x}-{rand_part:04x}")
}
```

If `uuid` is already a dependency, prefer `Uuid::new_v4().to_string()` instead.

#### 3. Add `write_shared_run` to `crates/roko-cli/src/run.rs`

```rust
/// Write a RunReport to `.roko/shared/{token}.json` and return the token.
pub fn write_shared_run(workdir: &std::path::Path, report: &RunReport) -> anyhow::Result<String> {
    let token = roko_core::generate_share_token();
    let dir = workdir.join(".roko").join("shared");
    std::fs::create_dir_all(&dir)?;
    let transcript = serde_json::json!({
        "id": &token, "agent": &report.agent, "role": &report.role,
        "prompt": &report.prompt, "success": report.overall_success,
        "gates": report.gate_results.iter().map(|g| (&g.name, g.passed)).collect::<Vec<_>>(),
        "output": &report.agent_output, "cost_usd": report.cost_usd,
        "input_tokens": report.input_tokens, "output_tokens": report.output_tokens,
        "model": &report.model, "duration_s": report.duration.as_secs_f64(),
        "episode_id": &report.episode_id,
        "timestamp": chrono::Utc::now().to_rfc3339(),
    });
    std::fs::write(dir.join(format!("{token}.json")), serde_json::to_string_pretty(&transcript)?)?;
    Ok(token)
}
```

#### 5. Unit test

```rust
#[test]
fn write_shared_run_creates_file() {
    let tmp = std::env::temp_dir().join("roko-test-share");
    let _ = std::fs::remove_dir_all(&tmp);
    let report = RunReport {
        agent: "test".into(), role: "coder".into(), prompt: "hi".into(),
        overall_success: true, gate_results: vec![],
        agent_output: Some("done".into()), cost_usd: Some(0.01),
        input_tokens: Some(100), output_tokens: Some(50),
        model: Some("claude-sonnet-4-6".into()),
        duration: std::time::Duration::from_secs(2), episode_id: Some("ep-1".into()),
    };
    let token = write_shared_run(&tmp, &report).unwrap();
    assert!(tmp.join(".roko/shared").join(format!("{token}.json")).exists());
    let _ = std::fs::remove_dir_all(&tmp);
}
```
