```rust
pub fn print_agent_table(agents: &[(&str, &str, &str, &str)]) {
    intro("Agents");
    divider();
    bar(&format!("  {:<14}{:<10}{:<25}{}",
        bold("NAME"), bold("STATUS"), bold("MODEL"), bold("LAST ACTIVE")));
    bar(&format!("  {}", dim(&"\u{2500}".repeat(60))));
    for (name, status, model, last_active) in agents {
        let status_display = match *status {
            "active" => green(status),
            "idle" => yellow(status),
            _ => dim(status),
        };
        bar(&format!("  {:<14}{:<10}{:<25}{}",
            cyan(name), status_display, dim(model), dim(last_active)));
    }
    divider();
    let active = agents.iter().filter(|(_, s, _, _)| *s == "active").count();
    end(&dim(&format!("{} agents ({} active)", agents.len(), active)));
}
```
