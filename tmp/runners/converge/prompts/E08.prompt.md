pub async fn run(&self, config: WorkflowRunConfig) -> Result<WorkflowResult> {
    let run_id = generate_run_id();
    let mut pipeline = PipelineStateV2::new(config.workflow.clone(), config.prompt.clone());
    let driver = EffectDriver::new(
        EffectServices { ... },
        run_id.clone(),
        config.workdir.clone(),
    );

    self.emit(WorkflowEvent::WorkflowStarted { ... });

    let mut output = pipeline.step(PipelineInput::Start);

    loop {
        let old_phase = pipeline.phase.label();
        let input = match &output {
            PipelineOutput::SpawnStrategist { prompt } => { ... }
            PipelineOutput::SpawnImplementer { prompt, context } => { ... }
            PipelineOutput::SpawnAutoFixer { error_output } => { ... }
            PipelineOutput::SpawnReviewer { diff_context } => { ... }
            PipelineOutput::RunGates => driver.run_gates(&config.enabled_gates).await,
            PipelineOutput::Commit => { ... }
            PipelineOutput::Done { outcome } => {
                self.emit(WorkflowEvent::WorkflowCompleted { ... });
                return Ok(WorkflowResult { run_id, outcome: outcome.clone(), iterations: pipeline.iteration });
            }
            PipelineOutput::Halt { reason } => {
                self.emit(WorkflowEvent::WorkflowCompleted { ... });
                return Ok(WorkflowResult { run_id, outcome, iterations: pipeline.iteration });
            }
        };
        output = pipeline.step(input);
        let new_phase = pipeline.phase.label();
        if old_phase != new_phase {
            self.emit(WorkflowEvent::PhaseTransition { ... });
        }
    }
}

#### 2. Implementation

a. Deserialize the checkpoint:

```rust
let pipeline = PipelineStateV2::from_checkpoint(checkpoint)
    .map_err(|e| crate::effect_driver::WorkflowError::Internal(format!(
        "invalid checkpoint: {e}"
    )))?;
```

b. Check for terminal phase â€” if the checkpoint is already `Complete`, `Halted`, or
`Cancelled`, return immediately without entering the loop:

```rust
if pipeline.phase.is_terminal() {
    let outcome = match &pipeline.phase {
        Phase::Complete => WorkflowOutcome::Success {
            commit_hash: pipeline.commit_hash.clone(),
        },
        Phase::Halted { reason } => WorkflowOutcome::Halted {
            reason: reason.clone(),
        },
        Phase::Cancelled => WorkflowOutcome::Cancelled,
        _ => unreachable!(),
    };
    return Ok(WorkflowResult {
        run_id: generate_run_id(),
        outcome,
        iterations: pipeline.iteration,
    });
}
```

c. Generate a new `run_id` for the resumed run (do NOT reuse the original â€” each
execution attempt gets its own id for logging):

```rust

```

d. Create the `EffectDriver` with the new run_id and the workdir from `config`:

```rust
let driver = EffectDriver::new(
    EffectServices {
        model_caller: Arc::clone(&self.services.model_caller),
        prompt_assembler: Arc::clone(&self.services.prompt_assembler),
        feedback_sink: Arc::clone(&self.services.feedback_sink),
        gate_runner: Arc::clone(&self.services.gate_runner),

```

e. Emit `WorkflowStarted` with `template` set to `"resumed"`:

```rust

    run_id: run_id.clone(),
    template: "resumed".to_string(),
    prompt: pipeline.original_prompt.clone(),

```

f. Re-enter the run loop. The key difference from `run()` is how the first `output`
is produced. Instead of `pipeline.step(PipelineInput::Start)`, step with a
`PipelineInput::Resume` if it exists, or simply re-derive the output from the
current phase:

```rust
let mut pipeline = pipeline; // make mutable
let mut output = pipeline.step(PipelineInput::Start);
```

Note: `PipelineInput::Start` on a non-Pending pipeline is harmless â€” the state
machine's `step` method checks the current phase and produces the correct output
for re-entry (e.g., `Implementing` produces `SpawnImplementer`). If this does NOT
work correctly, use the following approach instead:

```rust
// Derive the first output directly from the saved phase.
let mut output = match &pipeline.phase {
    Phase::Implementing => PipelineOutput::SpawnImplementer {
        prompt: pipeline.original_prompt.clone(),
        context: pipeline.strategist_brief.clone(),
    },
    Phase::Gating => PipelineOutput::RunGates,
    Phase::AutoFixing => PipelineOutput::SpawnAutoFixer {
        error_output: pipeline.last_gate_failure.clone().unwrap_or_default(),
    },
    Phase::Reviewing => PipelineOutput::SpawnReviewer {
        diff_context: None,
    },
    Phase::Committing => PipelineOutput::Commit,
    Phase::Strategizing => PipelineOutput::SpawnStrategist {
        prompt: pipeline.original_prompt.clone(),
    },
    _ => PipelineOutput::Halt {
        reason: format!("cannot resume from phase {:?}", pipeline.phase),
    },
};
```

g. The rest of the loop body is identical to `run()`: match on `output`, execute
effects, feed results back via `pipeline.step(input)`, emit phase transitions.
