### 3. Add private knowledge query helpers

Add two private methods:

```rust
/// Query knowledge store for relevant technique insights.
fn query_techniques(&self, task: &str) -> Vec<String> {
    let Some(store) = self.knowledge_store.as_ref() else {
        return Vec::new();
    };
    let entries = store.query(task, 5).unwrap_or_default();
    entries
        .into_iter()
        .filter(|e| e.kind != roko_neuro::KnowledgeKind::AntiKnowledge)
        .filter(|e| e.confidence >= 0.3)
        .map(|e| e.content)
        .collect()
}

/// Query knowledge store for anti-knowledge warnings.
fn query_anti_patterns(&self, task: &str) -> Vec<String> {
    let Some(store) = self.knowledge_store.as_ref() else {
        return Vec::new();
    };
    let entries = store.query(&format!("{task} warning anti-pattern"), 5).unwrap_or_default();
    entries
        .into_iter()
        .filter(|e| e.kind == roko_neuro::KnowledgeKind::AntiKnowledge)
        .filter(|e| e.confidence >= 0.2)
        .map(|e| format!("WARNING: {}", e.content))
        .collect()
}
```

### 5. Track which knowledge entries were used

Add a field to record which knowledge entry IDs were consulted:

```rust

/// Cleared on each `assemble()` call.

```

Add a public accessor:

```rust
/// Return the knowledge entry IDs used in the most recent assembly.
pub fn last_knowledge_ids(&self) -> Vec<String> {
```
