### Step 3 — Create CFactorSparkline.tsx

Create `demo/demo-app/src/components/Charts/CFactorSparkline.tsx`:

```tsx
import { useRef, useEffect, useCallback } from 'react';
import './Charts.css';

interface TrendBucket {
  start: string;
  samples: number;
  avg: number;
  p50: number;
  p95: number;
}

interface CFactorSparklineProps {
  trend: TrendBucket[];
  woolley?: Record<string, number[]>;
  height?: number;
}

const WOOLLEY_KEYS: { key: string; label: string; color: string }[] = [
  { key: 'turn_taking_equality',  label: 'Turn-Taking',   color: '#C8B890' },
  { key: 'social_perceptiveness', label: 'Perception',    color: '#AA7088' },
  { key: 'citation_reciprocity',  label: 'Reciprocity',   color: '#8A9C86' },
  { key: 'delivery_rate',         label: 'Delivery',      color: '#D8A878' },
  { key: 'hdc_diversity',         label: 'HDC Diversity',  color: '#9A8AB8' },
];

/** Draws a mini sparkline into a sub-region of the canvas. */
function drawSparkline(
  ctx: CanvasRenderingContext2D,
  values: number[],
  x: number, y: number, w: number, h: number,
  color: string, label: string,
) {
  if (values.length < 2) return;
  const min = Math.min(...values) - 0.01;
  const max = Math.max(...values) + 0.01;
  const range = max - min || 1;

  // Label
  ctx.fillStyle = '#6a5a68';
  ctx.font = '9px "JetBrains Mono", monospace';
  ctx.textAlign = 'left';
  ctx.fillText(label, x, y - 4);

  // Current value
  ctx.textAlign = 'right';
  ctx.fillStyle = color;
  ctx.fillText(values[values.length - 1].toFixed(3), x + w, y - 4);

  // Line
  ctx.beginPath();
  ctx.strokeStyle = color;
  ctx.lineWidth = 1.5;
  ctx.lineJoin = 'round';
  for (let i = 0; i < values.length; i++) {
    const px = x + (i / (values.length - 1)) * w;
    const py = y + h - ((values[i] - min) / range) * h;
    if (i === 0) ctx.moveTo(px, py);
    else ctx.lineTo(px, py);
  }
  ctx.stroke();
}

/** C-Factor trend sparkline with overall line + 5 Woolley mini-sparklines. */
export default function CFactorSparkline({ trend, woolley, height = 320 }: CFactorSparklineProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas || trend.length === 0) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    const rect = canvas.getBoundingClientRect();
    canvas.width = rect.width * dpr;
    canvas.height = rect.height * dpr;
    ctx.scale(dpr, dpr);

    const w = rect.width;
    const h = rect.height;
    const pad = { top: 28, right: 16, bottom: 12, left: 52 };

    ctx.clearRect(0, 0, w, h);

    // Title
    ctx.fillStyle = '#8a7a88';
    ctx.font = '11px "General Sans", sans-serif';
    ctx.fillText('C-FACTOR TREND (24h)', pad.left, 16);

    // --- Main trend area: top 45% ---
    const mainH = (h - pad.top - pad.bottom) * 0.40;
    const plotW = w - pad.left - pad.right;
    const avgValues = trend.map((b) => b.avg);
    const p95Values = trend.map((b) => b.p95);
    const allVals = [...avgValues, ...p95Values];
    const minVal = Math.min(...allVals) - 0.01;
    const maxVal = Math.max(...allVals) + 0.01;
    const range = maxVal - minVal || 1;

    // Grid lines for main chart
    ctx.strokeStyle = 'rgba(255,255,255,0.05)';
    ctx.lineWidth = 1;
    for (let i = 0; i <= 3; i++) {
      const yy = pad.top + mainH * (1 - i / 3);
      ctx.beginPath();
      ctx.moveTo(pad.left, yy);
      ctx.lineTo(pad.left + plotW, yy);
      ctx.stroke();

      ctx.fillStyle = '#6a5a68';
      ctx.font = '9px "JetBrains Mono", monospace';
      ctx.textAlign = 'right';
      ctx.fillText((minVal + (range * i) / 3).toFixed(3), pad.left - 6, yy + 3);
    }

    // p95 fill
    ctx.beginPath();
    for (let i = 0; i < p95Values.length; i++) {
      const x = pad.left + (i / (p95Values.length - 1)) * plotW;
      const y = pad.top + mainH * (1 - (p95Values[i] - minVal) / range);
      if (i === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    for (let i = avgValues.length - 1; i >= 0; i--) {
      const x = pad.left + (i / (avgValues.length - 1)) * plotW;
      const y = pad.top + mainH * (1 - (avgValues[i] - minVal) / range);
      ctx.lineTo(x, y);
    }
    ctx.closePath();
    ctx.fillStyle = 'rgba(220,165,189,0.08)';
    ctx.fill();

    // p95 line (dimmer)
    ctx.beginPath();
    ctx.strokeStyle = 'rgba(220,165,189,0.35)';
    ctx.lineWidth = 1;
    ctx.setLineDash([4, 3]);
    for (let i = 0; i < p95Values.length; i++) {
      const x = pad.left + (i / (p95Values.length - 1)) * plotW;
      const y = pad.top + mainH * (1 - (p95Values[i] - minVal) / range);
      if (i === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.stroke();
    ctx.setLineDash([]);

    // avg line (bright)
    ctx.beginPath();
    ctx.strokeStyle = '#DCA5BD';
    ctx.lineWidth = 2;
    ctx.lineJoin = 'round';
    for (let i = 0; i < avgValues.length; i++) {
      const x = pad.left + (i / (avgValues.length - 1)) * plotW;
      const y = pad.top + mainH * (1 - (avgValues[i] - minVal) / range);
      if (i === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.stroke();

    // Endpoint dot
    const lastX = pad.left + plotW;
    const lastY = pad.top + mainH * (1 - (avgValues[avgValues.length - 1] - minVal) / range);
    ctx.beginPath();
    ctx.arc(lastX, lastY, 3, 0, Math.PI * 2);
    ctx.fillStyle = '#DCA5BD';
    ctx.fill();

    // --- Woolley mini-sparklines: bottom 55% ---
    if (woolley) {
      const sparkTop = pad.top + mainH + 24;
      const sparkH = 20;
      const sparkGap = 10;

      WOOLLEY_KEYS.forEach((wk, i) => {
        const vals = woolley[wk.key];
        if (!vals || vals.length < 2) return;
        const sy = sparkTop + i * (sparkH + sparkGap + 14);
        drawSparkline(ctx, vals, pad.left, sy, plotW, sparkH, wk.color, wk.label);
      });
    }
  }, [trend, woolley]);

  useEffect(() => {
    draw();
    const ro = new ResizeObserver(draw);
    if (canvasRef.current) ro.observe(canvasRef.current);
    return () => ro.disconnect();
  }, [draw]);

  return (
    <div className="chart-container" style={{ height }}>
      <canvas ref={canvasRef} className="chart-canvas" />
    </div>
  );
}
```
