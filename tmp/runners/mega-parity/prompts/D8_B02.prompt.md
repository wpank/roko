```tsx
/* ── Topology types ─────────────────────────────────────── */

interface TopoNode { agent_id: string; role: string; endpoints: string[] }
interface TopoEdge { from: string; to: string; weight?: number }
interface TopoData { nodes: TopoNode[]; edges: TopoEdge[] }

const ROLE_COLORS: Record<string, string> = {
  implementer: '#C8B890',
  researcher:  '#9A8AB8',
  reviewer:    '#8A9C86',
  planner:     '#D8A878',
  auditor:     '#AA7088',
};

/* ── Force-directed graph canvas ────────────────────────── */

function TopologyGraph({ data, height = 280 }: { data: TopoData; height?: number }) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const simRef = useRef<{
    positions: { x: number; y: number; vx: number; vy: number }[];
    frame: number;
  } | null>(null);

  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas || data.nodes.length === 0) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    const rect = canvas.getBoundingClientRect();
    canvas.width = rect.width * dpr;
    canvas.height = rect.height * dpr;
    ctx.scale(dpr, dpr);

    const w = rect.width;
    const h = rect.height;
    const cx = w / 2;
    const cy = h / 2;

    // Initialize positions in a circle if not yet done
    if (!simRef.current || simRef.current.positions.length !== data.nodes.length) {
      const positions = data.nodes.map((_, i) => {
        const angle = (i / data.nodes.length) * Math.PI * 2;
        const r = Math.min(w, h) * 0.28;
        return { x: cx + Math.cos(angle) * r, y: cy + Math.sin(angle) * r, vx: 0, vy: 0 };
      });
      simRef.current = { positions, frame: 0 };
    }

    const sim = simRef.current;
    const pos = sim.positions;
    const nodeMap: Record<string, number> = {};
    data.nodes.forEach((n, i) => { nodeMap[n.agent_id] = i; });

    // Run 3 simulation ticks per frame for settling speed
    for (let tick = 0; tick < 3; tick++) {
      const damping = 0.85;
      const springK = 0.008;
      const springLen = Math.min(w, h) * 0.22;
      const chargeK = 2000;
      const centerPull = 0.002;

      // Reset forces
      for (const p of pos) { p.vx *= damping; p.vy *= damping; }

      // Charge repulsion (all pairs)
      for (let i = 0; i < pos.length; i++) {
        for (let j = i + 1; j < pos.length; j++) {
          let dx = pos[j].x - pos[i].x;
          let dy = pos[j].y - pos[i].y;
          const dist = Math.sqrt(dx * dx + dy * dy) || 1;
          const force = chargeK / (dist * dist);
          dx = (dx / dist) * force;
          dy = (dy / dist) * force;
          pos[i].vx -= dx;
          pos[i].vy -= dy;
          pos[j].vx += dx;
          pos[j].vy += dy;
        }
      }

      // Spring attraction (edges)
      for (const edge of data.edges) {
        const si = nodeMap[edge.from];
        const ti = nodeMap[edge.to];
        if (si === undefined || ti === undefined) continue;
        const dx = pos[ti].x - pos[si].x;
        const dy = pos[ti].y - pos[si].y;
        const dist = Math.sqrt(dx * dx + dy * dy) || 1;
        const displacement = dist - springLen;
        const fx = (dx / dist) * displacement * springK;
        const fy = (dy / dist) * displacement * springK;
        pos[si].vx += fx;
        pos[si].vy += fy;
        pos[ti].vx -= fx;
        pos[ti].vy -= fy;
      }

      // Center gravity
      for (const p of pos) {
        p.vx += (cx - p.x) * centerPull;
        p.vy += (cy - p.y) * centerPull;
      }

      // Apply velocities with boundary clamping
      for (const p of pos) {
        p.x = Math.max(30, Math.min(w - 30, p.x + p.vx));
        p.y = Math.max(30, Math.min(h - 30, p.y + p.vy));
      }
    }

    // --- Render ---
    ctx.clearRect(0, 0, w, h);

    // Edges
    for (const edge of data.edges) {
      const si = nodeMap[edge.from];
      const ti = nodeMap[edge.to];
      if (si === undefined || ti === undefined) continue;
      ctx.beginPath();
      ctx.moveTo(pos[si].x, pos[si].y);
      ctx.lineTo(pos[ti].x, pos[ti].y);
      ctx.strokeStyle = `rgba(200,184,144,${0.08 + (edge.weight ?? 1) * 0.04})`;
      ctx.lineWidth = 0.5 + (edge.weight ?? 1) * 0.3;
      ctx.stroke();
    }

    // Nodes
    data.nodes.forEach((node, i) => {
      const p = pos[i];
      const color = ROLE_COLORS[node.role] ?? '#AA7088';
      const radius = 14;

      // Glow
      ctx.beginPath();
      ctx.arc(p.x, p.y, radius + 6, 0, Math.PI * 2);
      ctx.fillStyle = color.replace(')', ',0.08)').replace('rgb', 'rgba').replace('#', '');
      // Use hex-to-rgba for glow
      const r = parseInt(color.slice(1, 3), 16);
      const g = parseInt(color.slice(3, 5), 16);
      const b = parseInt(color.slice(5, 7), 16);
      ctx.fillStyle = `rgba(${r},${g},${b},0.1)`;
      ctx.fill();

      // Circle
      ctx.beginPath();
      ctx.arc(p.x, p.y, radius, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(${r},${g},${b},0.25)`;
      ctx.fill();
      ctx.strokeStyle = color;
      ctx.lineWidth = 1.5;
      ctx.stroke();

      // Label
      const label = data.nodes[i].agent_id;
      ctx.fillStyle = '#c4b4c4';
      ctx.font = '9px "JetBrains Mono", monospace';
      ctx.textAlign = 'center';
      ctx.fillText(label, p.x, p.y + radius + 14);

      // Role label
      ctx.fillStyle = color;
      ctx.font = '8px "JetBrains Mono", monospace';
      ctx.fillText(node.role, p.x, p.y + 3);
    });

    sim.frame++;
    // Animate for first 120 frames to let the layout settle
    if (sim.frame < 120) {
      requestAnimationFrame(draw);
    }
  }, [data]);

  useEffect(() => {
    simRef.current = null; // Reset simulation on data change
    draw();
    const ro = new ResizeObserver(() => {
      simRef.current = null;
      draw();
    });
    if (canvasRef.current) ro.observe(canvasRef.current);
    return () => ro.disconnect();
  }, [draw]);

  return (
    <div style={{ position: 'relative', width: '100%', height, overflow: 'hidden' }}>
      <canvas ref={canvasRef} style={{ width: '100%', height: '100%', display: 'block' }} />
    </div>
  );
}
```
