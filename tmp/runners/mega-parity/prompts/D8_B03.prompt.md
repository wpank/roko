We need a waterfall timeline that shows how gate rungs execute sequentially for each task: compile(0), clippy(1), test(2), diff(3), fmt(4), custom(5), judge(6). ### Step 3 — Create GateWaterfall.tsx

Create `demo/demo-app/src/components/GateWaterfall.tsx`:

```tsx
import { useRef, useEffect, useCallback } from 'react';

interface Rung {
  name: string;
  rung: number;
  status: 'passed' | 'failed' | 'skipped';
  duration_ms: number;
}

interface GateRun {
  task_id: string;
  timestamp: string;
  rungs: Rung[];
}

interface GateWaterfallProps {
  runs: GateRun[];
  height?: number;
}

const STATUS_COLORS = {
  passed:  '#8A9C86',
  failed:  '#CC90A8',
  skipped: 'rgba(255,255,255,0.06)',
};

const STATUS_BORDER = {
  passed:  'rgba(138,156,134,0.6)',
  failed:  'rgba(204,144,168,0.6)',
  skipped: 'rgba(255,255,255,0.08)',
};

/** Gate waterfall: each row = one task run, bars = rungs left-to-right proportional to duration. */
export default function GateWaterfall({ runs, height = 360 }: GateWaterfallProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas || runs.length === 0) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    const rect = canvas.getBoundingClientRect();
    canvas.width = rect.width * dpr;
    canvas.height = rect.height * dpr;
    ctx.scale(dpr, dpr);

    const w = rect.width;
    const h = rect.height;
    const pad = { top: 32, right: 16, bottom: 12, left: 130 };
    const plotW = w - pad.left - pad.right;
    const rowH = Math.min(32, (h - pad.top - pad.bottom) / runs.length - 4);

    ctx.clearRect(0, 0, w, h);

    // Find max total duration across all runs for scaling
    const maxTotal = Math.max(
      ...runs.map((r) => r.rungs.reduce((s, rg) => s + rg.duration_ms, 0)),
      1
    );

    // Header: rung names
    ctx.fillStyle = '#8a7a88';
    ctx.font = '11px "General Sans", sans-serif';
    ctx.textAlign = 'left';
    ctx.fillText('GATE WATERFALL', pad.left, 16);

    // Time scale labels
    ctx.fillStyle = '#6a5a68';
    ctx.font = '8px "JetBrains Mono", monospace';
    ctx.textAlign = 'center';
    for (let i = 0; i <= 4; i++) {
      const x = pad.left + (i / 4) * plotW;
      const ms = (maxTotal * i) / 4;
      ctx.fillText(ms >= 1000 ? `${(ms / 1000).toFixed(1)}s` : `${Math.round(ms)}ms`, x, pad.top - 6);

      // Vertical grid line
      ctx.strokeStyle = 'rgba(255,255,255,0.04)';
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.moveTo(x, pad.top);
      ctx.lineTo(x, pad.top + runs.length * (rowH + 4));
      ctx.stroke();
    }

    // Rows
    runs.forEach((run, ri) => {
      const y = pad.top + ri * (rowH + 4);

      // Task label (truncated)
      ctx.fillStyle = '#8a7a88';
      ctx.font = '9px "JetBrains Mono", monospace';
      ctx.textAlign = 'right';
      const label = run.task_id.length > 18 ? run.task_id.slice(0, 18) + '...' : run.task_id;
      ctx.fillText(label, pad.left - 8, y + rowH / 2 + 3);

      // Row background
      ctx.fillStyle = 'rgba(255,255,255,0.02)';
      ctx.beginPath();
      ctx.roundRect(pad.left, y, plotW, rowH, 3);
      ctx.fill();

      // Rung bars (waterfall: each bar starts where the previous ended)
      let xOffset = 0;
      for (const rung of run.rungs) {
        const barW = (rung.duration_ms / maxTotal) * plotW;
        if (barW < 1 && rung.status === 'skipped') {
          xOffset += barW;
          continue;
        }

        const drawW = Math.max(barW, rung.status !== 'skipped' ? 3 : 0);
        const bx = pad.left + (xOffset / maxTotal) * plotW;

        // Bar fill
        ctx.fillStyle = STATUS_COLORS[rung.status];
        ctx.beginPath();
        ctx.roundRect(bx, y + 1, drawW, rowH - 2, 2);
        ctx.fill();

        // Bar border
        ctx.strokeStyle = STATUS_BORDER[rung.status];
        ctx.lineWidth = 0.5;
        ctx.stroke();

        // Rung label inside bar if wide enough
        if (drawW > 30) {
          ctx.fillStyle = rung.status === 'skipped' ? '#6a5a68' : '#1a1418';
          ctx.font = '8px "JetBrains Mono", monospace';
          ctx.textAlign = 'left';
          ctx.fillText(rung.name, bx + 4, y + rowH / 2 + 3);
        }

        xOffset += rung.duration_ms;
      }

      // Total duration label
      const totalMs = run.rungs.reduce((s, rg) => s + rg.duration_ms, 0);
      const totalLabel = totalMs >= 1000 ? `${(totalMs / 1000).toFixed(1)}s` : `${totalMs}ms`;
      const totalX = pad.left + (xOffset / maxTotal) * plotW + 6;
      ctx.fillStyle = '#6a5a68';
      ctx.font = '9px "JetBrains Mono", monospace';
      ctx.textAlign = 'left';
      if (totalX + 40 < w) {
        ctx.fillText(totalLabel, totalX, y + rowH / 2 + 3);
      }
    });

    // Legend at bottom
    const legendY = pad.top + runs.length * (rowH + 4) + 16;
    const legends = [
      { label: 'passed', color: STATUS_COLORS.passed },
      { label: 'failed', color: STATUS_COLORS.failed },
      { label: 'skipped', color: 'rgba(255,255,255,0.12)' },
    ];
    let lx = pad.left;
    for (const leg of legends) {
      ctx.fillStyle = leg.color;
      ctx.fillRect(lx, legendY, 10, 10);
      ctx.strokeStyle = 'rgba(255,255,255,0.1)';
      ctx.lineWidth = 0.5;
      ctx.strokeRect(lx, legendY, 10, 10);
      ctx.fillStyle = '#8a7a88';
      ctx.font = '9px "JetBrains Mono", monospace';
      ctx.textAlign = 'left';
      ctx.fillText(leg.label, lx + 14, legendY + 9);
      lx += 70;
    }
  }, [runs]);

  useEffect(() => {
    draw();
    const ro = new ResizeObserver(draw);
    if (canvasRef.current) ro.observe(canvasRef.current);
    return () => ro.disconnect();
  }, [draw]);

  return (
    <div style={{ position: 'relative', width: '100%', height, overflow: 'hidden' }}>
      <canvas ref={canvasRef} style={{ width: '100%', height: '100%', display: 'block' }} />
    </div>
  );
}
```
