## Problem
Roko has a dream consolidation system (crate `roko-dreams`) that processes episodes offline through 4 phases: Hypnagogia, NREM, REM, and Integration. There is no visualization of this process in the demo app. We need a component showing the 4 phases as styled cards with per-phase Canvas 2D sparklines, displaying metrics like episodes processed, clusters formed, knowledge entries written, and playbooks created.

## Architecture Contract
- All charts MUST use Canvas 2D with DPR scaling (match CostChart.tsx pattern exactly: useRef, useCallback, DPR scaling, ResizeObserver)
- All API calls MUST use useApiWithFallback with demo data fallback
- NO external visualization libraries (no d3, recharts, chart.js)
- Components must render without a live server (demo data fallback)
- Follow existing project conventions (functional components, hooks, CSS modules)

### Step 3 — Create DreamPhaseViz.tsx

Create `demo/demo-app/src/components/DreamPhaseViz.tsx`:

```tsx
import { useRef, useEffect, useCallback, useState } from 'react';
import { useApiWithFallback } from '../hooks/useApiWithFallback';

/* ── Types ───────────────────────────────────────────────── */

interface Phase {
  name: string;
  status: string;
  episodes_processed: number;
  clusters_formed: number;
  knowledge_entries_written: number;
  playbooks_created: number;
  duration_secs: number;
  trend: number[];
}

interface DreamJournal {
  last_cycle: string;
  cycle_count: number;
  phases: Phase[];
}

/* ── Phase palette ───────────────────────────────────────── */

const PHASE_STYLES: Record<string, { color: string; bg: string; border: string; glow: string }> = {
  Hypnagogia:  { color: '#9A8AB8', bg: 'rgba(154,138,184,0.06)', border: 'rgba(154,138,184,0.15)', glow: 'rgba(154,138,184,0.5)' },
  NREM:        { color: '#7A8AA8', bg: 'rgba(122,138,168,0.06)', border: 'rgba(122,138,168,0.15)', glow: 'rgba(122,138,168,0.5)' },
  REM:         { color: '#CC90A8', bg: 'rgba(204,144,168,0.06)', border: 'rgba(204,144,168,0.15)', glow: 'rgba(204,144,168,0.5)' },
  Integration: { color: '#8A9C86', bg: 'rgba(138,156,134,0.06)', border: 'rgba(138,156,134,0.15)', glow: 'rgba(138,156,134,0.5)' },
};

/* ── Mini sparkline (Canvas 2D, per-phase card) ──────────── */

function PhaseSparkline({ data, color, width = 120, height = 32 }: {
  data: number[];
  color: string;
  width?: number;
  height?: number;
}) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas || data.length < 2) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    canvas.width = width * dpr;
    canvas.height = height * dpr;
    ctx.scale(dpr, dpr);

    ctx.clearRect(0, 0, width, height);

    const min = Math.min(...data);
    const max = Math.max(...data);
    const range = max - min || 1;
    const pad = 2;
    const plotW = width - pad * 2;
    const plotH = height - pad * 2;

    // Fill area
    ctx.beginPath();
    for (let i = 0; i < data.length; i++) {
      const x = pad + (i / (data.length - 1)) * plotW;
      const y = pad + plotH * (1 - (data[i] - min) / range);
      if (i === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.lineTo(pad + plotW, pad + plotH);
    ctx.lineTo(pad, pad + plotH);
    ctx.closePath();

    // Parse hex color for rgba
    const r = parseInt(color.slice(1, 3), 16);
    const g = parseInt(color.slice(3, 5), 16);
    const b = parseInt(color.slice(5, 7), 16);
    ctx.fillStyle = `rgba(${r},${g},${b},0.12)`;
    ctx.fill();

    // Line
    ctx.beginPath();
    ctx.strokeStyle = color;
    ctx.lineWidth = 1.5;
    ctx.lineJoin = 'round';
    for (let i = 0; i < data.length; i++) {
      const x = pad + (i / (data.length - 1)) * plotW;
      const y = pad + plotH * (1 - (data[i] - min) / range);
      if (i === 0) ctx.moveTo(x, y);
      else ctx.lineTo(x, y);
    }
    ctx.stroke();

    // Endpoint dot
    const lastX = pad + plotW;
    const lastY = pad + plotH * (1 - (data[data.length - 1] - min) / range);
    ctx.beginPath();
    ctx.arc(lastX, lastY, 2, 0, Math.PI * 2);
    ctx.fillStyle = color;
    ctx.fill();
  }, [data, color, width, height]);

  useEffect(() => {
    draw();
  }, [draw]);

  return (
    <canvas
      ref={canvasRef}
      style={{ width, height, display: 'block' }}
    />
  );
}

/* ── Stat row helper ─────────────────────────────────────── */

function PhaseStat({ label, value }: { label: string; value: string | number }) {
  return (
    <div style={{ display: 'flex', justifyContent: 'space-between', padding: '3px 0' }}>
      <span style={{
        fontFamily: 'var(--mono)',
        fontSize: '0.6rem',
        letterSpacing: '.06em',
        color: 'var(--text-dim)',
      }}>
        {label}
      </span>
      <span style={{
        fontFamily: 'var(--mono)',
        fontSize: '0.7rem',
        fontWeight: 500,
        color: 'var(--text-primary)',
      }}>
        {value}
      </span>
    </div>
  );
}

/* ── Arrow connector between phase cards ─────────────────── */

function PhaseArrow({ color }: { color: string }) {
  return (
    <div style={{
      display: 'flex',
      alignItems: 'center',
      justifyContent: 'center',
      minWidth: 24,
      color,
      fontSize: 16,
      opacity: 0.5,
    }}>
      &#x2192;
    </div>
  );
}

/* ── Main component ──────────────────────────────────────── */

export default function DreamPhaseViz() {
  const { get } = useApiWithFallback();
  const [journal, setJournal] = useState<DreamJournal | null>(null);

  useEffect(() => {
    get<DreamJournal>('/api/dream/journal').then(setJournal).catch(() => {});
  }, [get]);

  const phases = journal?.phases ?? [];
  const cycleCount = journal?.cycle_count ?? 0;
  const lastCycle = journal?.last_cycle ?? '';

  return (
    <div style={{ display: 'flex', flexDirection: 'column', gap: 12 }}>
      {/* Header stats */}
      <div style={{
        display: 'flex',
        justifyContent: 'space-between',
        alignItems: 'baseline',
        padding: '0 4px',
      }}>
        <span style={{
          fontFamily: 'var(--mono)',
          fontSize: '0.6rem',
          letterSpacing: '.1em',
          textTransform: 'uppercase',
          color: 'var(--text-dim)',
        }}>
          {cycleCount} cycles completed
        </span>
        {lastCycle && (
          <span style={{
            fontFamily: 'var(--mono)',
            fontSize: '0.6rem',
            color: 'var(--text-ghost)',
          }}>
            last: {new Date(lastCycle).toLocaleString()}
          </span>
        )}
      </div>

      {/* Phase cards row */}
      <div style={{
        display: 'flex',
        gap: 0,
        alignItems: 'stretch',
        overflowX: 'auto',
      }}>
        {phases.map((phase, i) => {
          const style = PHASE_STYLES[phase.name] ?? PHASE_STYLES.Hypnagogia;
          return (
            <div key={phase.name} style={{ display: 'flex', alignItems: 'center' }}>
              {/* Phase card */}
              <div style={{
                background: style.bg,
                border: `1px solid ${style.border}`,
                borderRadius: 10,
                padding: '14px 16px',
                minWidth: 180,
                flex: 1,
                display: 'flex',
                flexDirection: 'column',
                gap: 10,
                transition: 'box-shadow .3s ease, transform .25s ease',
              }}>
                {/* Phase name + status dot */}
                <div style={{ display: 'flex', alignItems: 'center', gap: 8 }}>
                  <span style={{
                    width: 6,
                    height: 6,
                    borderRadius: '50%',
                    background: phase.status === 'completed' ? style.color : 'var(--text-ghost)',
                    boxShadow: phase.status === 'completed' ? `0 0 6px ${style.glow}` : 'none',
                  }} />
                  <span style={{
                    fontFamily: 'var(--display)',
                    fontSize: 13,
                    fontWeight: 500,
                    color: style.color,
                    letterSpacing: '.02em',
                  }}>
                    {phase.name}
                  </span>
                </div>

                {/* Sparkline */}
                <PhaseSparkline data={phase.trend} color={style.color} width={148} height={28} />

                {/* Stats */}
                <div style={{
                  display: 'flex',
                  flexDirection: 'column',
                  gap: 2,
                  borderTop: '1px solid rgba(255,255,255,0.04)',
                  paddingTop: 8,
                }}>
                  <PhaseStat label="episodes" value={phase.episodes_processed} />
                  <PhaseStat label="clusters" value={phase.clusters_formed} />
                  <PhaseStat label="knowledge" value={phase.knowledge_entries_written} />
                  <PhaseStat label="playbooks" value={phase.playbooks_created} />
                  <PhaseStat label="duration" value={`${phase.duration_secs}s`} />
                </div>
              </div>

              {/* Arrow between cards (not after last) */}
              {i < phases.length - 1 && (
                <PhaseArrow color={style.color} />
              )}
            </div>
          );
        })}
      </div>
    </div>
  );
}
```
