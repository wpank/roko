## Problem
The adaptive gate threshold system adjusts pass-rate thresholds per rung based on EMA of historical results. There is no visual indicator of where each rung's current performance sits relative to its threshold. We need semi-circular gauge widgets that show the EMA pass rate, the threshold line, and green/amber/red zones for each of the 7 gate rungs.

### Step 3 — Create ThresholdGauge.tsx

Create `demo/demo-app/src/components/ThresholdGauge.tsx`:

```tsx
import { useRef, useEffect, useCallback } from 'react';

interface RungThreshold {
  mean_pass_rate: number;
  ema_threshold: number;
  rung_count: number;
  consecutive_passes: number;
}

interface ThresholdGaugeProps {
  rung: string;
  data: RungThreshold;
  size?: number;
}

/**
 * Semi-circular gauge for a single gate rung.
 * Arc sweeps from 180deg (left) to 0deg (right) = PI to 0.
 * Green zone: > threshold, Amber zone: threshold-0.05..threshold, Red zone: < threshold-0.05
 */
export default function ThresholdGauge({ rung, data, size = 120 }: ThresholdGaugeProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);

  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    canvas.width = size * dpr;
    canvas.height = (size * 0.65) * dpr;
    ctx.scale(dpr, dpr);

    const w = size;
    const h = size * 0.65;
    const cx = w / 2;
    const cy = h - 8;
    const radius = w / 2 - 12;
    const lineW = 8;

    ctx.clearRect(0, 0, w, h);

    // Arc goes from PI (left) to 0 (right) = 180 degrees
    const startAngle = Math.PI;
    const endAngle = 0;
    const totalSweep = Math.PI;

    // Zone boundaries (angles)
    const threshAngle = startAngle - data.ema_threshold * totalSweep;
    const amberAngle = startAngle - (data.ema_threshold - 0.05) * totalSweep;

    // Background track
    ctx.beginPath();
    ctx.arc(cx, cy, radius, startAngle, endAngle, false);
    ctx.strokeStyle = 'rgba(255,255,255,0.06)';
    ctx.lineWidth = lineW;
    ctx.lineCap = 'round';
    ctx.stroke();

    // Red zone: 0% to (threshold - 5%)
    ctx.beginPath();
    ctx.arc(cx, cy, radius, startAngle, amberAngle, false);
    ctx.strokeStyle = 'rgba(204,144,168,0.2)';
    ctx.lineWidth = lineW;
    ctx.lineCap = 'butt';
    ctx.stroke();

    // Amber zone: (threshold - 5%) to threshold
    ctx.beginPath();
    ctx.arc(cx, cy, radius, amberAngle, threshAngle, false);
    ctx.strokeStyle = 'rgba(216,168,120,0.25)';
    ctx.lineWidth = lineW;
    ctx.stroke();

    // Green zone: threshold to 100%
    ctx.beginPath();
    ctx.arc(cx, cy, radius, threshAngle, endAngle, false);
    ctx.strokeStyle = 'rgba(138,156,134,0.25)';
    ctx.lineWidth = lineW;
    ctx.stroke();

    // Value arc (bright overlay)
    const valueAngle = startAngle - Math.min(data.mean_pass_rate, 1) * totalSweep;
    const valueColor = data.mean_pass_rate >= data.ema_threshold
      ? '#8A9C86'
      : data.mean_pass_rate >= data.ema_threshold - 0.05
        ? '#D8A878'
        : '#CC90A8';

    ctx.beginPath();
    ctx.arc(cx, cy, radius, startAngle, valueAngle, false);
    ctx.strokeStyle = valueColor;
    ctx.lineWidth = lineW;
    ctx.lineCap = 'round';
    ctx.stroke();

    // Glow on the value arc endpoint
    const endX = cx + Math.cos(valueAngle) * radius;
    const endY = cy + Math.sin(valueAngle) * radius;
    ctx.beginPath();
    ctx.arc(endX, endY, lineW / 2 + 2, 0, Math.PI * 2);
    // Parse hex for rgba
    const vr = parseInt(valueColor.slice(1, 3), 16);
    const vg = parseInt(valueColor.slice(3, 5), 16);
    const vb = parseInt(valueColor.slice(5, 7), 16);
    ctx.fillStyle = `rgba(${vr},${vg},${vb},0.3)`;
    ctx.fill();

    // Threshold marker line
    const thX1 = cx + Math.cos(threshAngle) * (radius - lineW / 2 - 2);
    const thY1 = cy + Math.sin(threshAngle) * (radius - lineW / 2 - 2);
    const thX2 = cx + Math.cos(threshAngle) * (radius + lineW / 2 + 2);
    const thY2 = cy + Math.sin(threshAngle) * (radius + lineW / 2 + 2);
    ctx.beginPath();
    ctx.moveTo(thX1, thY1);
    ctx.lineTo(thX2, thY2);
    ctx.strokeStyle = 'rgba(255,255,255,0.5)';
    ctx.lineWidth = 1.5;
    ctx.stroke();

    // Center value text
    ctx.fillStyle = valueColor;
    ctx.font = `bold 16px "JetBrains Mono", monospace`;
    ctx.textAlign = 'center';
    ctx.textBaseline = 'bottom';
    ctx.fillText(`${(data.mean_pass_rate * 100).toFixed(1)}%`, cx, cy - 6);

    // Rung name
    ctx.fillStyle = '#8a7a88';
    ctx.font = '9px "JetBrains Mono", monospace';
    ctx.textBaseline = 'top';
    ctx.fillText(rung.toUpperCase(), cx, cy + 2);

    // Threshold label (small)
    ctx.fillStyle = '#6a5a68';
    ctx.font = '7px "JetBrains Mono", monospace';
    ctx.textBaseline = 'top';
    ctx.fillText(`thr: ${(data.ema_threshold * 100).toFixed(0)}%`, cx, cy + 14);
  }, [rung, data, size]);

  useEffect(() => {
    draw();
  }, [draw]);

  return (
    <div style={{
      display: 'flex',
      flexDirection: 'column',
      alignItems: 'center',
      gap: 4,
    }}>
      <canvas
        ref={canvasRef}
        style={{ width: size, height: size * 0.65, display: 'block' }}
      />
      <div style={{
        display: 'flex',
        gap: 8,
        justifyContent: 'center',
      }}>
        <span style={{
          fontFamily: 'var(--mono)',
          fontSize: '0.55rem',
          color: 'var(--text-ghost)',
        }}>
          {data.rung_count} runs
        </span>
        <span style={{
          fontFamily: 'var(--mono)',
          fontSize: '0.55rem',
          color: data.consecutive_passes >= 10 ? 'var(--success)' : 'var(--text-ghost)',
        }}>
          {data.consecutive_passes} streak
        </span>
      </div>
    </div>
  );
}
```

### Step 4 — Create ThresholdGaugeRow wrapper

Add a helper component for rendering all 7 gauges in a row. Add this to the same file, or create a small wrapper. For simplicity, add it to `ThresholdGauge.tsx`:

```tsx
export function ThresholdGaugeRow({ thresholds }: {
  thresholds: Record<string, RungThreshold>;
}) {
  const rungs = ['compile', 'clippy', 'test', 'diff', 'fmt', 'custom', 'judge'];
  return (
    <div style={{
      display: 'flex',
      flexWrap: 'wrap',
      gap: 8,
      justifyContent: 'center',
    }}>
      {rungs.map((rung) => {
        const data = thresholds[rung];
        if (!data) return null;
        return <ThresholdGauge key={rung} rung={rung} data={data} size={110} />;
      })}
    </div>
  );
}
```
