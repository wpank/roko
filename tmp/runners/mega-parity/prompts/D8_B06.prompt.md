### Step 3 — Create CostRace.tsx

Create `demo/demo-app/src/components/CostRace.tsx`:

```tsx
import { useRef, useEffect, useCallback, useState } from 'react';
import { SERVE_URL } from '../lib/serve-url';

/* ── Types ───────────────────────────────────────────────── */

interface ModelCost {
  model: string;
  cost_usd: number;
  tokens: number;
  tasks: number;
  color?: string;
}

interface CostRaceProps {
  /** Static data to display (used in demo/offline mode). */
  models?: ModelCost[];
  /** If true, subscribe to SSE for live updates. */
  live?: boolean;
  height?: number;
}

/* ── Default colors for models ───────────────────────────── */

const MODEL_COLORS: Record<string, string> = {
  'claude-sonnet-4': '#C8B890',
  'claude-haiku-3':  '#8A9C86',
  'claude-opus-4':   '#AA7088',
  'gpt-4o':          '#D8A878',
  'gemini-2.5-pro':  '#9A8AB8',
};

function getModelColor(model: string, explicit?: string): string {
  if (explicit) return explicit;
  for (const [key, color] of Object.entries(MODEL_COLORS)) {
    if (model.includes(key)) return color;
  }
  return '#AA7088';
}

/* ── Component ───────────────────────────────────────────── */

export default function CostRace({ models: initialModels, live = false, height = 260 }: CostRaceProps) {
  const canvasRef = useRef<HTMLCanvasElement>(null);
  const [models, setModels] = useState<ModelCost[]>(initialModels ?? []);
  const animRef = useRef<{ targets: number[]; current: number[]; frame: number }>({
    targets: [],
    current: [],
    frame: 0,
  });

  // Sync prop changes
  useEffect(() => {
    if (initialModels) setModels(initialModels);
  }, [initialModels]);

  // SSE subscription for live mode
  useEffect(() => {
    if (!live) return;

    let es: EventSource | null = null;
    try {
      es = new EventSource(`${SERVE_URL}/api/bench/events`);
      es.addEventListener('BenchTaskCompleted', (event) => {
        try {
          const data = JSON.parse(event.data);
          if (!data.model || data.cost_usd === undefined) return;
          setModels((prev) => {
            const idx = prev.findIndex((m) => m.model === data.model);
            if (idx >= 0) {
              const updated = [...prev];
              updated[idx] = {
                ...updated[idx],
                cost_usd: updated[idx].cost_usd + (data.cost_usd ?? 0),
                tokens: updated[idx].tokens + (data.tokens ?? 0),
                tasks: updated[idx].tasks + 1,
              };
              return updated;
            }
            return [...prev, {
              model: data.model,
              cost_usd: data.cost_usd ?? 0,
              tokens: data.tokens ?? 0,
              tasks: 1,
            }];
          });
        } catch { /* ignore parse errors */ }
      });
    } catch { /* SSE not available */ }

    return () => {
      if (es) es.close();
    };
  }, [live]);

  const draw = useCallback(() => {
    const canvas = canvasRef.current;
    if (!canvas || models.length === 0) return;

    const ctx = canvas.getContext('2d');
    if (!ctx) return;

    const dpr = window.devicePixelRatio || 1;
    const rect = canvas.getBoundingClientRect();
    canvas.width = rect.width * dpr;
    canvas.height = rect.height * dpr;
    ctx.scale(dpr, dpr);

    const w = rect.width;
    const h = rect.height;
    const pad = { top: 12, right: 80, bottom: 12, left: 120 };
    const plotW = w - pad.left - pad.right;

    // Sort by cost descending for ranking
    const sorted = [...models].sort((a, b) => b.cost_usd - a.cost_usd);
    const maxCost = Math.max(...sorted.map((m) => m.cost_usd), 0.01);
    const barH = Math.min(28, (h - pad.top - pad.bottom) / sorted.length - 6);
    const barGap = 6;

    // Animation: lerp current toward targets
    const anim = animRef.current;
    if (anim.targets.length !== sorted.length) {
      anim.targets = sorted.map((m) => m.cost_usd);
      anim.current = sorted.map((m) => m.cost_usd);
      anim.frame = 0;
    } else {
      anim.targets = sorted.map((m) => m.cost_usd);
      for (let i = 0; i < anim.current.length; i++) {
        anim.current[i] += (anim.targets[i] - anim.current[i]) * 0.12;
      }
    }

    ctx.clearRect(0, 0, w, h);

    sorted.forEach((m, i) => {
      const y = pad.top + i * (barH + barGap);
      const animCost = anim.current[i] ?? m.cost_usd;
      const barW = Math.max((animCost / maxCost) * plotW, 2);
      const color = getModelColor(m.model, m.color);

      // Parse hex for rgba
      const r = parseInt(color.slice(1, 3), 16);
      const g = parseInt(color.slice(3, 5), 16);
      const b = parseInt(color.slice(5, 7), 16);

      // Model label
      ctx.fillStyle = '#8a7a88';
      ctx.font = '10px "JetBrains Mono", monospace';
      ctx.textAlign = 'right';
      ctx.textBaseline = 'middle';
      const label = m.model.length > 16 ? m.model.slice(0, 16) : m.model;
      ctx.fillText(label, pad.left - 10, y + barH / 2);

      // Bar track background
      ctx.fillStyle = 'rgba(255,255,255,0.03)';
      ctx.beginPath();
      ctx.roundRect(pad.left, y, plotW, barH, 4);
      ctx.fill();

      // Bar fill with gradient
      const grad = ctx.createLinearGradient(pad.left, 0, pad.left + barW, 0);
      grad.addColorStop(0, `rgba(${r},${g},${b},0.4)`);
      grad.addColorStop(1, `rgba(${r},${g},${b},0.8)`);
      ctx.fillStyle = grad;
      ctx.beginPath();
      ctx.roundRect(pad.left, y, barW, barH, 4);
      ctx.fill();

      // Glow at the bar tip
      ctx.beginPath();
      ctx.arc(pad.left + barW, y + barH / 2, barH / 2, 0, Math.PI * 2);
      ctx.fillStyle = `rgba(${r},${g},${b},0.15)`;
      ctx.fill();

      // Cost label to the right
      ctx.fillStyle = color;
      ctx.font = 'bold 10px "JetBrains Mono", monospace';
      ctx.textAlign = 'left';
      ctx.textBaseline = 'middle';
      ctx.fillText(`$${m.cost_usd.toFixed(3)}`, pad.left + plotW + 8, y + barH / 2);

      // Tokens/tasks below the cost (smaller)
      ctx.fillStyle = '#6a5a68';
      ctx.font = '8px "JetBrains Mono", monospace';
      ctx.fillText(`${(m.tokens / 1000).toFixed(1)}K tok  ${m.tasks} tasks`, pad.left + plotW + 8, y + barH / 2 + 12);
    });

    // If animating, continue
    const needsAnim = anim.current.some((c, i) => Math.abs(c - anim.targets[i]) > 0.0001);
    if (needsAnim) {
      anim.frame++;
      requestAnimationFrame(draw);
    }
  }, [models]);

  useEffect(() => {
    animRef.current = { targets: [], current: [], frame: 0 };
    draw();
    const ro = new ResizeObserver(draw);
    if (canvasRef.current) ro.observe(canvasRef.current);
    return () => ro.disconnect();
  }, [draw]);

  return (
    <div style={{ position: 'relative', width: '100%', height, overflow: 'hidden' }}>
      <canvas ref={canvasRef} style={{ width: '100%', height: '100%', display: 'block' }} />
    </div>
  );
}
```
