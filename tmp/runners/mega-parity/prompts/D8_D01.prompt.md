### Step 1 — Define the scenario object
Add this scenario constant before the `SCENARIOS` array export:

```typescript
const providerRace: Scenario = {
  id: 'provider-race',
  title: 'Provider Race',
  subtitle: '4 providers race on the same prompt. First to pass gates wins.',
  panes: 4,
  labels: ['anthropic (haiku)', 'openai (gpt-4o)', 'gemini (flash)', 'moonshot (v1)'],
  panel: true,
  promptBar: false,
  steps: [
    { label: 'Setup', sublabel: 'init workspaces' },
    { label: 'Race start', sublabel: 'dispatch all 4' },
    { label: 'Live tracking', sublabel: 'cost + gates' },
    { label: 'Winner', sublabel: 'first to pass' },
    { label: 'Cost summary', sublabel: 'compare totals' },
  ],
  async run({ entries, playback, timeline, setMetric, setGate, logCommand }) {
    const providerNames = ['anthropic', 'openai', 'gemini', 'moonshot'];
    const providerModels = ['haiku', 'gpt-4o', 'flash', 'v1'];

    // Step 0: Setup — create workspaces
    timeline.init(this.steps);
    timeline.setActive(0);
    playback.setProgress(0, 5, 'initializing workspaces...');

    const dir = await setupWorkspace(entries[0], 'roko-race4');
    for (const e of entries.slice(1)) {
      await joinWorkspace(e, dir);
    }
    const ROKO = getRoko();
    setMetric('model', 'race mode');

    // Step 1: Race start — dispatch simultaneously
    await playback.waitForStep();
    timeline.setActive(1);
    const prompt = 'Build a Rust CLI that converts Celsius to Fahrenheit with tests';
    playback.setProgress(1, 5, 'dispatching to all providers...');
    logCommand('race start', `Dispatching "${prompt}" to ${providerNames.join(', ')} simultaneously.`);

    // Track individual costs per provider
    const costs: (string | null)[] = [null, null, null, null];
    const finishOrder: number[] = [];

    const trackers = entries.map((e, i) =>
      trackMetrics(e, {
        onCost: (c) => {
          costs[i] = c;
          setMetric(`cost`, costs.filter(Boolean).join(' / '));
        },
        onTokens: (t) => setMetric('tokens', `${providerNames[i]}: ${t}`),
      }),
    );

    // Step 2: Live tracking — all running in parallel
    timeline.setActive(2);
    playback.setProgress(2, 5, 'tracking live output...');

    const raceResults = await Promise.all(
      entries.map(async (e, i) => {
        const result = await showCmd(e, `${ROKO} run "${prompt}" --provider ${providerNames[i]}`, {
          timeout: 240000,
          customDesc: `Racing ${providerNames[i]} (${providerModels[i]}) against the field.`,
          onLog: logCommand,
          onGate: (name, status) => {
            setGate(`${providerNames[i]}:${name}`, status);
            if (status === 'pass' && !finishOrder.includes(i)) {
              finishOrder.push(i);
            }
          },
        });
        if (!finishOrder.includes(i)) finishOrder.push(i);
        return { provider: providerNames[i], ...result };
      }),
    );

    // Clean up trackers
    trackers.forEach((t) => clearInterval(t));

    // Step 3: Winner
    timeline.setActive(3);
    playback.setProgress(3, 5, 'determining winner...');
    const winner = finishOrder.length > 0
      ? raceResults[finishOrder[0]]
      : raceResults[0];
    setMetric('model', `winner: ${winner.provider}`);
    logCommand('winner', `${winner.provider} finished first in ${winner.elapsed.toFixed(1)}s — cost: ${winner.cost ?? 'unknown'}`);

    // Step 4: Cost summary
    await playback.waitForStep();
    timeline.setActive(4);
    playback.setProgress(4, 5, 'summarizing costs...');
    const summary = raceResults
      .map((r) => `${r.provider}: ${r.cost ?? '?'} (${r.elapsed.toFixed(1)}s)`)
      .join(' | ');
    setMetric('cost', summary);
    logCommand('summary', `Race results: ${summary}`);

    timeline.setActive(5);
  },
};
```
