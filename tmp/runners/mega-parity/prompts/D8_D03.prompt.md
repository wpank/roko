## Problem
Roko's dream consolidation cycle (hypnagogia, NREM clustering, REM synthesis, integration) is one of the most distinctive features — it processes episodes offline into durable knowledge. No existing scenario shows this cycle running. Users need to see dream phases progressing, knowledge being distilled, and the resulting stats afterwards.

## Architecture Contract
- Scenarios MUST implement the `Scenario` interface from `scenarios.ts`
- Use the existing terminal handle API (`execCmd`, `typeCmd`, `waitForPrompt`) via `showCmd`
- Use `playback.waitForStep()` for timing between steps
- All scenarios must work with both live server and simulated output
- Add the scenario to the `SCENARIOS` array export
- Use `setupWorkspace` / `joinWorkspace` from `useTerminalSession`
- 2-pane layout: left runs dream commands, right monitors knowledge/metrics

## Current State
The `explore` scenario (scenarios.ts ~line 562) runs `knowledge stats` and `knowledge query` but never triggers dream consolidation. The selfhost scenario runs a PRD pipeline but has no offline processing.

The dream CLI commands are:
- `roko knowledge dream run` — triggers a full dream cycle
- `roko knowledge dream report` — shows latest dream report
- `roko knowledge dream schedule` — shows/sets dream schedule

The `showCmd` helper returns `CommandResult` with `{ ok, elapsed, gates, cost, tokens }`. The `setMetric` and `setGate` context methods update the sidebar.

### Step 1 — Define the scenario object
Add this scenario constant before the `SCENARIOS` array export:

```typescript
const dreamConsolidation: Scenario = {
  id: 'dream-consolidation',
  title: 'Dream Cycle',
  subtitle: 'Offline consolidation — episodes distilled into durable knowledge.',
  panes: 2,
  labels: ['dream engine', 'knowledge monitor'],
  panel: true,
  promptBar: false,
  steps: [
    { label: 'Trigger check', sublabel: 'dream schedule' },
    { label: 'Seed episodes', sublabel: 'roko run' },
    { label: 'Hypnagogia', sublabel: 'replay selection' },
    { label: 'NREM clustering', sublabel: 'pattern extraction' },
    { label: 'REM synthesis', sublabel: 'creative linking' },
    { label: 'Integration', sublabel: 'knowledge merge' },
    { label: 'Report', sublabel: 'dream report' },
  ],
  async run({ entries, playback, timeline, setMetric, setGate, logCommand }) {
    const [dream, monitor] = entries;

    const dir = await setupWorkspace(dream, 'roko-dream');
    await joinWorkspace(monitor, dir);
    const ROKO = getRoko();

    timeline.init(this.steps);
    setMetric('model', 'dream engine');

    // Step 0: Trigger check — show dream schedule and pre-dream state
    await playback.waitForStep();
    timeline.setActive(0);
    playback.setProgress(1, 7, `${ROKO} knowledge dream schedule`);
    logCommand('dream schedule', 'Checking when the next dream cycle is due.');
    await showCmd(dream, `${ROKO} knowledge dream schedule`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Shows the dream consolidation schedule. Dreams are triggered by episode count thresholds or cron.',
    });
    await showCmd(monitor, `${ROKO} knowledge stats`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Pre-dream knowledge store state — baseline for measuring what the dream cycle adds.',
    });
    setGate('dream-ready', 'pending');

    // Step 1: Seed episodes — run a task to generate material for dreaming
    await playback.waitForStep();
    timeline.setActive(1);
    playback.setProgress(2, 7, `${ROKO} run "..."`);
    logCommand('seed episodes', 'Running a task to generate episodes that the dream cycle will consolidate.');
    await showCmd(dream, `${ROKO} run "Build a Rust function that validates email addresses with regex"`, {
      timeout: 180000,
      onLog: logCommand,
      customDesc: 'Generates agent episodes — turn-by-turn records of what the agent did, what worked, what failed. These are the raw material for dream consolidation.',
    });
    setMetric('cost', '$ seeded');
    setGate('episodes', 'pass');

    // Step 2: Hypnagogia — dream cycle begins
    await playback.waitForStep();
    timeline.setActive(2);
    playback.setProgress(3, 7, `${ROKO} knowledge dream run`);
    logCommand('dream run', 'Starting the full dream consolidation cycle. Phase 1: Hypnagogia — replaying and selecting salient episodes.');
    setGate('dream-ready', 'pass');
    setGate('hypnagogia', 'pending');

    // Start the dream run (long-running) — we monitor on the right pane
    const dreamPromise = showCmd(dream, `${ROKO} knowledge dream run`, {
      timeout: 300000,
      onLog: logCommand,
      customDesc: 'Runs the full dream cycle: hypnagogia (replay selection), NREM (clustering), REM (creative synthesis), integration (knowledge merge).',
    });

    // Monitor knowledge changes while dream runs
    await rawSleep(3000);
    setGate('hypnagogia', 'pass');
    setMetric('tokens', 'dreaming...');

    // Step 3: NREM clustering
    timeline.setActive(3);
    playback.setProgress(4, 7, 'NREM clustering...');
    logCommand('NREM', 'Phase 2: NREM clustering — extracting patterns and grouping related episodes into clusters.');
    setGate('nrem', 'pending');
    await showCmd(monitor, `${ROKO} knowledge stats`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Mid-dream knowledge check — distillation may already be updating entries.',
    });
    setGate('nrem', 'pass');

    // Step 4: REM synthesis
    timeline.setActive(4);
    playback.setProgress(5, 7, 'REM synthesis...');
    logCommand('REM', 'Phase 3: REM synthesis — creative linking across clusters, generating novel associations.');
    setGate('rem', 'pending');
    monitor.clearTerminal();
    await showCmd(monitor, `${ROKO} knowledge query "validation patterns"`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Querying mid-dream to see if synthesized knowledge is appearing.',
    });
    setGate('rem', 'pass');

    // Step 5: Integration — wait for dream to finish
    timeline.setActive(5);
    playback.setProgress(6, 7, 'integration...');
    logCommand('integration', 'Phase 4: Integration — merging dream-distilled knowledge into the durable store.');
    setGate('integration', 'pending');
    await dreamPromise;
    setGate('integration', 'pass');
    setMetric('tokens', 'consolidated');

    // Step 6: Report — show what the dream produced
    await playback.waitForStep();
    timeline.setActive(6);
    playback.setProgress(7, 7, `${ROKO} knowledge dream report`);
    dream.clearTerminal();
    monitor.clearTerminal();
    logCommand('dream report', 'Viewing the dream consolidation report — what was learned, what was pruned, what was synthesized.');
    await showCmd(dream, `${ROKO} knowledge dream report`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'The dream report shows entries processed, clusters formed, novel associations, and knowledge entries created or updated.',
    });
    await showCmd(monitor, `${ROKO} knowledge stats`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Post-dream knowledge stats — compare with pre-dream baseline to see what consolidation added.',
    });
    setMetric('model', 'consolidated');

    timeline.setActive(7); // all completed
  },
};
```
