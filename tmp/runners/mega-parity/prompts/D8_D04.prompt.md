## Problem
Roko's gate-failure replan loop is a core differentiator — when a task fails gates, the system classifies the failure (transient vs structural), adjusts its retry strategy, and retries with a revised plan. No existing scenario demonstrates this cycle end-to-end. The `race` scenario shows two strategies but never shows failure recovery. Users need to see the full classify-adjust-retry flow.

- Use `playback.waitForStep()` for timing between steps
- All scenarios must work with both live server and simulated output
- Add the scenario to the `SCENARIOS` array export
- Use `setupWorkspace` from `useTerminalSession`
- 2-pane layout: left runs the task, right shows gate status

## Current State
The existing `race` scenario (scenarios.ts ~line 455) compares naive vs cascade execution but does not show gate failure handling. The `prdPipeline` scenario (~line 194) runs `plan run --max-retries 1` but treats failure as a terminal state.

### Step 1 — Define the scenario object
Add this scenario constant before the `SCENARIOS` array export:

```typescript
const gateRetry: Scenario = {
  id: 'gate-retry',
  title: 'Gate Retry',
  subtitle: 'Watch a task fail gates, get classified, and retry with an adjusted strategy.',
  panes: 2,
  labels: ['task execution', 'gate status'],
  panel: true,
  promptBar: false,
  steps: [
    { label: 'First attempt', sublabel: 'roko run' },
    { label: 'Gate failure', sublabel: 'compile/test/clippy' },
    { label: 'Classification', sublabel: 'transient vs structural' },
    { label: 'Strategy adjust', sublabel: 'replan' },
    { label: 'Retry', sublabel: 'second attempt' },
    { label: 'Pass', sublabel: 'gates green' },
  ],
  async run({ entries, playback, timeline, setMetric, setGate, logCommand }) {
    const [task, gates] = entries;

    const dir = await setupWorkspace(task, 'roko-retry');
    await joinWorkspace(gates, dir);
    const ROKO = getRoko();

    timeline.init(this.steps);
    setMetric('model', 'cascade');

    // Ensure replan-on-failure is enabled
    await task.execCmd(`${ROKO} config set learning.replan_on_gate_failure true`, 10000);

    // Step 0: First attempt — run a task that is likely to need iteration
    await playback.waitForStep();
    timeline.setActive(0);
    playback.setProgress(1, 6, `${ROKO} run "..." --max-retries 2`);
    logCommand('first attempt', 'Running a moderately complex task with --max-retries 2. The first attempt may fail gates.');

    // Set all gates to pending
    setGate('compile', 'pending');
    setGate('test', 'pending');
    setGate('clippy', 'pending');

    // Show gate config on the right pane
    await showCmd(gates, `${ROKO} learn tune gates`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Current adaptive gate thresholds — these influence pass/fail decisions and retry strategy.',
    });

    // Start the main run (will potentially fail and retry)
    const runResult = await showCmd(
      task,
      `${ROKO} run "Build a Rust async HTTP client with retry logic and exponential backoff" --max-retries 2`,
      {
        timeout: 360000,
        onLog: logCommand,
        onGate: (name, status) => {
          setGate(name, status);
          if (status === 'fail') {
            logCommand(`gate:${name}`, `Gate "${name}" failed — will trigger classification and replan.`);
          } else {
            logCommand(`gate:${name}`, `Gate "${name}" passed.`);
          }
        },
        customDesc: 'Runs with --max-retries 2. If gates fail, the system classifies the failure and retries with adjusted strategy.',
      },
    );

    // Step 1: Gate failure — detected via onGate callback above
    // The timeline advances as gates report
    timeline.setActive(1);
    playback.setProgress(2, 6, 'analyzing gate results...');

    // Check output for failure classification markers
    const buf = task.getOutputBuffer();
    const hasReplan = /replan|retry|adjusted|structural|transient/i.test(buf);
    const hasFailure = /fail|error|[✖✗]/i.test(buf);

    if (hasFailure && hasReplan) {
      // Step 2: Classification detected
      timeline.setActive(2);
      playback.setProgress(3, 6, 'failure classified');
      logCommand('classification', 'Gate failure classified — system determines if failure is transient (retry same approach) or structural (revise plan).');
      setMetric('tokens', 'classifying');

      // Step 3: Strategy adjustment
      timeline.setActive(3);
      playback.setProgress(4, 6, 'adjusting strategy...');
      logCommand('strategy adjust', 'Replan triggered: the system revises the approach based on failure classification.');
      setMetric('cost', '$ attempt 1 + replan');

      // Step 4: Retry — show updated gate state
      timeline.setActive(4);
      playback.setProgress(5, 6, 'retrying...');
      gates.clearTerminal();
      await showCmd(gates, `${ROKO} learn all`, {
        timeout: 30000,
        onLog: logCommand,
        customDesc: 'Learning state after failure classification — cascade router and gate thresholds may have adjusted.',
      });
    } else if (hasFailure) {
      // Failed without replan
      timeline.setActive(2);
      logCommand('no replan', 'Gates failed but replan did not trigger — check learning.replan_on_gate_failure config.');
    } else {
      // Passed on first try
      timeline.setActive(4);
      logCommand('first-try pass', 'Task passed all gates on the first attempt — no retry needed.');
    }

    // Step 5: Final state — show result
    await playback.waitForStep();
    timeline.setActive(5);
    playback.setProgress(6, 6, 'final gate status');
    gates.clearTerminal();
    await showCmd(gates, `${ROKO} status`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Final workspace status after gate retry cycle.',
    });
    await showCmd(gates, `${ROKO} learn efficiency`, {
      timeout: 30000,
      onLog: logCommand,
      customDesc: 'Efficiency metrics show per-turn cost across attempts — retry overhead is visible here.',
    });

    if (runResult.ok) {
      setMetric('model', 'gates passed');
      logCommand('result', `Task completed in ${runResult.elapsed.toFixed(1)}s — cost: ${runResult.cost ?? 'unknown'}`);
    } else {
      setMetric('model', 'gates failed');
      logCommand('result', 'Task did not pass all gates within retry budget.');
    }

    timeline.setActive(6); // all completed
  },
};
```
