Key pattern:
```tsx
const canvasRef = useRef<HTMLCanvasElement>(null);
const draw = useCallback(() => {
  const canvas = canvasRef.current;
  if (!canvas || data.length === 0) return;
  const ctx = canvas.getContext('2d');
  if (!ctx) return;
  const dpr = window.devicePixelRatio || 1;
  const rect = canvas.getBoundingClientRect();
  canvas.width = rect.width * dpr;
  canvas.height = rect.height * dpr;
  ctx.scale(dpr, dpr);
  // ... draw grid, lines, fill ...
}, [data, title, color]);
useEffect(() => {
  draw();
  const ro = new ResizeObserver(draw);
  if (canvasRef.current) ro.observe(canvasRef.current);
  return () => ro.disconnect();
}, [draw]);
```

### Step 1 -- Add token cost estimation helper
```tsx
function tokenCost(model: string, tokensIn: number, tokensOut: number): [number, number] {
  const rates: Record<string, [number, number]> = {
    haiku: [0.00025, 0.00125], sonnet: [0.003, 0.015], opus: [0.015, 0.075],
    'gpt-4o-mini': [0.00015, 0.0006], 'gpt-4o': [0.005, 0.015],
    'o3-mini': [0.0011, 0.0044], gemini: [0.00125, 0.01],
  };
  const key = Object.keys(rates).find((k) => model.includes(k)) ?? 'sonnet';
  const [inRate, outRate] = rates[key];
  return [(tokensIn * inRate) / 1000, (tokensOut * outRate) / 1000];
}
```

### Step 2 -- Define CostBreakdownChart inline component
Place before the default export. Props: `results: BenchTaskResult[]`, `height?: number`.

Internal state: `groupBy: 'task' | 'model' | 'difficulty'` (default `'task'`).

Grouping logic via `useMemo`:
- **task**: one segment per result. Label = task_name (truncated 25 chars).
- **model**: aggregate by `result.model`. Sum tokens per unique model.
- **difficulty**: bucket by difficulty. Since `BenchTaskResult` lacks difficulty, use a
  simple heuristic: split by task_id prefix or divide results into thirds (easy/medium/hard).

Each segment: `{ label: string; inputCost: number; outputCost: number }`.
