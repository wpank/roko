### Add test module to playbook.rs

```rust
#[cfg(test)]
mod extraction_tests {
    use super::*;

    #[test]
    fn test_extract_from_tool_calls() {
        let tool_calls = vec![
            ("shell".to_string(), "cargo init".to_string()),
            ("write_file".to_string(), "Write src/lib.rs with function stub".to_string()),
            ("shell".to_string(), "cargo test".to_string()),
        ];

        let playbook = extract_playbook_from_episode("task-1", "Implement add function", &tool_calls);
        let pb = playbook.expect("should produce playbook");

        assert_eq!(pb.steps.len(), 3);
        assert_eq!(pb.steps[0].action_kind, "shell");
        assert_eq!(pb.steps[0].index, 0);
        assert_eq!(pb.steps[1].action_kind, "write_file");
        assert_eq!(pb.steps[2].action_kind, "shell");
        assert_eq!(pb.success_count, 1);
        assert_eq!(pb.failure_count, 0);
        assert!(pb.id.starts_with("ep-task-1-"));
        assert!(pb.goal.contains("Implement add function"));
    }

    #[test]
    fn test_extract_empty_tool_calls_returns_none() {
        let result = extract_playbook_from_episode("task-2", "Empty task", &[]);
        assert!(result.is_none());
    }

    #[test]
    fn test_extract_truncates_long_prompt() {
        let long_prompt = "x".repeat(500);
        let tool_calls = vec![("shell".to_string(), "echo hello".to_string())];
        let pb = extract_playbook_from_episode("task-3", &long_prompt, &tool_calls).unwrap();
        assert!(pb.goal.len() <= 210); // 200 + "..."
    }

    #[tokio::test]
    async fn test_save_or_merge_creates_new() {
        let dir = tempfile::tempdir().unwrap();
        let store = PlaybookStore::new(dir.path());

        let tool_calls = vec![("shell".to_string(), "cargo test".to_string())];
        let pb = extract_playbook_from_episode("t1", "Run tests", &tool_calls).unwrap();
        store.save_or_merge(&pb).await.unwrap();

        let loaded = store.load(&pb.id).await.unwrap();
        assert!(loaded.is_some());
        assert_eq!(loaded.unwrap().success_count, 1);
    }

    #[tokio::test]
    async fn test_save_or_merge_increments_existing() {
        let dir = tempfile::tempdir().unwrap();
        let store = PlaybookStore::new(dir.path());

        let tool_calls = vec![("shell".to_string(), "cargo test".to_string())];
        let pb1 = extract_playbook_from_episode("t1", "Run tests", &tool_calls).unwrap();
        store.save_or_merge(&pb1).await.unwrap();

        // Save again with same task — should merge
        let pb2 = extract_playbook_from_episode("t1", "Run tests", &tool_calls).unwrap();
        store.save_or_merge(&pb2).await.unwrap();

        let all = store.list().await.unwrap();
        // Either 1 merged entry with count=2, or 2 entries — depends on merge logic
        let total_success: u64 = all.iter().map(|p| p.success_count).sum();
        assert!(total_success >= 2);
    }
}
```
