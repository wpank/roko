### Step 4 — Derive learning curve data from history

```typescript
const learningCurveData: LearningCurvePoint[] = useMemo(() => {
  return history
    .filter(run => run.summary)
    .sort((a, b) => new Date(a.started_at).getTime() - new Date(b.started_at).getTime())
    .map((run, i) => ({
      runIndex: i + 1,
      runId: run.id,
      passRate: run.summary!.pass_rate * 100,
      strategy: run.config?.strategy || 'minimal',
      playbooks_created: run.summary?.learning?.playbooks_created ?? 0,
      anti_patterns_created: run.summary?.learning?.anti_patterns_created ?? 0,
      label: run.label,
    }));
}, [history]);
```

### Step 5 — Derive knowledge deltas from history

```typescript
const knowledgeDeltas: KnowledgeDelta[] = useMemo(() => {
  return history
    .filter(run => run.summary?.learning)
    .sort((a, b) => new Date(a.started_at).getTime() - new Date(b.started_at).getTime())
    .map((run, i) => ({
      runId: run.id,
      runIndex: i + 1,
      strategy: run.config?.strategy || 'minimal',
      ...run.summary!.learning!,
    }));
}, [history]);
```
