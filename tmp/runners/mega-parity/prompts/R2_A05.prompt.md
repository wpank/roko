## Problem

The init template calls `Config::default_toml_template(cloud)` at line 168 of
`crates/roko-cli/src/commands/util.rs`. `Config::default()` sets
`gates: vec![GateConfig::default_shell_true()]` (line 85 of `crates/roko-cli/src/config.rs`),
which serializes to:

```toml
[[gate]]
kind = "shell"
program = "true"
```

This always passes, defeating the purpose of gates. This batch adds profile-aware gate generation:
when `--profile rust` is given, emit real cargo gates; when `--profile typescript`, emit real
tsc/npm gates; when no profile, emit no `[[gate]]` entries with a comment block.

```rust
/// Render the default `roko.toml` template used by `roko init --profile <profile>`.
///
/// When `profile` is `Some("rust")`, the generated config includes real
/// `cargo check`, `cargo test`, and `cargo clippy` gates.  When `profile` is
/// `Some("typescript")`, it includes `npx tsc --noEmit` and `npm test` gates.
/// When profile is absent or unrecognised, no `[[gate]]` entries are emitted.
pub fn default_toml_template_with_profile(cloud: bool, profile: Option<&str>) -> Result<String> {
    let mut config = Self::default();
    config.agent.command = "claude".into();
    // Profile-specific gates (no "true" placeholder).
    config.gates = match profile {
        Some("rust") => vec![
            GateConfig::Compile { build_system: "cargo".into(), timeout_ms: 600_000 },
            GateConfig::Test    { build_system: "cargo".into(), timeout_ms: 600_000 },
            GateConfig::Clippy  { build_system: "cargo".into(), timeout_ms: 600_000 },
        ],
        Some("typescript") => vec![
            GateConfig::Shell {
                program: "npx".into(),
                args: vec!["tsc".into(), "--noEmit".into()],
                timeout_ms: 120_000,
            },
            GateConfig::Shell {
                program: "npm".into(),
                args: vec!["test".into()],
                timeout_ms: 300_000,
            },
        ],
        _ => Vec::new(),
    };
    if cloud {
        config.log_format = Some("json".to_string());
        config.bind = Some("0.0.0.0".to_string());
        config.data_dir = Some(PathBuf::from("/data/.roko"));
    }
    let rendered = config.to_toml()?;
    let cloud_deploy = if cloud {
        "\n# Auto-register webhooks after deploy\n\
         [[serve.deploy.webhooks]]\n\
         provider = \"github\"\n\
         owner = \"nunchi\"\n\
         repo = \"roko\"\n\
         \n\
         [[serve.deploy.webhooks]]\n\
         provider = \"github\"\n\
         owner = \"nunchi\"\n\
         repo = \"collaboration\"\n"
    } else {
        ""
    };

    let provider_block = if crate::auth_detect::claude_on_path() {
        "\n[providers.claude_cli]\nkind = \"claude_cli\"\ncommand = \"claude\"\n".to_string()
    } else {
        concat!(
            "\n# [providers.claude_cli]",
            "  # uncomment after installing the claude CLI\n",
            "# kind = \"claude_cli\"\n",
            "# command = \"claude\"\n",
        )
        .to_string()
    };
    let model_block =
        "\n[models.claude-sonnet-4-6]\nprovider = \"claude_cli\"\n".to_string();

    // For profiles with no gates, add a comment explaining how to configure.
    let gates_comment = if config.gates.is_empty() {
        "\n# Gates: add [[gate]] entries to validate agent output.\n\
         # Example for Rust:\n\
         # [[gate]]\n\
         # kind = \"compile\"\n\
         # build_system = \"cargo\"\n"
    } else {
        ""
    };

    Ok(format!(
        "schema_version = 2\n\
         \n\
         # REQUIRED_ENV\n\
         # Required environment variables (set in .env or shell):\n\
         # GITHUB_TOKEN       — GitHub personal access token (for MCP GitHub server)\n\
         # GITHUB_WEBHOOK_SECRET — GitHub webhook secret for deploy registration\n\
         # SLACK_BOT_TOKEN    — Slack bot token (for MCP Slack server)\n\
         # SLACK_SIGNING_SECRET — Slack webhook signing secret\n\
         # ANTHROPIC_API_KEY  — Claude API key (for direct API agents, not needed for CLI agents)\n\
         \n\
         {rendered}{cloud_deploy}{provider_block}{model_block}{gates_comment}\n\
         # PRD settings (parsed by `RokoConfig`)\n\
         [prd]\n\
         auto_plan = false\n"
    ))
}
}   // ← end of impl Config
```
