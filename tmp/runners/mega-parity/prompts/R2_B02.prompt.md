```rust
//! Unified model selection with a single precedence chain.
//!
//! Precedence: CLI override > task hint > role config > cascade router > project default > builtin.

use anyhow::{Result, anyhow};
use roko_core::agent::resolve_model;
use roko_core::config::schema::RokoConfig;

// ─── Public types ────────────────────────────────────────────────────────────

/// Which level of the precedence chain resolved the model.
#[derive(Debug, Clone, PartialEq, serde::Serialize, serde::Deserialize)]
pub enum SelectionSource {
    /// `--model` flag was provided and validated against configured providers.
    CliOverride,
    /// `model_hint` in tasks.toml selected this model (no CLI override).
    TaskModel,
    /// `[agent.roles.<role>]` in roko.toml selected this model.
    RoleConfig,
    /// CascadeRouter selected this model based on bandit observations.
    CascadeRouter,
    /// `agent.default_model` in roko.toml or global config.
    ProjectDefault,
    /// Hardcoded fallback used when no configuration is present.
    BuiltInDefault,
}

impl std::fmt::Display for SelectionSource {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        match self {
            SelectionSource::CliOverride => write!(f, "cli-override"),
            SelectionSource::TaskModel => write!(f, "task-model"),
            SelectionSource::RoleConfig => write!(f, "role-config"),
            SelectionSource::CascadeRouter => write!(f, "cascade-router"),
            SelectionSource::ProjectDefault => write!(f, "project-default"),
            SelectionSource::BuiltInDefault => write!(f, "builtin-default"),
        }
    }
}

/// The result of applying the full precedence chain for model selection.
#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
pub struct EffectiveModelSelection {
    /// Raw value from `--model` flag, if provided.
    pub requested_model: Option<String>,
    /// The model key that will actually be used (config key or slug).
    pub effective_model_key: String,
    /// Provider name as configured in roko.toml `[providers]`.
    pub provider_key: String,
    /// Provider kind string (e.g. `"claude-cli"`, `"anthropic-api"`).
    pub provider_kind: String,
    /// Backend slug for display and logging.
    pub backend_slug: String,
    /// Which level of the precedence chain resolved this.
    pub source: SelectionSource,
    /// Human-readable explanation of why this model was chosen.
    pub reason: String,
}

impl EffectiveModelSelection {
    /// Format for stderr display.
    ///
    /// Output format: `model: <key> via <provider> (source: <source>)`
    pub fn display_line(&self) -> String {
        format!(
            "model: {} via {} (source: {})",
            self.effective_model_key, self.provider_key, self.source
        )
    }

    /// Print the selection to stderr.
    pub fn print_stderr(&self) {
        eprintln!("{}", self.display_line());
    }
}

// ─── Resolution inputs ───────────────────────────────────────────────────────

/// All inputs needed to resolve the effective model.
pub struct ModelResolutionInput<'a> {
    /// Value from `--model` CLI flag.
    pub cli_model: Option<&'a str>,
    /// `model_hint` field from tasks.toml task definition.
    pub task_hint: Option<&'a str>,
    /// Agent role label (e.g. `"implementer"`, `"strategist"`).
    pub role: Option<&'a str>,
    /// Reference to the loaded `RokoConfig`.
    pub config: &'a RokoConfig,
    /// Optional cascade router for bandit-based selection.
    /// Pass `None` to skip cascade routing and fall through to project default.
    pub cascade_router: Option<&'a roko_learn::cascade_router::CascadeRouter>,
}

// ─── Core resolution function ─────────────────────────────────────────────────

/// Resolve the effective model applying the full precedence chain.
///
/// Precedence order:
/// 1. `cli_model` (--model flag) — hard override, error if provider missing
/// 2. `task_hint` (tasks.toml model_hint) — honored when no CLI override
/// 3. Role config override ([agent.roles.<role>] in roko.toml)
/// 4. CascadeRouter selection (if cascade_router provided)
/// 5. `config.agent.default_model` (project or global default)
/// 6. Builtin hardcoded default ("claude-sonnet-4-6")
///
/// # Errors
///
/// Returns `Err` if:
/// - `cli_model` is provided but the resolved provider is not configured.
/// - `cli_model` is provided but the model slug is completely unknown.
pub fn resolve_effective_model(input: ModelResolutionInput<'_>) -> Result<EffectiveModelSelection> {
    // Level 1: CLI override — hard error if provider not configured.
    if let Some(model_key) = input.cli_model {
        return resolve_cli_override(model_key, input.config);
    }

    // Level 2: Task hint.
    if let Some(hint) = input.task_hint {
        if !hint.trim().is_empty() {
            let resolved = resolve_model(input.config, hint);
            let provider_key = resolved
                .profile
                .as_ref()
                .map(|p| p.provider.clone())
                .unwrap_or_else(|| resolved.provider_kind.label().to_string());
            return Ok(EffectiveModelSelection {
                requested_model: None,
                effective_model_key: resolved.model_key.clone(),
                provider_key,
                provider_kind: resolved.provider_kind.label().to_string(),
                backend_slug: resolved.slug,
                source: SelectionSource::TaskModel,
                reason: format!("task model_hint = {}", hint),
            });
        }
    }

    // Level 3: Role config override.
    if let Some(role) = input.role {
        if let Some(model_key) = role_config_model(input.config, role) {
            let resolved = resolve_model(input.config, &model_key);
            let provider_key = resolved
                .profile
                .as_ref()
                .map(|p| p.provider.clone())
                .unwrap_or_else(|| resolved.provider_kind.label().to_string());
            return Ok(EffectiveModelSelection {
                requested_model: None,
                effective_model_key: resolved.model_key.clone(),
                provider_key,
                provider_kind: resolved.provider_kind.label().to_string(),
                backend_slug: resolved.slug,
                source: SelectionSource::RoleConfig,
                reason: format!("role_override for role={}", role),
            });
        }
    }

    // Level 4: CascadeRouter.
    if let Some(router) = input.cascade_router {
        use roko_learn::cascade_router::RoutingContext;
        let ctx = RoutingContext::default();
        let explanation = router.explain_route(&ctx, None);
        let slug = &explanation.selected_slug;
        if !slug.is_empty() {
            let resolved = resolve_model(input.config, slug);
            let provider_key = resolved
                .profile
                .as_ref()
                .map(|p| p.provider.clone())
                .unwrap_or_else(|| resolved.provider_kind.label().to_string());
            return Ok(EffectiveModelSelection {
                requested_model: None,
                effective_model_key: resolved.model_key.clone(),
                provider_key,
                provider_kind: resolved.provider_kind.label().to_string(),
                backend_slug: resolved.slug,
                source: SelectionSource::CascadeRouter,
                reason: format!("cascade router stage={:?}", explanation.stage),
            });
        }
    }

    // Level 5: Project default.
    let default_key = input.config.agent.default_model.trim().to_string();
    if !default_key.is_empty() {
        let resolved = resolve_model(input.config, &default_key);
        let provider_key = resolved
            .profile
            .as_ref()
            .map(|p| p.provider.clone())
            .unwrap_or_else(|| resolved.provider_kind.label().to_string());
        return Ok(EffectiveModelSelection {
            requested_model: None,
            effective_model_key: resolved.model_key.clone(),
            provider_key,
            provider_kind: resolved.provider_kind.label().to_string(),
            backend_slug: resolved.slug,
            source: SelectionSource::ProjectDefault,
            reason: "agent.default_model from config".to_string(),
        });
    }

    // Level 6: Builtin default.
    let builtin = "claude-sonnet-4-6";
    let resolved = resolve_model(input.config, builtin);
    let provider_key = resolved
        .profile
        .as_ref()
        .map(|p| p.provider.clone())
        .unwrap_or_else(|| "claude".to_string());
    Ok(EffectiveModelSelection {
        requested_model: None,
        effective_model_key: builtin.to_string(),
        provider_key,
        provider_kind: resolved.provider_kind.label().to_string(),
        backend_slug: resolved.slug,
        source: SelectionSource::BuiltInDefault,
        reason: "hardcoded builtin default".to_string(),
    })
}

// ─── Private helpers ──────────────────────────────────────────────────────────

fn resolve_cli_override(model_key: &str, config: &RokoConfig) -> Result<EffectiveModelSelection> {
    let resolved = resolve_model(config, model_key);
    // Error if the provider config is entirely missing AND the slug looks
    // like it should be a configured model (contains a dot or dash).
    // Pure heuristic: if config has any providers and this one isn't found,
    // treat it as a hard error so the user gets clear feedback.
    let provider_key = match &resolved.profile {
        Some(p) => {
            // Provider must exist in config.
            if !config.providers.contains_key(&p.provider) && !config.providers.is_empty() {
                return Err(anyhow!(
                    "model '{}' references provider '{}' which is not configured in roko.toml. \
                     Run `roko config providers list` to see available providers.",
                    model_key,
                    p.provider
                ));
            }
            p.provider.clone()
        }
        None => resolved.provider_kind.label().to_string(),
    };
    Ok(EffectiveModelSelection {
        requested_model: Some(model_key.to_string()),
        effective_model_key: resolved.model_key.clone(),
        provider_key,
        provider_kind: resolved.provider_kind.label().to_string(),
        backend_slug: resolved.slug,
        source: SelectionSource::CliOverride,
        reason: format!("--model flag = {}", model_key),
    })
}

fn role_config_model(config: &RokoConfig, role: &str) -> Option<String> {
    // The per-role model overrides live in config.agent.roles (type HashMap<String, RoleOverride>
    // in AgentConfig). NOT config.routing.role_overrides — RoutingConfig has no such field.
    config
        .agent
        .roles
        .get(role)
        .and_then(|ro| ro.model.clone())
        .filter(|m| !m.trim().is_empty())
}
```
