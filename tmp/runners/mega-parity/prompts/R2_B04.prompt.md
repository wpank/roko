```rust
pub(crate) async fn cmd_prd(cli: &Cli, cmd: PrdCmd) -> Result<i32> {
    use roko_cli::agent_config::{command_from_config, load_gateway_env};
    use roko_cli::agent_exec::{AgentExecOpts, run_agent_capture_silent};

    let workdir = resolve_workdir(cli);
    let gw = load_gateway_env(&workdir);
    let effort = cli.effort.map(|effort| effort.to_string());
    let effort_ref = effort.as_deref();
    let resume_session = cli.resume.as_deref();
    let agent_command = command_from_config(&workdir).unwrap_or_else(|| "claude".to_string());

    // Resolve model via unified precedence chain.
    let roko_cfg = {
        let mut rc = roko_core::config::load_config(&workdir).unwrap_or_default();
        rc.apply_process_env();
        roko_cli::config::merge_global_providers(&mut rc);
        rc
    };
    let model_selection = roko_cli::model_selection::resolve_effective_model(
        roko_cli::model_selection::ModelResolutionInput {
            cli_model: cli.model.as_deref(),
            task_hint: None,
            role: cli.role.as_deref(),
            config: &roko_cfg,
            cascade_router: None,
        },
    )?;
    model_selection.print_stderr();
    let model_str = model_selection.effective_model_key.clone();
    let model_ref = Some(model_str.as_str());
```

Note: `model_ref` type is `Option<&str>` which is what `AgentExecOpts.model` expects.
