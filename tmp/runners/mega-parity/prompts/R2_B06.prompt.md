**Add a new wrapper function** after `cmd_provider_test`:
```rust
/// Test a provider, using `cli_model` to select the model when provided.
///
/// When `cli_model` is `Some`, `resolve_effective_model()` is called to validate
/// the model and find its provider. If the resolved provider matches `provider_name`,
/// that model is used for the test. Otherwise, the standard `select_provider_test_model`
/// fallback is used.
pub(crate) async fn cmd_provider_test_with_model(
    workdir: &Path,
    provider_name: &str,
    cli_model: Option<&str>,
) -> Result<()> {
    let config = load_roko_config(workdir)?;
    let providers = configured_providers(&config);
    let provider = providers
        .get(provider_name)
        .ok_or_else(|| anyhow!("provider '{provider_name}' is not configured"))?;

    // If cli_model is given, resolve it and use the resulting slug for the test.
    let test_model: Option<String> = if let Some(model_key) = cli_model {
        let selection = roko_cli::model_selection::resolve_effective_model(
            roko_cli::model_selection::ModelResolutionInput {
                cli_model: Some(model_key),
                task_hint: None,
                role: None,
                config: &config,
                cascade_router: None,
            },
        )?;
        // Warn if the resolved provider doesn't match what the user asked to test.
        if selection.provider_key != provider_name {
            eprintln!(
                "warning: model '{}' belongs to provider '{}', not '{}'. \
                 Testing provider '{}' with its default model instead.",
                model_key, selection.provider_key, provider_name, provider_name
            );
            None
        } else {
            selection.print_stderr();
            Some(selection.backend_slug)
        }
    } else {
        None
    };

    match provider.kind {
        ProviderKind::OpenAiCompat | ProviderKind::AnthropicApi
        | ProviderKind::GeminiApi | ProviderKind::PerplexityApi
        | ProviderKind::CerebrasApi => {
            let model = if let Some(ref m) = test_model {
                // Build a (key, profile) tuple matching what select_provider_test_model returns.
                let profile = config.models.get(m)
                    .cloned()
                    .unwrap_or_else(|| {
                        let mut p = roko_core::config::schema::ModelProfile::default();
                        p.slug = m.clone();
                        p.provider = provider_name.to_string();
                        p
                    });
                (m.clone(), profile)
            } else {
                select_provider_test_model(&config, provider_name)
                    .ok_or_else(|| {
                        anyhow!(
                            "provider '{provider_name}' has no configured models. \
                             Add a model referencing this provider in roko.toml, or use \
                             `roko config models route` to see available models."
                        )
                    })?
            };
            match provider.kind {
                ProviderKind::GeminiApi => run_gemini_provider_test(provider_name, provider, &model.1).await,
                ProviderKind::AnthropicApi => run_anthropic_provider_test(provider_name, provider, &model.1).await,
                _ => run_openai_compat_provider_test(provider_name, provider, &model.1).await,
            }
        }
        ProviderKind::ClaudeCli => run_claude_cli_provider_test(provider_name, provider).await,
        ProviderKind::CursorAcp => run_cursor_provider_test(provider_name, provider).await,
    }
}
```
