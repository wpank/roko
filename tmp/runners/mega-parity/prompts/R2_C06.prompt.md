### Test 3: adaptive-skip produces skipped verdict (not a pass)

```rust
#[tokio::test]
async fn adaptive_skip_produces_skipped_verdict_not_pass() {
    use roko_gate::adaptive_threshold::AdaptiveThresholds;
    use roko_core::foundation::{GateConfig, GateRunner};
    use roko_gate::gate_service::GateService;

    let mut thresholds = AdaptiveThresholds::new();
    // Observe 25 consecutive passes on rung 1 (clippy) to trigger adaptive skip.
    for _ in 0..25 {
        thresholds.observe(1, true);
    }
    assert!(
        thresholds.should_skip_rung(1),
        "threshold must trigger skip after 25 consecutive passes"
    );

    let svc = GateService::new().with_adaptive_thresholds(thresholds);
    let config = GateConfig {
        workdir: std::env::current_dir().unwrap(),
        enabled_gates: vec!["clippy".to_string()],
        max_rung: None,
    };
    let report = svc.run_gates(config).await.unwrap();
    assert_eq!(report.verdicts.len(), 1);
    let v = &report.verdicts[0];
    assert!(!v.passed, "adaptive-skipped gate must NOT be counted as pass");
    assert!(v.skipped, "adaptive-skipped gate must be marked skipped");
    assert!(
        v.skip_reason.as_deref().is_some_and(|r| r.contains("adaptive")),
        "skip_reason must mention adaptive"
    );
}
```

### Test 5: GateVerdict serde round-trip preserves skipped field

```rust
#[test]
fn gate_verdict_serde_roundtrip_preserves_skipped_fields() {
    use roko_core::foundation::GateVerdict;

    // Test that old JSON without "skipped" deserializes correctly (backward compat).
    let old_json = r#"{"gate_name":"compile","passed":true,"output":"ok","duration_ms":100}"#;
    let v: GateVerdict = serde_json::from_str(old_json).expect("must deserialize old format");
    assert!(!v.skipped, "old verdicts must default skipped=false");
    assert!(v.skip_reason.is_none(), "old verdicts must default skip_reason=None");

    // Test that new skipped verdict round-trips correctly.
    let skipped = GateVerdict {
        gate_name: "test".to_string(),
        passed: false,
        output: "Skipped".to_string(),
        duration_ms: 0,
        skipped: true,
        skip_reason: Some("adaptive skip: rung 2".to_string()),
    };
    let json = serde_json::to_string(&skipped).expect("must serialize");
    let back: GateVerdict = serde_json::from_str(&json).expect("must deserialize");
    assert!(back.skipped);
    assert_eq!(back.skip_reason.as_deref(), Some("adaptive skip: rung 2"));
}
```
