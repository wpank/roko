### Step 1: Add a validation helper function

Add a new private function in `plan.rs` that runs validation and returns an exit code
if validation fails (returns `Some(i32)`) or `None` if validation passed:

```rust
/// Run plan validation as a prerequisite for `plan run`.
/// Returns `Some(exit_code)` if validation failed (caller should return early),
/// or `None` if validation passed.
fn validate_before_run(plans_dir: &Path) -> Option<i32> {
    let current_dir = match std::env::current_dir() {
        Ok(d) => d,
        Err(e) => {
            eprintln!("error: cannot resolve cwd for validation: {e}");
            return Some(1);
        }
    };
    let config_path = current_dir.join("roko.toml");
    let models = if config_path.is_file() {
        std::fs::read_to_string(&config_path)
            .ok()
            .and_then(|s| toml::from_str::<RokoConfig>(&s).ok())
            .map(|c| crate::commands::config_cmd::configured_models(&c))
    } else {
        None
    };
    let report = match plan_validate::validate_plans_dir(plans_dir, models.as_ref()) {
        Ok(r) => r,
        Err(e) => {
            eprintln!("error: plan validation failed: {e:#}");
            return Some(1);
        }
    };
    let code = report.exit_code(false);
    if code != 0 {
        eprintln!("{}", plan_validate::render_text(&report));
        eprintln!("error: plan validation failed — fix the errors above before running");
        Some(1)
    } else {
        None
    }
}
```

`mod plan_validate;` and `plan.rs` uses `use crate::*;` (line 4).
