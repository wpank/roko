### Step 1: Extract a helper to read executor state

Add a helper function in `commands/plan.rs` that reads the executor state and returns
task progress per plan. This helper will be used by both `plan list` and can inform
`status`.

```rust
/// Read the executor state file and return per-plan task completion info.
///
/// Returns `None` if the state file does not exist.
/// Returns `Some(vec)` where each tuple is (plan_id, tasks_completed, tasks_total).
fn read_executor_state(
    workdir: &std::path::Path,
) -> Option<Vec<(String, usize, usize)>> {
    let path = workdir.join(".roko").join("state").join("executor.json");
    if !path.is_file() {
        return None;
    }
    let contents = std::fs::read_to_string(&path).ok()?;
    let val: serde_json::Value = serde_json::from_str(&contents).ok()?;
    let plans = val.get("plans")?.as_array()?;
    let mut result = Vec::new();
    for plan in plans {
        let id = plan
            .get("plan_id")
            .or_else(|| plan.get("id"))
            .and_then(|v| v.as_str())
            .unwrap_or("unknown")
            .to_string();
        let completed = plan
            .get("tasks_completed")
            .or_else(|| plan.get("completed_tasks"))
            .and_then(|v| v.as_u64())
            .unwrap_or(0) as usize;
        let total = plan
            .get("tasks_total")
            .or_else(|| plan.get("total_tasks"))
            .and_then(|v| v.as_u64())
            .unwrap_or(0) as usize;
        result.push((id, completed, total));
    }
    Some(result)
}
```

Place this helper near the bottom of `commands/plan.rs` (after the existing helper
functions).

            let wd = workdir.unwrap_or_else(|| resolve_workdir(cli));

            let state_map: std::collections::HashMap<String, (usize, usize)> = executor_state

                        let (tasks_done, tasks_total) = state_map
                            .get(&p.id)
                            .copied()
                            .unwrap_or((p.tasks_done, p.task_count));
                        serde_json::json!({
                            "id": p.id,
                            "title": p.title,
                            "task_count": tasks_total,
                            "tasks_done": tasks_done,
                            "tasks_failed": p.tasks_failed,
                            "completed": tasks_total > 0 && tasks_done == tasks_total,
                            "has_run_state": state_exists,
                        })
                    })
                    .collect();
                println!("{}", serde_json::to_string_pretty(&entries)?);
            } else {
                if summaries.is_empty() {
                    if !state_exists {
                        println!("no run state found");
                    } else {
                        println!("no plans found in discovery path");
                    }
                } else {
                    // Header
                    println!(
                        "{:<16} {:<40} {:<12} {}",
                        "ID", "TITLE", "PROGRESS", "STATUS"
                    );
                    for p in &summaries {
                        let (tasks_done, tasks_total) = state_map
                            .get(&p.id)
                            .copied()
                            .unwrap_or((p.tasks_done, p.task_count));
                        let status = if tasks_total > 0 && tasks_done == tasks_total {
                            "done"
                        } else if tasks_done > 0 {
                            "in-progress"
                        } else {
                            "pending"
                        };
                        let progress = format!("{}/{}", tasks_done, tasks_total);
                        println!(
                            "{:<16} {:<40} {:<12} {}",
                            &p.id, &p.title, progress, status
                        );
                    }
                    if !state_exists {
                        println!("(no run state found — counts from tasks.toml files)");
                    }
                }
            }
            Ok(EXIT_SUCCESS)
