```rust
//! Fixture tests that verify learn read functions find data written to the expected paths.

use tempfile::TempDir;

// Pull in the read functions directly from the learn crate.
use roko_learn::runtime_feedback::{
    read_project_efficiency_events, read_project_episodes_lossy,

// link against the library crate and cannot access private modules. Instead, we test

    let roko = tmp.path().join(".roko");
    fs::create_dir_all(&roko).unwrap();
    roko
}
```

### Step 1: efficiency events test

```rust
#[tokio::test]
async fn efficiency_events_found_at_expected_path() {
    let tmp = TempDir::new().unwrap();
    let roko = make_roko_dir(&tmp);

    // Write two fake efficiency events to the EXACT path the reader expects.
    let learn_dir = roko.join("learn");
    fs::create_dir_all(&learn_dir).unwrap();
    let efficiency_path = learn_dir.join("efficiency.jsonl");

    // Minimal valid AgentEfficiencyEvent JSONL (all required fields).
    let events_json = concat!(
        r#"{"task_id":"t1","model":"claude-test","role":"implementer","gate_passed":true,"input_tokens":100,"output_tokens":50,"cost_usd":0.001,"wall_ms":500,"timestamp":"2024-01-01T00:00:00Z","run_id":"run_1"}"#,
        "\n",
        r#"{"task_id":"t2","model":"claude-test","role":"reviewer","gate_passed":false,"input_tokens":80,"output_tokens":30,"cost_usd":0.0008,"wall_ms":300,"timestamp":"2024-01-01T00:01:00Z","run_id":"run_2"}"#,
        "\n",
    );
    fs::write(&efficiency_path, events_json).unwrap();

    // Read function must find both events.
    let events = read_project_efficiency_events(tmp.path())
        .await
        .expect("read efficiency events");

    assert_eq!(events.len(), 2, "expected 2 efficiency events");
    assert_eq!(events[0].task_id, "t1");
    assert_eq!(events[1].task_id, "t2");
    assert!(events[0].gate_passed);
    assert!(!events[1].gate_passed);
}
```

```rust
#[tokio::test]
async fn efficiency_summary_counts_match_written_events() {
    let tmp = TempDir::new().unwrap();
    let roko = make_roko_dir(&tmp);
    let learn_dir = roko.join("learn");
    fs::create_dir_all(&learn_dir).unwrap();

    // Write 3 events: 2 pass, 1 fail.
    let lines: Vec<&str> = vec![
        r#"{"task_id":"t1","model":"m","role":"r","gate_passed":true,"input_tokens":10,"output_tokens":5,"cost_usd":0.0001,"wall_ms":100,"timestamp":"2024-01-01T00:00:00Z","run_id":"r1"}"#,
        r#"{"task_id":"t2","model":"m","role":"r","gate_passed":true,"input_tokens":20,"output_tokens":10,"cost_usd":0.0002,"wall_ms":200,"timestamp":"2024-01-01T00:01:00Z","run_id":"r2"}"#,
        r#"{"task_id":"t3","model":"m","role":"r","gate_passed":false,"input_tokens":30,"output_tokens":15,"cost_usd":0.0003,"wall_ms":300,"timestamp":"2024-01-01T00:02:00Z","run_id":"r3"}"#,
    ];
    fs::write(learn_dir.join("efficiency.jsonl"), lines.join("\n") + "\n").unwrap();

    let events = read_project_efficiency_events(tmp.path())
        .await
        .expect("read events");

    assert_eq!(events.len(), 3);
    let pass_count = events.iter().filter(|e| e.gate_passed).count();
    assert_eq!(pass_count, 2, "2 of 3 events should be passing");
    let total_cost: f64 = events.iter().map(|e| e.cost_usd).sum();
    assert!(
        (total_cost - 0.0006).abs() < 1e-9,
        "total cost should be 0.0006, got {total_cost}"
    );
}
```
