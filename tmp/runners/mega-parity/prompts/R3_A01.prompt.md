```rust
//! `ChatAgentSession` — unified agent session for interactive and one-shot CLI modes.
//!
//! Delegates to `ClaudeCliAgent` for Claude CLI turns, instead of duplicating
//! command construction in `dispatch_direct::dispatch_claude_cli`.

use std::path::PathBuf;
use std::time::Duration;

use anyhow::Result;
use roko_core::foundation::ChatMessage;

/// Unified agent session for interactive and one-shot CLI modes.
///
/// Delegates to `ClaudeCliAgent` for Claude CLI turns and to provider
/// adapters for API turns, instead of duplicating command construction.
pub struct ChatAgentSession {
    /// Working directory for the agent subprocess.
    pub workdir: PathBuf,
    /// Model slug passed to `--model`, e.g. "claude-sonnet-4-6".
    pub model: String,
    /// Reasoning effort level: "low", "medium", "high", "max".
    pub effort: String,
    /// System prompt built by SystemPromptBuilder.
    pub system_prompt: String,
    /// Tool allowlist as comma-separated names for --tools flag, e.g. "Read,Edit,Bash".
    pub allowed_tools_csv: String,
    /// Path to MCP config file, if discovered.
    pub mcp_config: Option<PathBuf>,
    /// Session ID from previous turn (for --resume).
    pub session_id: Option<String>,
    /// API message history for non-CLI providers.
    pub api_history: Vec<ChatMessage>,
    /// Shared HTTP client for API providers.
    pub http_client: reqwest::Client,
    /// Path to Claude CLI settings JSON file (None = use build_settings_json() default).
    pub settings_json: Option<String>,
    /// Per-turn timeout.
    pub timeout: Option<Duration>,
}

impl ChatAgentSession {
    /// Create a new session from CLI config and working directory.
    ///
    /// Resolves effort and timeout from `config.agent`. All other fields are
    /// placeholders filled by R3_A02 (system_prompt), R3_A03 (allowed_tools_csv),
    /// and R3_A04 (mcp_config).
    pub fn new(config: &crate::config::Config, workdir: PathBuf) -> Result<Self> {
        // 1. Build system prompt via SystemPromptBuilder (placeholder — R3_A02 fills this in)
        let system_prompt = String::new();

        // 2. Resolve tool policy (placeholder — R3_A03 fills this in)
        let allowed_tools_csv = String::new();

        // 3. Discover MCP config (placeholder — R3_A04 fills this in)
        let mcp_config: Option<PathBuf> = None;

        // 4. Read effort from config (String with serde default, not Option)
        let effort = config.agent.effort.clone();

        // 5. Read model from config (defaults to "claude-sonnet-4-6" if not set)
        let model = config
            .agent
            .model
            .clone()
            .unwrap_or_else(|| "claude-sonnet-4-6".to_string());

        // 6. Read timeout from config
        let timeout = if config.agent.timeout_ms > 0 {
            Some(Duration::from_millis(config.agent.timeout_ms))
        } else {
            None
        };

        Ok(Self {
            workdir,
            model,
            effort,
            system_prompt,
            allowed_tools_csv,
            mcp_config,
            session_id: None,
            api_history: Vec::new(),
            http_client: reqwest::Client::new(),
            settings_json: None,
            timeout,
        })
    }
}
```
