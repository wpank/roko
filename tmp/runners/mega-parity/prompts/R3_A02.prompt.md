```rust
/// Build a system prompt for interactive/one-shot chat using SystemPromptBuilder.
///
/// Uses workspace context from the working directory. Falls back to a minimal
/// prompt if the builder encounters errors.
fn build_chat_system_prompt(workdir: &Path) -> String {
    const ROLE_IDENTITY: &str = "You are an expert software engineer working in an \
        interactive chat session. You have access to tools for reading, editing, \
        and searching code. Be concise, accurate, and prefer showing code over \
        explaining it.";

    let mut builder = SystemPromptBuilder::new(ROLE_IDENTITY);

    // Layer 3: Domain context from working directory
    let project_name = workdir
        .file_name()
        .and_then(|n| n.to_str())
        .unwrap_or("unknown");
    let domain = format!(
        "Working directory: {}\nProject: {}",
        workdir.display(),
        project_name,
    );
    builder = builder.with_domain(&domain);

    // Layer 3b: Workspace context (git branch, language hints)
    if let Ok(context) = gather_workspace_context(workdir) {
        if !context.is_empty() {
            builder = builder.with_context(&context);
        }
    }

    let prompt = builder.build();

    // Fallback if builder produces empty
    if prompt.trim().is_empty() {
        ROLE_IDENTITY.to_string()
    } else {
        prompt
    }
}

/// Gather lightweight workspace context: git branch, language hints.
fn gather_workspace_context(workdir: &Path) -> anyhow::Result<String> {
    let mut parts = Vec::new();

    // Git branch
    if let Ok(output) = std::process::Command::new("git")
        .args(["branch", "--show-current"])
        .current_dir(workdir)
        .output()
    {
        if output.status.success() {
            let branch = String::from_utf8_lossy(&output.stdout).trim().to_string();
            if !branch.is_empty() {
                parts.push(format!("Git branch: {branch}"));
            }
        }
    }

    // Language hint from Cargo.toml or package.json
    if workdir.join("Cargo.toml").exists() {
        parts.push("Language: Rust".to_string());
    } else if workdir.join("package.json").exists() {
        parts.push("Language: TypeScript/JavaScript".to_string());
    } else if workdir.join("go.mod").exists() {
        parts.push("Language: Go".to_string());
    } else if workdir.join("pyproject.toml").exists() || workdir.join("setup.py").exists() {
        parts.push("Language: Python".to_string());
    }

    Ok(parts.join("\n"))
}
```

## Actual Code — Key Facts

### SystemPromptBuilder API (system_prompt_builder.rs lines 131–244)

The builder is constructed and used like this:
```rust
// Constructor — takes role identity string (layer 1)
pub fn new(role_identity: impl Into<String>) -> Self

// Layer 2: project conventions
pub fn with_conventions(mut self, conventions: impl Into<String>) -> Self

// Layer 3: domain context (project-specific knowledge)
pub fn with_domain(mut self, domain: impl Into<String>) -> Self

// Layer 3b: assembled context for current task
pub fn with_context(mut self, context: impl Into<String>) -> Self

// Layer 4: current task details
pub fn with_task(mut self, task: impl Into<String>) -> Self

// Layer 5: tool instructions
pub fn with_tools(mut self, tools: impl Into<String>) -> Self

// Layer 7: anti-patterns
pub fn with_anti_patterns(mut self, patterns: Vec<String>) -> Self

// Build and return the assembled prompt string
pub fn build(self) -> String
```

The builder takes ownership (fluent/builder pattern with `mut self`). All `with_*` methods return `Self`.
