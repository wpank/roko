- Stores `Option<PathBuf>` — None means no MCP
- Never fails on missing MCP config (it is optional)
- Never parses MCP config contents (just passes the path)

```rust
/// Discover MCP config file using the same resolution order as orchestrate.rs.
///
/// Priority:
/// 1. Explicit path in `config.agent.mcp_config`
/// 2. Workspace `.roko/mcp.json`
/// 3. Global `~/.claude/mcp-config.json`
///
/// Returns None if no MCP config is found (not an error).
fn resolve_mcp_config(workdir: &Path, config: &crate::config::Config) -> Option<PathBuf> {
    // 1. Explicit config path from roko.toml [agent] mcp_config = "..."
    if let Some(ref path) = config.agent.mcp_config {
        let resolved = if path.is_absolute() {
            path.clone()
        } else {
            workdir.join(path)
        };
        if resolved.exists() {
            tracing::debug!("MCP config from roko.toml: {}", resolved.display());
            return Some(resolved);
        }
        tracing::debug!(
            "MCP config in roko.toml does not exist: {}",
            resolved.display()
        );
    }

    // 2. Workspace .roko/mcp.json
    let workspace_mcp = workdir.join(".roko/mcp.json");
    if workspace_mcp.exists() {
        tracing::debug!("MCP config from workspace: {}", workspace_mcp.display());
        return Some(workspace_mcp);
    }

    // 3. Global ~/.claude/mcp-config.json
    if let Some(home) = home_dir() {
        let global_mcp = home.join(".claude/mcp-config.json");
        if global_mcp.exists() {
            tracing::debug!("MCP config from global: {}", global_mcp.display());
            return Some(global_mcp);
        }
    }

    tracing::debug!("no MCP config found");
    None
}

/// Get the user home directory.
fn home_dir() -> Option<PathBuf> {
    std::env::var_os("HOME")
        .map(PathBuf::from)
        .or_else(|| {
            // Fallback for systems where HOME is not set
            #[cfg(unix)]
            {
                use std::ffi::CStr;
                // SAFETY: getpwuid_r would be safer but this is a best-effort fallback
                let uid = unsafe { libc::getuid() };
                let _ = uid; // suppress unused warning — we use env fallback only
                None
            }
            #[cfg(not(unix))]
            {
                None
            }
        })
}
```

**Note**: If the `libc` crate is not available or causes issues, simplify `home_dir` to just:
```rust
fn home_dir() -> Option<PathBuf> {
    std::env::var_os("HOME").map(PathBuf::from)
}
```

The `libc`-based fallback is optional. The `HOME` env var approach is sufficient for macOS and Linux.
