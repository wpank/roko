#[cfg(test)]
mod tests {
    use super::*;

    /// Create a minimal ChatAgentSession for testing.
    ///
    /// Uses defaults that avoid needing real config or filesystem.
    fn test_session() -> ChatAgentSession {
        ChatAgentSession {
            workdir: PathBuf::from("/tmp/test-workspace"),
            model_selection: /* minimal EffectiveModelSelection */,
            effort: "medium".to_string(),
            system_prompt: "Test system prompt".to_string(),
            allowed_tools_csv: "Read,Bash".to_string(),
            mcp_config: None,
            session_id: None,
            api_history: Vec::new(),
            http_client: reqwest::Client::new(),
            settings_json: None,
            timeout: Some(Duration::from_secs(30)),
        }
    }

    #[test]
    fn first_turn_no_resume() {
        let session = test_session();
        assert!(session.session_id.is_none());
        // Verify build_agent does NOT include --resume
        let agent = session.build_agent().unwrap();
        // Assert resume field is None
    }

    #[test]
    fn second_turn_with_resume() {
        let mut session = test_session();
        session.session_id = Some("sess-abc-123".to_string());
        // Verify build_agent includes --resume sess-abc-123
        let agent = session.build_agent().unwrap();
        // Assert resume field is Some("sess-abc-123")
    }

    #[test]
    fn reset_clears_session_id() {
        let mut session = test_session();

        session.api_history.push(/* test message */);
        session.handle_slash_command("/reset");
        assert!(session.session_id.is_none());
        assert!(session.api_history.is_empty());
    }

    #[test]
    fn model_change_updates_build() {
        let mut session = test_session();
        session.handle_slash_command("/model claude-haiku-4-5");

    }

    fn effort_change_updates_build() {
        let mut session = test_session();
        session.handle_slash_command("/effort high");
        assert_eq!(session.effort, "high");
    }

    fn cancelled_turn_result() {

        assert!(result.session_id.is_none());
    }

    }
}
