```rust
async fn runs_fake_claude_binary_and_passes_flags() {

set -eu
args_file="{args_file}"
printf '%s\n' "$@" > "$args_file"

        perms.set_mode(0o755);
```

### Step 2: Add tests that verify session state mutations

```rust
    #[test]
    fn first_turn_no_session_id() {
        let session = test_session();
        // First turn: session_id is None, so --resume is NOT passed.
        assert!(session.session_id.is_none());
        // build_agent() should not panic when session_id is None.
        let result = session.build_agent();
        assert!(result.is_ok());
    }

    #[test]
    fn second_turn_with_session_id() {
        let mut session = test_session();
        // Set session_id as if a previous turn returned one.
        session.session_id = Some("sess-abc-123".to_string());
        // build_agent() should include --resume when session_id is Some.
        // We verify by checking the session state (fields are private on ClaudeCliAgent).
        let agent = session.build_agent().unwrap();
        // Agent was built without error with a session_id set.
        // Full flag verification requires running the agent against a fake binary
        // (see ClaudeCliAgent tests in claude_cli_agent.rs lines 728-785).
        let _ = agent; // agent is built
        assert_eq!(session.session_id.as_deref(), Some("sess-abc-123"));
    }

    #[test]
    fn reset_clears_session_id_and_history() {
        let mut session = test_session();
        session.session_id = Some("sess-abc-123".to_string());
        // Handle the /reset slash command (defined in R3_A05).
        session.handle_slash_command("/reset");
        assert!(session.session_id.is_none());
        // api_history should also be cleared by /reset (from R3_A05).
        assert!(session.api_history.is_empty());
    }

    #[test]
    fn model_change_updates_model() {
        let mut session = test_session();
        session.handle_slash_command("/model claude-haiku-4-5");
        // The model should reflect the new model.
        assert_eq!(session.model, "claude-haiku-4-5");
        // build_agent() should still succeed after model change.
        let result = session.build_agent();
        assert!(result.is_ok());
    }

    #[test]
    fn effort_change_updates_session_state() {
        let mut session = test_session();
        session.handle_slash_command("/effort high");
        assert_eq!(session.effort, "high");
        // build_agent() uses self.effort, so it should still succeed.
        let result = session.build_agent();
        assert!(result.is_ok());
    }

    #[test]
    fn system_prompt_change_updates_session_state() {
        let mut session = test_session();
        session.handle_slash_command("/system You are a Go expert");
        assert_eq!(session.system_prompt, "You are a Go expert");
        let result = session.build_agent();
        assert!(result.is_ok());
    }

    #[test]
    fn cancelled_turn_result_has_correct_fields() {
        let result = TurnResult::cancelled(Duration::from_secs(5));
        assert!(result.cancelled);
        assert!(result.text.is_empty());
        assert!(result.session_id.is_none());
        assert_eq!(result.input_tokens, 0);
        assert_eq!(result.output_tokens, 0);
        assert!(result.tool_calls.is_empty());
        assert_eq!(result.duration, Duration::from_secs(5));
    }

    #[test]
    fn tool_summary_abbreviation() {
        // Verify that accumulate_tool_event truncates output to 200 chars.
        use roko_agent::runtime_events::AgentRuntimeEvent;
        let mut tool_calls: Vec<ToolCallSummary> = Vec::new();
        let mut pending_ids: Vec<(String, usize)> = Vec::new();

        // Add a ToolCall event.
        accumulate_tool_event(
            &mut tool_calls,
            &mut pending_ids,
            &AgentRuntimeEvent::ToolCall {
                id: "tu_1".to_string(),
                name: "Bash".to_string(),
            },
        );
        assert_eq!(tool_calls.len(), 1);
        assert_eq!(tool_calls[0].name, "Bash");

        // Add a ToolOutput event with long content.
        let long_output = "x".repeat(250);
        accumulate_tool_event(
            &mut tool_calls,
            &mut pending_ids,
            &AgentRuntimeEvent::ToolOutput {
                id: "tu_1".to_string(),
                output: long_output.clone(),
            },
        );
        // Input abbrev should be truncated to 200 chars + "...".
        assert_eq!(tool_calls[0].input_abbrev.len(), 203); // 200 + "..."
        assert!(tool_calls[0].input_abbrev.ends_with("..."));
    }

    #[test]
    fn oneshot_does_not_accumulate_session_id() {
        // send_turn_oneshot should temporarily clear session_id.
        // After the call, session_id should be restored to what it was before.
        let mut session = test_session();
        session.session_id = Some("sess-old".to_string());
        // We can't easily call send_turn_oneshot without a real agent,
        // but we can verify the contract by inspecting the implementation:
        // send_turn_oneshot takes session_id, calls send_turn, then restores.
        let saved = session.session_id.take();
        // During the "turn" (simulated), session_id is None.
        assert!(session.session_id.is_none());
        // After the turn, restore.
        session.session_id = saved;
        assert_eq!(session.session_id.as_deref(), Some("sess-old"));
    }

    #[test]
    fn unmatched_tool_output_is_ignored() {
        use roko_agent::runtime_events::AgentRuntimeEvent;
        let mut tool_calls: Vec<ToolCallSummary> = Vec::new();
        let mut pending_ids: Vec<(String, usize)> = Vec::new();

        // ToolOutput without a matching ToolCall — should not panic.
        accumulate_tool_event(
            &mut tool_calls,
            &mut pending_ids,
            &AgentRuntimeEvent::ToolOutput {
                id: "nonexistent-id".to_string(),
                output: "some output".to_string(),
            },
        );
        // No crash, tool_calls remains empty.
        assert!(tool_calls.is_empty());
    }

} // end mod tests
```
