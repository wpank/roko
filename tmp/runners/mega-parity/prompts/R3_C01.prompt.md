## Problem

`dispatch_direct.rs` buffers ALL stream-json output then processes it, meaning the user sees nothing until the full response completes. `ClaudeCliAgent` processes stream events but returns a single `AgentResult`. For interactive chat, we need incremental parsing that emits events as they arrive so the caller can display text deltas immediately and show tool activity in real-time.

## Architecture Contract

- Line-by-line parsing of Claude CLI `--output-format stream-json`
- Each line yields zero or more `StreamEvent` variants
- Caller consumes events incrementally (via channel or async iterator)
- Stream events include: text deltas, tool start/end, final result with metadata, errors
- Does NOT replace non-streaming path (both coexist)

```rust
/// Events emitted incrementally during a streaming agent turn.
#[derive(Debug, Clone)]
pub enum StreamEvent {
    /// A chunk of text from the assistant's response.
    TextDelta(String),
    /// A tool invocation has started.
    ToolStart { name: String },
    /// A tool invocation has completed.
    ToolEnd { name: String, success: bool },
    /// The turn is complete. Contains final metadata.
    Result {
        session_id: Option<String>,
        usage: Option<StreamUsage>,
    },
    /// An error occurred during the turn.
    Error(String),
}

/// Token usage from a completed turn.
#[derive(Debug, Clone)]
pub struct StreamUsage {
    pub input_tokens: u64,
    pub output_tokens: u64,
}

/// Parse a single line of Claude CLI stream-json output into stream events.
///
/// Returns an empty vec for unrecognized or empty lines.
pub fn parse_stream_line(line: &str) -> Vec<StreamEvent> {
    let trimmed = line.trim();
    if trimmed.is_empty() {
        return Vec::new();
    }

    let event: serde_json::Value = match serde_json::from_str(trimmed) {
        Ok(v) => v,
        Err(_) => return Vec::new(),
    };

    let event_type = event.get("type").and_then(|v| v.as_str());

    match event_type {
        Some("assistant") => {
            // Text content from assistant message
            let mut events = Vec::new();
            if let Some(content) = event.get("message")
                .and_then(|m| m.get("content"))
                .and_then(|c| c.as_array())
            {
                for block in content {
                    if let Some(text) = block.get("text").and_then(|t| t.as_str()) {
                        if !text.is_empty() {
                            events.push(StreamEvent::TextDelta(text.to_string()));
                        }
                    }
                }
            }
            events
        }
        Some("content_block_delta") => {
            // Incremental text delta
            if let Some(delta) = event.get("delta")
                .and_then(|d| d.get("text"))
                .and_then(|t| t.as_str())
            {
                vec![StreamEvent::TextDelta(delta.to_string())]
            } else {
                Vec::new()
            }
        }
        Some("tool") | Some("tool_use") => {
            let name = event.get("tool")
                .or_else(|| event.get("name"))
                .and_then(|v| v.as_str())
                .unwrap_or("unknown")
                .to_string();
            vec![StreamEvent::ToolStart { name }]
        }
        Some("tool_result") => {
            let name = event.get("tool")
                .and_then(|v| v.as_str())
                .unwrap_or("unknown")
                .to_string();
            let success = !event.get("is_error")
                .and_then(|v| v.as_bool())
                .unwrap_or(false);
            vec![StreamEvent::ToolEnd { name, success }]
        }
        Some("result") => {
            let session_id = event.get("session_id")
                .and_then(|v| v.as_str())
                .map(String::from);
            let usage = event.get("usage").map(|u| StreamUsage {
                input_tokens: u.get("input_tokens")
                    .and_then(|v| v.as_u64())
                    .unwrap_or(0),
                output_tokens: u.get("output_tokens")
                    .and_then(|v| v.as_u64())
                    .unwrap_or(0),
            });
            vec![StreamEvent::Result { session_id, usage }]
        }
        Some("error") => {
            let msg = event.get("error")
                .or_else(|| event.get("message"))
                .and_then(|v| v.as_str())
                .unwrap_or("unknown error")
                .to_string();
            vec![StreamEvent::Error(msg)]
        }
        _ => Vec::new(),
    }
}
```
