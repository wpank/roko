### Step 2: Test `parse_stream_line` with mock data

```rust
    #[test]
    fn parse_text_deltas_incrementally() {
        let events: Vec<AgentRuntimeEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        let text_deltas: Vec<&str> = events.iter()
            .filter_map(|e| match e {
                AgentRuntimeEvent::MessageDelta { text } => Some(text.as_str()),
                _ => None,
            })
            .collect();

        assert!(!text_deltas.is_empty(), "should have text deltas");
        assert!(text_deltas.iter().any(|t| t.contains("Hello")));
        assert!(text_deltas.iter().any(|t| t.contains("world")));
        // Verify incremental: "Hello, " and "world!" should be SEPARATE events
        assert_eq!(text_deltas.len(), 2);
    }

    #[test]
    fn parse_session_id_from_system_event() {
        let events: Vec<AgentRuntimeEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        let system_event = events.iter().find(|e| matches!(e, AgentRuntimeEvent::SystemInit { .. }));
        assert!(system_event.is_some(), "should have SystemInit event");

        if let Some(AgentRuntimeEvent::SystemInit { session_id, model }) = system_event {
            assert_eq!(session_id, "sess-test-123");
            assert_eq!(model, "claude-sonnet-4-6");
        }
    }

    #[test]
    fn parse_session_id_from_result_event() {
        let events: Vec<AgentRuntimeEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        let turn_completed = events.iter().find(|e| matches!(e, AgentRuntimeEvent::TurnCompleted { .. }));
        assert!(turn_completed.is_some(), "should have TurnCompleted event");

        if let Some(AgentRuntimeEvent::TurnCompleted { session_id, is_error, .. }) = turn_completed {
            assert_eq!(session_id.as_deref(), Some("sess-test-123"));
            assert!(!is_error);
        }
    }

    #[test]
    fn parse_token_usage_from_result() {
        let events: Vec<AgentRuntimeEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        // Find the TokenUsage event (emitted after TurnCompleted for the result event).
        let usage_events: Vec<_> = events.iter()
            .filter(|e| matches!(e, AgentRuntimeEvent::TokenUsage { .. }))
            .collect();

        assert!(!usage_events.is_empty(), "should have TokenUsage event from result");

        // The last TokenUsage is from the result event (authoritative).
        if let Some(AgentRuntimeEvent::TokenUsage { input_tokens, output_tokens, .. }) =
            usage_events.last()
        {
            assert_eq!(*input_tokens, 150);
            assert_eq!(*output_tokens, 42);
        }
    }

    #[test]
    fn parse_tool_events() {
        let events: Vec<AgentRuntimeEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        let tool_calls: Vec<_> = events.iter()
            .filter(|e| matches!(e, AgentRuntimeEvent::ToolCall { .. }))
            .collect();
        let tool_outputs: Vec<_> = events.iter()
            .filter(|e| matches!(e, AgentRuntimeEvent::ToolOutput { .. }))
            .collect();

        assert_eq!(tool_calls.len(), 1, "expected 1 ToolCall");
        assert_eq!(tool_outputs.len(), 1, "expected 1 ToolOutput");

        if let AgentRuntimeEvent::ToolCall { id, name } = &tool_calls[0] {
            assert_eq!(id, "tu_1");
            assert_eq!(name, "Read");
        }
        if let AgentRuntimeEvent::ToolOutput { id, output } = &tool_outputs[0] {
            assert_eq!(id, "tu_1");
            assert_eq!(output, "file contents here");
        }
    }

    #[test]
    fn empty_lines_produce_no_events() {
        assert!(parse_stream_line("").is_empty());
        assert!(parse_stream_line("   ").is_empty());
        assert!(parse_stream_line("{not json}").is_empty());
    }

    #[test]
    fn error_event_parsed() {
        let events = parse_stream_line(r#"{"type":"error","message":"rate limited"}"#);
        assert_eq!(events.len(), 1, "expected 1 Error event");
        if let AgentRuntimeEvent::Error { message } = &events[0] {
            assert!(message.contains("rate limited"));
        } else {
            panic!("expected AgentRuntimeEvent::Error");
        }
    }

    #[tokio::test]
    async fn streaming_channel_receives_events() {
        let (tx, mut rx) = tokio::sync::mpsc::channel::<AgentRuntimeEvent>(100);

        // Simulate sending events through channel (no subprocess needed).
        let tx_clone = tx.clone();
        let handle = tokio::spawn(async move {
            for line in MOCK_STREAM.lines() {
                for event in parse_stream_line(line) {
                    let _ = tx_clone.send(event).await;
                }
            }
        });

        drop(tx);  // close sender after spawning

        let _ = handle.await;

        let mut received = Vec::new();
        while let Some(event) = rx.recv().await {
            received.push(event);
        }

        assert!(!received.is_empty(), "channel should receive events");
        assert!(
            received.iter().any(|e| matches!(e, AgentRuntimeEvent::MessageDelta { .. })),
            "should have MessageDelta"
        );
        assert!(
            received.iter().any(|e| matches!(e, AgentRuntimeEvent::TurnCompleted { .. })),
            "should have TurnCompleted"
        );
        assert!(
            received.iter().any(|e| matches!(e, AgentRuntimeEvent::SystemInit { .. })),
            "should have SystemInit"
        );
    }

    #[test]
    fn accumulate_tool_event_matches_by_id() {
        let mut tool_calls: Vec<ToolCallSummary> = Vec::new();
        let mut pending_ids: Vec<(String, usize)> = Vec::new();

        // Add ToolCall for "Bash" with id "tu_1".
        accumulate_tool_event(
            &mut tool_calls,
            &mut pending_ids,
            &AgentRuntimeEvent::ToolCall {
                id: "tu_1".to_string(),
                name: "Bash".to_string(),
            },
        );
        assert_eq!(tool_calls.len(), 1);
        assert_eq!(tool_calls[0].name, "Bash");
        assert!(tool_calls[0].input_abbrev.is_empty()); // not yet populated

        // Add ToolOutput for "tu_1".
        accumulate_tool_event(
            &mut tool_calls,
            &mut pending_ids,
            &AgentRuntimeEvent::ToolOutput {
                id: "tu_1".to_string(),
                output: "exit code 0".to_string(),
            },
        );
        assert_eq!(tool_calls[0].input_abbrev, "exit code 0");
        assert!(pending_ids.is_empty()); // id consumed
    }

    #[test]
    fn tool_output_truncated_at_200_chars() {
        let mut tool_calls: Vec<ToolCallSummary> = Vec::new();
        let mut pending_ids: Vec<(String, usize)> = Vec::new();

        accumulate_tool_event(
            &mut tool_calls,
            &mut pending_ids,
            &AgentRuntimeEvent::ToolCall { id: "tu_x".to_string(), name: "Read".to_string() },
        );
        let long_output = "a".repeat(300);
        accumulate_tool_event(
            &mut tool_calls,
            &mut pending_ids,
            &AgentRuntimeEvent::ToolOutput { id: "tu_x".to_string(), output: long_output },
        );
        assert_eq!(tool_calls[0].input_abbrev.len(), 203); // 200 + "..."
        assert!(tool_calls[0].input_abbrev.ends_with("..."));
    }

} // end streaming_tests
```
