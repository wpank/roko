```rust
#[cfg(test)]
mod streaming_tests {
    use super::*;

    /// Sample Claude CLI stream-json output for testing.
    const MOCK_STREAM: &str = r#"
{"type":"system","subtype":"init","session_id":"sess-test-123"}
{"type":"assistant","message":{"content":[{"type":"text","text":"Hello, "}]}}
{"type":"content_block_delta","delta":{"text":"world!"}}
{"type":"tool","tool":"Read","content":"file contents here"}
{"type":"tool_result","tool":"Read","is_error":false}
{"type":"result","session_id":"sess-test-123","usage":{"input_tokens":150,"output_tokens":42}}
"#;

    #[test]
    fn parse_text_deltas_incrementally() {
        let events: Vec<StreamEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        let text_deltas: Vec<&str> = events.iter()
            .filter_map(|e| match e {
                StreamEvent::TextDelta(t) => Some(t.as_str()),
                _ => None,
            })
            .collect();

        assert!(!text_deltas.is_empty(), "should have text deltas");
        assert!(text_deltas.iter().any(|t| t.contains("Hello")));
        assert!(text_deltas.iter().any(|t| t.contains("world")));
    }

    #[test]
    fn parse_result_metadata() {
        let events: Vec<StreamEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        let result_event = events.iter().find(|e| matches!(e, StreamEvent::Result { .. }));
        assert!(result_event.is_some(), "should have Result event");

        if let Some(StreamEvent::Result { session_id, usage }) = result_event {
            assert_eq!(session_id.as_deref(), Some("sess-test-123"));
            assert!(usage.is_some());
            let u = usage.as_ref().unwrap();
            assert_eq!(u.input_tokens, 150);
            assert_eq!(u.output_tokens, 42);
        }
    }

    #[test]
    fn parse_tool_events() {
        let events: Vec<StreamEvent> = MOCK_STREAM
            .lines()
            .flat_map(parse_stream_line)
            .collect();

        let tool_starts: Vec<_> = events.iter()
            .filter(|e| matches!(e, StreamEvent::ToolStart { .. }))
            .collect();
        let tool_ends: Vec<_> = events.iter()
            .filter(|e| matches!(e, StreamEvent::ToolEnd { .. }))
            .collect();

        assert_eq!(tool_starts.len(), 1);
        assert_eq!(tool_ends.len(), 1);

        if let StreamEvent::ToolStart { name } = &tool_starts[0] {
            assert_eq!(name, "Read");
        }
        if let StreamEvent::ToolEnd { name, success } = &tool_ends[0] {
            assert_eq!(name, "Read");
            assert!(*success);
        }
    }

    #[test]
    fn empty_lines_produce_no_events() {
        assert!(parse_stream_line("").is_empty());
        assert!(parse_stream_line("   ").is_empty());
        assert!(parse_stream_line("not json").is_empty());
    }

    #[test]
    fn error_event_parsed() {
        let events = parse_stream_line(r#"{"type":"error","error":"rate limited"}"#);
        assert_eq!(events.len(), 1);
        if let StreamEvent::Error(msg) = &events[0] {
            assert!(msg.contains("rate limited"));
        } else {
            panic!("expected Error event");
        }
    }

    #[tokio::test]
    async fn streaming_channel_receives_events() {
        let (tx, mut rx) = tokio::sync::mpsc::channel(100);

        // Simulate sending events through channel
        for line in MOCK_STREAM.lines() {
            for event in parse_stream_line(line) {
                tx.send(event).await.unwrap();
            }
        }
        drop(tx);

        let mut received = Vec::new();
        while let Some(event) = rx.recv().await {
            received.push(event);
        }

        assert!(!received.is_empty());
        // Should have text deltas, tool events, and result
        assert!(received.iter().any(|e| matches!(e, StreamEvent::TextDelta(_))));
        assert!(received.iter().any(|e| matches!(e, StreamEvent::Result { .. })));
    }
}
```
