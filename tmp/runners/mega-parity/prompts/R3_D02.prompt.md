```rust
use crate::chat_session::ChatAgentSession;

pub async fn cmd_oneshot_inline(prompt: &str, quiet: bool) -> Result<i32> {
    let auth = detect_auth();
    if matches!(auth, AuthMethod::NeedsSetup) {
        print_setup_instructions();
        return Ok(1);
    }

    let workdir = std::env::current_dir().unwrap_or_else(|_| PathBuf::from("."));

    if !quiet {
        eprintln!("roko — auth: {}", auth.label());
    }

    // Load config
    let config = load_config_or_defaults(None, &workdir)?;

    // Resolve model selection
    let model_selection = /* resolve from config + auth */;

    // Create session (same struct as interactive)
    let mut session = ChatAgentSession::new(&config, workdir, model_selection)?;

    // Single-turn: send prompt, no --resume
    let result = session.send_turn_oneshot(prompt).await?;

    // Show tool outputs before response text
    for tool_call in &result.tool_calls {
        let status = if tool_call.success { "done" } else { "failed" };
        eprintln!("[{}] {}", tool_call.name, status);
    }

    // Print response to stdout
    println!("{}", result.text);

    if !quiet {
        eprintln!(
            "\n[{} | {} in / {} out tokens | {:.1}s]",
            result.model,
            result.input_tokens,
            result.output_tokens,
            result.duration.as_secs_f64(),
        );
    }

    Ok(0)
}
```
