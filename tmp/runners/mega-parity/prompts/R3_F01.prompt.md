```rust
/// Run a single agent phase using ClaudeCliAgent (wraps the claude subprocess with safety hooks).
/// Replaces the direct Command::new("claude") approach.
async fn run_claude_cli_via_agent(
    prompt: &str,
    workdir: &Path,
    model_slug: &str,
    cancel_token: &CancelToken,
) -> anyhow::Result<String> {
    if cancel_token.is_cancelled() {
        return Err(anyhow::anyhow!("cancelled"));
    }

    let agent = ClaudeCliAgent::new(CLAUDE_CLI_BIN, workdir, model_slug);

    // Build input Engram for the agent.
    let input = Engram::builder(Kind::Task)
        .body(Body::text(prompt))
        .build();
    let ctx = roko_core::Context::at(chrono::Utc::now().timestamp_millis());

    // Run with cancellation support.
    let result = tokio::select! {
        biased;
        _ = cancel_token.cancelled() => {
            return Err(anyhow::anyhow!("cancelled"));
        }
        result = agent.run(&input, &ctx) => result,
    };

    if result.success {
        Ok(result.output.body.as_text().unwrap_or("").to_string())
    } else {
        let error_text = result.output.body.as_text().unwrap_or("agent failed").to_string();
        Err(anyhow::anyhow!("agent failed: {error_text}"))
    }
}
```
