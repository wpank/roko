// Add at the top of session.rs with other use imports:
use roko_compose::SystemPromptBuilder;

// Add method to AcpSession impl block after system_prompt_for_mode():
/// Build a context-rich system prompt using the 9-layer SystemPromptBuilder.
/// Replaces the static CODE_MODE/PLAN_MODE/RESEARCH_MODE strings.
/// `gate_feedback_text` is pre-rendered text from prior gate failures (may be empty).
/// `conventions` is the content of workspace CLAUDE.md (may be None).
pub fn build_system_prompt(
    &self,
    workdir: &std::path::Path,
    gate_feedback_text: &[String],
    conventions: Option<&str>,
) -> String {
    let role_identity = match self.config_state.agent_mode.as_str() {
        "plan" => "You are a software architect and strategist. Your role is to plan, not implement. Decompose tasks into clear actionable steps, identify files that need changes, and produce structured plans with numbered steps. Do NOT write implementation code directly.",
        "research" => "You are a technical researcher. Your role is to gather context and analyze options. Search broadly, cite specific files and line numbers, compare alternatives with tradeoffs. Do NOT make changes — report what you find.",
        _ => "You are an expert code implementer. Your role is to write and edit code directly. Make minimal targeted changes, read existing code before modifying it, follow existing patterns, and write correct working code.",
    };

    let mut builder = SystemPromptBuilder::new(role_identity);

    // Layer 2: Conventions — from workspace CLAUDE.md (cached by caller).
    if let Some(conv) = conventions {
        builder = builder.with_conventions(conv);
    }

    // Layer 3: Domain context.
    let domain = format!(
        "Working directory: {}\nSession: {} (mode: {}, model: {})",
        workdir.display(),
        self.session_id,
        self.config_state.agent_mode,
        self.config_state.model,
    );
    builder = builder.with_domain(&domain);

    // Layer 4b: Gate feedback (from previous iteration failures).
    for text in gate_feedback_text {
        if !text.is_empty() {
            builder = builder.with_gate_feedback_text(text.as_str());
        }
    }

    builder.build()
}
```

/// Load workspace CLAUDE.md content, truncated to 4096 chars.
/// Returns None if the file does not exist.
pub fn load_conventions(workdir: &std::path::Path) -> Option<String> {
    let claude_md = workdir.join("CLAUDE.md");
    std::fs::read_to_string(&claude_md).ok().map(|s| {
        if s.len() > 4096 {
            s[..4096].to_string()
        } else {
            s
        }
    })
}

/// Prompt assembler for the ACP WorkflowEngine path.
/// Uses SystemPromptBuilder to produce proper 9-layer prompts.
struct AcpPromptAssembler {
    workdir: PathBuf,
    conventions: Option<String>,
}

impl AcpPromptAssembler {
    fn new(workdir: PathBuf, conventions: Option<String>) -> Self {
        Self { workdir, conventions }
    }
}

impl PromptAssembler for AcpPromptAssembler {
    fn assemble<'life0, 'async_trait>(
        &'life0 self,
        spec: PromptSpec,
    ) -> Pin<Box<dyn Future<Output = roko_core::Result<String>> + Send + 'async_trait>>
    where
        'life0: 'async_trait,
        Self: 'async_trait,
    {
        use roko_compose::SystemPromptBuilder;

        let role = spec.role.clone().unwrap_or_else(|| {
            "You are an expert software engineer. Make minimal, correct changes following existing patterns.".to_string()
        });
        let conventions = self.conventions.clone();
        let workdir = self.workdir.clone();

        Box::pin(async move {
            let mut builder = SystemPromptBuilder::new(&role);

            if let Some(conv) = &conventions {
                builder = builder.with_conventions(conv.as_str());
            }

            let domain = format!("Working directory: {}", workdir.display());
            builder = builder.with_domain(&domain);

            if let Some(task) = spec.task.filter(|t| !t.trim().is_empty()) {
                builder = builder.with_task(&task);
            }

            for feedback in &spec.gate_feedback {
                if !feedback.is_empty() {
                    builder = builder.with_gate_feedback_text(feedback.as_str());
                }
            }

            if !spec.anti_patterns.is_empty() {
                builder = builder.with_anti_patterns(spec.anti_patterns.clone());
            }

            Ok(builder.build())
        })
    }
}
```
