## Problem
PRD and plan generation prompts receive no structured information about the repository they target. Agents hallucinate crate names, reference nonexistent files, and propose greenfield rewrites of existing code. We need a bounded, serializable struct that carries repository context into generation prompts.

## Architecture Contract
- `RepoContextPack` carries bounded repo context for generation prompts
- Lives in `crates/roko-cli/src/repo_context.rs` (new file)
- Must be serializable to JSON (for sidecar persistence) and to bounded markdown (for prompt injection)
- `to_prompt_section()` output must be <=32000 chars, truncating large workspaces gracefully
- No dependency on code-intel index (`roko-index`), no file contents, no generic utilities

```rust
//! Repository context packing for grounding PRD/plan generation prompts.
//!
//! `RepoContextPack` is a bounded, serializable snapshot of the workspace
//! that is injected into agent system prompts to prevent hallucination of
//! crate names, file paths, and symbol names.

use std::fmt::Write as _;
use std::path::{Path, PathBuf};

use serde::{Deserialize, Serialize};

/// Bounded repository context for grounding PRD/plan generation prompts.
///
/// Serializes to JSON for sidecar persistence (`.validation.json` files).
/// `to_prompt_section()` renders bounded markdown (<=32000 chars) for
/// injection into agent system prompts.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct RepoContextPack {
    /// Workspace root directory (absolute path).
    pub root: PathBuf,
    /// Detected project kind (from manifest files at root).
    pub project_kind: ProjectKind,
    /// Workspace member names: crate names for Rust, package names for TS/Go.
    pub workspace_members: Vec<String>,
    /// Key files relevant to the feature being generated (max 20).
    /// Paths are relative to workspace root.
    pub key_files: Vec<PathBuf>,
    /// Symbol matches from content grep (max 30).
    pub matching_symbols: Vec<SymbolHit>,
    /// Related PRD paths found in `.roko/prd/drafts/` (max 5).
    /// Paths are relative to workspace root.
    pub related_prds: Vec<PathBuf>,
    /// Related plan paths found in `plans/` or `.roko/plans/` (max 5).
    /// Paths are relative to workspace root.
    pub related_plans: Vec<PathBuf>,
    /// Names that already exist and must not be re-created.
    /// Populated from workspace_members + crate directory names.
    pub do_not_create: Vec<String>,
    /// True when at least one feature keyword matches a workspace member
    /// AND real source files exist for that member on disk.
    pub context_root_verified: bool,
}

/// A single symbol match from repository content search.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct SymbolHit {
    /// File path relative to workspace root.
    pub file: PathBuf,
    /// Line number (1-indexed).
    pub line: u32,
    /// Matched line text, trimmed, max 200 chars.
    pub text: String,
}

/// Detected project kind based on manifest files present at the workspace root.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
pub enum ProjectKind {
    Rust,
    TypeScript,
    Go,
    Python,
    Mixed,
    Unknown,
}

impl std::fmt::Display for ProjectKind {
    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
        let label = match self {
            Self::Rust => "Rust",
            Self::TypeScript => "TypeScript",
            Self::Go => "Go",
            Self::Python => "Python",
            Self::Mixed => "Mixed",
            Self::Unknown => "Unknown",
        };
        f.write_str(label)
    }
}

impl RepoContextPack {
    /// Construct an empty pack for use as a fallback when context collection fails.
    #[must_use]
    pub fn empty(root: &Path) -> Self {
        Self {
            root: root.to_path_buf(),
            project_kind: ProjectKind::Unknown,
            workspace_members: Vec::new(),
            key_files: Vec::new(),
            matching_symbols: Vec::new(),
            related_prds: Vec::new(),
            related_plans: Vec::new(),
            do_not_create: Vec::new(),
            context_root_verified: false,
        }
    }

    /// Render as a markdown section suitable for injection into agent system prompts.
    ///
    /// Output is hard-bounded to <=32000 characters. When the full content
    /// would exceed this limit, entries are dropped in this priority order:
    /// 1. Symbol text truncated to 120 chars
    /// 2. Symbols past 15 entries dropped
    /// 3. Key files past 10 entries dropped
    /// 4. Workspace members truncated with "... and N more"
    /// 5. Final hard truncate at 32000 chars with "[truncated]"
    #[must_use]
    pub fn to_prompt_section(&self) -> String {
        const MAX_CHARS: usize = 32000;
        const MAX_SYMBOLS_IN_PROMPT: usize = 15;
        const MAX_FILES_IN_PROMPT: usize = 10;
        const MAX_MEMBERS_IN_PROMPT: usize = 50;
        const MAX_SYMBOL_TEXT_CHARS: usize = 120;

        let mut out = String::new();

        if !self.context_root_verified {
            let members_preview = if self.workspace_members.is_empty() {
                "(none detected)".to_string()
            } else {
                let shown: Vec<&str> = self.workspace_members.iter()
                    .take(10)
                    .map(String::as_str)
                    .collect();
                let rest = self.workspace_members.len().saturating_sub(10);
                if rest > 0 {
                    format!("{}, ... and {rest} more", shown.join(", "))
                } else {
                    shown.join(", ")
                }
            };
            let _ = writeln!(
                out,
                "> **WARNING: Repository context not verified.**\n\
                 > The feature keywords do not match any known workspace members.\n\
                 > Known members: {members_preview}\n\
                 > Generated artifacts may reference nonexistent code. Review carefully before accepting.\n"
            );
        }

        let _ = writeln!(out, "## Repository Context");
        let _ = writeln!(
            out,
            "**Project**: {} workspace at `{}`",
            self.project_kind,
            self.root.display()
        );

        // Workspace members
        let member_count = self.workspace_members.len();
        let shown_members: Vec<&str> = self.workspace_members.iter()
            .take(MAX_MEMBERS_IN_PROMPT)
            .map(String::as_str)
            .collect();
        let members_line = if member_count > MAX_MEMBERS_IN_PROMPT {
            format!(
                "{}, ... and {} more",
                shown_members.join(", "),
                member_count - MAX_MEMBERS_IN_PROMPT
            )
        } else {
            shown_members.join(", ")
        };
        let _ = writeln!(out, "**Workspace members** ({member_count}): {members_line}");

        // Key files
        if !self.key_files.is_empty() {
            let _ = writeln!(out, "\n**Key files**:");
            for path in self.key_files.iter().take(MAX_FILES_IN_PROMPT) {
                let _ = writeln!(out, "- `{}`", path.display());
            }
            if self.key_files.len() > MAX_FILES_IN_PROMPT {
                let rest = self.key_files.len() - MAX_FILES_IN_PROMPT;
                let _ = writeln!(out, "- ... and {rest} more");
            }
        }

        // Symbol matches
        if !self.matching_symbols.is_empty() {
            let _ = writeln!(out, "\n**Symbol matches**:");
            for hit in self.matching_symbols.iter().take(MAX_SYMBOLS_IN_PROMPT) {
                let text = if hit.text.len() > MAX_SYMBOL_TEXT_CHARS {
                    &hit.text[..MAX_SYMBOL_TEXT_CHARS]
                } else {
                    &hit.text
                };
                let _ = writeln!(
                    out,
                    "- `{}:{}`: {}",
                    hit.file.display(),
                    hit.line,
                    text
                );
            }
            if self.matching_symbols.len() > MAX_SYMBOLS_IN_PROMPT {
                let rest = self.matching_symbols.len() - MAX_SYMBOLS_IN_PROMPT;
                let _ = writeln!(out, "- ... and {rest} more");
            }
        }

        // Related PRDs
        if !self.related_prds.is_empty() {
            let _ = writeln!(out, "\n**Related PRDs** (already exist — do NOT duplicate):");
            for path in &self.related_prds {
                let _ = writeln!(out, "- `{}`", path.display());
            }
        }

        // Related plans
        if !self.related_plans.is_empty() {
            let _ = writeln!(out, "\n**Related plans** (already exist — do NOT duplicate):");
            for path in &self.related_plans {
                let _ = writeln!(out, "- `{}`", path.display());
            }
        }

        // Do-not-create list
        if !self.do_not_create.is_empty() {
            let shown: Vec<&str> = self.do_not_create.iter()
                .take(50)
                .map(String::as_str)
                .collect();
            let rest = self.do_not_create.len().saturating_sub(50);
            let dnc_line = if rest > 0 {
                format!("{}, ... and {rest} more", shown.join(", "))
            } else {
                shown.join(", ")
            };
            let _ = writeln!(
                out,
                "\n**Do NOT create these** (they already exist): {dnc_line}"
            );
        }

        // Verification status
        let verified_label = if self.context_root_verified {
            "VERIFIED — keywords match workspace members and real source files exist"
        } else {
            "UNVERIFIED — see WARNING above"
        };
        let _ = writeln!(out, "\n**Context verification**: {verified_label}");

        // Hard truncate
        if out.len() > MAX_CHARS {
            out.truncate(MAX_CHARS - "[truncated]".len());
            out.push_str("[truncated]");
        }

        out
    }
}
```
