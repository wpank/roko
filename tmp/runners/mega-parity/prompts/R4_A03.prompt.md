/// Directories to exclude from file and symbol search.
pub const EXCLUDED_DIRS: &[&str] = &[
    "target", "node_modules", ".git", "tmp", "vendor", ".roko",
];

/// Source file extensions to include in symbol search.
const SOURCE_EXTENSIONS: &[&str] = &[
    "rs", "ts", "tsx", "js", "jsx", "go", "py", "toml", "json",
];

/// Find key files whose names match any of the feature keywords.
///
/// Scoring:
/// - exact stem match (lowercase) = 3
/// - stem contains keyword = 2
/// - any path component contains keyword = 1
///
/// Returns up to `max_files` paths relative to workspace root, sorted by
/// score descending. Bails with partial results after 5 seconds.
#[must_use]
pub fn find_key_files(root: &Path, keywords: &[&str], max_files: usize) -> Vec<PathBuf> {
    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(5);
    let mut scored: Vec<(u8, PathBuf)> = Vec::new();
    walk_source_dir(root, root, &deadline, &mut |rel_path: &Path| {
        if is_binary_path(&root.join(rel_path)) {
            return;
        }
        let stem = rel_path
            .file_stem()
            .and_then(|s| s.to_str())
            .unwrap_or("")
            .to_lowercase();
        let rel_str = rel_path.to_string_lossy().to_lowercase();

        let score: u8 = keywords.iter().map(|kw| {
            let kw_lower = kw.to_lowercase();
            if stem == kw_lower {
                3u8
            } else if stem.contains(kw_lower.as_str()) {
                2u8
            } else if rel_str.contains(kw_lower.as_str()) {
                1u8
            } else {
                0u8
            }
        }).max().unwrap_or(0);

        if score > 0 {
            scored.push((score, rel_path.to_path_buf()));
        }
    });

    scored.sort_by(|a, b| b.0.cmp(&a.0).then_with(|| a.1.cmp(&b.1)));
    scored.truncate(max_files);
    scored.into_iter().map(|(_, p)| p).collect()
}

/// Find symbol matches (lines containing keywords) in source files.
///
/// Prioritizes definition lines (fn/struct/enum/trait/type/impl/pub) over
/// other matching lines. Skips blank lines and pure comment lines.
///
/// Only searches files with source extensions: .rs, .ts, .tsx, .js, .go,
/// .py, .toml, .json.
///
/// Returns up to `max_hits` SymbolHit entries with text truncated to 200 chars.
/// Bails with partial results after 5 seconds.
#[must_use]
pub fn find_symbol_matches(root: &Path, keywords: &[&str], max_hits: usize) -> Vec<SymbolHit> {
    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(5);
    let mut definition_hits: Vec<SymbolHit> = Vec::new();
    let mut other_hits: Vec<SymbolHit> = Vec::new();

    walk_source_dir(root, root, &deadline, &mut |rel_path: &Path| {
        // Only search source extensions
        let ext = rel_path
            .extension()
            .and_then(|e| e.to_str())
            .unwrap_or("");
        if !SOURCE_EXTENSIONS.contains(&ext) {
            return;
        }
        // Stop accumulating once we have ample candidates
        if definition_hits.len() + other_hits.len() >= max_hits * 3 {
            return;
        }
        let abs_path = root.join(rel_path);
        let Ok(content) = std::fs::read_to_string(&abs_path) else {
            return;
        };
        // Skip binary-looking files
        if content.as_bytes().iter().take(512).any(|&b| b == 0) {
            return;
        }
        for (line_idx, line) in content.lines().enumerate() {
            let trimmed = line.trim();
            if trimmed.is_empty()
                || trimmed.starts_with("//")
                || trimmed.starts_with('#')
            {
                continue;
            }
            let lower = trimmed.to_lowercase();
            let matches_kw = keywords.iter().any(|kw| lower.contains(&kw.to_lowercase()));
            if !matches_kw {
                continue;
            }
            let text: String = trimmed.chars().take(200).collect();
            let hit = SymbolHit {
                file: rel_path.to_path_buf(),
                line: (line_idx + 1) as u32,
                text,
            };
            let is_def = trimmed.contains("fn ")
                || trimmed.contains("struct ")
                || trimmed.contains("enum ")
                || trimmed.contains("trait ")
                || trimmed.contains("type ")
                || trimmed.contains("impl ")
                || trimmed.starts_with("pub ");
            if is_def {
                definition_hits.push(hit);
            } else {
                other_hits.push(hit);
            }
            if definition_hits.len() + other_hits.len() >= max_hits * 3 {
                break;
            }
        }
    });

    // Definitions first, then other hits
    let mut result = definition_hits;
    result.extend(other_hits);
    result.truncate(max_hits);
    result
}

/// Recursive directory walker that skips EXCLUDED_DIRS and respects a deadline.
/// Calls `cb` for each file (not directory), passing the relative path.
fn walk_source_dir<F>(root: &Path, dir: &Path, deadline: &std::time::Instant, cb: &mut F)
where
    F: FnMut(&Path),
{
    if std::time::Instant::now() >= *deadline {
        return;
    }
    let Ok(entries) = std::fs::read_dir(dir) else {
        return;
    };
    let mut sorted_entries: Vec<_> = entries.flatten().collect();
    sorted_entries.sort_by_key(|e| e.path());

    for entry in sorted_entries {
        if std::time::Instant::now() >= *deadline {
            return;
        }
        let abs = entry.path();
        let name = entry.file_name();
        let name_str = name.to_string_lossy();

        if abs.is_dir() {
            if EXCLUDED_DIRS.contains(&name_str.as_ref()) {
                continue;
            }
            walk_source_dir(root, &abs, deadline, cb);
        } else if abs.is_file() {
            if let Ok(rel) = abs.strip_prefix(root) {
                cb(rel);
            }
        }
    }
}

/// Returns true if the first 512 bytes of the file contain a null byte.
fn is_binary_path(path: &Path) -> bool {
    let Ok(bytes) = std::fs::read(path) else {
        return false;
    };
    bytes.iter().take(512).any(|&b| b == 0)
}
