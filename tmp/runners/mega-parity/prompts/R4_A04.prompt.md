```rust
/// Find related PRDs by scanning `.roko/prd/drafts/` for keyword matches.
///
/// Matches by:
/// - Filename stem contains any keyword (score 2)
/// - First 500 bytes of file content contains any keyword (score 1)
///
/// Returns up to `max_results` paths, relative to `root`. Returns an empty
/// vec (no error) when `.roko/prd/drafts/` does not exist.
pub fn find_related_prds(root: &Path, keywords: &[&str], max_results: usize) -> Vec<PathBuf> {
    let prd_dir = root.join(".roko").join("prd").join("drafts");
    if !prd_dir.exists() {
        return vec![];
    }
    let Ok(entries) = std::fs::read_dir(&prd_dir) else {
        return vec![];
    };

    let mut scored: Vec<(u8, PathBuf)> = Vec::new();
    for entry in entries.flatten() {
        let abs = entry.path();
        if abs.is_dir() {
            continue;
        }
        let ext = abs.extension().and_then(|e| e.to_str()).unwrap_or("");
        if ext != "md" {
            continue;
        }
        let stem = abs
            .file_stem()
            .and_then(|s| s.to_str())
            .unwrap_or("")
            .to_lowercase();
        let rel = abs.strip_prefix(root).unwrap_or(&abs).to_path_buf();

        let mut best_score = 0u8;
        for kw in keywords {
            let kw_lower = kw.to_lowercase();
            if stem.contains(kw_lower.as_str()) {
                best_score = best_score.max(2);
                break; // Filename match is best possible score
            }
        }
        // If no filename match, check first 500 bytes of content
        if best_score < 2 {
            if let Ok(mut file) = std::fs::File::open(&abs) {
                use std::io::Read;
                let mut buf = [0u8; 500];
                if let Ok(n) = file.read(&mut buf) {
                    let preview = String::from_utf8_lossy(&buf[..n]).to_lowercase();
                    for kw in keywords {
                        if preview.contains(&kw.to_lowercase()) {
                            best_score = best_score.max(1);
                            break;
                        }
                    }
                }
            }
        }
        if best_score > 0 {
            scored.push((best_score, rel));
        }
    }

    scored.sort_by(|a, b| b.0.cmp(&a.0));
    scored.truncate(max_results);
    scored.into_iter().map(|(_, p)| p).collect()
}

/// Find related plans by scanning plan directories for keyword matches.
///
/// Searches both `.roko/plans/` and `plans/` (whichever exists). For each
/// subdirectory, matches by:
/// - Directory name contains any keyword (score 2)
/// - First 500 bytes of `tasks.toml` in that directory contains any keyword (score 1)
///
/// Returns up to `max_results` paths (relative to `root`). Returns empty
/// vec when neither plan directory exists.
pub fn find_related_plans(root: &Path, keywords: &[&str], max_results: usize) -> Vec<PathBuf> {
    let plan_dirs = [
        root.join(".roko").join("plans"),
        root.join("plans"),
    ];
    let mut scored: Vec<(u8, PathBuf)> = Vec::new();

    for plan_dir in &plan_dirs {
        if !plan_dir.exists() {
            continue;
        }
        let Ok(entries) = std::fs::read_dir(plan_dir) else {
            continue;
        };
        for entry in entries.flatten() {
            let abs = entry.path();
            if !abs.is_dir() {
                continue;
            }
            let dir_name = entry.file_name();
            let dir_name_lower = dir_name.to_string_lossy().to_lowercase();
            let rel = abs.strip_prefix(root).unwrap_or(&abs).to_path_buf();

            let mut best_score = 0u8;
            for kw in keywords {
                if dir_name_lower.contains(&kw.to_lowercase()) {
                    best_score = best_score.max(2);
                    break;
                }
            }
            // Check first 500 bytes of tasks.toml
            if best_score < 2 {
                let tasks_toml = abs.join("tasks.toml");
                if let Ok(mut file) = std::fs::File::open(&tasks_toml) {
                    use std::io::Read;
                    let mut buf = [0u8; 500];
                    if let Ok(n) = file.read(&mut buf) {
                        let preview = String::from_utf8_lossy(&buf[..n]).to_lowercase();
                        for kw in keywords {
                            if preview.contains(&kw.to_lowercase()) {
                                best_score = best_score.max(1);
                                break;
                            }
                        }
                    }
                }
            }
            if best_score > 0 {
                scored.push((best_score, rel));
            }
        }
    }

    scored.sort_by(|a, b| b.0.cmp(&a.0));
    scored.truncate(max_results);
    scored.into_iter().map(|(_, p)| p).collect()
}
```

```rust
    fn find_related_prds_returns_empty_when_dir_missing() {
        let tmp = tempdir().unwrap();

        let results = find_related_prds(tmp.path(), &["usage"], 5);

    #[test]
    fn find_related_prds_matches_by_filename() {
        let tmp = tempdir().unwrap();
        let drafts = tmp.path().join(".roko").join("prd").join("drafts");
        fs::create_dir_all(&drafts).unwrap();
        fs::write(drafts.join("usage-tracking.md"), "# Usage Tracking PRD").unwrap();
        fs::write(drafts.join("unrelated.md"), "# Something Else").unwrap();

        let results = find_related_prds(tmp.path(), &["usage"], 5);
        assert!(!results.is_empty());
        assert!(results.iter().any(|p| p.to_string_lossy().contains("usage-tracking")));
        // Relative paths only
        assert!(results.iter().all(|p| !p.is_absolute()));
    }

    #[test]
    fn find_related_plans_returns_empty_when_dirs_missing() {
        let tmp = tempdir().unwrap();
        // Neither plans/ nor .roko/plans/ exists

        assert!(results.is_empty());
```
