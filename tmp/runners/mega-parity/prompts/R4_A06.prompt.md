```rust
/// Build a complete RepoContextPack for the given workspace and feature keywords.
///
/// Time budget: 10 seconds. On timeout, returns partial results with warnings.
/// Missing files or directories produce incomplete packs, never errors.
pub async fn build_repo_context(
    workdir: &Path,
    feature_keywords: &[&str],
) -> Result<RepoContextPack> {
    let start = std::time::Instant::now();
    let time_budget = std::time::Duration::from_secs(10);

    // 1. Detect project kind (fast, <100ms)
    let project_kind = ProjectKind::detect(workdir);

    // 2. Collect workspace members (moderate, <2s)
    let (workspace_members, mut do_not_create) =
        collect_workspace_members(workdir, &project_kind);

    // 3. Find key files (bounded, <3s with internal timeout)
    let key_files = if start.elapsed() < time_budget {
        find_key_files(workdir, feature_keywords, 20)
    } else {
        vec![]
    };

    // 4. Find symbol matches (bounded, <3s with internal timeout)
    let matching_symbols = if start.elapsed() < time_budget {
        find_symbol_matches(workdir, feature_keywords, 30)
    } else {
        vec![]
    };

    // 5. Find related PRDs (fast, <1s)
    let related_prds = if start.elapsed() < time_budget {
        find_related_prds(workdir, feature_keywords, 5)
    } else {
        vec![]
    };

    // 6. Find related plans (fast, <1s)
    let related_plans = if start.elapsed() < time_budget {
        find_related_plans(workdir, feature_keywords, 5)
    } else {
        vec![]
    };

    // 7. Verify context root
    let context_root_verified =
        verify_context_root(workdir, feature_keywords, &workspace_members, &key_files);

    // 8. Extend do_not_create with any additional known names
    // (workspace_members are already there; add crate directory names as fallback)
    extend_do_not_create(&mut do_not_create, workdir);

    Ok(RepoContextPack {
        root: workdir.to_path_buf(),
        project_kind,
        workspace_members,
        key_files,
        matching_symbols,
        related_prds,
        related_plans,
        do_not_create,
        context_root_verified,
    })
}

/// Extend do_not_create with crate directory names that might not be in workspace members.
/// For example, if a crate dir exists but isn't in Cargo.toml members yet.
fn extend_do_not_create(do_not_create: &mut Vec<String>, root: &Path) {
    // Check common crate directories: crates/, packages/, libs/
    for dir_name in &["crates", "packages", "libs"] {
        let dir = root.join(dir_name);
        if dir.is_dir() {
            if let Ok(entries) = std::fs::read_dir(&dir) {
                for entry in entries.flatten() {
                    if entry.path().is_dir() {
                        if let Some(name) = entry.file_name().to_str() {
                            let name = name.to_string();
                            if !do_not_create.contains(&name) {
                                do_not_create.push(name);
                            }
                        }
                    }
                }
            }
        }
    }
    do_not_create.sort();
    do_not_create.dedup();
}
```
