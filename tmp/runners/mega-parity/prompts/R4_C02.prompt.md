```rust
/// Known model aliases and their canonical names.
fn normalize_model_alias(alias: &str) -> Option<String> {
    let alias_lower = alias.to_lowercase().trim().to_string();
    match alias_lower.as_str() {
        "sonnet" | "claude-sonnet" => Some("claude-sonnet-4-6".to_string()),
        "opus" | "claude-opus" => Some("claude-opus-4-6".to_string()),
        "haiku" | "claude-haiku" => Some("claude-haiku-3.5".to_string()),
        "gpt4" | "gpt-4" => Some("gpt-4.1".to_string()),
        // Already a full name? Return as-is
        s if s.contains('-') && s.len() > 8 => Some(s.to_string()),
        _ => None, // Unknown alias, cannot normalize
    }
}

/// Validate and normalize model fields in plan tasks.
fn validate_task_models(tasks: &[Task]) -> Vec<ValidationIssue> {
    let mut issues = Vec::new();
    for task in tasks {
        if let Some(model) = &task.model {
            let model_str = model.trim();
            if model_str.is_empty() {
                continue; // Empty model = use default, that's fine
            }

            // Check if it's a known alias
            match normalize_model_alias(model_str) {
                Some(canonical) if canonical != model_str.to_lowercase() => {
                    issues.push(ValidationIssue {
                        severity: Severity::Warning,
                        category: "model_alias".to_string(),
                        message: format!(
                            "Task '{}' uses model alias '{}', normalized to '{}'",
                            task.id, model_str, canonical
                        ),
                        location: Some(task.id.clone()),
                    });
                    // Optionally: mutate task.model to canonical
                    // (depends on whether tasks are mutable at this point)
                }
                None => {
                    issues.push(ValidationIssue {
                        severity: Severity::Error,
                        category: "invalid_model".to_string(),
                        message: format!(
                            "Task '{}' uses unknown model '{}'. Use full model names.",
                            task.id, model_str
                        ),
                        location: Some(task.id.clone()),
                    });
                }
                _ => {} // Already canonical
            }
        }
    }
    issues
}
```
