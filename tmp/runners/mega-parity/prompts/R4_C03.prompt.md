## Problem
Generated plan tasks reference file paths that may not exist in the repository. A task that says "modify crates/roko-prompt/src/lib.rs" when "roko-prompt" doesn't exist will fail at execution. We need validation that checks task file references against the actual repository, distinguishing between files that should exist (WARNING if missing) and new crates that conflict with existing ones (ERROR).

/// Extract crate name from a path like "crates/roko-foo/src/lib.rs" → Some("roko-foo").
/// Returns None if the path does not start with "crates/" or "packages/".
fn extract_crate_from_path(path: &str) -> Option<&str> {
    let parts: Vec<&str> = path.splitn(3, '/').collect();
    if parts.len() >= 2 && (parts[0] == "crates" || parts[0] == "packages") {
        Some(parts[1])
    } else {
        None
    }
}

/// Collect all file paths from a tasks.toml Value (from `files` or `write_files` arrays).
fn collect_task_file_paths(parsed: &toml::Value) -> Vec<(String, String)> {
    // Returns Vec<(task_id, file_path)>
    let mut out = Vec::new();
    let tasks = parsed
        .get("task")
        .and_then(toml::Value::as_array)
        .map(Vec::as_slice)
        .unwrap_or(&[]);

    for task in tasks {
        let table = match task.as_table() {
            Some(t) => t,
            None => continue,
        };
        let task_id = table
            .get("id")
            .and_then(toml::Value::as_str)
            .unwrap_or("unknown")
            .to_string();

        for field in &["files", "write_files"] {
            if let Some(files) = table.get(*field).and_then(toml::Value::as_array) {
                for file in files {
                    if let Some(path) = file.as_str().map(str::trim).filter(|p| !p.is_empty()) {
                        out.push((task_id.clone(), path.to_string()));
                    }
                }
            }
        }
    }
    out
}

/// Validate that file references in a tasks.toml exist on disk.
/// workdir: the repository root (parent of crates/, plans/, etc.)
/// Returns diagnostics for missing files and duplicate crate proposals.
pub fn validate_file_references(
    tasks_path: &Path,
    workdir: &Path,
) -> Result<Vec<Diagnostic>> {
    let content = std::fs::read_to_string(tasks_path)
        .with_context(|| format!("read {}", tasks_path.display()))?;
    let parsed: toml::Value = toml::from_str(&content)
        .with_context(|| format!("parse TOML {}", tasks_path.display()))?;

    let plan_id = parsed
        .get("meta")
        .and_then(toml::Value::as_table)
        .and_then(|meta| meta.get("plan"))
        .and_then(toml::Value::as_str)
        .map(str::trim)
        .filter(|v| !v.is_empty())
        .unwrap_or("unknown-plan")
        .to_string();

    // Collect existing crate names from crates/ directory.
    let existing_crates: std::collections::HashSet<String> = {
        let crates_dir = workdir.join("crates");
        std::fs::read_dir(&crates_dir)
            .ok()
            .into_iter()
            .flatten()
            .flatten()
            .filter_map(|entry| {
                let p = entry.path();
                if p.is_dir() {
                    p.file_name().map(|n| n.to_string_lossy().to_string())
                } else {
                    None
                }
            })
            .collect()
    };

    let file_refs = collect_task_file_paths(&parsed);
    let mut diagnostics = Vec::new();

    for (task_id, file_path) in &file_refs {
        let full_path = workdir.join(file_path);

        // Check if this references a crate that doesn't exist.
        if let Some(crate_name) = extract_crate_from_path(file_path) {
            if !existing_crates.contains(crate_name) {
                // Check if this looks like a create operation (path doesn't exist yet but crate name is in task)
                // We always warn here — creation of new crate paths is expected.
                if !full_path.exists() {
                    diagnostics.push(Diagnostic {
                        severity: Severity::Warning,
                        rule_id: "PLAN_030".to_string(),
                        plan_id: Some(plan_id.clone()),
                        task_id: Some(task_id.clone()),
                        message: format!(
                            "task '{}' references file in crate '{}' which does not exist in crates/",
                            task_id, crate_name
                        ),
                    });
                }
            }
            // Check if the crate name matches an existing crate but the file doesn't exist yet —
            // that's acceptable (it's being created by the task).
        } else if !full_path.exists() {
            // File reference outside crates/ that doesn't exist.
            diagnostics.push(Diagnostic {
                severity: Severity::Warning,
                rule_id: "PLAN_031".to_string(),
                plan_id: Some(plan_id.clone()),
                task_id: Some(task_id.clone()),
                message: format!(
                    "task '{}' references '{}' which does not exist on disk",
                    task_id, file_path
                ),
            });
        }
    }

    Ok(diagnostics)
}

/// Validate plans in `dir` with optional file-reference checking against `workdir`.
///
/// When `workdir` is Some, each tasks.toml's file references are checked against
/// the actual filesystem. When None, file-reference checks are skipped.
pub fn validate_plans_dir_with_workdir(
    dir: &Path,
    models: Option<&HashMap<String, ModelProfile>>,
    workdir: Option<&Path>,
) -> Result<ValidationReport> {
    let tasks_files = collect_tasks_files(dir)?;
    let mut plans = Vec::with_capacity(tasks_files.len());
    let mut totals = Totals {
        plans_checked: tasks_files.len(),
        ..Totals::default()
    };

    for tasks_path in tasks_files {
        let mut plan = validate_tasks_file(&tasks_path, models)
            .with_context(|| format!("validate {}", tasks_path.display()))?;

        // Optionally add file-reference diagnostics.
        if let Some(wd) = workdir {
            if let Ok(ref_diagnostics) = validate_file_references(&tasks_path, wd) {
                plan.diagnostics.extend(ref_diagnostics);
                plan.diagnostics.sort_by(|left, right| {
                    left.severity
                        .cmp(&right.severity)
                        .then_with(|| left.rule_id.cmp(&right.rule_id))
                        .then_with(|| left.task_id.cmp(&right.task_id))
                });
            }
        }

        for diagnostic in &plan.diagnostics {
            match diagnostic.severity {
                Severity::Error => totals.errors += 1,
                Severity::Warning => totals.warnings += 1,
            }
        }
        if !plan.diagnostics.is_empty() {
            plans.push(plan);
        }
    }

    Ok(ValidationReport { plans, totals })
}
