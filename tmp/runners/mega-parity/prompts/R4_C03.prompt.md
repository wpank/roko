## Task
Validate referenced files against repo context

## Runner Context

## Problem
Generated plan tasks reference file paths that may not exist in the repository. A task that says "modify crates/roko-prompt/src/lib.rs" when "roko-prompt" doesn't exist will fail at execution. We need validation that checks task file references against the actual repository, distinguishing between files that should exist (WARNING if missing) and new crates that conflict with existing ones (ERROR).

## Architecture Contract
- Generated artifacts validated before acceptance
- Task file references checked against repo filesystem and `RepoContextPack`
- File doesn't exist + task is not "create" type: WARNING
- New crate name matches `do_not_create` list: ERROR
- Distinguish ERROR / WARNING / INFO severity levels
- Validation runs during `plan validate` and pre-execution

/// Validate file references in plan tasks against the actual repository.
fn validate_task_file_references(
    tasks: &[Task],
    context: &RepoContextPack,
    workdir: &Path,
) -> Vec<ValidationIssue> {
    let mut issues = Vec::new();

    for task in tasks {
        let file_refs = extract_file_references(&task.prompt);

        for file_ref in &file_refs {
            let full_path = workdir.join(file_ref);

            // Check if the file reference is in a "create" context
            let is_create = is_create_context(&task.prompt, file_ref);

            if !full_path.exists() && !is_create {
                issues.push(ValidationIssue {
                    severity: Severity::Warning,
                    category: "nonexistent_file".to_string(),
                    message: format!(
                        "Task '{}' references '{}' which does not exist",
                        task.id, file_ref
                    ),
                    location: Some(task.id.clone()),
                });
            }

            // Check if the file reference implies a new crate that conflicts with existing
            if let Some(crate_name) = extract_crate_name_from_path(file_ref) {
                if is_create && context.do_not_create.iter().any(|existing| {
                    existing.to_lowercase() == crate_name.to_lowercase()
                }) {
                    issues.push(ValidationIssue {
                        severity: Severity::Error,
                        category: "duplicate_crate_path".to_string(),
                        message: format!(
                            "Task '{}' creates files in '{}' which already exists as '{}'",
                            task.id, file_ref, crate_name
                        ),
                        location: Some(task.id.clone()),
                    });
                }
            }
        }
    }
    issues
}

fn extract_crate_name_from_path(path: &str) -> Option<String> {
    // Pattern: "crates/{name}/..." or "packages/{name}/..."
    let parts: Vec<&str> = path.split('/').collect();
    if parts.len() >= 2 && (parts[0] == "crates" || parts[0] == "packages") {
        Some(parts[1].to_string())
