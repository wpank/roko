#[cfg(test)]
mod tests {
    use super::*;
    use std::fs;
    use std::path::Path;
    use tempfile::TempDir;

    /// Helper: create a minimal Rust workspace with N crates under `dir`.
    /// Writes root Cargo.toml with `[workspace] members = ["crates/*"]`,
    /// creates crates/<name>/Cargo.toml with `[package] name = "<name>"`,
    /// and creates crates/<name>/src/lib.rs with `// placeholder\n`.
    fn create_rust_workspace(dir: &Path, crate_names: &[&str]) {
        let workspace_toml = format!(
            "[workspace]\nmembers = [\"crates/*\"]\nresolver = \"2\"\n"
        );
        fs::write(dir.join("Cargo.toml"), workspace_toml).unwrap();
        fs::create_dir_all(dir.join("crates")).unwrap();

        for name in crate_names {
            let crate_dir = dir.join("crates").join(name);
            fs::create_dir_all(crate_dir.join("src")).unwrap();
            let pkg_toml = format!(
                "[package]\nname = \"{name}\"\nversion = \"0.1.0\"\nedition = \"2021\"\n"
            );
            fs::write(crate_dir.join("Cargo.toml"), pkg_toml).unwrap();
            fs::write(crate_dir.join("src/lib.rs"), "// placeholder\n").unwrap();
        }
    }

    // ────────────────────────────────────────────────────────────────
    // Test: workspace member detection
    // ────────────────────────────────────────────────────────────────

    #[test]
    fn test_5_crate_workspace_members_and_do_not_create() {
        let tmp = TempDir::new().unwrap();
        let crates = [
            "roko-core",
            "roko-agent",
            "roko-compose",
            "roko-gate",
            "roko-learn",
        ];
        create_rust_workspace(tmp.path(), &crates);

        let kind = ProjectKind::detect(tmp.path());
        assert_eq!(kind, ProjectKind::Rust, "Should detect Rust project kind");

        let (members, do_not_create) = collect_workspace_members(tmp.path(), &kind);
        for name in &crates {
            assert!(
                members.contains(&name.to_string()),
                "Expected '{}' in workspace members, got: {:?}",
                name,
                members
            );
            assert!(
                do_not_create.contains(&name.to_string()),
                "Expected '{}' in do_not_create, got: {:?}",
                name,
                do_not_create
            );
        }
    }

    // ────────────────────────────────────────────────────────────────
    // Test: key files and symbol matches
    // ────────────────────────────────────────────────────────────────

    #[test]
    fn test_prompt_keywords_find_compose() {
        let tmp = TempDir::new().unwrap();
        let crates = [
            "roko-core",
            "roko-agent",
            "roko-compose",
            "roko-gate",
            "roko-learn",
        ];
        create_rust_workspace(tmp.path(), &crates);

        // Add content to roko-compose/src/lib.rs that contains symbol definitions
        let compose_lib = tmp.path().join("crates/roko-compose/src/lib.rs");
        fs::write(
            &compose_lib,
            "pub struct PromptAssembly;\npub fn build_prompt() {}\n",
        )
        .unwrap();

        let key_files = find_key_files(tmp.path(), &["prompt", "compose"], 20);
        let has_compose = key_files
            .iter()
            .any(|f| f.to_str().unwrap_or("").contains("roko-compose"));
        assert!(
            has_compose,
            "Expected roko-compose files in key_files, got: {:?}",
            key_files
        );

        let symbols = find_symbol_matches(tmp.path(), &["prompt"], 30);
        let has_prompt_assembly = symbols
            .iter()
            .any(|s| s.text.contains("PromptAssembly") || s.text.contains("build_prompt"));
        assert!(
            has_prompt_assembly,
            "Expected PromptAssembly or build_prompt in symbol matches, got: {:?}",
            symbols
        );
    }

    // ────────────────────────────────────────────────────────────────
    // Test: empty directory
    // ────────────────────────────────────────────────────────────────

    #[test]
    fn test_empty_dir_returns_unknown_unverified() {
        let tmp = TempDir::new().unwrap();

        let kind = ProjectKind::detect(tmp.path());
        assert_eq!(
            kind,
            ProjectKind::Unknown,
            "Empty dir should detect as Unknown"
        );

        let (members, _do_not_create) = collect_workspace_members(tmp.path(), &kind);
        assert!(
            members.is_empty(),
            "Empty dir should have no workspace members"
        );

        let verified = verify_context_root(tmp.path(), &["anything"], &members, &[]);
        assert!(
            !verified,
            "Empty dir should not be verified (no members to match)"
        );
    }

    // ────────────────────────────────────────────────────────────────
    // Test: to_prompt_section output size
    // ────────────────────────────────────────────────────────────────

    #[test]
    fn test_to_prompt_section_under_32000_chars() {
        let tmp = TempDir::new().unwrap();
        // Large workspace: 60 crates to exercise truncation logic
        let crate_names: Vec<String> = (0..60).map(|i| format!("crate-{i:03}")).collect();
        let crate_refs: Vec<&str> = crate_names.iter().map(|s| s.as_str()).collect();
        create_rust_workspace(tmp.path(), &crate_refs);

        let pack = RepoContextPack {
            root: tmp.path().to_path_buf(),
            project_kind: ProjectKind::Rust,
            workspace_members: crate_names.clone(),
            key_files: (0..20)
                .map(|i| PathBuf::from(format!("crates/crate-{i:03}/src/lib.rs")))
                .collect(),
            matching_symbols: (0..30)
                .map(|i| SymbolHit {
                    file: PathBuf::from(format!("crates/crate-{i:03}/src/lib.rs")),
                    line: (i as u32) + 1,
                    text: format!(
                        "pub struct LongStructName{i} {{ field_one: String, field_two: u64 }}"
                    ),
                })
                .collect(),
            related_prds: vec![],
            related_plans: vec![],
            do_not_create: crate_names,
            context_root_verified: true,
        };

        let section = pack.to_prompt_section();
        assert!(
            section.len() <= 32_000,
            "to_prompt_section() output is {} chars, exceeds 32000 limit",
            section.len()
        );
        assert!(
            section.contains("Repository Context"),
            "Output should contain 'Repository Context' heading, got: {}",
            &section[..section.len().min(200)]
        );
    }

    // ────────────────────────────────────────────────────────────────
    // Test: build_repo_context integration
    // ────────────────────────────────────────────────────────────────

    #[tokio::test]
    async fn test_build_repo_context_integration() {
        let tmp = TempDir::new().unwrap();
        let crates = ["roko-core", "roko-compose", "roko-agent"];
        create_rust_workspace(tmp.path(), &crates);

        // Add source with "compose" keyword to help verify the context
        let compose_src = tmp.path().join("crates/roko-compose/src/lib.rs");
        fs::write(&compose_src, "pub fn compose_prompt() {}\n").unwrap();

        let pack = build_repo_context(tmp.path(), &["compose", "prompt"])
            .await
            .unwrap();

        assert_eq!(pack.project_kind, ProjectKind::Rust);
        assert!(
            pack.workspace_members.len() >= 3,
            "Should have at least 3 workspace members, got: {:?}",
            pack.workspace_members
        );
        // context_root_verified should be true because "compose" matches "roko-compose"
        // AND source files exist (we wrote one above)
        assert!(
            pack.context_root_verified,
            "Context should be verified: 'compose' matches 'roko-compose' and source exists"
        );
    }

    // ────────────────────────────────────────────────────────────────
    // Test: verify_context_root edge cases
    // ────────────────────────────────────────────────────────────────

    #[test]
    fn test_verify_context_root_no_keyword_match_returns_false() {
        let tmp = TempDir::new().unwrap();
        let crates = ["roko-core", "roko-agent"];
        create_rust_workspace(tmp.path(), &crates);

        let members = vec!["roko-core".to_string(), "roko-agent".to_string()];
        // "blockchain" doesn't match any member name
        let verified =
            verify_context_root(tmp.path(), &["blockchain", "solidity"], &members, &[]);
        assert!(
            !verified,
            "Keywords with no member match should not be verified"
        );
    }

    #[test]
    fn test_verify_context_root_empty_keywords_returns_false() {
        let tmp = TempDir::new().unwrap();
        let members = vec!["roko-core".to_string()];
        let verified = verify_context_root(tmp.path(), &[], &members, &[]);
        assert!(!verified, "Empty keywords should return false");
    }

    #[test]
    fn test_verify_context_root_empty_members_returns_false() {
        let tmp = TempDir::new().unwrap();
        let verified = verify_context_root(tmp.path(), &["core"], &[], &[]);
        assert!(!verified, "Empty workspace members should return false");
    }
}
