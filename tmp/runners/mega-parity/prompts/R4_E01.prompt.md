#[cfg(test)]
mod tests {
    use super::*;
    use tempfile::TempDir;
    use std::fs;

    /// Helper: create a minimal Rust workspace with N crates.
    fn create_rust_workspace(dir: &Path, crate_names: &[&str]) {
        // Write root Cargo.toml with [workspace] members = ["crates/*"]
        let cargo_toml = format!(
            r#"[workspace]
members = ["crates/*"]
resolver = "2"
"#
        );
        fs::write(dir.join("Cargo.toml"), cargo_toml).unwrap();
        fs::create_dir_all(dir.join("crates")).unwrap();

        for name in crate_names {
            let crate_dir = dir.join("crates").join(name);
            fs::create_dir_all(crate_dir.join("src")).unwrap();
            fs::write(
                crate_dir.join("Cargo.toml"),
                format!(
                    r#"[package]
name = "{}"
version = "0.1.0"
edition = "2021"
"#,
                    name
                ),
            ).unwrap();
            fs::write(crate_dir.join("src/lib.rs"), "// placeholder\n").unwrap();
        }
    }

    #[test]
    fn test_5_crate_workspace_members_and_do_not_create() {
        let tmp = TempDir::new().unwrap();
        let crates = ["roko-core", "roko-agent", "roko-compose", "roko-gate", "roko-learn"];
        create_rust_workspace(tmp.path(), &crates);

        let kind = ProjectKind::detect(tmp.path());
        assert_eq!(kind, ProjectKind::Rust);

        let (members, do_not_create) = collect_workspace_members(tmp.path(), &kind);
        for name in &crates {
            assert!(members.contains(&name.to_string()),
                "Expected '{}' in workspace members", name);
            assert!(do_not_create.contains(&name.to_string()),
                "Expected '{}' in do_not_create", name);
        }
    }

    #[test]
    fn test_prompt_keywords_find_compose() {
        let tmp = TempDir::new().unwrap();
        let crates = ["roko-core", "roko-agent", "roko-compose", "roko-gate", "roko-learn"];
        create_rust_workspace(tmp.path(), &crates);

        // Add some content to roko-compose for symbol matching
        let compose_lib = tmp.path().join("crates/roko-compose/src/lib.rs");
        fs::write(&compose_lib, "pub struct PromptAssembly;\npub fn build_prompt() {}\n").unwrap();

        let key_files = find_key_files(tmp.path(), &["prompt", "compose"], 20);
        let has_compose = key_files.iter().any(|f| {
            f.to_str().unwrap_or("").contains("roko-compose")
        });
        assert!(has_compose, "Expected roko-compose files in key_files: {:?}", key_files);

        let symbols = find_symbol_matches(tmp.path(), &["prompt"], 30);
        let has_prompt_assembly = symbols.iter().any(|s| s.text.contains("PromptAssembly"));
        assert!(has_prompt_assembly,
            "Expected PromptAssembly in symbol matches: {:?}", symbols);
    }

    #[test]
    fn test_empty_dir_returns_unknown_unverified() {
        let tmp = TempDir::new().unwrap();

        let kind = ProjectKind::detect(tmp.path());
        assert_eq!(kind, ProjectKind::Unknown);

        let (members, _) = collect_workspace_members(tmp.path(), &kind);
        assert!(members.is_empty());

        let verified = verify_context_root(tmp.path(), &["anything"], &members, &[]);
        assert!(!verified, "Empty dir should not be verified");
    }

    #[test]
    fn test_to_prompt_section_under_32000_chars() {
        let tmp = TempDir::new().unwrap();
        // Create a large workspace with 60 crates
        let crate_names: Vec<String> = (0..60).map(|i| format!("crate-{:03}", i)).collect();
        let crate_refs: Vec<&str> = crate_names.iter().map(|s| s.as_str()).collect();
        create_rust_workspace(tmp.path(), &crate_refs);

        let pack = RepoContextPack {
            root: tmp.path().to_path_buf(),
            project_kind: ProjectKind::Rust,
            workspace_members: crate_names.clone(),
            key_files: (0..20).map(|i| PathBuf::from(format!("crates/crate-{:03}/src/lib.rs", i))).collect(),
            matching_symbols: (0..30).map(|i| SymbolHit {
                file: PathBuf::from(format!("crates/crate-{:03}/src/lib.rs", i)),
                line: i as u32 + 1,
                text: format!("pub struct LongStructName{} {{ field: String }}", i),
            }).collect(),
            related_prds: vec![],
            related_plans: vec![],
            do_not_create: crate_names,
            context_root_verified: true,
        };

        let section = pack.to_prompt_section();
        assert!(
            section.len() <= 32000,
            "to_prompt_section() output is {} chars, exceeds 32000 limit",
            section.len()
        );
        assert!(section.contains("Repository Context"),
            "Output should contain 'Repository Context' heading");
    }

    #[tokio::test]
    async fn test_build_repo_context_integration() {
        let tmp = TempDir::new().unwrap();
        let crates = ["roko-core", "roko-compose", "roko-agent"];
        create_rust_workspace(tmp.path(), &crates);

        let pack = build_repo_context(tmp.path(), &["compose", "prompt"]).await.unwrap();

        assert_eq!(pack.project_kind, ProjectKind::Rust);
        assert!(pack.workspace_members.len() >= 3);
        // context_root_verified should be true because "compose" matches "roko-compose"
        // and source files exist
        assert!(pack.context_root_verified,
            "Context should be verified: compose matches roko-compose");
    }
}
