#[cfg(test)]
mod plan_validation_tests {
    use super::*;

    /// Helper: create a mock Task with given fields.
    fn mock_task(id: &str, role: Option<&str>, model: Option<&str>, prompt: &str) -> Task {
        Task {
            id: id.to_string(),
            title: format!("Task {}", id),
            role: role.map(|r| r.to_string()),
            model: model.map(|m| m.to_string()),
            prompt: prompt.to_string(),
            depends_on: vec![],
            // ... other required fields with defaults
        }
    }

    // === R4_C01: Role validation ===

    #[test]
    fn test_missing_role_produces_error() {
        let tasks = vec![mock_task("T01", None, None, "do something")];
        let issues = validate_task_roles(&tasks);
        assert_eq!(issues.len(), 1);
        assert_eq!(issues[0].severity, Severity::Error);
        assert_eq!(issues[0].category, "missing_role");
        assert!(issues[0].message.contains("T01"));
    }

    #[test]
    fn test_known_role_passes() {
        let tasks = vec![mock_task("T01", Some("implementer"), None, "do something")];
        let issues = validate_task_roles(&tasks);
        assert!(issues.is_empty(), "Known role should produce no issues");
    }

    #[test]
    fn test_unknown_role_produces_warning() {
        let tasks = vec![mock_task("T01", Some("wizard"), None, "do magic")];
        let issues = validate_task_roles(&tasks);
        assert_eq!(issues.len(), 1);
        assert_eq!(issues[0].severity, Severity::Warning);
        assert_eq!(issues[0].category, "unknown_role");
    }

    // === R4_C02: Model alias normalization ===

    #[test]
    fn test_alias_sonnet_normalized() {
        let result = normalize_model_alias("sonnet");
        assert_eq!(result, Some("claude-sonnet-4-6".to_string()));
    }

    #[test]
    fn test_alias_sonnet_in_task_produces_warning() {
        let tasks = vec![mock_task("T01", Some("implementer"), Some("sonnet"), "code")];
        let issues = validate_task_models(&tasks);
        assert!(issues.iter().any(|i| i.category == "model_alias"),
            "Should produce model_alias warning");
    }

    #[test]
    fn test_unknown_model_produces_error() {
        let tasks = vec![mock_task("T01", Some("implementer"), Some("gpt2"), "code")];
        let issues = validate_task_models(&tasks);
        assert!(issues.iter().any(|i| i.severity == Severity::Error && i.category == "invalid_model"),
            "Unknown short model name should produce error");
    }

    #[test]
    fn test_full_model_name_passes() {
        let tasks = vec![mock_task("T01", Some("implementer"), Some("claude-sonnet-4-6"), "code")];
        let issues = validate_task_models(&tasks);
        assert!(issues.is_empty(), "Full model name should produce no issues");
    }

    // === R4_C03: File reference validation ===

    #[test]
    fn test_nonexistent_file_produces_warning() {
        let tmp = tempfile::TempDir::new().unwrap();
        let context = RepoContextPack {
            root: tmp.path().to_path_buf(),
            workspace_members: vec!["roko-core".to_string()],
            do_not_create: vec!["roko-core".to_string()],
            ..Default::default() // or construct fully
        };

        let tasks = vec![mock_task(
            "T01", Some("implementer"), None,
            "Modify `crates/roko-nonexistent/src/lib.rs` to add feature"
        )];

        let issues = validate_task_file_references(&tasks, &context, tmp.path());
        assert!(issues.iter().any(|i| i.severity == Severity::Warning && i.category == "nonexistent_file"),
            "Nonexistent file should produce warning");
    }

    // === R4_C04: Greenfield duplicate detection ===

    #[test]
    fn test_duplicate_crate_produces_error() {
        let context = RepoContextPack {
            workspace_members: vec!["roko-core".to_string(), "roko-compose".to_string()],
            do_not_create: vec!["roko-core".to_string(), "roko-compose".to_string()],
            ..Default::default()
        };

        let tasks = vec![mock_task(
            "T01", Some("architect"), None,
            "Create crate roko-compose with prompt assembly logic"
        )];

        let issues = validate_no_greenfield_duplicates(&tasks, &context, "");
        assert!(issues.iter().any(|i| i.severity == Severity::Error && i.category == "greenfield_duplicate"),
            "Duplicate crate proposal should produce error: {:?}", issues);
    }

    #[test]
    fn test_new_crate_in_empty_workspace_ok() {
        let context = RepoContextPack {
            workspace_members: vec![],
            do_not_create: vec![],
            ..Default::default()
        };

        let tasks = vec![mock_task(
            "T01", Some("architect"), None,
            "Create crate my-new-lib with utilities"
        )];

        let issues = validate_no_greenfield_duplicates(&tasks, &context, "");
        assert!(issues.is_empty(),
            "New crate in empty workspace should be OK: {:?}", issues);
    }

    #[test]
    fn test_greenfield_phrase_in_existing_workspace_produces_error() {
        let context = RepoContextPack {
            workspace_members: vec!["roko-core".to_string()],
            do_not_create: vec!["roko-core".to_string()],
            ..Default::default()
        };

        let plan_content = "This is a greenfield project starting from scratch.";
        let issues = validate_no_greenfield_duplicates(&[], &context, plan_content);
        assert!(issues.iter().any(|i| i.category == "greenfield_claim"),
            "Greenfield phrase in existing workspace should produce error: {:?}", issues);
    }
}
