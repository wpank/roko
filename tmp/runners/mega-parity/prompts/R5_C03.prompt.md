```rust
pub(crate) fn print_learn_gate_thresholds(workdir: &std::path::Path) {
    let path = workdir.join(".roko/learn/gate-thresholds.json");
    println!("  path: {}", path.display());
    if !path.exists() {
        println!("Gate thresholds: No data at {}", path.display());
        return;
    }
    let Ok(content) = std::fs::read_to_string(&path) else {
        println!("Gate thresholds: unreadable at {}", path.display());
        return;
    };
    let entries = content.lines().filter(|l| !l.trim().is_empty()).count();
    if entries == 0 {
        println!("Gate thresholds: 0 entries at {}", path.display());
    } else {
        println!("Gate thresholds: present at {}", path.display());
    }
}
```
