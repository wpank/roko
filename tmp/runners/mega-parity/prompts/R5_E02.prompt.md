## Problem
No integration-level test verifies the invariant "one efficiency event per dispatch attempt." If the emit path is called twice for the same dispatch, cost will be double-counted in learning data and dashboards. This test locks in the correct behavior.

## Architecture Contract
- One `AgentEfficiencyEvent` per dispatch attempt (one call to `emit_efficiency_event` or `emit_failure_efficiency_event` per agent invocation)
- A gate failure does not trigger a second efficiency event for the same attempt
- Total cost in efficiency log = sum of individual attempt costs

### Full test implementation

```rust
//! Integration test: exactly one efficiency event is emitted per dispatch
//! attempt. No duplicates from gate failures or retries.

mod common;

use std::collections::HashSet;
use std::fs;

use roko_learn::runtime_feedback::read_efficiency_events;
use tempfile::TempDir;

/// Verify that a single-task plan with a passing gate produces exactly one
/// efficiency event.
#[tokio::test]
async fn single_task_plan_emits_exactly_one_efficiency_event() {
    let tmp = TempDir::new().expect("tempdir");
    let workdir = tmp.path();

    common::setup_sample_plan_workspace(workdir);

    // Run the plan (non-blocking; kills on timeout).
    // Uses the mock claude binary from setup_sample_plan_workspace.
    let _output = common::run_sample_plan(workdir);

    // Read back the efficiency log.
    let efficiency_path = workdir.join(".roko").join("learn").join("efficiency.jsonl");

    // File may not exist if no dispatch happened (e.g. dry-run or all-skip).
    if !efficiency_path.exists() {
        // No events emitted — acceptable for a plan that was entirely skipped.
        return;
    }

    let events = read_efficiency_events(&efficiency_path)
        .await
        .expect("read efficiency events");

    // Filter to events for our plan.
    let plan_events: Vec<_> = events
        .iter()
        .filter(|e| e.plan_id == common::SAMPLE_PLAN_ID)
        .collect();

    if plan_events.is_empty() {
        // Plan may have been skipped (no tasks ready) — not an error.
        return;
    }

    // Assert: one event per task attempt.
    // The sample plan has exactly one task "T1". A single success path
    // should produce exactly one efficiency event.
    assert!(
        plan_events.len() >= 1,
        "expected at least 1 efficiency event for plan {}, got 0",
        common::SAMPLE_PLAN_ID
    );

    // Assert: no duplicate (plan_id, task_id) pairs in the events.
    // A duplicate would indicate the emit function was called twice for
    // the same dispatch.
    let mut seen: HashSet<(&str, &str)> = HashSet::new();
    for ev in &plan_events {
        let key = (ev.plan_id.as_str(), ev.task_id.as_str());
        assert!(
            seen.insert(key),
            "duplicate efficiency event for ({}, {}) — cost would be double-counted",
            ev.plan_id,
            ev.task_id,
        );
    }

    // Assert: total cost = sum (not zero, not inflated).
    let total_cost: f64 = plan_events.iter().map(|e| e.cost_usd).sum();
    // Cost should be non-negative.
    assert!(
        total_cost >= 0.0,
        "total cost must be non-negative, got {total_cost}"
    );
}

/// Verify that when multiple distinct tasks run, each task gets its own
/// efficiency event and none are duplicated.
#[tokio::test]
async fn efficiency_events_are_not_duplicated_across_attempts() {
    let tmp = TempDir::new().expect("tempdir");
    let workdir = tmp.path();

    common::setup_sample_plan_workspace(workdir);
    let _output = common::run_sample_plan(workdir);

    let efficiency_path = workdir.join(".roko").join("learn").join("efficiency.jsonl");

    if !efficiency_path.exists() {
        return; // No events — acceptable
    }

    let events = read_efficiency_events(&efficiency_path)
        .await
        .expect("read efficiency events");

    // Verify the basic structure of each event.
    for ev in &events {
        assert!(
            !ev.plan_id.is_empty(),
            "efficiency event must have a non-empty plan_id"
        );
        assert!(
            !ev.task_id.is_empty(),
            "efficiency event must have a non-empty task_id"
        );
        assert!(
            ev.cost_usd >= 0.0,
            "cost_usd must be >= 0, got {} for task {}",
            ev.cost_usd,
            ev.task_id
        );
    }

    // Verify total count is reasonable (not wildly inflated).
    // The mock plan has 1 task and max_retries=1, so at most 2 events.
    let plan_count = events
        .iter()
        .filter(|e| e.plan_id == common::SAMPLE_PLAN_ID)
        .count();
    assert!(
        plan_count <= 2,
        "expected at most 2 efficiency events (1 attempt + 1 retry) for plan {}, got {}",
        common::SAMPLE_PLAN_ID,
        plan_count
    );
}
```
