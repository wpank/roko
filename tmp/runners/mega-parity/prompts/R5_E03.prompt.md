### How to write JSONL files

```rust
use std::io::Write;

fn write_jsonl<T: serde::Serialize>(path: &std::path::Path, records: &[T]) {
    std::fs::create_dir_all(path.parent().unwrap()).unwrap();
    let mut file = std::fs::File::create(path).unwrap();
    for record in records {
        let line = serde_json::to_string(record).unwrap();
        writeln!(file, "{line}").unwrap();
    }
}
```

/// Efficiency log: orchestrate writes to `.roko/learn/efficiency.jsonl`;
/// `read_project_efficiency_events` reads from the same path.
#[tokio::test]
async fn efficiency_write_path_matches_read_path() {
    let tmp = TempDir::new().expect("tempdir");
    let workdir = tmp.path();

    let efficiency_path = workdir.join(".roko").join("learn").join("efficiency.jsonl");

    // Write 3 fake efficiency events.
    let events = vec![
        sample_efficiency_event("plan-a", "T1"),
        sample_efficiency_event("plan-a", "T2"),
        sample_efficiency_event("plan-b", "T1"),
    ];
    write_jsonl(&efficiency_path, &events);

    // Read back using the same function the `learn efficiency` command uses.
    let read_back = read_efficiency_events(&efficiency_path)
        .await
        .expect("read efficiency events");

    assert_eq!(
        read_back.len(),
        3,
        "efficiency reader found {} events, expected 3",
        read_back.len()
    );
    assert_eq!(read_back[0].plan_id, "plan-a");
    assert_eq!(read_back[0].task_id, "T1");
    assert_eq!(read_back[1].task_id, "T2");
    assert_eq!(read_back[2].plan_id, "plan-b");
}

/// Cascade-router path: `print_learn_router` reads from `.roko/learn/cascade-router.json`.
/// Verify the file exists at that path after we write it.
#[test]
fn cascade_router_path_is_correct() {
    let tmp = TempDir::new().expect("tempdir");
    let workdir = tmp.path();

    // Write a minimal cascade-router JSON.
    let router_path = workdir
        .join(".roko")
        .join("learn")
        .join("cascade-router.json");
    std::fs::create_dir_all(router_path.parent().unwrap()).expect("create dirs");

    let router_json = serde_json::json!({
        "model_slugs": ["claude-sonnet-4-6"],
        "arms": [
            {
                "model_slug": "claude-sonnet-4-6",
                "observations": 5,
                "mean_reward": 0.8
            }
        ]
    });
    std::fs::write(&router_path, serde_json::to_string_pretty(&router_json).unwrap())
        .expect("write cascade-router.json");

    // The read side uses the same path.
    assert!(
        router_path.exists(),
        "cascade-router.json must be at .roko/learn/cascade-router.json"
    );

    let content = std::fs::read_to_string(&router_path).expect("read cascade-router.json");
    let parsed: serde_json::Value = serde_json::from_str(&content).expect("parse json");

    let slugs = parsed["model_slugs"].as_array().expect("model_slugs array");
    assert_eq!(slugs.len(), 1);
    assert_eq!(slugs[0].as_str().unwrap(), "claude-sonnet-4-6");

    let arms = parsed["arms"].as_array().expect("arms array");
    let total_obs: u64 = arms
        .iter()
        .filter_map(|arm| arm.get("observations").and_then(|v| v.as_u64()))
        .sum();
    assert_eq!(total_obs, 5, "total_obs mismatch");
}

/// Experiments path: `print_learn_experiments` reads from `.roko/learn/experiments.json`.
#[test]
fn experiments_path_is_correct() {
    use roko_learn::prompt_experiment::ExperimentStore;

    let tmp = TempDir::new().expect("tempdir");
    let workdir = tmp.path();

    let experiments_path = workdir
        .join(".roko")
        .join("learn")
        .join("experiments.json");
    std::fs::create_dir_all(experiments_path.parent().unwrap()).expect("create dirs");

    // Write a minimal experiments JSON (empty store).
    let store = ExperimentStore::load_or_new(&experiments_path);
    // Just having loaded+created the store should not error.
    // The file may or may not exist on disk at this point.
    // Force creation:
    std::fs::write(&experiments_path, serde_json::to_string(&store).unwrap_or_default())
        .ok(); // ok if this errors — the default store has nothing to serialize

    // Verify the path is readable.
    let read_store = ExperimentStore::load_or_new(&experiments_path);
    assert_eq!(
        read_store.running_count(),
        0,
        "empty store should have 0 running experiments"
    );
    assert_eq!(
        read_store.concluded_count(),
        0,
        "empty store should have 0 concluded experiments"
    );
}
