**After**:
```rust
/// Build the CORS layer from configured origins.
///
/// Behaviour:
/// - `cors_origins` non-empty → allow exactly those origins.
/// - `cors_origins` empty + `unsafe_public=false` → allow localhost:* and 127.0.0.1:* only.
/// - `cors_origins` empty + `unsafe_public=true` → allow all origins (*); emits a warning.
pub fn cors_layer(cors_origins: &[String], unsafe_public: bool) -> CorsLayer {
    if !cors_origins.is_empty() {
        let allowed: Vec<axum::http::HeaderValue> =
            cors_origins.iter().filter_map(|o| o.parse().ok()).collect();
        return CorsLayer::new()
            .allow_origin(allowed)
            .allow_methods(Any)
            .allow_headers(Any);
    }

    if unsafe_public {
        tracing::warn!(
            "CORS is unrestricted (allow *) because server.unsafe_public_cors = true \
             and no cors_origins are configured. Set cors_origins to limit access."
        );
        return CorsLayer::permissive();
    }

    // Default: local-only. Allow any port on localhost / 127.0.0.1.
    // tower-http CorsLayer does not support wildcard port matching natively,
    // so we use a closure predicate.
    CorsLayer::new()
        .allow_origin(tower_http::cors::AllowOrigin::predicate(
            |origin: &axum::http::HeaderValue, _parts: &axum::http::request::Parts| {
                let Ok(s) = origin.to_str() else {
                    return false;
                };
                is_local_origin(s)
            },
        ))
        .allow_methods(Any)
        .allow_headers(Any)
}

/// Returns `true` when the origin is a localhost or loopback origin on any port.
fn is_local_origin(origin: &str) -> bool {
    // Strip scheme prefix.
    let host = if let Some(rest) = origin.strip_prefix("https://") {
        rest
    } else if let Some(rest) = origin.strip_prefix("http://") {
        rest
    } else {
        return false;
    };
    // Strip optional port suffix.
    let host_no_port = host.split(':').next().unwrap_or(host);
    matches!(host_no_port, "localhost" | "127.0.0.1" | "::1")
}
```
