### Create `tests/security-smoke.sh`

**File:** `tests/security-smoke.sh` (new file, mark executable)

```bash
#!/usr/bin/env bash
set -euo pipefail

# Security smoke test for roko serve.
# Run after cargo build. Requires a free port.
# Exits 0 if all checks pass, nonzero otherwise.

PASS=0
FAIL=0

pass() { echo "  PASS: $1"; PASS=$((PASS + 1)); }
fail() { echo "  FAIL: $1"; FAIL=$((FAIL + 1)); }

echo "=== Security Smoke Test ==="
echo ""

# --- Test 1: Default bind is localhost ---
echo "[1] Default bind is localhost"
# Start server with defaults, capture startup output.
WORKDIR=$(mktemp -d)
mkdir -p "$WORKDIR/.roko"
cat > "$WORKDIR/roko.toml" <<'TOML'
[agent]
model = "test-model"
TOML

# Start in background, capture output.
cargo run -p roko-cli -- serve --workdir "$WORKDIR" 2>&1 &
SERVER_PID=$!
sleep 3

# Check that localhost responds.
if curl -sf http://127.0.0.1:6677/health > /dev/null 2>&1; then
    pass "Server responds on 127.0.0.1:6677"
else
    fail "Server does not respond on 127.0.0.1:6677"
fi

# --- Test 2: Terminal disabled by default ---
echo ""
echo "[2] Terminal disabled by default"
HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" http://127.0.0.1:6677/api/terminal/create -X POST 2>/dev/null || echo "000")
if [ "$HTTP_CODE" = "403" ] || [ "$HTTP_CODE" = "404" ]; then
    pass "Terminal creation rejected (HTTP $HTTP_CODE)"
else
    fail "Terminal creation NOT rejected (HTTP $HTTP_CODE, expected 403 or 404)"
fi

# --- Test 3: CORS does not allow arbitrary origins ---
echo ""
echo "[3] CORS restricts arbitrary origins"
CORS_HEADER=$(curl -s -H "Origin: http://evil.com" -X OPTIONS \
    -D - http://127.0.0.1:6677/api/status 2>/dev/null \
    | grep -i "access-control-allow-origin" || true)
if echo "$CORS_HEADER" | grep -qi "evil.com"; then
    fail "CORS allows evil.com origin"
else
    pass "CORS does not allow evil.com origin"
fi

# --- Test 4: Public bind without auth is rejected ---
echo ""
echo "[4] Public bind without auth is rejected"
# Kill current server first.
kill $SERVER_PID 2>/dev/null || true
wait $SERVER_PID 2>/dev/null || true
sleep 1

# Attempt to bind to 0.0.0.0 without auth.
PUBLIC_OUTPUT=$(cargo run -p roko-cli -- serve --workdir "$WORKDIR" --bind 0.0.0.0 2>&1 || true) &
PUBLIC_PID=$!
sleep 3

# The server should have exited with an error or refused to start.
if ! kill -0 $PUBLIC_PID 2>/dev/null; then
    # Process exited — check if error message mentions auth/unsafe.
    if echo "$PUBLIC_OUTPUT" | grep -qi "auth\|unsafe.public\|acknowledge"; then
        pass "Public bind rejected with auth/unsafe message"
    else
        pass "Public bind rejected (process exited)"
    fi
else
    fail "Server started on 0.0.0.0 without auth — SECURITY ISSUE"
    kill $PUBLIC_PID 2>/dev/null || true
fi

# --- Cleanup ---
echo ""
kill $SERVER_PID 2>/dev/null || true
kill $PUBLIC_PID 2>/dev/null || true
rm -rf "$WORKDIR"

# --- Summary ---
echo "=== Results: $PASS passed, $FAIL failed ==="
if [ "$FAIL" -gt 0 ]; then
    exit 1
fi
exit 0
```
