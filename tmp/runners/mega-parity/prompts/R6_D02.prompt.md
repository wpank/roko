```rust
//! Tests for server bind safety validation.
//!
//! These tests verify that binding to public addresses (0.0.0.0, ::)
//! without authentication is rejected at startup, preventing accidental
//! exposure of the terminal and API.

use roko_core::config::schema::RokoConfig;
use roko_core::config::serve::{ServeAuthConfig, ServerConfig};

/// Helper: build a RokoConfig with the given bind address and auth setting.
fn config_with_bind(bind: &str, auth_enabled: bool) -> RokoConfig {
    let mut config = RokoConfig::default();
    config.server.bind = bind.to_string();
    config.server.port = 0; // OS-assigned port, never actually binds in these tests
    config.serve.auth.enabled = auth_enabled;
    config
}

#[test]
fn loopback_bind_without_auth_succeeds() {
    // 127.0.0.1 is always safe regardless of auth setting.
    let config = config_with_bind("127.0.0.1", false);
    // Call the validation function added by R6_A02.
    // Adjust the function name to match what R6_A02 actually adds:
    let result = roko_serve::validate_bind_safety(
        &config.server.bind,
        &config.serve.auth,
    );
    assert!(result.is_ok(), "loopback bind should always succeed");
}

#[test]
fn public_bind_without_auth_fails() {
    // 0.0.0.0 + auth=false → must be rejected.
    let config = config_with_bind("0.0.0.0", false);
    let result = roko_serve::validate_bind_safety(
        &config.server.bind,
        &config.serve.auth,
    );
    assert!(result.is_err(), "public bind without auth must fail");

    // Error message should mention auth or unsafe-public.
    let err_msg = result.unwrap_err().to_string();
    assert!(
        err_msg.contains("auth") || err_msg.contains("unsafe") || err_msg.contains("public"),
        "error should mention auth or unsafe-public, got: {err_msg}"
    );
}

#[test]
fn public_bind_with_auth_succeeds() {
    // 0.0.0.0 + auth=true → should be allowed.
    let config = config_with_bind("0.0.0.0", true);
    let result = roko_serve::validate_bind_safety(
        &config.server.bind,
        &config.serve.auth,
    );
    assert!(result.is_ok(), "public bind with auth should succeed");
}

#[test]
fn ipv6_wildcard_without_auth_fails() {
    // [::] is the IPv6 equivalent of 0.0.0.0.
    let config = config_with_bind("::", false);
    let result = roko_serve::validate_bind_safety(
        &config.server.bind,
        &config.serve.auth,
    );
    assert!(result.is_err(), "IPv6 wildcard bind without auth must fail");
}

#[test]
fn localhost_string_without_auth_succeeds() {
    // "localhost" should be treated as loopback.
    let config = config_with_bind("localhost", false);
    let result = roko_serve::validate_bind_safety(
        &config.server.bind,
        &config.serve.auth,
    );
    assert!(result.is_ok(), "localhost bind should always succeed");
}
```

### Step 3 — If R6_A02 validation function doesn't exist yet

If `validate_bind_safety` (or equivalent) hasn't been added by R6_A02, create a minimal one in `crates/roko-serve/src/lib.rs`:

```rust
/// Validate that the bind address is safe given the auth configuration.
///
/// Public addresses (0.0.0.0, ::) require authentication to be enabled.
/// Loopback addresses (127.0.0.1, localhost, ::1) are always allowed.
pub fn validate_bind_safety(
    bind: &str,
    auth: &roko_core::config::serve::ServeAuthConfig,
) -> anyhow::Result<()> {
    let is_loopback = matches!(bind, "127.0.0.1" | "localhost" | "::1");
    let is_public = matches!(bind, "0.0.0.0" | "::");

    if is_public && !auth.enabled {
        anyhow::bail!(
            "Refusing to bind to {bind} without authentication enabled. \
             Either enable auth (serve.auth.enabled = true) or use \
             127.0.0.1 for local-only access. To override, set \
             ROKO_ACKNOWLEDGE_PUBLIC_RISK=1."
        );
    }

    Ok(())
}
```

Then wire `validate_bind_safety()` into `run_server_with_state()` before the `TcpListener::bind()` call:

