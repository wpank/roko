fn push_tool_outputs(
    term: &mut InlineTerminal,
    theme: &Theme,
    tool_outputs: &[ToolOutput],
) -> std::io::Result<()> {
    if tool_outputs.is_empty() {
        return Ok(());
    }
    for output in tool_outputs {
        let tool_label = output.tool_name.as_deref().unwrap_or("tool");
        // Show first line of output as preview, truncated
        let preview = output
            .content
            .lines()
            .next()
            .unwrap_or("")
            .chars()
            .take(80)
            .collect::<String>();
        let line_count = output.content.lines().count();
        let suffix = if line_count > 1 {
            format!(" (+{} lines)", line_count - 1)
        } else {
            String::new()
        };

        let tool_sym = symbols::TOOL;
        term.push_lines(&[Line::from(vec![
            Span::styled(
                format!("  {tool_sym} "),
                Style::default().fg(Theme::TEXT_DIM),
            ),
            Span::styled(
                tool_label.to_string(),
                Style::default().fg(theme.info).add_modifier(Modifier::BOLD),
            ),
            Span::styled(
                format!("  {preview}{suffix}"),
                Style::default().fg(Theme::TEXT_DIM),

    Ok(())
}

Replace the ENTIRE function body (keeping the signature) with improved display:

```rust
fn push_tool_outputs(
    term: &mut InlineTerminal,
    theme: &Theme,
    tool_outputs: &[ToolOutput],
) -> std::io::Result<()> {
    if tool_outputs.is_empty() {
        return Ok(());
    }
    for output in tool_outputs {
        let tool_label = output.tool_name.as_deref().unwrap_or("tool");

        // Infer success/failure from content
        let is_error = output.content.is_empty()
            || output.content.starts_with("Error:")
            || output.content.starts_with("error:");

        // Summary: line count + brief preview for success; first line for error
        let summary = if is_error {
            let first = output.content.lines().next().unwrap_or("failed");
            let first = if first.len() > 60 {
                format!("{}...", &first[..60])
            } else {
                first.to_string()
            };
            first
        } else {
            let line_count = output.content.lines().count();
            let char_count = output.content.len();
            if line_count > 1 {
                format!("{line_count} lines")
            } else if char_count > 0 {
                format!("{char_count} bytes")
            } else {
                "ok".to_string()
            }
        };

        // Choose indicator symbol
        let (indicator, indicator_style) = if is_error {
            (symbols::FAIL, Style::default().fg(Theme::EMBER))
        } else {
            (symbols::PASS, Style::default().fg(Theme::SAGE))
        };

        term.push_lines(&[Line::from(vec![
            Span::styled(format!("  {indicator} "), indicator_style),
            Span::styled(
                format!("[{tool_label}]"),
                Style::default().fg(theme.info).add_modifier(Modifier::BOLD),
            ),
            Span::styled(
                format!("  {summary}"),
                Style::default().fg(if is_error { Theme::EMBER } else { Theme::TEXT_DIM }),
            ),
        ])])?;
    }
    term.push_lines(&[Line::raw("")])?;
    Ok(())
}
```

**Important**: `Theme::EMBER` is the red/error color and `Theme::SAGE` is the green/success color. Check the `Theme` struct in the TUI theme file to confirm these constants exist. They are used throughout `chat_inline.rs`.
