//!
//! Writes realistic-looking but clearly-labelled (source:"seed") data to
//! all learning store files so dashboards are populated on first launch.

use anyhow::{Context as _, Result};
use chrono::{Duration, Utc};
use serde_json::json;
use std::path::Path;

/// Write seed data into `workdir/.roko/`.
///
/// Safe to call on an existing workspace — skips any file that already
/// contains at least one non-empty line.
pub fn seed_demo_data(workdir: &Path) -> Result<()> {
    let roko_dir = workdir.join(".roko");
    seed_efficiency_events(&roko_dir)?;
    seed_episodes(&roko_dir)?;
    seed_cascade_router(&roko_dir)?;
    seed_knowledge(&roko_dir)?;
    Ok(())
}

fn already_has_data(path: &Path) -> bool {
    if !path.exists() {
        return false;
    }
    std::fs::read_to_string(path)
        .map(|s| s.lines().any(|l| !l.trim().is_empty()))
        .unwrap_or(false)
}

fn append_jsonl(path: &Path, lines: &[serde_json::Value]) -> Result<()> {
    if already_has_data(path) {
        return Ok(());
    }
    std::fs::create_dir_all(path.parent().unwrap())?;
    let mut out = String::new();
    for v in lines {
        out.push_str(&serde_json::to_string(v)?);
        out.push('\n');
    }
    std::fs::write(path, out).with_context(|| format!("write {}", path.display()))
}
