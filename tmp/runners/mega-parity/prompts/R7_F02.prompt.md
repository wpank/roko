### Step 2: Add `detect_file_changes()` to `crates/roko-acp/src/runner.rs`

Add this free function near the top of `runner.rs` (after the imports, before `run_with_workflow_engine`):

```rust
use crate::types::{FileChangeNotification, FileChangeType};

/// Detect files changed in the most recent commit via `git diff --name-status HEAD~1 HEAD`.
/// Returns empty vec if git is unavailable or the repo has no prior commits (e.g. initial commit).
fn detect_file_changes(workdir: &Path) -> Vec<FileChangeNotification> {
    let output = match std::process::Command::new("git")
        .args(["diff", "--name-status", "HEAD~1", "HEAD"])
        .current_dir(workdir)
        .output()
    {
        Ok(o) if o.status.success() => o,
        _ => return Vec::new(),
    };

    let stdout = String::from_utf8_lossy(&output.stdout);
    let mut changes = Vec::new();

    for line in stdout.lines() {
        let mut parts = line.splitn(2, '\t');
        let status = match parts.next() {
            Some(s) => s.trim(),
            None => continue,
        };
        let path = match parts.next() {
            Some(p) => p.trim().to_string(),
            None => continue,
        };

        // Skip binary/lock files by extension.
        if path.ends_with(".lock")
            || path.ends_with(".png")
            || path.ends_with(".jpg")
            || path.ends_with(".jpeg")
            || path.ends_with(".gif")
            || path.ends_with(".ico")
        {
            continue;
        }

        let change_type = match status.chars().next() {
            Some('A') => FileChangeType::Added,
            Some('M') => FileChangeType::Modified,
            Some('D') => FileChangeType::Deleted,
            Some('R') => FileChangeType::Renamed,
            _ => FileChangeType::Modified,
        };

        changes.push(FileChangeNotification { path, change_type });
    }

    changes
}
```

Note: uses `std::process::Command` (sync), NOT `tokio::process::Command`, because this is called inside an `async fn` but the git command completes quickly (committed files only).

### Step 4: Emit file change notifications in the `PipelineAction::Commit` handler

Find the `PipelineAction::Commit` block (lines 721-741). Replace it with:

```rust
PipelineAction::Commit => {
    let commit_result = run_commit(
        session_id,
        workdir,
        &run.pipeline.original_prompt,
        &cancel_token,
        &event_sender,
    )
    .await;
    action = match commit_result {
        Ok(msg) => {
            // Detect actually changed files from git after commit succeeds.
            let changes = detect_file_changes(workdir);
            let real_count = changes.len() as u32;

            // Emit one ToolCallStart + ToolCallComplete per changed file.
            for change in &changes {
                let type_char = match change.change_type {
                    FileChangeType::Added => "+",
                    FileChangeType::Modified => "~",
                    FileChangeType::Deleted => "-",
                    FileChangeType::Renamed => ">",
                };
                let tool_id = format!(
                    "file-change-{}",
                    change.path.replace('/', "-").replace('.', "_")
                );
                let _ = event_sender.send(CognitiveEvent::ToolCallStart {
                    tool_call_id: tool_id.clone(),
                    title: format!("{type_char} {}", change.path),
                    kind: match change.change_type {
                        FileChangeType::Added => ToolCallKind::Create,
                        FileChangeType::Deleted => ToolCallKind::Delete,
                        _ => ToolCallKind::Edit,
                    },
                }).await;
                let _ = event_sender.send(CognitiveEvent::ToolCallComplete {
                    tool_call_id: tool_id,
                    status: ToolCallStatus::Completed,
                    content: vec![],
                }).await;
            }

            // Update pipeline state with real file count.
            run.pipeline.files_changed = real_count;
            run.pipeline.step(PipelineEvent::CommitDone { message: msg })
        }
        Err(e) => {
            error!(error = %e, "commit failed");
            run.pipeline.step(PipelineEvent::CommitDone {
                message: format!("(commit failed: {e})"),
            })
        }
    };
}
```
