### Change 2: Add `phase_badge()` function to `runner.rs`

**File:** `crates/roko-acp/src/runner.rs`

Add this function after the `emit_plan_update()` function (around line 800):

```rust
/// Generate an inline phase badge string for display between agent outputs.
/// Returns `None` for `Pending` (nothing has happened yet).
fn phase_badge(phase: &PipelinePhase, iteration: u32) -> Option<String> {
    let (icon, name) = match phase {
        PipelinePhase::Strategizing => ("⊕", "Strategizing"),
        PipelinePhase::Implementing => ("♦", "Implementing"),
        PipelinePhase::Gating => ("⊕", "Running gates"),
        PipelinePhase::AutoFixing => ("🔧", "Auto-fixing"),
        PipelinePhase::Reviewing => ("🔍", "Reviewing"),
        PipelinePhase::Committing => ("✓", "Committing"),
        PipelinePhase::Complete => ("✅", "Complete"),
        PipelinePhase::Halted { .. } => ("⛔", "Halted"),
        PipelinePhase::Cancelled => ("⊘", "Cancelled"),
        PipelinePhase::Pending => return None,
    };

    // Show iteration number only when > 1 and phase is retry-capable.
    let badge = if iteration > 1
        && matches!(
            phase,
            PipelinePhase::Implementing | PipelinePhase::AutoFixing | PipelinePhase::Gating
        )
    {
        format!("\n\n{icon} **{name} (iter {iteration})**\n\n")
    } else {
        format!("\n\n{icon} **{name}**\n\n")
    };

    Some(badge)
}
```
