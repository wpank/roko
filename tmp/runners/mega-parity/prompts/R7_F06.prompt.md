```rust
fn read_file_context(uris: &[String], workdir: &Path) -> String {
    let mut context = String::new();
    let workdir_canonical = workdir.canonicalize().unwrap_or_else(|_| workdir.to_path_buf());
    for uri in uris {
        let path_str = uri.strip_prefix("file://").unwrap_or(uri);

        let canonical = match path.canonicalize() { Ok(p) => p, Err(_) => continue };

            continue;

                    format!("{}... [truncated at 32KB]", &contents[..32_768])
                } else { contents };
                let rel_path = canonical.strip_prefix(&workdir_canonical).unwrap_or(&canonical);
                context.push_str(&format!("<file path=\"{}\">\n{}\n</file>\n", rel_path.display(), truncated));
            }
            Err(e) => warn!(path = %canonical.display(), error = %e, "failed to read file for context"),
        }
    }
    context
}
```

## Problem Statement (refined)

**File context from `file://` URIs is already wired** via `extract_resource_uris` + `read_file_context` + injection into both dispatch paths.

**What is NOT wired:**
1. `@branch-diff`, `@diff` — git diff context
2. `@recent-commits`, `@git-log` — git log context
3. `@git-status`, `@status` — git status context
4. General `@<path>` file path mentions embedded in Text blocks (not Resource blocks)

These @-mentions appear as `ContentBlock::Text { text: "@branch-diff" }` or as part of `ContentBlock::Text { text: "fix the issue in @src/main.rs" }`.

The current code ignores all `ContentBlock::Text` blocks for context purposes (they become part of `extract_prompt_text()` which just joins them).

**The `ContextProvider` types** from the original prompt do not need to be added to `session.rs`. The resolution is purely a dispatch-time function.

```rust
/// Resolve all context items from prompt blocks into a single context string.
///
/// Handles:
/// - `ContentBlock::Resource { ResourceRef::File { uri } }` → file content
/// - `ContentBlock::Text` starting with `@` → @-mention resolution (git, files)
///
/// Returns empty string if no context blocks are present or all resolution fails.
/// Never blocks dispatch — failed resolutions are logged and skipped.
pub(crate) async fn resolve_context_items(
    prompt: &[ContentBlock],
    workdir: &Path,
) -> String {
    let mut parts: Vec<String> = Vec::new();

    for block in prompt {
        match block {
            ContentBlock::Resource {
                resource: crate::types::ResourceRef::File { uri },
            } => {
                match resolve_file_uri(uri, workdir).await {
                    Ok(content) => parts.push(content),
                    Err(e) => {
                        tracing::warn!(uri, error = %e, "failed to resolve file resource URI");
                    }
                }
            }
            ContentBlock::Text { text } if text.trim_start().starts_with('@') => {
                // Extract the first @-mention token from the text block.
                let mention = text.trim_start().trim_start_matches('@').trim();
                // Take only the first word (handle "@branch-diff fix the bug")
                let label = mention.split_whitespace().next().unwrap_or(mention);
                match resolve_at_mention(label, workdir).await {
                    Ok(content) => parts.push(content),
                    Err(e) => {
                        tracing::warn!(label, error = %e, "failed to resolve @-mention");
                    }
                }
            }
            _ => {} // Non-resource Text and Diff blocks pass through as user text
        }
    }

    parts.join("\n\n")
}

/// Resolve a `file://` URI to XML-tagged file content.
async fn resolve_file_uri(uri: &str, workdir: &Path) -> anyhow::Result<String> {
    let path_str = uri.strip_prefix("file://").unwrap_or(uri);
    let path = std::path::Path::new(path_str);

    let full_path = if path.is_absolute() {
        path.to_path_buf()
    } else {
        workdir.join(path)
    };

    // Security: verify path stays within workdir.
    let workdir_canonical = workdir.canonicalize().unwrap_or_else(|_| workdir.to_path_buf());
    let canonical = full_path
        .canonicalize()
        .map_err(|e| anyhow::anyhow!("cannot canonicalize {}: {e}", full_path.display()))?;
    if !canonical.starts_with(&workdir_canonical) {
        return Err(anyhow::anyhow!("path {} is outside workdir", canonical.display()));
    }

    let content = tokio::fs::read_to_string(&canonical).await?;
    let rel_path = canonical.strip_prefix(&workdir_canonical).unwrap_or(&canonical);
    let truncated = if content.len() > 32_768 {
        format!("{}... [truncated at 32KB]", &content[..32_768])
    } else {
        content
    };
    Ok(format!(
        "<file path=\"{}\">\n{}\n</file>",
        rel_path.display(),
        truncated
    ))
}

/// Resolve an @-mention label to context text.
async fn resolve_at_mention(label: &str, workdir: &Path) -> anyhow::Result<String> {
    match label {
        "branch-diff" | "diff" => {
            let output = tokio::process::Command::new("git")
                .args(["diff"])
                .current_dir(workdir)
                .output()
                .await?;
            let diff = String::from_utf8_lossy(&output.stdout);
            let truncated = if diff.len() > 10_240 {
                format!("{}...\n[truncated]", &diff[..10_240])
            } else {
                diff.to_string()
            };
            Ok(format!("--- branch diff ---\n{truncated}"))
        }
        "recent-commits" | "git-log" | "log" => {
            let output = tokio::process::Command::new("git")
                .args(["log", "--oneline", "-20"])
                .current_dir(workdir)
                .output()
                .await?;
            Ok(format!(
                "--- recent commits ---\n{}",
                String::from_utf8_lossy(&output.stdout)
            ))
        }
        "status" | "git-status" => {
            let output = tokio::process::Command::new("git")
                .args(["status", "--short"])
                .current_dir(workdir)
                .output()
                .await?;
            Ok(format!(
                "--- git status ---\n{}",
                String::from_utf8_lossy(&output.stdout)
            ))
        }
        _ => {
            // Treat as a relative file path.
            let path = workdir.join(label);
            if path.exists() {
                let content = tokio::fs::read_to_string(&path).await?;
                let truncated = if content.len() > 10_240 {
                    format!("{}...\n[truncated]", &content[..10_240])
                } else {
                    content
                };
                Ok(format!("--- {} ---\n{truncated}", label))
            } else {
                Err(anyhow::anyhow!("unknown @-mention and no matching file: @{label}"))
            }
        }
    }
}
```
