```rust
/// Classification of the gate failure cause.
#[derive(Debug)]
enum GateErrorType {
    CompileError {
        file: String,
        line: u32,
        message: String,
    },
    TestFailure {
        test_name: String,
        expected: Option<String>,
        actual: Option<String>,
    },
    ClippyWarning {
        lint: String,
        location: String,
    },
    RuntimePanic {
        message: String,
    },
    Unknown,
}

impl GateErrorType {
    fn description(&self) -> &'static str {
        match self {
            Self::CompileError { .. } => "compile error",
            Self::TestFailure { .. } => "test failure",
            Self::ClippyWarning { .. } => "clippy warning",
            Self::RuntimePanic { .. } => "runtime panic",
            Self::Unknown => "unknown error",
        }
    }
}
```

### Step 2: Add error classifier function

```rust
fn classify_gate_error(output: &str) -> GateErrorType {
    // Rust compiler errors: "src/foo.rs:42:5: error[E0308]: ..."
    // or "error[E0308]: mismatched types"
    for line in output.lines() {
        if line.contains("error[E") {
            // Try to extract file:line from surrounding context
            let file = output
                .lines()
                .find(|l| l.contains("-->") && l.contains(".rs:"))
                .and_then(|l| l.split("-->").nth(1))
                .map(|s| s.trim().split(':').next().unwrap_or("").to_string())
                .unwrap_or_default();
            let line_no: u32 = output
                .lines()
                .find(|l| l.contains("-->") && l.contains(".rs:"))
                .and_then(|l| l.split(':').nth(1))
                .and_then(|s| s.parse().ok())
                .unwrap_or(0);
            let message = line
                .split("error[E")
                .nth(1)
                .and_then(|s| s.split("]: ").nth(1))
                .unwrap_or(line)
                .chars()
                .take(120)
                .collect();
            return GateErrorType::CompileError { file, line: line_no, message };
        }
    }

    // Test failures: "test module::test_name ... FAILED"
    if output.contains("test result: FAILED") || output.contains("FAILED") {
        let test_name = output
            .lines()
            .find(|l| l.contains("FAILED") && !l.contains("test result:"))
            .and_then(|l| l.split_whitespace().nth(1))
            .unwrap_or("unknown")
            .trim_end_matches("...")
            .to_string();
        let expected = output
            .lines()
            .find(|l| l.trim().starts_with("left =") || l.trim().starts_with("left:"))
            .map(|l| l.trim().chars().take(80).collect());
        let actual = output
            .lines()
            .find(|l| l.trim().starts_with("right =") || l.trim().starts_with("right:"))
            .map(|l| l.trim().chars().take(80).collect());
        return GateErrorType::TestFailure { test_name, expected, actual };
    }

    // Clippy warnings treated as errors (via -D warnings): "warning: <lint>"
    if output.contains("warning:") && output.contains("error: could not compile") {
        let lint = output
            .lines()
            .find(|l| l.trim().starts_with("warning:"))
            .and_then(|l| l.strip_prefix("warning:"))
            .map(|s| s.trim().chars().take(60).collect())
            .unwrap_or_default();
        let location = output
            .lines()
            .find(|l| l.contains("-->") && l.contains(".rs:"))
            .and_then(|l| l.split("-->").nth(1))
            .map(|s| s.trim().chars().take(60).collect())
            .unwrap_or_default();
        return GateErrorType::ClippyWarning { lint, location };
    }

    // Runtime panic
    if output.contains("panicked at") {
        let message = output
            .lines()
            .find(|l| l.contains("panicked at"))
            .map(|l| l.chars().take(120).collect())
            .unwrap_or_default();
        return GateErrorType::RuntimePanic { message };
    }

    GateErrorType::Unknown
}

fn extract_root_cause(output: &str) -> String {
    output
        .lines()
        .find(|l| {
            l.contains("error[E")
                || l.contains("error:")
                || l.contains("FAILED")
                || l.contains("panicked at")
        })
        .unwrap_or("could not determine root cause")
        .trim()
        .chars()
        .take(200)
        .collect()
}
```

### Step 3: Add the autopsy analysis function

```rust
async fn analyze_gate_failure(
    error_output: &str,
    workdir: &Path,
    iteration: u32,
) -> GateAutopsy {
    let error_type = classify_gate_error(error_output);
    let root_cause = extract_root_cause(error_output);
    let mut causal_chain: Vec<String> = Vec::new();

    // 1. What did the agent change? (git diff --stat HEAD)
    if let Ok(output) = tokio::process::Command::new("git")
        .args(["diff", "--stat", "HEAD"])
        .current_dir(workdir)
        .output()
        .await
    {
        if output.status.success() {
            let stat = String::from_utf8_lossy(&output.stdout);
            if let Some(summary) = stat.lines().last() {
                let summary = summary.trim();
                if !summary.is_empty() {
                    causal_chain.push(format!("Changes: {summary}"));
                }
            }
        }
    }

    // 2. Error-specific chain entries
    match &error_type {
        GateErrorType::CompileError { file, line, message } => {
            if !file.is_empty() {
                causal_chain.push(format!("Compile error at {file}:{line}"));
            }
            causal_chain.push(format!("Message: {message}"));
        }
        GateErrorType::TestFailure { test_name, expected, actual } => {
            causal_chain.push(format!("Test `{test_name}` failed"));
            if let (Some(exp), Some(act)) = (expected, actual) {
                causal_chain.push(format!("Expected: {exp}"));
                causal_chain.push(format!("Got: {act}"));
            }
        }
        GateErrorType::ClippyWarning { lint, location } => {
            causal_chain.push(format!("Clippy: {lint}"));
            if !location.is_empty() {
                causal_chain.push(format!("In: {location}"));
            }
        }
        GateErrorType::RuntimePanic { message } => {
            causal_chain.push(format!("Panic: {message}"));
        }
        GateErrorType::Unknown => {
            causal_chain.push("Error type: unclassified".to_string());
        }
    }

    // 3. Cross-reference with past episodes
    let mut similar_past: Option<(String, String)> = None;
    let episodes_path = workdir.join(".roko").join("episodes.jsonl");
    if episodes_path.exists() {
        use roko_learn::episode_logger::EpisodeLogger;
        if let Ok(episodes) = EpisodeLogger::read_all_lossy(&episodes_path).await {
            let error_sig: String = root_cause.chars().take(60).collect();
            // Scan last 50 episodes most-recent-first
            for ep in episodes.iter().rev().take(50) {
                if !ep.success {
                    if let Some(reason) = &ep.failure_reason {
                        if reason.contains(&error_sig) || similar_strings(reason, &error_sig) {
                            let resolution = ep.reflection
                                .as_deref()
                                .unwrap_or("(resolution not recorded)");
                            similar_past = Some((ep.task_id.clone(), resolution.chars().take(120).collect()));
                            break;
                        }
                    }
                }
            }
        }
    }

    // 4. Confidence score
    let confidence = match (&error_type, &similar_past) {
        (GateErrorType::Unknown, None) => 0.30,
        (GateErrorType::Unknown, Some(_)) => 0.60,
        (_, None) => 0.65,
        (_, Some(_)) => 0.89,
    };

    GateAutopsy {
        error_type,
        root_cause,
        causal_chain,
        similar_past,
        confidence,
    }
}

/// Simple word-overlap similarity for error matching (>50% overlap = similar).
fn similar_strings(a: &str, b: &str) -> bool {
    use std::collections::HashSet;
    let words_a: HashSet<&str> = a.split_whitespace().collect();
    let words_b: HashSet<&str> = b.split_whitespace().collect();
    let overlap = words_a.intersection(&words_b).count();
    let total = words_a.len().max(words_b.len());
    total > 3 && (overlap as f64 / total as f64) > 0.5
}
```

```rust
PipelineAction::RunGates => {
    let gate_result = run_gates(
        session_id,
        workdir,
        config.clippy_enabled,
        config.tests_enabled,
        &cancel_token,
        &event_sender,
    )
    .await;
    action = match gate_result {
        Ok(()) => run.pipeline.step(PipelineEvent::GatesPassed),
        Err(e) => {
            let error_str = e.to_string();
            // Run forensic autopsy before transitioning to auto-fixer.
            let autopsy = analyze_gate_failure(&error_str, workdir, run.pipeline.iteration).await;

            // Build and emit the autopsy card.
            let mut card_text = format!(
                "**Root cause:** {}\n\n**Causal chain:**\n",
                autopsy.root_cause
            );
            for (i, step) in autopsy.causal_chain.iter().enumerate() {
                card_text.push_str(&format!("  {}. {}\n", i + 1, step));
            }
            if let Some((ep_id, resolution)) = &autopsy.similar_past {
                card_text.push_str(&format!(
                    "\n**Similar past failure:** episode `{ep_id}`\n  Resolution: {resolution}\n"
                ));
            }
            card_text.push_str(&format!(
                "\n**Confidence:** {:.0}% ({})",
                autopsy.confidence * 100.0,
                autopsy.error_type.description()
            ));

            let autopsy_id = format!("gate-autopsy-{}", run.pipeline.iteration);
            let _ = event_sender.send(CognitiveEvent::ToolCallStart {
                tool_call_id: autopsy_id.clone(),
                title: format!("Gate autopsy — {}", autopsy.error_type.description()),
                kind: ToolCallKind::Other,
            }).await;
            let _ = event_sender.send(CognitiveEvent::ToolCallComplete {
                tool_call_id: autopsy_id,
                status: ToolCallStatus::Completed,
                content: vec![ContentBlock::Text { text: card_text }],
            }).await;

            // Enhance the auto-fixer prompt with autopsy context.
            let enhanced_error = if let Some((_, resolution)) = &autopsy.similar_past {
                format!(
                    "Fix the following {} error. A similar past failure was resolved by: {resolution}\n\nError output:\n{error_str}",
                    autopsy.error_type.description()
                )
            } else {
                format!(
                    "Fix the following {} error:\n\n{error_str}",
                    autopsy.error_type.description()
                )
            };

            run.pipeline.step(PipelineEvent::GateFailed {
                gate: "gate".into(),
                output: enhanced_error,  // Enhanced prompt for auto-fixer
            })
        }
    };
}
```
