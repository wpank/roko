### Step 1: Add a formatter module

New file `crates/roko-acp/src/dream_card.rs`:

```rust
//! Dream-journal card formatting (AD_02).
//!
//! Renders a `roko_dreams::DreamReport` as a ToolCall card content
//! payload suitable for the ACP `session/update` notification emitted
//! at session start.

use chrono::{DateTime, Utc};
use roko_dreams::DreamReport;

use crate::types::{ContentBlock, SessionUpdate, ToolCallKind, ToolCallStatus};

/// Stable tool-call id so the editor can dedupe across multiple session/news
/// in the same process lifetime.
pub fn dream_card_id(report: &DreamReport) -> String {
    format!(
        "dream-{}",
        report.completed_at.timestamp() as u64
    )
}

/// Render a dream report as a ToolCall SessionUpdate.
pub fn render_dream_card(report: &DreamReport) -> SessionUpdate {
    let body = render_body(report);
    SessionUpdate::ToolCall {
        tool_call_id: dream_card_id(report),
        title: format!(
            "🌙 Dream report — consolidated {}",
            relative_when(report.completed_at)
        ),
        kind: ToolCallKind::Other,
        status: ToolCallStatus::Completed,
        content: vec![ContentBlock::Text { text: body }],
        locations: None,
    }
}

fn render_body(r: &DreamReport) -> String {
    let mut out = String::with_capacity(512);

    // Header summary.
    out.push_str(&format!(
        "**Replayed:** {} episodes (of {} total)\n",
        r.processed_episodes, r.total_episodes,
    ));

    // Clusters.
    if !r.clusters.is_empty() {
        out.push_str(&format!("\n**Clusters:** {}\n", r.clusters.len()));
        for c in r.clusters.iter().take(5) {
            // DreamClusterReport fields — adapt to actual struct
            out.push_str(&format!("  - {}\n", cluster_label(c)));
        }
    }

    // Knowledge & playbooks promoted.
    if r.knowledge_entries_written > 0 || r.playbooks_created > 0 {
        out.push_str(&format!(
            "\n**Promoted:** {} knowledge entr{}, {} playbook{}\n",
            r.knowledge_entries_written,
            if r.knowledge_entries_written == 1 { "y" } else { "ies" },
            r.playbooks_created,
            if r.playbooks_created == 1 { "" } else { "s" },
        ));
    }

    // Regressions / threat rehearsal.
    if !r.regressions_detected.is_empty() {
        out.push_str(&format!(
            "\n**Regressions detected:** {}\n",
            r.regressions_detected.len()
        ));
        for entry in r.regressions_detected.iter().take(3) {
            out.push_str(&format!("  ⚠ {}\n", knowledge_entry_label(entry)));
        }
    }

    // Strategy hypotheses (cross-domain learning).
    if !r.strategy_hypotheses.is_empty() {
        out.push_str(&format!(
            "\n**Hypotheses:** {} strategies under evaluation\n",
            r.strategy_hypotheses.len()
        ));
    }

    // Routing recommendations.
    if r.routing_recommendations > 0 {
        out.push_str(&format!(
            "\n💡 **Routing updated:** {} recommendation{}\n",
            r.routing_recommendations,
            if r.routing_recommendations == 1 { "" } else { "s" },
        ));
    }

    // Performance notes (truncations, budget exhaustion).
    if !r.performance_notes.is_empty() {
        out.push_str("\n_Notes:_\n");
        for note in r.performance_notes.iter().take(3) {
            out.push_str(&format!("  - {note}\n"));
        }
    }

    out
}

fn relative_when(when: DateTime<Utc>) -> String {
    let now = Utc::now();
    let delta = now.signed_duration_since(when);
    let hours = delta.num_hours();
    match hours {
        h if h < 1 => "just now".to_owned(),
        h if h < 24 => format!("{h} hours ago"),
        h if h < 48 => "yesterday".to_owned(),
        h => format!("{} days ago", h / 24),
    }
}

fn cluster_label(_c: &roko_dreams::cycle::DreamClusterReport) -> String {
    // Adapt to actual fields — `name` / `summary` / `episode_count`.
    // If the struct exposes a Display impl, prefer that.
    String::from("(cluster)")
}

fn knowledge_entry_label(_e: &roko_neuro::KnowledgeEntry) -> String {
    String::from("(regression)")
}
```

"session/new" => {
    let params: SessionNewParams = match parse_params(params, &method) {
        Ok(params) => params,
        Err(error) => return send_error_response(transport, id, error).await,
    };
    let pending_report = sessions.pending_dream_report.clone();
    let result = sessions.create_session(params);
    let session_id = result.session_id.clone();
    send_success(transport, id, result).await?;
    send_slash_commands_notification(transport, &session_id).await?;
    if let Some(report) = pending_report {
        emit_dream_card(transport, &session_id, &report).await?;
    }
    Ok(())
}

### Step 4: `emit_dream_card` helper

In `handler.rs`:

```rust
async fn emit_dream_card<R, W>(
    transport: &mut StdioTransport<R, W>,
    session_id: &str,
    report: &roko_dreams::DreamReport,
) -> Result<()>
where
    R: AsyncRead + Unpin,
    W: AsyncWrite + Unpin,
{
    let update = crate::dream_card::render_dream_card(report);
    let notification = JsonRpcNotification {
        jsonrpc: JSON_RPC_VERSION.to_owned(),
        method: "session/update".to_owned(),
        params: Some(serde_json::json!({
            "sessionId": session_id,
            "update": update,
        })),
    };
    transport.send_notification(notification).await
        .context("send dream report card")?;
    tracing::info!(
        session_id,
        completed_at = %report.completed_at,
        "emitted dream report card"
    );
    Ok(())
}
```

(The exact `JsonRpcNotification` shape and the existing `send_notification` helper name should be verified in `crates/roko-acp/src/handler.rs` and `crates/roko-acp/src/transport.rs`.)
