/// Extract `affect.enabled` from the loaded RokoConfig.
///
/// Returns false when the `[affect]` table is absent or malformed so that
/// affect-unaware repos see no behavior change.
fn affect_enabled_from_config(cfg: &roko_core::config::schema::RokoConfig) -> bool {
    // Adapt the field path to whatever the schema uses; verify with:
    //   rg "struct AffectConfig" crates/roko-core/src/config/
    cfg.affect.as_ref().map(|a| a.enabled).unwrap_or(false)
}
