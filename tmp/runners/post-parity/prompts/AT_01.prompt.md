### Step 3: Add tournament configuration

```rust
// crates/roko-acp/src/config.rs

/// Tournament configuration. Only used when WorkflowTemplate::Tournament
/// is selected. All fields have sensible defaults for first-time users.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
pub struct TournamentConfig {
    /// Number of competing arms. Must be 2..=4.
    /// Higher counts cost linearly more in tokens and worktree disk space.
    pub arms: u32,

    /// Strategy hint per arm. Each becomes the strategist's brief.
    /// MUST have len() == arms when explicit, else defaults are used.
    /// Defaults (3 arms): ["conservative", "incremental", "aggressive"].
    pub arm_strategies: Vec<String>,

    /// Maximum total spend across all arms. Enforces $-budget; arms that
    /// exceed are halted before completion.
    pub max_cost_usd: f64,

    /// Minimum gates required to be eligible for selection.
    /// Default: ["compile", "test"].
    pub required_gates: Vec<String>,

    /// Auto-apply the winning arm without asking the user.
    /// Default: false (always prompt).
    pub auto_apply_winner: bool,
}

impl Default for TournamentConfig {
    fn default() -> Self {
        Self {
            arms: 3,
            arm_strategies: vec![
                "conservative — minimal change, preserve existing structure".into(),
                "incremental — small directional refactors".into(),
                "aggressive — full rewrite for the desired property".into(),
            ],
            max_cost_usd: 1.50,
            required_gates: vec!["compile".into(), "test".into()],
            auto_apply_winner: false,
        }
    }
}

impl TournamentConfig {
    /// Validate the config. Returns Err with a human-readable message.
    pub fn validate(&self) -> Result<(), String> {
        if !(2..=4).contains(&self.arms) {
            return Err(format!("tournament.arms must be 2..=4, got {}", self.arms));
        }
        if !self.arm_strategies.is_empty() && self.arm_strategies.len() != self.arms as usize {
            return Err(format!(
                "tournament.arm_strategies has {} entries but arms={}",
                self.arm_strategies.len(), self.arms
            ));
        }
        if !(0.10..=10.00).contains(&self.max_cost_usd) {
            return Err(format!(
                "tournament.max_cost_usd must be 0.10..=10.00, got {}",
                self.max_cost_usd
            ));
        }
        Ok(())
    }
}
```

fn tournament_validate_rejects_bad_arm_count() {
    let mut c = TournamentConfig::default();
