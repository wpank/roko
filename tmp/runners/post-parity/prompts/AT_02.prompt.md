//! Tournament data model (AT_02).
//!
//! `TournamentArm` represents a single competing agent: its strategy, its
//! isolated worktree, its WorkflowRun, gate verdicts, and final patch.
//!
//! `TournamentRun` is the aggregate: N arms + bookkeeping + selection
//! state.

use std::path::PathBuf;

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};

use crate::config::TournamentConfig;
use crate::workflow::WorkflowRun;

/// Verdict of a single arm after all gates have run.
#[derive(Debug, Clone, Serialize, Deserialize, PartialEq)]
#[serde(rename_all = "snake_case")]
pub enum ArmStatus {
    /// Not started yet.
    Pending,
    /// Currently running (strategist / implementer / gating / etc).
    Running,
    /// All required gates passed; eligible for selection.
    Passed,
    /// One or more required gates failed; not eligible.
    Failed,
    /// Halted because the cost ceiling was hit before completion.
    BudgetExhausted,
    /// Crashed mid-run (panic, IO, killed).
    Errored,
    /// Cancelled by the user.
    Cancelled,
}

/// A single competing agent in a tournament.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TournamentArm {
    /// Stable arm id, e.g. "arm-1" / "arm-2" / "arm-3".
    pub arm_id: String,

    /// Index in the tournament (0-based).
    pub index: u32,

    /// Strategy hint passed to the strategist (e.g. "conservative — minimal change").
    pub strategy: String,

    /// Branch name in the host repo. Format: `roko/tournament/{run_id}-{arm_id}`.
    pub branch: String,

    /// Path to the dedicated git worktree. Lives under `<workdir>/.roko/worktrees/`.
    pub worktree_path: PathBuf,

    /// Current status.
    pub status: ArmStatus,

    /// Embedded WorkflowRun for this arm. None until execution begins.
    pub workflow: Option<WorkflowRun>,

    /// Per-gate pass/fail outcomes keyed by gate name.
    #[serde(default)]
    pub gate_results: std::collections::HashMap<String, bool>,

    /// Cumulative cost across all dispatches in this arm.
    #[serde(default)]
    pub cost_usd: f64,

    /// Cumulative tokens used by this arm (input + output).
    #[serde(default)]
    pub tokens_used: u64,

    /// Files modified by this arm. Populated after commit.
    #[serde(default)]
    pub files_changed: Vec<String>,

    /// SHA of the commit produced by this arm. None until committed.
    #[serde(default)]
    pub commit_sha: Option<String>,
}

/// Aggregate state of an N-arm tournament.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TournamentRun {
    pub run_id: String,
    pub session_id: String,
    pub user_prompt: String,
    pub config: TournamentConfig,
    pub arms: Vec<TournamentArm>,
    pub started_at: DateTime<Utc>,
    pub completed_at: Option<DateTime<Utc>>,

    /// Arm chosen by the user. None until selection completes.
    pub selected_arm: Option<String>,
}

impl TournamentRun {
    /// Build a fresh tournament from config. Validates the config first.
    pub fn build(
        run_id: impl Into<String>,
        session_id: impl Into<String>,
        user_prompt: impl Into<String>,
        worktrees_root: &std::path::Path,
        config: TournamentConfig,
    ) -> Result<Self, String> {
        config.validate()?;
        let run_id = run_id.into();
        let session_id = session_id.into();
        let user_prompt = user_prompt.into();
        let strategies = if config.arm_strategies.is_empty() {
            TournamentConfig::default().arm_strategies
                .into_iter()
                .take(config.arms as usize)
                .collect::<Vec<_>>()
        } else {
            config.arm_strategies.clone()
        };

        let arms: Vec<TournamentArm> = (0..config.arms).map(|i| {
            let arm_id = format!("arm-{}", i + 1);
            let branch = format!("roko/tournament/{}-{}", run_id, arm_id);
            let worktree_id = format!("tournament-{}-{}", run_id, arm_id);
            TournamentArm {
                arm_id: arm_id.clone(),
                index: i,
                strategy: strategies.get(i as usize).cloned()
                    .unwrap_or_else(|| format!("strategy-{}", i + 1)),
                branch,
                worktree_path: worktrees_root.join(&worktree_id),
                status: ArmStatus::Pending,
                workflow: None,
                gate_results: Default::default(),
                cost_usd: 0.0,
                tokens_used: 0,
                files_changed: Vec::new(),
                commit_sha: None,
            }
        }).collect();

        Ok(Self {
            run_id,
            session_id,
            user_prompt,
            config,
            arms,
            started_at: Utc::now(),
            completed_at: None,
            selected_arm: None,
        })
    }

    /// Eligible arms = those that Passed AND meet required_gates.
    pub fn eligible_arms(&self) -> Vec<&TournamentArm> {
        self.arms.iter().filter(|a| {
            if a.status != ArmStatus::Passed { return false }
            self.config.required_gates.iter().all(|gate| {
                a.gate_results.get(gate).copied().unwrap_or(false)
            })
        }).collect()
    }

    /// Sum of arm costs.
    pub fn total_cost_usd(&self) -> f64 {
        self.arms.iter().map(|a| a.cost_usd).sum()
    }

    /// Whether the cumulative cost exceeds the configured cap.
    pub fn budget_exhausted(&self) -> bool {
        self.total_cost_usd() > self.config.max_cost_usd
    }

    /// A simple recommendation: lowest cost among eligible arms; ties broken by
    /// fewest files_changed (= least invasive).
    pub fn recommended_arm(&self) -> Option<&TournamentArm> {
        let mut eligible = self.eligible_arms();
        eligible.sort_by(|a, b| {
            a.cost_usd.partial_cmp(&b.cost_usd).unwrap_or(std::cmp::Ordering::Equal)
                .then_with(|| a.files_changed.len().cmp(&b.files_changed.len()))
        });
        eligible.into_iter().next()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    fn cfg() -> TournamentConfig { TournamentConfig::default() }

    #[test]
    fn build_creates_n_arms_with_unique_ids() {
        let tmp = tempfile::tempdir().unwrap();
        let run = TournamentRun::build("run-1", "sess-1", "test", tmp.path(), cfg()).unwrap();
        assert_eq!(run.arms.len(), 3);
        assert_eq!(run.arms[0].arm_id, "arm-1");
        assert_eq!(run.arms[2].arm_id, "arm-3");
        // Branch names are unique
        let branches: std::collections::HashSet<_> = run.arms.iter().map(|a| &a.branch).collect();
        assert_eq!(branches.len(), 3);
    }

    #[test]
    fn build_rejects_invalid_config() {
        let tmp = tempfile::tempdir().unwrap();
        let mut bad = cfg();
        bad.arms = 1;
        assert!(TournamentRun::build("r", "s", "p", tmp.path(), bad).is_err());
    }

    #[test]
    fn eligible_filters_by_status_and_gates() {
        let tmp = tempfile::tempdir().unwrap();
        let mut run = TournamentRun::build("r", "s", "p", tmp.path(), cfg()).unwrap();
        run.arms[0].status = ArmStatus::Passed;
        run.arms[0].gate_results.insert("compile".into(), true);
        run.arms[0].gate_results.insert("test".into(), true);

        run.arms[1].status = ArmStatus::Passed;
        run.arms[1].gate_results.insert("compile".into(), true);
        run.arms[1].gate_results.insert("test".into(), false);  // failed

        run.arms[2].status = ArmStatus::Failed;

        let eligible = run.eligible_arms();
        assert_eq!(eligible.len(), 1);
        assert_eq!(eligible[0].arm_id, "arm-1");
    }

    #[test]
    fn recommended_picks_cheapest_eligible() {
        let tmp = tempfile::tempdir().unwrap();
        let mut run = TournamentRun::build("r", "s", "p", tmp.path(), cfg()).unwrap();
        for arm in &mut run.arms {
            arm.status = ArmStatus::Passed;
            arm.gate_results.insert("compile".into(), true);
            arm.gate_results.insert("test".into(), true);
        }
        run.arms[0].cost_usd = 0.50;
        run.arms[1].cost_usd = 0.20;
        run.arms[2].cost_usd = 0.80;
        assert_eq!(run.recommended_arm().unwrap().arm_id, "arm-2");
    }
}
