//! Evidence collection traits and implementations.
//!
//! Collectors gather raw evidence (process output, diffs, metrics) that
//! criteria use for evaluation.

use crate::types::EvidenceKind;
use async_trait::async_trait;
use std::collections::HashMap;
use std::path::Path;

/// Collected evidence from a single source.
#[derive(Clone, Debug)]
pub struct Evidence {
    /// What kind of evidence this is.
    pub kind: EvidenceKind,
    /// Raw text content.
    pub content: String,
    /// Exit code if from a process.
    pub exit_code: Option<i32>,
    /// Arbitrary metadata.
    pub metadata: HashMap<String, String>,
}

impl Evidence {
    /// Create evidence from process output.
    pub fn from_process(kind: EvidenceKind, stdout: String, stderr: String, exit_code: i32) -> Self {
        let content = if stderr.is_empty() {
            stdout
        } else {
            format!("{stdout}\n--- stderr ---\n{stderr}")
        };
        Self {
            kind,
            content,
            exit_code: Some(exit_code),
            metadata: HashMap::new(),
        }
    }

    /// Create evidence from text content.
    pub fn from_text(kind: EvidenceKind, content: String) -> Self {
        Self {
            kind,
            content,
            exit_code: None,
            metadata: HashMap::new(),
        }
    }

    /// Whether the evidence indicates success (exit code 0 or no exit code).
    pub fn is_success(&self) -> bool {
        self.exit_code.map_or(true, |c| c == 0)
    }
}

/// Trait for collecting evidence from external sources.
#[async_trait]
pub trait EvidenceCollector: Send + Sync {
    /// Collect evidence. Returns an error only for infrastructure failures,
    /// not for "the process exited non-zero" (which is captured in Evidence).
    async fn collect(&self, workdir: &Path) -> Result<Evidence, CollectorError>;

    /// Human-readable name for this collector.
    fn name(&self) -> &str;
}

/// Error from evidence collection (infrastructure, not evaluation).
#[derive(Debug, thiserror::Error)]
pub enum CollectorError {
    #[error("process failed to start: {0}")]
    SpawnFailed(String),
    #[error("IO error: {0}")]
    Io(#[from] std::io::Error),
    #[error("timeout after {0}ms")]
    Timeout(u64),
}

/// Collects evidence by running a shell process.
pub struct ProcessCollector {
    /// Command to run.
    pub command: String,
    /// Arguments.
    pub args: Vec<String>,
    /// Evidence kind to tag the output with.
    pub evidence_kind: EvidenceKind,
    /// Timeout in milliseconds.
    pub timeout_ms: u64,
}

impl ProcessCollector {
    /// Create a collector for `cargo build`.
    pub fn cargo_build() -> Self {
        Self {
            command: "cargo".into(),
            args: vec!["build".into(), "--workspace".into()],
            evidence_kind: EvidenceKind::CompilerOutput,
            timeout_ms: 120_000,
        }
    }

    /// Create a collector for `cargo clippy`.
    pub fn cargo_clippy() -> Self {
        Self {
            command: "cargo".into(),
            args: vec!["clippy".into(), "--workspace".into(), "--no-deps".into(), "--".into(), "-D".into(), "warnings".into()],
            evidence_kind: EvidenceKind::LintOutput,
            timeout_ms: 120_000,
        }
    }

    /// Create a collector for `cargo test`.
    pub fn cargo_test() -> Self {
        Self {
            command: "cargo".into(),
            args: vec!["test".into(), "--workspace".into()],
            evidence_kind: EvidenceKind::TestOutput,
            timeout_ms: 300_000,
        }
    }
}

#[async_trait]
impl EvidenceCollector for ProcessCollector {
    async fn collect(&self, workdir: &Path) -> Result<Evidence, CollectorError> {
        let output = tokio::process::Command::new(&self.command)
            .args(&self.args)
            .current_dir(workdir)
            .output()
            .await
            .map_err(|e| CollectorError::SpawnFailed(e.to_string()))?;

        let stdout = String::from_utf8_lossy(&output.stdout).to_string();
        let stderr = String::from_utf8_lossy(&output.stderr).to_string();
        let exit_code = output.status.code().unwrap_or(-1);

        Ok(Evidence::from_process(self.evidence_kind.clone(), stdout, stderr, exit_code))
    }

    fn name(&self) -> &str {
        &self.command
    }
}

/// Collects evidence from git diff.
pub struct DiffCollector {
    /// Base ref to diff against.
    pub base_ref: String,
}

impl DiffCollector {
    /// Diff against HEAD~1.
    pub fn last_commit() -> Self {
        Self { base_ref: "HEAD~1".into() }
    }

    /// Diff against a named branch.
    pub fn against(base: impl Into<String>) -> Self {
        Self { base_ref: base.into() }
    }
}

#[async_trait]
impl EvidenceCollector for DiffCollector {
    async fn collect(&self, workdir: &Path) -> Result<Evidence, CollectorError> {
        let output = tokio::process::Command::new("git")
            .args(["diff", &self.base_ref])
            .current_dir(workdir)
            .output()
            .await
            .map_err(|e| CollectorError::SpawnFailed(e.to_string()))?;

        let stdout = String::from_utf8_lossy(&output.stdout).to_string();
        let stderr = String::from_utf8_lossy(&output.stderr).to_string();
        let exit_code = output.status.code().unwrap_or(-1);

        Ok(Evidence::from_process(EvidenceKind::Diff, stdout, stderr, exit_code))
    }

    fn name(&self) -> &str {
        "git-diff"
    }
}
