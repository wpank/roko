//! Criterion trait — the evaluation interface.
//!
//! A criterion takes collected evidence, inspects it, and produces
//! a `CriterionResult` with findings and a pass/fail verdict.

use crate::collector::Evidence;
use crate::types::{CriterionResult, EvidenceKind, Finding, Severity};
use async_trait::async_trait;
use std::time::Instant;

/// A single evaluation criterion.
///
/// Implementations inspect evidence and produce findings. The framework
/// calls `evaluate()` with evidence that matches the criterion's
/// `required_evidence()` kinds.
#[async_trait]
pub trait Criterion: Send + Sync {
    /// Human-readable name (e.g., "compile", "lint", "test").
    fn name(&self) -> &str;

    /// What evidence kinds this criterion needs.
    fn required_evidence(&self) -> Vec<EvidenceKind>;

    /// Evaluate the evidence and produce a result.
    ///
    /// The `evidence` slice contains all evidence collected for this
    /// evaluation run. The criterion should filter for its relevant
    /// evidence kinds.
    async fn evaluate(&self, evidence: &[Evidence]) -> CriterionResult;

    /// Optional weight for scoring (default 1.0).
    fn weight(&self) -> f64 {
        1.0
    }

    /// Whether this criterion can be skipped by adaptive thresholds.
    fn skippable(&self) -> bool {
        true
    }
}

/// Helper for criteria that just check process exit codes.
pub fn exit_code_criterion(
    name: &str,
    evidence: &[Evidence],
    target_kind: &EvidenceKind,
) -> CriterionResult {
    let start = Instant::now();

    let relevant: Vec<&Evidence> = evidence
        .iter()
        .filter(|e| &e.kind == target_kind)
        .collect();

    if relevant.is_empty() {
        return CriterionResult {
            criterion_name: name.to_string(),
            passed: false,
            duration_ms: start.elapsed().as_millis() as u64,
            findings: vec![Finding {
                id: format!("{name}-no-evidence"),
                message: format!("No {target_kind:?} evidence collected"),
                severity: Severity::Error,
                evidence_kind: target_kind.clone(),
                file: None,
                line: None,
                evidence_text: None,
            }],
            score: Some(0.0),
            metadata: Default::default(),
        };
    }

    let all_passed = relevant.iter().all(|e| e.is_success());
    let mut findings = Vec::new();

    for ev in &relevant {
        if !ev.is_success() {
            // Extract first N lines of output as a finding
            let excerpt: String = ev.content.lines().take(50).collect::<Vec<_>>().join("\n");
            findings.push(Finding {
                id: format!("{name}-exit-{}", ev.exit_code.unwrap_or(-1)),
                message: format!("{name} exited with code {}", ev.exit_code.unwrap_or(-1)),
                severity: Severity::Error,
                evidence_kind: target_kind.clone(),
                file: None,
                line: None,
                evidence_text: Some(excerpt),
            });
        }
    }

    CriterionResult {
        criterion_name: name.to_string(),
        passed: all_passed,
        duration_ms: start.elapsed().as_millis() as u64,
        findings,
        score: Some(if all_passed { 1.0 } else { 0.0 }),
        metadata: Default::default(),
    }
}

/// Parse compiler error lines into individual findings.
pub fn parse_compiler_findings(output: &str) -> Vec<Finding> {
    let mut findings = Vec::new();
    let mut current_id = 0u32;

    for line in output.lines() {
        // Match lines like "error[E0308]: mismatched types"
        if line.starts_with("error") {
            current_id += 1;
            let (file, line_num) = extract_file_line(line);
            findings.push(Finding {
                id: format!("compile-{current_id}"),
                message: line.to_string(),
                severity: Severity::Error,
                evidence_kind: EvidenceKind::CompilerOutput,
                file,
                line: line_num,
                evidence_text: None,
            });
        } else if line.starts_with("warning") {
            current_id += 1;
            let (file, line_num) = extract_file_line(line);
            findings.push(Finding {
                id: format!("compile-warn-{current_id}"),
                message: line.to_string(),
                severity: Severity::Warning,
                evidence_kind: EvidenceKind::CompilerOutput,
                file,
                line: line_num,
                evidence_text: None,
            });
        }
    }

    findings
}

/// Extract file path and line number from a compiler error line.
fn extract_file_line(line: &str) -> (Option<String>, Option<u32>) {
    // Match patterns like " --> src/foo.rs:42:5"
    if let Some(arrow_pos) = line.find("-->") {
        let after = &line[arrow_pos + 4..];
        let parts: Vec<&str> = after.trim().splitn(3, ':').collect();
        if parts.len() >= 2 {
            let file = parts[0].to_string();
            let line_num = parts[1].parse().ok();
            return (Some(file), line_num);
        }
    }
    (None, None)
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn parse_compiler_errors() {
        let output = "error[E0308]: mismatched types\n  --> src/main.rs:10:5\n";
        let findings = parse_compiler_findings(output);
        assert_eq!(findings.len(), 1);
        assert_eq!(findings[0].severity, Severity::Error);
    }

    #[test]
    fn exit_code_missing_evidence() {
        let result = exit_code_criterion("test", &[], &EvidenceKind::TestOutput);
        assert!(!result.passed);
        assert_eq!(result.findings.len(), 1);
    }
}
