# GE_04: Define EvalService orchestrator

## Task
Define the `EvalService` that orchestrates evaluation: loads a profile, collects evidence, runs criteria, and aggregates into an `EvalReport`.

## Runner Context
Runner GE, batch 4 of 8. Depends on GE_02, GE_03.

## Problem
Individual collectors and criteria exist but there is no orchestrator that ties them together. `EvalService` is the entry point that takes an `EvalProfile`, runs all collectors, feeds evidence to criteria, handles fail-fast logic, and produces an `EvalReport`.

## Current Code

**EvalProfile** (GE_01): Profile defining which criteria to run.
**EvidenceCollector** (GE_02): Trait + implementations.
**Criterion** (GE_03): Evaluation trait.
**EvalReport** (GE_01): Aggregated results.

//! EvalService — orchestrates evidence collection and criterion evaluation.

use crate::collector::{Evidence, EvidenceCollector};
use crate::criterion::Criterion;
use crate::types::{EvalProfile, EvalReport};
use std::collections::HashMap;
use std::path::{Path, PathBuf};
use std::sync::Arc;
use std::time::Instant;
use tracing::{debug, info, warn};

/// The main evaluation orchestrator.
///
/// Flow: profile -> collect evidence -> evaluate criteria -> aggregate report.
pub struct EvalService {
    /// Registered collectors, keyed by name.
    collectors: HashMap<String, Arc<dyn EvidenceCollector>>,
    /// Registered criteria, keyed by name.
    criteria: HashMap<String, Arc<dyn Criterion>>,
    /// Working directory for evaluations.
    workdir: PathBuf,
}

impl EvalService {
    /// Create a new service with the given working directory.
    pub fn new(workdir: PathBuf) -> Self {
        Self {
            collectors: HashMap::new(),
            criteria: HashMap::new(),
            workdir,
        }
    }

    /// Register an evidence collector.
    pub fn register_collector(
        &mut self,
        name: impl Into<String>,
        collector: Arc<dyn EvidenceCollector>,
    ) {
        self.collectors.insert(name.into(), collector);
    }

    /// Register a criterion.
    pub fn register_criterion(
        &mut self,
        name: impl Into<String>,
        criterion: Arc<dyn Criterion>,
    ) {
        self.criteria.insert(name.into(), criterion);
    }

    /// Builder: add a collector.
    pub fn with_collector(
        mut self,
        name: impl Into<String>,
        collector: Arc<dyn EvidenceCollector>,
    ) -> Self {
        self.register_collector(name, collector);
        self
    }

    /// Builder: add a criterion.
    pub fn with_criterion(
        mut self,
        name: impl Into<String>,
        criterion: Arc<dyn Criterion>,
    ) -> Self {
        self.register_criterion(name, criterion);
        self
    }

    /// Run a full evaluation against the given profile.
    pub async fn evaluate(&self, profile: &EvalProfile) -> EvalReport {
        let start = Instant::now();
        info!(profile = %profile.name, criteria = profile.criteria.len(), "starting evaluation");

        // Step 1: Collect all evidence
        let evidence = self.collect_all_evidence().await;
        debug!(evidence_count = evidence.len(), "evidence collected");

        // Step 2: Run each criterion in profile order
        let mut results = Vec::new();
        for criterion_name in &profile.criteria {
            let criterion = match self.criteria.get(criterion_name) {
                Some(c) => c,
                None => {
                    warn!(criterion = %criterion_name, "criterion not registered, skipping");
                    continue;
                }
            };

            let result = criterion.evaluate(&evidence).await;
            let passed = result.passed;
            debug!(
                criterion = %criterion_name,
                passed,
                findings = result.findings.len(),
                "criterion evaluated"
            );

            results.push(result);

            // Fail-fast: stop at first failure if configured
            if profile.fail_fast && !passed {
                info!(criterion = %criterion_name, "fail-fast: stopping evaluation");
                break;
            }
        }

        // Step 3: Aggregate
        let all_passed = results.iter().all(|r| r.passed);
        let total_findings: usize = results.iter().map(|r| r.findings.len()).sum();

        let report = EvalReport {
            profile: profile.name.clone(),
            results,
            passed: all_passed,
            total_duration_ms: start.elapsed().as_millis() as u64,
            total_findings,
        };

        info!(
            passed = report.passed,
            duration_ms = report.total_duration_ms,
            findings = report.total_findings,
            "evaluation complete"
        );

        report
    }

    /// Collect evidence from all registered collectors.
    async fn collect_all_evidence(&self) -> Vec<Evidence> {
        let mut all_evidence = Vec::new();

        for (name, collector) in &self.collectors {
            match collector.collect(&self.workdir).await {
                Ok(evidence) => {
                    debug!(collector = %name, kind = ?evidence.kind, "evidence collected");
                    all_evidence.push(evidence);
                }
                Err(e) => {
                    warn!(collector = %name, error = %e, "evidence collection failed");
                }
            }
        }

        all_evidence
    }

    /// Create a standard Rust evaluation service with compile, lint, and test.
    pub fn rust_standard(workdir: PathBuf) -> Self {
        use crate::collector::ProcessCollector;
        Self::new(workdir)
            .with_collector("cargo-build", Arc::new(ProcessCollector::cargo_build()))
            .with_collector("cargo-clippy", Arc::new(ProcessCollector::cargo_clippy()))
            .with_collector("cargo-test", Arc::new(ProcessCollector::cargo_test()))
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use crate::types::EvalProfile;

    #[tokio::test]
    async fn empty_profile_passes() {
        let svc = EvalService::new(PathBuf::from("."));
        let profile = EvalProfile {
            name: "empty".into(),
            criteria: vec![],
            fail_fast: false,
            timeout_ms: None,
            config: HashMap::new(),
        };
        let report = svc.evaluate(&profile).await;
        assert!(report.passed);
        assert_eq!(report.total_findings, 0);
    }
}
