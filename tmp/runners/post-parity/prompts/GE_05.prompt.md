# GE_05: Create BridgeGateService wrapping legacy GateService

## Task
Create a bridge that wraps the legacy `GateService` as an `EvalService` criterion, allowing gradual migration from the old 7-rung pipeline to the new evaluation framework.

## Runner Context
Runner GE, batch 5 of 8. Depends on GE_04.

## Problem
The legacy `GateService` in `roko-gate` works and is wired. We need a bridge so the new `EvalService` can delegate to existing gates during the transition period rather than reimplementing everything at once.

//! Bridge from legacy GateService to the new EvalService criterion interface.

use crate::collector::Evidence;
use crate::criterion::Criterion;
use crate::types::{CriterionResult, EvidenceKind, Finding, Severity};
use async_trait::async_trait;
use std::path::PathBuf;
use std::time::Instant;

/// A criterion that delegates to the legacy GateService for a single gate.
///
/// This allows gradual migration: register a `BridgeCriterion` for each
/// legacy gate, then replace them one by one with native criteria.
pub struct BridgeCriterion {
    /// Name of the legacy gate (e.g., "compile", "clippy", "test").
    gate_name: String,
    /// Working directory for the gate.
    workdir: PathBuf,
    /// Shell command the legacy gate runs.
    command: String,
    /// Arguments for the command.
    args: Vec<String>,
}

impl BridgeCriterion {
    /// Create a bridge for a named legacy gate.
    pub fn new(gate_name: impl Into<String>, workdir: PathBuf) -> Self {
        let gate_name = gate_name.into();
        let (command, args) = Self::resolve_gate_command(&gate_name);
        Self {
            gate_name,
            workdir,
            command,
            args,
        }
    }

    /// Map gate name to the command the legacy gate would run.
    fn resolve_gate_command(name: &str) -> (String, Vec<String>) {
        match name {
            "compile" | "compile:cargo" => (
                "cargo".into(),
                vec!["build".into(), "--workspace".into()],
            ),
            "clippy" | "clippy:cargo" => (
                "cargo".into(),
                vec!["clippy".into(), "--workspace".into(), "--no-deps".into(), "--".into(), "-D".into(), "warnings".into()],
            ),
            "test" | "test:cargo" => (
                "cargo".into(),
                vec!["test".into(), "--workspace".into()],
            ),
            "fmt" | "fmt:cargo" => (
                "cargo".into(),
                vec!["fmt".into(), "--all".into(), "--check".into()],
            ),
            _ => (
                "echo".into(),
                vec![format!("unknown gate: {name}")],
            ),
        }
    }
}

#[async_trait]
impl Criterion for BridgeCriterion {
    fn name(&self) -> &str {
        &self.gate_name
    }

    fn required_evidence(&self) -> Vec<EvidenceKind> {
        // Bridge collects its own evidence by running the command
        vec![]
    }

    async fn evaluate(&self, _evidence: &[Evidence]) -> CriterionResult {
        let start = Instant::now();

        // Run the legacy gate command directly
        let output = match tokio::process::Command::new(&self.command)
            .args(&self.args)
            .current_dir(&self.workdir)
            .output()
            .await
        {
            Ok(out) => out,
            Err(e) => {
                return CriterionResult::fail(
                    &self.gate_name,
                    start.elapsed(),
                    vec![Finding {
                        id: format!("{}-spawn-error", self.gate_name),
                        message: format!("Failed to run gate: {e}"),
                        severity: Severity::Critical,
                        evidence_kind: EvidenceKind::ShellOutput,
                        file: None,
                        line: None,
                        evidence_text: None,
                    }],
                );
            }
        };

        let exit_code = output.status.code().unwrap_or(-1);
        let passed = exit_code == 0;
        let duration = start.elapsed();

        if passed {
            CriterionResult::pass(&self.gate_name, duration)
        } else {
            let stderr = String::from_utf8_lossy(&output.stderr);
            let excerpt: String = stderr.lines().take(30).collect::<Vec<_>>().join("\n");

            CriterionResult::fail(
                &self.gate_name,
                duration,
                vec![Finding {
                    id: format!("{}-exit-{exit_code}", self.gate_name),
                    message: format!("{} exited with code {exit_code}", self.gate_name),
                    severity: Severity::Error,
                    evidence_kind: EvidenceKind::ShellOutput,
                    file: None,
                    line: None,
                    evidence_text: Some(excerpt),
                }],
            )
        }
    }

    fn skippable(&self) -> bool {
        // compile gate is never skippable, matching legacy behavior
        self.gate_name != "compile"
    }
}

/// Register bridge criteria for all standard gates.
pub fn register_standard_bridges(
    service: &mut crate::service::EvalService,
    workdir: &std::path::Path,
) {
    use std::sync::Arc;
    for gate in &["compile", "clippy", "test", "fmt"] {
        let bridge = BridgeCriterion::new(*gate, workdir.to_path_buf());
        service.register_criterion(*gate, Arc::new(bridge));
    }
}
