# GE_06: Implement CompileCriterion

## Task
Create a native `CompileCriterion` that evaluates `CompilerOutput` evidence and parses error/warning findings.

## Runner Context
Runner GE, batch 6 of 8. Depends on GE_03.

## Problem
The bridge criterion (GE_05) runs commands directly. A native `CompileCriterion` should work with pre-collected evidence, parsing compiler output into structured findings with file paths and line numbers.

## Current Code

**parse_compiler_findings** (GE_03): Helper for parsing error lines.
**Evidence** (GE_02): Contains `kind`, `content`, `exit_code`.

//! CompileCriterion — evaluates compiler output evidence.

use crate::collector::Evidence;
use crate::criterion::{Criterion, exit_code_criterion, parse_compiler_findings};
use crate::types::{CriterionResult, EvidenceKind, Finding, Severity};
use async_trait::async_trait;
use std::time::Instant;

/// Criterion that evaluates compilation output.
///
/// Looks for `CompilerOutput` evidence, checks exit code, and parses
/// error/warning messages into structured findings with file/line info.
pub struct CompileCriterion {
    /// Whether warnings should be treated as errors.
    pub warnings_as_errors: bool,
}

impl CompileCriterion {
    /// Standard compile check (warnings are warnings, not errors).
    pub fn standard() -> Self {
        Self { warnings_as_errors: false }
    }

    /// Strict compile check (warnings are errors).
    pub fn strict() -> Self {
        Self { warnings_as_errors: true }
    }
}

impl Default for CompileCriterion {
    fn default() -> Self {
        Self::standard()
    }
}

#[async_trait]
impl Criterion for CompileCriterion {
    fn name(&self) -> &str {
        "compile"
    }

    fn required_evidence(&self) -> Vec<EvidenceKind> {
        vec![EvidenceKind::CompilerOutput]
    }

    async fn evaluate(&self, evidence: &[Evidence]) -> CriterionResult {
        let start = Instant::now();

        let compiler_evidence: Vec<&Evidence> = evidence
            .iter()
            .filter(|e| e.kind == EvidenceKind::CompilerOutput)
            .collect();

        if compiler_evidence.is_empty() {
            return CriterionResult::fail(
                "compile",
                start.elapsed(),
                vec![Finding {
                    id: "compile-no-evidence".into(),
                    message: "No compiler output evidence collected".into(),
                    severity: Severity::Error,
                    evidence_kind: EvidenceKind::CompilerOutput,
                    file: None,
                    line: None,
                    evidence_text: None,
                }],
            );
        }

        let mut all_findings = Vec::new();
        let mut any_failed = false;

        for ev in &compiler_evidence {
            if !ev.is_success() {
                any_failed = true;
            }

            let mut findings = parse_compiler_findings(&ev.content);

            // Optionally promote warnings to errors
            if self.warnings_as_errors {
                for f in &mut findings {
                    if f.severity == Severity::Warning {
                        f.severity = Severity::Error;
                        any_failed = true;
                    }
                }
            }

            all_findings.extend(findings);
        }

        let passed = !any_failed;
        let error_count = all_findings.iter().filter(|f| f.severity >= Severity::Error).count();

        CriterionResult {
            criterion_name: "compile".into(),
            passed,
            duration_ms: start.elapsed().as_millis() as u64,
            findings: all_findings,
            score: Some(if passed { 1.0 } else { 0.0 }),
            metadata: [("error_count".to_string(), error_count.to_string())].into(),
        }
    }

    fn skippable(&self) -> bool {
        false // compile is never skippable
    }

    fn weight(&self) -> f64 {
        2.0 // compile errors are the most critical
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[tokio::test]
    async fn compile_success() {
        let evidence = vec![Evidence::from_process(
            EvidenceKind::CompilerOutput,
            "Compiling roko v0.1.0\n    Finished".into(),
            String::new(),
            0,
        )];
        let result = CompileCriterion::standard().evaluate(&evidence).await;
        assert!(result.passed);
    }

    #[tokio::test]
    async fn compile_failure() {
        let evidence = vec![Evidence::from_process(
            EvidenceKind::CompilerOutput,
            "error[E0308]: mismatched types\n  --> src/main.rs:10:5".into(),
            String::new(),
            1,
        )];
        let result = CompileCriterion::standard().evaluate(&evidence).await;
        assert!(!result.passed);
        assert!(!result.findings.is_empty());
    }
}
