//! LintCriterion — evaluates clippy/lint output evidence.

use crate::collector::Evidence;
use crate::criterion::Criterion;
use crate::types::{CriterionResult, EvidenceKind, Finding, Severity};
use async_trait::async_trait;
use std::time::Instant;

/// Criterion that evaluates linter output (clippy).
pub struct LintCriterion {
    /// Maximum allowed warnings before failing.
    pub max_warnings: usize,
}

impl LintCriterion {
    /// Standard lint check: zero warnings allowed (matches `-D warnings`).
    pub fn strict() -> Self {
        Self { max_warnings: 0 }
    }

    /// Relaxed lint check: allow up to N warnings.
    pub fn relaxed(max_warnings: usize) -> Self {
        Self { max_warnings }
    }
}

impl Default for LintCriterion {
    fn default() -> Self {
        Self::strict()
    }
}

#[async_trait]
impl Criterion for LintCriterion {
    fn name(&self) -> &str {
        "lint"
    }

    fn required_evidence(&self) -> Vec<EvidenceKind> {
        vec![EvidenceKind::LintOutput]
    }

    async fn evaluate(&self, evidence: &[Evidence]) -> CriterionResult {
        let start = Instant::now();

        let lint_evidence: Vec<&Evidence> = evidence
            .iter()
            .filter(|e| e.kind == EvidenceKind::LintOutput)
            .collect();

        if lint_evidence.is_empty() {
            return CriterionResult::fail(
                "lint",
                start.elapsed(),
                vec![Finding {
                    id: "lint-no-evidence".into(),
                    message: "No lint output evidence collected".into(),
                    severity: Severity::Error,
                    evidence_kind: EvidenceKind::LintOutput,
                    file: None,
                    line: None,
                    evidence_text: None,
                }],
            );
        }

        let mut findings = Vec::new();
        let mut any_errors = false;
        let mut warning_count = 0u32;

        for ev in &lint_evidence {
            if !ev.is_success() {
                any_errors = true;
            }

            let mut id_counter = 0u32;
            for line in ev.content.lines() {
                let trimmed = line.trim();

                if trimmed.starts_with("error") {
                    id_counter += 1;
                    let (file, line_num) = extract_clippy_location(trimmed);
                    findings.push(Finding {
                        id: format!("lint-error-{id_counter}"),
                        message: trimmed.to_string(),
                        severity: Severity::Error,
                        evidence_kind: EvidenceKind::LintOutput,
                        file,
                        line: line_num,
                        evidence_text: None,
                    });
                    any_errors = true;
                } else if trimmed.starts_with("warning") && !trimmed.starts_with("warning: ") || trimmed.starts_with("warning[") || trimmed.starts_with("warning: ") {
                    // Skip the "N warnings generated" summary line
                    if trimmed.contains("generated") || trimmed.contains("emitted") {
                        continue;
                    }
                    id_counter += 1;
                    warning_count += 1;
                    let (file, line_num) = extract_clippy_location(trimmed);
                    findings.push(Finding {
                        id: format!("lint-warn-{id_counter}"),
                        message: trimmed.to_string(),
                        severity: Severity::Warning,
                        evidence_kind: EvidenceKind::LintOutput,
                        file,
                        line: line_num,
                        evidence_text: None,
                    });
                }
            }
        }

        let too_many_warnings = warning_count as usize > self.max_warnings;
        let passed = !any_errors && !too_many_warnings;

        CriterionResult {
            criterion_name: "lint".into(),
            passed,
            duration_ms: start.elapsed().as_millis() as u64,
            findings,
            score: Some(if passed { 1.0 } else { 0.0 }),
            metadata: [
                ("warning_count".to_string(), warning_count.to_string()),
                ("max_warnings".to_string(), self.max_warnings.to_string()),
            ].into(),
        }
    }

    fn weight(&self) -> f64 {
        1.5
    }
}

/// Extract file path and line number from a clippy warning/error line.
fn extract_clippy_location(line: &str) -> (Option<String>, Option<u32>) {
    if let Some(arrow_pos) = line.find("-->") {
        let after = &line[arrow_pos + 4..];
        let parts: Vec<&str> = after.trim().splitn(3, ':').collect();
        if parts.len() >= 2 {
            return (Some(parts[0].to_string()), parts[1].parse().ok());
        }
    }
    (None, None)
}

#[cfg(test)]
mod tests {
    use super::*;

    #[tokio::test]
    async fn lint_clean() {
        let evidence = vec![Evidence::from_process(
            EvidenceKind::LintOutput,
            "Checking roko v0.1.0\n    Finished".into(),
            String::new(),
            0,
        )];
        let result = LintCriterion::strict().evaluate(&evidence).await;
        assert!(result.passed);
    }

    #[tokio::test]
    async fn lint_with_warnings() {
        let evidence = vec![Evidence::from_process(
            EvidenceKind::LintOutput,
            "warning: unused variable `x`\n  --> src/main.rs:5:9".into(),
            String::new(),
            0,
        )];
        let result = LintCriterion::strict().evaluate(&evidence).await;
        assert!(!result.passed); // strict: 0 warnings allowed

        let result = LintCriterion::relaxed(5).evaluate(&evidence).await;
        assert!(result.passed); // relaxed: 5 warnings allowed
    }
}
