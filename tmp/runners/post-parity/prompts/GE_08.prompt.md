//! TestCriterion — evaluates test output evidence.

use crate::collector::Evidence;
use crate::criterion::Criterion;
use crate::types::{CriterionResult, EvidenceKind, Finding, Severity};
use async_trait::async_trait;
use std::time::Instant;

/// Criterion that evaluates test output.
pub struct TestCriterion {
    /// Minimum test count required for a pass.
    pub min_tests: usize,
}

impl TestCriterion {
    /// Standard: at least 1 test must exist and run.
    pub fn standard() -> Self {
        Self { min_tests: 1 }
    }

    /// No minimum test count (pure pass/fail check).
    pub fn any() -> Self {
        Self { min_tests: 0 }
    }
}

impl Default for TestCriterion {
    fn default() -> Self {
        Self::standard()
    }
}

#[async_trait]
impl Criterion for TestCriterion {
    fn name(&self) -> &str {
        "test"
    }

    fn required_evidence(&self) -> Vec<EvidenceKind> {
        vec![EvidenceKind::TestOutput]
    }

    async fn evaluate(&self, evidence: &[Evidence]) -> CriterionResult {
        let start = Instant::now();

        let test_evidence: Vec<&Evidence> = evidence
            .iter()
            .filter(|e| e.kind == EvidenceKind::TestOutput)
            .collect();

        if test_evidence.is_empty() {
            return CriterionResult::fail(
                "test",
                start.elapsed(),
                vec![Finding {
                    id: "test-no-evidence".into(),
                    message: "No test output evidence collected".into(),
                    severity: Severity::Error,
                    evidence_kind: EvidenceKind::TestOutput,
                    file: None,
                    line: None,
                    evidence_text: None,
                }],
            );
        }

        let mut findings = Vec::new();
        let mut total_passed = 0u32;
        let mut total_failed = 0u32;
        let mut total_ignored = 0u32;
        let mut any_failed = false;

        for ev in &test_evidence {
            if !ev.is_success() {
                any_failed = true;
            }

            // Parse "test result: ok/FAILED. N passed; N failed; N ignored"
            for line in ev.content.lines() {
                let trimmed = line.trim();

                if trimmed.starts_with("test result:") {
                    let counts = parse_test_result_line(trimmed);
                    total_passed += counts.0;
                    total_failed += counts.1;
                    total_ignored += counts.2;
                }

                // Capture individual failed test names
                if trimmed.starts_with("---- ") && trimmed.ends_with(" ----") {
                    let test_name = trimmed.trim_start_matches("---- ").trim_end_matches(" ----");
                    findings.push(Finding {
                        id: format!("test-fail-{test_name}"),
                        message: format!("Test failed: {test_name}"),
                        severity: Severity::Error,
                        evidence_kind: EvidenceKind::TestOutput,
                        file: None,
                        line: None,
                        evidence_text: None,
                    });
                }

                // Capture "thread panicked" messages
                if trimmed.contains("thread") && trimmed.contains("panicked") {
                    findings.push(Finding {
                        id: format!("test-panic-{}", findings.len()),
                        message: trimmed.to_string(),
                        severity: Severity::Critical,
                        evidence_kind: EvidenceKind::TestOutput,
                        file: None,
                        line: None,
                        evidence_text: None,
                    });
                }
            }
        }

        let total_tests = total_passed + total_failed;
        let below_minimum = (total_tests as usize) < self.min_tests;

        if below_minimum {
            findings.push(Finding {
                id: "test-below-minimum".into(),
                message: format!(
                    "Only {total_tests} tests ran, minimum is {}",
                    self.min_tests
                ),
                severity: Severity::Warning,
                evidence_kind: EvidenceKind::TestOutput,
                file: None,
                line: None,
                evidence_text: None,
            });
        }

        let passed = !any_failed && !below_minimum && total_failed == 0;
        let score = if total_tests > 0 {
            total_passed as f64 / total_tests as f64
        } else {
            0.0
        };

        CriterionResult {
            criterion_name: "test".into(),
            passed,
            duration_ms: start.elapsed().as_millis() as u64,
            findings,
            score: Some(score),
            metadata: [
                ("passed".to_string(), total_passed.to_string()),
                ("failed".to_string(), total_failed.to_string()),
                ("ignored".to_string(), total_ignored.to_string()),
            ].into(),
        }
    }

    fn weight(&self) -> f64 {
        1.0
    }
}

/// Parse a "test result:" line and extract (passed, failed, ignored) counts.
fn parse_test_result_line(line: &str) -> (u32, u32, u32) {
    let mut passed = 0u32;
    let mut failed = 0u32;
    let mut ignored = 0u32;

    for part in line.split(';') {
        let part = part.trim();
        if let Some(num_str) = part.strip_suffix(" passed") {
            passed += num_str.trim().parse::<u32>().unwrap_or(0);
        } else if let Some(num_str) = part.strip_suffix(" failed") {
            failed += num_str.trim().parse::<u32>().unwrap_or(0);
        } else if let Some(num_str) = part.strip_suffix(" ignored") {
            ignored += num_str.trim().parse::<u32>().unwrap_or(0);
        }
    }

    (passed, failed, ignored)
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn parse_result_line() {
        let (p, f, i) = parse_test_result_line("test result: ok. 42 passed; 0 failed; 3 ignored");
        assert_eq!(p, 42);
        assert_eq!(f, 0);
        assert_eq!(i, 3);
    }

    #[tokio::test]
    async fn test_all_pass() {
        let evidence = vec![Evidence::from_process(
            EvidenceKind::TestOutput,
            "test result: ok. 10 passed; 0 failed; 0 ignored".into(),
            String::new(),
            0,
        )];
        let result = TestCriterion::standard().evaluate(&evidence).await;
        assert!(result.passed);
        assert_eq!(result.score, Some(1.0));
    }

    #[tokio::test]
    async fn test_some_fail() {
        let evidence = vec![Evidence::from_process(
            EvidenceKind::TestOutput,
            "test result: FAILED. 8 passed; 2 failed; 0 ignored".into(),
            String::new(),
            1,
        )];
        let result = TestCriterion::standard().evaluate(&evidence).await;
        assert!(!result.passed);
        assert_eq!(result.score, Some(0.8));
    }
}
