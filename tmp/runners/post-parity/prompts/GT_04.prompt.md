# GT_04: Implement GitHubAdapter

## Task
Implement the `GitHubAdapter` struct that implements `RokoAdapter` and `WorkSourceAdapter` using the `octocrab` crate to interact with GitHub issues and pull requests.

## Runner Context
Runner GT_B (GTM: GitHub), batch 1 of 3. Depends on GT_01.

## Problem
There is no adapter implementation for GitHub. The webhook receiver exists (`webhooks.rs`) but there is no client-side adapter to list issues, create issues, or update issue status from roko.

## Current Code

**webhooks.rs** -- `crates/roko-serve/src/routes/webhooks.rs:29-34`:
Has `github_webhook` handler with HMAC verification.

```rust
//! GitHub adapter -- implements WorkSourceAdapter via octocrab (GT_04).

use std::sync::Arc;

use async_trait::async_trait;
use octocrab::Octocrab;
use roko_core::adapter::{
    AdapterCapability, AdapterError, AdapterInfo, RokoAdapter,
    WorkItem, WorkSourceAdapter,
};

/// GitHub adapter configuration.
#[derive(Debug, Clone)]
pub struct GitHubAdapterConfig {
    pub owner: String,
    pub repo: String,
    pub labels: Vec<String>,
}

/// GitHub integration adapter.
pub struct GitHubAdapter {
    config: GitHubAdapterConfig,
    client: Option<Arc<Octocrab>>,
}

impl GitHubAdapter {
    pub fn new(config: GitHubAdapterConfig) -> Self {
        Self {
            config,
            client: None,
        }
    }

    fn client(&self) -> Result<&Octocrab, AdapterError> {
        self.client
            .as_ref()
            .map(|c| c.as_ref())
            .ok_or_else(|| AdapterError::Connection("GitHub client not initialized".into()))
    }
}

#[async_trait]
impl RokoAdapter for GitHubAdapter {
    fn info(&self) -> AdapterInfo {
        AdapterInfo {
            name: "github".to_string(),
            description: format!("GitHub: {}/{}", self.config.owner, self.config.repo),
            capabilities: vec![
                AdapterCapability::WorkSource,
                AdapterCapability::WebhookReceiver,
            ],
            version: env!("CARGO_PKG_VERSION").to_string(),
        }
    }

    async fn init(&self) -> Result<(), AdapterError> {
        // Build client from GITHUB_TOKEN env var
        let token = std::env::var("GITHUB_TOKEN")
            .map_err(|_| AdapterError::Config("GITHUB_TOKEN not set".into()))?;

        let client = Octocrab::builder()
            .personal_token(token)
            .build()
            .map_err(|e| AdapterError::Connection(format!("octocrab build: {e}")))?;

        // Verify connectivity
        client
            .repos(&self.config.owner, &self.config.repo)
            .get()
            .await
            .map_err(|e| AdapterError::Connection(format!("GitHub API: {e}")))?;

        // Note: self is &self, so we cannot mutate. The actual implementation
        // should use interior mutability (OnceCell/OnceLock) for the client.
        Ok(())
    }

    async fn shutdown(&self) -> Result<(), AdapterError> {
        Ok(())
    }

    async fn health(&self) -> Result<(), AdapterError> {
        let client = self.client()?;
        client
            .ratelimit()
            .get()
            .await
            .map_err(|e| AdapterError::Connection(format!("GitHub health check: {e}")))?;
        Ok(())
    }
}

#[async_trait]
impl WorkSourceAdapter for GitHubAdapter {
    async fn list_work_items(&self) -> Result<Vec<WorkItem>, AdapterError> {
        let client = self.client()?;
        let issues = client
            .issues(&self.config.owner, &self.config.repo)
            .list()
            .state(octocrab::params::State::Open)
            .per_page(50)
            .send()
            .await
            .map_err(|e| AdapterError::Operation(format!("list issues: {e}")))?;

        Ok(issues
            .items
            .into_iter()
            .map(|issue| WorkItem {
                external_id: issue.number.to_string(),
                source: "github".to_string(),
                title: issue.title,
                body: issue.body.unwrap_or_default(),
                labels: issue.labels.iter().map(|l| l.name.clone()).collect(),
                priority: None,
                url: Some(issue.html_url.to_string()),
            })
            .collect())
    }

    async fn get_work_item(&self, id: &str) -> Result<Option<WorkItem>, AdapterError> {
        let number: u64 = id
            .parse()
            .map_err(|_| AdapterError::Operation(format!("invalid issue number: {id}")))?;

        let client = self.client()?;
        match client
            .issues(&self.config.owner, &self.config.repo)
            .get(number)
            .await
        {
            Ok(issue) => Ok(Some(WorkItem {
                external_id: issue.number.to_string(),
                source: "github".to_string(),
                title: issue.title,
                body: issue.body.unwrap_or_default(),
                labels: issue.labels.iter().map(|l| l.name.clone()).collect(),
                priority: None,
                url: Some(issue.html_url.to_string()),
            })),
            Err(_) => Ok(None),
        }
    }

    async fn update_work_item_status(
        &self,
        id: &str,
        status: &str,
        comment: Option<&str>,
    ) -> Result<(), AdapterError> {
        let number: u64 = id
            .parse()
            .map_err(|_| AdapterError::Operation(format!("invalid issue number: {id}")))?;

        let client = self.client()?;

        if let Some(comment_body) = comment {
            client
                .issues(&self.config.owner, &self.config.repo)
                .create_comment(number, comment_body)
                .await
                .map_err(|e| AdapterError::Operation(format!("add comment: {e}")))?;
        }

        if status == "closed" {
            client
                .issues(&self.config.owner, &self.config.repo)
                .update(number)
                .state(octocrab::models::IssueState::Closed)
                .send()
                .await
                .map_err(|e| AdapterError::Operation(format!("close issue: {e}")))?;
        }

        Ok(())
    }
}
```
