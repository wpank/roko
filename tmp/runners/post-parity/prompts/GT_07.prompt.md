# GT_07: Linear GraphQL client

## Task
Implement a `LinearAdapter` that implements `RokoAdapter` and `WorkSourceAdapter` using Linear's GraphQL API to list, get, and update tickets.

## Runner Context
Runner GT_C (GTM: Linear), batch 1 of 2. Depends on GT_01.

## Problem
Linear is a common project management tool for development teams. A Linear adapter would allow roko to pull work items from Linear and update ticket status when plans complete.

## Exact Changes

### Step 1: Create `crates/roko-cli/src/adapter/linear.rs`

```rust
//! Linear adapter -- implements WorkSourceAdapter via GraphQL (GT_07).

use async_trait::async_trait;
use roko_core::adapter::{
    AdapterCapability, AdapterError, AdapterInfo, RokoAdapter,
    WorkItem, WorkSourceAdapter,
};
use serde::{Deserialize, Serialize};

const LINEAR_API_URL: &str = "https://api.linear.app/graphql";

/// Linear adapter configuration.
#[derive(Debug, Clone)]
pub struct LinearAdapterConfig {
    pub team: String,
    pub project: Option<String>,
}

/// Linear integration adapter.
pub struct LinearAdapter {
    config: LinearAdapterConfig,
    api_key: Option<String>,
    client: reqwest::Client,
}

impl LinearAdapter {
    pub fn new(config: LinearAdapterConfig) -> Self {
        Self {
            config,
            api_key: None,
            client: reqwest::Client::new(),
        }
    }

    async fn graphql<T: for<'de> Deserialize<'de>>(
        &self,
        query: &str,
        variables: serde_json::Value,
    ) -> Result<T, AdapterError> {
        let key = self.api_key.as_ref()
            .ok_or_else(|| AdapterError::Connection("LINEAR_API_KEY not set".into()))?;

        let response = self.client
            .post(LINEAR_API_URL)
            .header("Authorization", key)
            .header("Content-Type", "application/json")
            .json(&serde_json::json!({
                "query": query,
                "variables": variables,
            }))
            .send()
            .await
            .map_err(|e| AdapterError::Connection(format!("Linear API: {e}")))?;

        let body: serde_json::Value = response.json().await
            .map_err(|e| AdapterError::Operation(format!("parse response: {e}")))?;

        if let Some(errors) = body.get("errors") {
            return Err(AdapterError::Operation(format!("GraphQL errors: {errors}")));
        }

        serde_json::from_value(body["data"].clone())
            .map_err(|e| AdapterError::Operation(format!("deserialize: {e}")))
    }
}

#[async_trait]
impl RokoAdapter for LinearAdapter {
    fn info(&self) -> AdapterInfo {
        AdapterInfo {
            name: "linear".to_string(),
            description: format!("Linear: team={}", self.config.team),
            capabilities: vec![AdapterCapability::WorkSource],
            version: env!("CARGO_PKG_VERSION").to_string(),
        }
    }

    async fn init(&self) -> Result<(), AdapterError> {
        let _key = std::env::var("LINEAR_API_KEY")
            .map_err(|_| AdapterError::Config("LINEAR_API_KEY not set".into()))?;
        Ok(())
    }

    async fn shutdown(&self) -> Result<(), AdapterError> {
        Ok(())
    }

    async fn health(&self) -> Result<(), AdapterError> {
        let _: serde_json::Value = self.graphql(
            "query { viewer { id name } }",
            serde_json::json!({}),
        ).await?;
        Ok(())
    }
}

#[derive(Deserialize)]
struct LinearIssuesResponse {
    issues: LinearIssueConnection,
}

#[derive(Deserialize)]
struct LinearIssueConnection {
    nodes: Vec<LinearIssue>,
}

#[derive(Deserialize)]
struct LinearIssue {
    id: String,
    identifier: String,
    title: String,
    description: Option<String>,
    url: String,
    priority: f64,
    labels: LinearLabelConnection,
}

#[derive(Deserialize)]
struct LinearLabelConnection {
    nodes: Vec<LinearLabel>,
}

#[derive(Deserialize)]
struct LinearLabel {
    name: String,
}

#[async_trait]
impl WorkSourceAdapter for LinearAdapter {
    async fn list_work_items(&self) -> Result<Vec<WorkItem>, AdapterError> {
        let query = r#"
            query($teamKey: String!) {
                issues(filter: { team: { key: { eq: $teamKey } }, state: { type: { nin: ["completed", "canceled"] } } }, first: 50) {
                    nodes {
                        id identifier title description url priority
                        labels { nodes { name } }
                    }
                }
            }
        "#;

        let response: LinearIssuesResponse = self.graphql(
            query,
            serde_json::json!({ "teamKey": self.config.team }),
        ).await?;

        Ok(response.issues.nodes.into_iter().map(|issue| {
            WorkItem {
                external_id: issue.identifier,
                source: "linear".to_string(),
                title: issue.title,
                body: issue.description.unwrap_or_default(),
                labels: issue.labels.nodes.into_iter().map(|l| l.name).collect(),
                priority: Some(issue.priority as u32),
                url: Some(issue.url),
            }
        }).collect())
    }

    async fn get_work_item(&self, id: &str) -> Result<Option<WorkItem>, AdapterError> {
        let items = self.list_work_items().await?;
        Ok(items.into_iter().find(|i| i.external_id == id))
    }

    async fn update_work_item_status(
        &self,
        id: &str,
        _status: &str,
        comment: Option<&str>,
    ) -> Result<(), AdapterError> {
        if let Some(body) = comment {
            let mutation = r#"
                mutation($issueId: String!, $body: String!) {
                    commentCreate(input: { issueId: $issueId, body: $body }) {
                        success
                    }
                }
            "#;
            let _: serde_json::Value = self.graphql(
                mutation,
                serde_json::json!({ "issueId": id, "body": body }),
            ).await?;
        }
        Ok(())
    }
}
```
