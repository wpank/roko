# IN_02: Wire SemanticCache for repeated query deduplication

## Task
Create a `SemanticCache` that detects and deduplicates repeated or near-identical queries, returning cached results instead of re-dispatching.

## Runner Context
Runner IN, batch 2 of 9. No dependencies.

## Problem
The same or very similar prompts are sometimes dispatched multiple times (retry loops, repeated tasks). A semantic cache that detects near-duplicates can return cached results instantly, saving cost and time.

## Current Code

**HDC vectors** -- `crates/roko-primitives/src/hdc.rs`:
Provides high-dimensional computing operations for similarity comparison.

//! SemanticCache — caches model call results by prompt similarity.
//!
//! Uses hash-based exact matching with an optional fuzzy fallback
//! based on character n-gram similarity.

use std::collections::HashMap;
use std::time::{Duration, Instant};
use serde::{Deserialize, Serialize};

/// Configuration for the semantic cache.
#[derive(Clone, Debug)]
pub struct SemanticCacheConfig {
    /// Maximum entries in the cache.
    pub max_entries: usize,
    /// TTL for cached entries.
    pub ttl: Duration,
    /// Minimum similarity threshold for fuzzy matching (0.0 to 1.0).
    pub similarity_threshold: f64,
}

impl Default for SemanticCacheConfig {
    fn default() -> Self {
        Self {
            max_entries: 100,
            ttl: Duration::from_secs(3600), // 1 hour
            similarity_threshold: 0.95,
        }
    }
}

/// A cached model call result.
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct CachedResult {
    /// The response text.
    pub text: String,
    /// Model that produced the result.
    pub model: String,
    /// Tokens used.
    pub tokens_in: u64,
    pub tokens_out: u64,
    /// Cost of the original call.
    pub cost_usd: f64,
}

struct CacheEntry {
    prompt_hash: u64,
    prompt_ngrams: Vec<u64>,
    result: CachedResult,
    created_at: Instant,
    hits: u32,
}

/// Cache for deduplicating model calls by prompt similarity.
pub struct SemanticCache {
    config: SemanticCacheConfig,
    entries: HashMap<u64, CacheEntry>,
    total_hits: u64,
    total_misses: u64,
    total_saved_cost: f64,
}

impl SemanticCache {
    /// Create a new cache with default configuration.
    pub fn new() -> Self {
        Self::with_config(SemanticCacheConfig::default())
    }

    /// Create with custom configuration.
    pub fn with_config(config: SemanticCacheConfig) -> Self {
        Self {
            config,
            entries: HashMap::new(),
            total_hits: 0,
            total_misses: 0,
            total_saved_cost: 0.0,
        }
    }

    /// Look up a prompt in the cache.
    ///
    /// Returns a cached result if an exact or near-duplicate match is found.
    pub fn lookup(&mut self, prompt: &str) -> Option<CachedResult> {
        let hash = hash_prompt(prompt);

        // Exact match
        if let Some(entry) = self.entries.get_mut(&hash) {
            if entry.created_at.elapsed() < self.config.ttl {
                entry.hits += 1;
                self.total_hits += 1;
                self.total_saved_cost += entry.result.cost_usd;
                return Some(entry.result.clone());
            } else {
                // Expired
                self.entries.remove(&hash);
            }
        }

        // Fuzzy match via n-gram similarity
        if self.config.similarity_threshold < 1.0 {
            let query_ngrams = compute_ngrams(prompt);
            for entry in self.entries.values_mut() {
                if entry.created_at.elapsed() >= self.config.ttl {
                    continue;
                }
                let similarity = ngram_similarity(&query_ngrams, &entry.prompt_ngrams);
                if similarity >= self.config.similarity_threshold {
                    entry.hits += 1;
                    self.total_hits += 1;
                    self.total_saved_cost += entry.result.cost_usd;
                    return Some(entry.result.clone());
                }
            }
        }

        self.total_misses += 1;
        None
    }

    /// Store a prompt->result mapping in the cache.
    pub fn store(&mut self, prompt: &str, result: CachedResult) {
        // Evict if at capacity
        if self.entries.len() >= self.config.max_entries {
            self.evict_oldest();
        }

        let hash = hash_prompt(prompt);
        let ngrams = compute_ngrams(prompt);

        self.entries.insert(hash, CacheEntry {
            prompt_hash: hash,
            prompt_ngrams: ngrams,
            result,
            created_at: Instant::now(),
            hits: 0,
        });
    }

    /// Total cost saved by cache hits.
    pub fn saved_cost(&self) -> f64 {
        self.total_saved_cost
    }

    /// Hit rate as a fraction.
    pub fn hit_rate(&self) -> f64 {
        let total = self.total_hits + self.total_misses;
        if total == 0 { 0.0 } else { self.total_hits as f64 / total as f64 }
    }

    /// Number of entries currently cached.
    pub fn len(&self) -> usize {
        self.entries.len()
    }

    /// Whether the cache is empty.
    pub fn is_empty(&self) -> bool {
        self.entries.is_empty()
    }

    fn evict_oldest(&mut self) {
        if let Some((&oldest_key, _)) = self.entries.iter()
            .min_by_key(|(_, e)| e.created_at)
        {
            self.entries.remove(&oldest_key);
        }
    }
}

impl Default for SemanticCache {
    fn default() -> Self {
        Self::new()
    }
}

fn hash_prompt(prompt: &str) -> u64 {
    use std::collections::hash_map::DefaultHasher;
    use std::hash::{Hash, Hasher};
    let mut hasher = DefaultHasher::new();
    prompt.hash(&mut hasher);
    hasher.finish()
}

fn compute_ngrams(text: &str) -> Vec<u64> {
    use std::collections::hash_map::DefaultHasher;
    use std::hash::{Hash, Hasher};

    let chars: Vec<char> = text.chars().collect();
    let n = 3;
    let mut ngrams = Vec::new();

    for window in chars.windows(n) {
        let mut hasher = DefaultHasher::new();
        window.hash(&mut hasher);
        ngrams.push(hasher.finish());
    }

    ngrams.sort();
    ngrams.dedup();
    ngrams
}

fn ngram_similarity(a: &[u64], b: &[u64]) -> f64 {
    if a.is_empty() || b.is_empty() {
        return 0.0;
    }

    let mut intersection = 0usize;
    let mut ai = 0;
    let mut bi = 0;

    while ai < a.len() && bi < b.len() {
        match a[ai].cmp(&b[bi]) {
            std::cmp::Ordering::Equal => {
                intersection += 1;
                ai += 1;
                bi += 1;
            }
            std::cmp::Ordering::Less => ai += 1,
            std::cmp::Ordering::Greater => bi += 1,
        }
    }

    let union = a.len() + b.len() - intersection;
    intersection as f64 / union as f64
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn exact_match_hit() {
        let mut cache = SemanticCache::new();
        cache.store("hello world", CachedResult {
            text: "response".into(),
            model: "test".into(),
            tokens_in: 10,
            tokens_out: 20,
            cost_usd: 0.01,
        });
        let result = cache.lookup("hello world");
        assert!(result.is_some());
        assert_eq!(result.unwrap().text, "response");
        assert_eq!(cache.total_hits, 1);
    }

    #[test]
    fn miss_on_different() {
        let mut cache = SemanticCache::new();
        cache.store("hello world", CachedResult {
            text: "response".into(),
            model: "test".into(),
            tokens_in: 10,
            tokens_out: 20,
            cost_usd: 0.01,
        });
        let result = cache.lookup("completely different prompt");
        assert!(result.is_none());
    }

    #[test]
    fn eviction_at_capacity() {
        let config = SemanticCacheConfig {
            max_entries: 2,
            ..Default::default()
        };
        let mut cache = SemanticCache::with_config(config);
        cache.store("a", CachedResult { text: "1".into(), model: "t".into(), tokens_in: 0, tokens_out: 0, cost_usd: 0.0 });
        cache.store("b", CachedResult { text: "2".into(), model: "t".into(), tokens_in: 0, tokens_out: 0, cost_usd: 0.0 });
        cache.store("c", CachedResult { text: "3".into(), model: "t".into(), tokens_in: 0, tokens_out: 0, cost_usd: 0.0 });
        assert_eq!(cache.len(), 2);
    }
}
