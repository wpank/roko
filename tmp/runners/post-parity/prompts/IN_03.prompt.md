# IN_03: Wire ContextBudgetManager for per-session allocation

## Task
Create a `ContextBudgetManager` that tracks context window usage across a session and allocates remaining budget to subsystems.

## Runner Context
Runner IN, batch 3 of 9. No dependencies.

## Problem
Each model has a fixed context window. Multiple subsystems compete for context space: system prompt, history, tools, knowledge, task context. Currently, allocation is ad-hoc. A `ContextBudgetManager` tracks total usage and distributes remaining budget fairly.

## Current Code

**BudgetGuardrail** -- `crates/roko-learn/src/budget.rs`:
Tracks cost budgets. No context window tracking.

## Exact Changes

### Step 1: Create `crates/roko-learn/src/context_budget.rs`

```rust
//! ContextBudgetManager — tracks and allocates context window tokens.
//!
//! Maintains per-section token counts and distributes remaining budget
//! to requesting subsystems proportionally.

use std::collections::HashMap;
use serde::{Deserialize, Serialize};

/// Per-section allocation with usage tracking.
#[derive(Clone, Debug, Default, Serialize, Deserialize)]
pub struct SectionAlloc {
    /// Maximum tokens allocated to this section.
    pub budget: usize,
    /// Tokens actually used.
    pub used: usize,
    /// Priority weight (higher = gets more budget).
    pub priority: f64,
}

/// Manages context window budget across sections.
pub struct ContextBudgetManager {
    /// Total context window size in tokens.
    total_budget: usize,
    /// Per-section allocations.
    sections: HashMap<String, SectionAlloc>,
    /// Reserved tokens (always reserved, not allocatable).
    reserved: usize,
}

impl ContextBudgetManager {
    /// Create a manager for a given total context window.
    pub fn new(total_budget: usize) -> Self {
        Self {
            total_budget,
            sections: HashMap::new(),
            reserved: 0,
        }
    }

    /// Reserve tokens that cannot be allocated to sections.
    ///
    /// Use for overhead like message framing, tool definitions, etc.
    pub fn reserve(&mut self, tokens: usize) {
        self.reserved = tokens;
    }

    /// Register a section with a priority weight.
    pub fn register_section(&mut self, name: impl Into<String>, priority: f64) {
        self.sections.insert(name.into(), SectionAlloc {
            budget: 0,
            used: 0,
            priority,
        });
    }

    /// Allocate budget to all registered sections proportionally.
    ///
    /// Each section gets budget proportional to its priority weight.
    pub fn allocate(&mut self) {
        let available = self.total_budget.saturating_sub(self.reserved);
        let total_priority: f64 = self.sections.values().map(|s| s.priority).sum();

        if total_priority <= 0.0 {
            return;
        }

        for section in self.sections.values_mut() {
            section.budget = ((available as f64) * (section.priority / total_priority)) as usize;
        }
    }

    /// Record token usage for a section.
    pub fn record_usage(&mut self, section: &str, tokens: usize) {
        if let Some(alloc) = self.sections.get_mut(section) {
            alloc.used = tokens;
        }
    }

    /// Get remaining budget for a section.
    pub fn remaining(&self, section: &str) -> usize {
        self.sections
            .get(section)
            .map(|s| s.budget.saturating_sub(s.used))
            .unwrap_or(0)
    }

    /// Get total remaining budget across all sections.
    pub fn total_remaining(&self) -> usize {
        let used: usize = self.sections.values().map(|s| s.used).sum();
        self.total_budget.saturating_sub(self.reserved + used)
    }

    /// Get the allocation for a section.
    pub fn get_allocation(&self, section: &str) -> Option<&SectionAlloc> {
        self.sections.get(section)
    }

    /// Reallocate unused budget from under-utilizing sections to
    /// sections that need more.
    pub fn rebalance(&mut self) {
        let mut surplus = 0usize;
        let mut deficit_total_priority = 0.0;

        // Collect surplus from under-utilizing sections
        for section in self.sections.values() {
            if section.used < section.budget {
                surplus += section.budget - section.used;
            } else if section.used >= section.budget {
                deficit_total_priority += section.priority;
            }
        }

        if surplus == 0 || deficit_total_priority <= 0.0 {
            return;
        }

        // Distribute surplus proportionally to deficit sections
        for section in self.sections.values_mut() {
            if section.used >= section.budget {
                let extra = ((surplus as f64) * (section.priority / deficit_total_priority)) as usize;
                section.budget += extra;
            }
        }
    }

    /// Total budget utilization as a fraction (0.0 to 1.0).
    pub fn utilization(&self) -> f64 {
        let used: usize = self.sections.values().map(|s| s.used).sum();
        let available = self.total_budget.saturating_sub(self.reserved);
        if available == 0 { 0.0 } else { used as f64 / available as f64 }
    }

    /// Summary of all section allocations.
    pub fn summary(&self) -> Vec<(String, usize, usize)> {
        let mut result: Vec<_> = self.sections
            .iter()
            .map(|(name, alloc)| (name.clone(), alloc.budget, alloc.used))
            .collect();
        result.sort_by(|a, b| a.0.cmp(&b.0));
        result
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn proportional_allocation() {
        let mut mgr = ContextBudgetManager::new(1000);
        mgr.reserve(200);
        mgr.register_section("system_prompt", 2.0);
        mgr.register_section("history", 1.0);
        mgr.register_section("tools", 1.0);
        mgr.allocate();

        // 800 available, system_prompt gets 2/4 = 400, others get 200 each
        assert_eq!(mgr.get_allocation("system_prompt").unwrap().budget, 400);
        assert_eq!(mgr.get_allocation("history").unwrap().budget, 200);
        assert_eq!(mgr.get_allocation("tools").unwrap().budget, 200);
    }

    #[test]
    fn rebalance_redistributes() {
        let mut mgr = ContextBudgetManager::new(1000);
        mgr.register_section("a", 1.0);
        mgr.register_section("b", 1.0);
        mgr.allocate();

        // a uses only 100 of 500, b needs more
        mgr.record_usage("a", 100);
        mgr.record_usage("b", 500);
        mgr.rebalance();

        // b should get some of a's surplus
        assert!(mgr.get_allocation("b").unwrap().budget > 500);
    }
}
```
