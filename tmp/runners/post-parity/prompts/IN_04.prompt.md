# IN_04: Wire failure pattern extraction from gate history

## Task
Create a `GateEvolver` module that analyzes gate failure history to extract recurring failure patterns.

## Runner Context
Runner IN, batch 4 of 9. No dependencies.

## Problem
Gate failures repeat. The same compilation errors, the same lint warnings, the same test failures recur across tasks. Extracting patterns from gate history lets the system generate targeted checks and hints.

## Current Code

**ErrorPatternStore** -- `crates/roko-learn/src/error_pattern_store.rs`:
Stores error patterns with frequency and recency data.

## Exact Changes

### Step 1: Create `crates/roko-learn/src/gate_evolver.rs`

```rust
//! GateEvolver — extracts failure patterns from gate history and
//! generates dynamic gate checks.

use serde::{Deserialize, Serialize};
use std::collections::HashMap;
use std::path::{Path, PathBuf};

/// A recurring failure pattern extracted from gate history.
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct FailurePattern {
    /// Human-readable label.
    pub label: String,
    /// Pattern category (compile, lint, test, logic).
    pub category: String,
    /// Frequency: how many times this pattern has been seen.
    pub frequency: u32,
    /// Last seen timestamp.
    pub last_seen: String,
    /// Example error text (truncated).
    pub example: String,
    /// File paths where this pattern occurs most.
    pub hot_files: Vec<String>,
    /// Suggested check command, if derivable.
    pub suggested_check: Option<String>,
}

/// Accumulated gate failure records for pattern extraction.
#[derive(Clone, Debug, Default, Serialize, Deserialize)]
pub struct GateHistory {
    /// Raw failure entries.
    pub entries: Vec<GateFailureEntry>,
}

/// One gate failure entry.
#[derive(Clone, Debug, Serialize, Deserialize)]
pub struct GateFailureEntry {
    /// Gate name (compile, clippy, test).
    pub gate: String,
    /// Task ID that failed.
    pub task_id: String,
    /// Error output (truncated).
    pub error_text: String,
    /// Timestamp.
    pub timestamp: String,
    /// Files involved.
    pub files: Vec<String>,
}

/// Extracts and manages failure patterns from gate history.
pub struct GateEvolver {
    /// Path to gate history data.
    data_dir: PathBuf,
    /// Extracted patterns.
    patterns: Vec<FailurePattern>,
    /// Minimum frequency to consider a pattern recurring.
    min_frequency: u32,
}

impl GateEvolver {
    /// Create a new evolver reading from the given data directory.
    pub fn new(data_dir: PathBuf) -> Self {
        Self {
            data_dir,
            patterns: Vec::new(),
            min_frequency: 3,
        }
    }

    /// Record a gate failure.
    pub fn record_failure(&mut self, entry: GateFailureEntry) {
        let history_path = self.data_dir.join("gate-failures.jsonl");
        if let Some(parent) = history_path.parent() {
            let _ = std::fs::create_dir_all(parent);
        }
        if let Ok(json) = serde_json::to_string(&entry) {
            use std::io::Write;
            if let Ok(mut f) = std::fs::OpenOptions::new()
                .create(true)
                .append(true)
                .open(&history_path)
            {
                let _ = writeln!(f, "{json}");
            }
        }
    }

    /// Extract patterns from accumulated gate history.
    pub fn extract_patterns(&mut self) -> &[FailurePattern] {
        let history_path = self.data_dir.join("gate-failures.jsonl");
        let history = load_gate_history(&history_path);

        // Group by normalized error signature
        let mut groups: HashMap<String, Vec<&GateFailureEntry>> = HashMap::new();
        for entry in &history.entries {
            let signature = normalize_error(&entry.error_text);
            groups.entry(signature).or_default().push(entry);
        }

        // Convert groups to patterns
        self.patterns = groups
            .into_iter()
            .filter(|(_, entries)| entries.len() as u32 >= self.min_frequency)
            .map(|(sig, entries)| {
                let mut hot_files: HashMap<String, u32> = HashMap::new();
                for entry in &entries {
                    for file in &entry.files {
                        *hot_files.entry(file.clone()).or_default() += 1;
                    }
                }
                let mut hot_files_sorted: Vec<String> = hot_files
                    .into_iter()
                    .collect::<Vec<_>>()
                    .into_iter()
                    .map(|(f, _)| f)
                    .collect();
                hot_files_sorted.truncate(5);

                let category = entries.first().map(|e| e.gate.clone()).unwrap_or_default();
                let suggested_check = derive_check(&category, &sig);

                FailurePattern {
                    label: sig.chars().take(80).collect(),
                    category,
                    frequency: entries.len() as u32,
                    last_seen: entries.last().map(|e| e.timestamp.clone()).unwrap_or_default(),
                    example: entries.first().map(|e| e.error_text.chars().take(200).collect()).unwrap_or_default(),
                    hot_files: hot_files_sorted,
                    suggested_check,
                }
            })
            .collect();

        self.patterns.sort_by(|a, b| b.frequency.cmp(&a.frequency));
        &self.patterns
    }

    /// Get the most frequent failure patterns.
    pub fn top_patterns(&self, n: usize) -> &[FailurePattern] {
        &self.patterns[..n.min(self.patterns.len())]
    }

    /// Number of extracted patterns.
    pub fn pattern_count(&self) -> usize {
        self.patterns.len()
    }
}

fn load_gate_history(path: &Path) -> GateHistory {
    let content = std::fs::read_to_string(path).unwrap_or_default();
    let entries: Vec<GateFailureEntry> = content
        .lines()
        .filter_map(|line| serde_json::from_str(line).ok())
        .collect();
    GateHistory { entries }
}

fn normalize_error(text: &str) -> String {
    // Strip line numbers, file paths, and variable names to create a signature
    let mut sig = text.to_string();
    // Remove line numbers
    sig = sig.replace(|c: char| c.is_ascii_digit(), "#");
    // Collapse whitespace
    sig = sig.split_whitespace().collect::<Vec<_>>().join(" ");
    // Truncate
    sig.chars().take(120).collect()
}

fn derive_check(category: &str, signature: &str) -> Option<String> {
    match category {
        "compile" => Some("cargo build --workspace 2>&1 | head -50".into()),
        "clippy" => Some("cargo clippy --workspace --no-deps -- -D warnings 2>&1 | head -50".into()),
        "test" => {
            // Try to extract test module name from signature
            if signature.contains("test_") {
                Some(format!("cargo test --workspace 2>&1 | head -50"))
            } else {
                Some("cargo test --workspace 2>&1 | head -50".into())
            }
        }
        _ => None,
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn normalize_strips_numbers() {
        let sig = normalize_error("error at line 42 in foo.rs");
        assert!(!sig.contains("42"));
    }

    #[test]
    fn derive_check_for_compile() {
        let check = derive_check("compile", "type mismatch");
        assert!(check.is_some());
        assert!(check.unwrap().contains("cargo build"));
    }
}
```
