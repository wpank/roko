# IN_06: Cost-aware Pareto routing in CascadeRouter

## Task
Add Pareto-optimal model selection to `CascadeRouter` that considers cost alongside quality and latency.

## Runner Context
Runner IN, batch 6 of 9. No dependencies.

## Problem
The CascadeRouter selects models via UCB1/Thompson based on quality reward. It does not consider cost-quality tradeoffs. For many tasks, a cheaper model achieves the same quality. Pareto routing identifies the cheapest model at each quality level and prefers it.

/// A model on the Pareto front (not dominated by any other).
#[derive(Clone, Debug)]
pub struct ParetoCandidate {
    /// Model slug.
    pub model: String,
    /// Quality score (higher is better).
    pub quality: f64,
    /// Cost per call in USD (lower is better).
    pub cost_per_call: f64,
    /// Is this model on the Pareto front?
    pub is_pareto: bool,
}

impl CascadeRouter {
    /// Compute the Pareto front across quality and cost.
    ///
    /// A model is Pareto-optimal if no other model is both cheaper
    /// AND higher quality.
    pub fn pareto_front(&self, cost_table: &CostTable) -> Vec<ParetoCandidate> {
        let candidates: Vec<ParetoCandidate> = self.model_slugs()
            .iter()
            .map(|slug| {
                let quality = self.confidence_snapshot()
                    .get(slug)
                    .map(|(successes, total)| {
                        if *total == 0 { 0.5 } else { *successes as f64 / *total as f64 }
                    })
                    .unwrap_or(0.5);

                let cost_per_call = crate::model_router::normalized_cost(slug, cost_table);

                ParetoCandidate {
                    model: slug.clone(),
                    quality,
                    cost_per_call,
                    is_pareto: false,
                }
            })
            .collect();

        mark_pareto(candidates)
    }

    /// Select the cheapest Pareto-optimal model at or above the
    /// given quality threshold.
    pub fn select_pareto(
        &self,
        cost_table: &CostTable,
        min_quality: f64,
    ) -> Option<String> {
        let front = self.pareto_front(cost_table);
        front
            .iter()
            .filter(|c| c.is_pareto && c.quality >= min_quality)
            .min_by(|a, b| a.cost_per_call.partial_cmp(&b.cost_per_call).unwrap_or(std::cmp::Ordering::Equal))
            .map(|c| c.model.clone())
    }

    /// Route with cost-awareness: prefer Pareto-optimal models when
    /// a quality threshold is met.
    pub fn route_cost_aware(
        &self,
        ctx: &RoutingContext,
        cost_table: &CostTable,
        min_quality: f64,
    ) -> CascadeModel {
        // First, try Pareto selection
        if let Some(pareto_model) = self.select_pareto(cost_table, min_quality) {
            tracing::debug!(
                model = %pareto_model,
                min_quality,
                "pareto-optimal model selected"
            );
            return CascadeModel {
                model_slug: pareto_model.clone(),
                confidence: min_quality,
                reason: "pareto-optimal".into(),
            };
        }

        // Fall back to standard routing
        self.route(ctx)
    }
}

/// Mark Pareto-optimal candidates.
///
/// A candidate is dominated if another candidate has both higher quality
/// and lower cost. Pareto-optimal candidates are not dominated by any other.
fn mark_pareto(mut candidates: Vec<ParetoCandidate>) -> Vec<ParetoCandidate> {
    let n = candidates.len();
    for i in 0..n {
        let mut dominated = false;
        for j in 0..n {
            if i == j { continue; }
            if candidates[j].quality >= candidates[i].quality
                && candidates[j].cost_per_call <= candidates[i].cost_per_call
                && (candidates[j].quality > candidates[i].quality
                    || candidates[j].cost_per_call < candidates[i].cost_per_call)
            {
                dominated = true;
                break;
            }
        }
        candidates[i].is_pareto = !dominated;
    }
    candidates
}

use crate::model_router::RoutingContext;
