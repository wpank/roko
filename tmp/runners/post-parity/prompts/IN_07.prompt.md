# IN_07: Define SteeringAction enum and controller

## Task
Define a `SteeringAction` enum and `SteeringController` that governs high-level agent behavior decisions: pause, resume, escalate, demote, split, merge.

## Runner Context
Runner IN, batch 7 of 9. No dependencies.

## Problem
The orchestrator makes ad-hoc decisions about agent behavior (retry, switch model, abort). A `SteeringController` provides a structured vocabulary of actions and a policy framework for making these decisions consistently.

## Exact Changes

### Step 1: Create `crates/roko-core/src/steering.rs`

```rust
//! Steering — high-level agent behavior control.
//!
//! The steering layer sits above individual agent dispatch and below
//! the plan executor. It provides a vocabulary of actions for
//! controlling agent behavior at the task level.

use serde::{Deserialize, Serialize};

/// A high-level steering action for agent control.
#[derive(Clone, Debug, PartialEq, Eq, Hash, Serialize, Deserialize)]
pub enum SteeringAction {
    /// Continue with current configuration.
    Continue,
    /// Pause the current task (save state, yield control).
    Pause { reason: String },
    /// Resume a paused task.
    Resume { task_id: String },
    /// Escalate to a more capable model.
    Escalate { from_model: String, to_model: String },
    /// Demote to a cheaper model (task is simpler than expected).
    Demote { from_model: String, to_model: String },
    /// Split a task into subtasks.
    Split { task_id: String, subtask_count: usize },
    /// Merge subtask results into a parent task.
    Merge { parent_task_id: String, subtask_ids: Vec<String> },
    /// Inject context into the next agent turn.
    InjectContext { context: String },
    /// Switch the agent's role.
    SwitchRole { from_role: String, to_role: String },
    /// Abort the task entirely.
    Abort { reason: String },
    /// Request human review before proceeding.
    RequestReview { task_id: String, question: String },
}

impl SteeringAction {
    /// Whether this action changes the active model.
    pub fn changes_model(&self) -> bool {
        matches!(self, Self::Escalate { .. } | Self::Demote { .. })
    }

    /// Whether this action terminates the task.
    pub fn is_terminal(&self) -> bool {
        matches!(self, Self::Abort { .. })
    }

    /// Whether this action requires human intervention.
    pub fn requires_human(&self) -> bool {
        matches!(self, Self::RequestReview { .. })
    }

    /// Short label for logging.
    pub fn label(&self) -> &'static str {
        match self {
            Self::Continue => "continue",
            Self::Pause { .. } => "pause",
            Self::Resume { .. } => "resume",
            Self::Escalate { .. } => "escalate",
            Self::Demote { .. } => "demote",
            Self::Split { .. } => "split",
            Self::Merge { .. } => "merge",
            Self::InjectContext { .. } => "inject-context",
            Self::SwitchRole { .. } => "switch-role",
            Self::Abort { .. } => "abort",
            Self::RequestReview { .. } => "request-review",
        }
    }
}

/// Input state for steering decisions.
#[derive(Clone, Debug, Default)]
pub struct SteeringContext {
    /// Current attempt number.
    pub attempt: u32,
    /// Consecutive failures.
    pub consecutive_failures: u32,
    /// Total cost spent on this task.
    pub cost_usd: f64,
    /// Time elapsed on this task.
    pub elapsed_ms: u64,
    /// Current model slug.
    pub model: String,
    /// Current role.
    pub role: String,
    /// Quality score from most recent gate (0.0 to 1.0).
    pub quality_score: f64,
    /// Whether the last attempt was a success.
    pub last_succeeded: bool,
}

/// Policy for making steering decisions.
pub trait SteeringPolicy: Send + Sync {
    /// Decide the next steering action based on context.
    fn decide(&self, ctx: &SteeringContext) -> SteeringAction;
}

/// Default rule-based steering policy.
pub struct DefaultSteeringPolicy {
    /// Maximum consecutive failures before escalation.
    pub escalate_after: u32,
    /// Maximum consecutive failures before abort.
    pub abort_after: u32,
    /// Cost threshold for demotion (switch to cheaper model).
    pub demote_cost_threshold: f64,
}

impl Default for DefaultSteeringPolicy {
    fn default() -> Self {
        Self {
            escalate_after: 2,
            abort_after: 5,
            demote_cost_threshold: 0.50,
        }
    }
}

impl SteeringPolicy for DefaultSteeringPolicy {
    fn decide(&self, ctx: &SteeringContext) -> SteeringAction {
        if ctx.last_succeeded {
            return SteeringAction::Continue;
        }

        if ctx.consecutive_failures >= self.abort_after {
            return SteeringAction::Abort {
                reason: format!("{} consecutive failures", ctx.consecutive_failures),
            };
        }

        if ctx.consecutive_failures >= self.escalate_after {
            return SteeringAction::Escalate {
                from_model: ctx.model.clone(),
                to_model: "strongest".into(), // placeholder
            };
        }

        if ctx.cost_usd > self.demote_cost_threshold && ctx.quality_score > 0.8 {
            return SteeringAction::Demote {
                from_model: ctx.model.clone(),
                to_model: "cheapest".into(), // placeholder
            };
        }

        SteeringAction::Continue
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn continue_on_success() {
        let policy = DefaultSteeringPolicy::default();
        let ctx = SteeringContext {
            last_succeeded: true,
            ..Default::default()
        };
        assert_eq!(policy.decide(&ctx), SteeringAction::Continue);
    }

    #[test]
    fn escalate_on_repeated_failure() {
        let policy = DefaultSteeringPolicy::default();
        let ctx = SteeringContext {
            consecutive_failures: 3,
            model: "sonnet".into(),
            ..Default::default()
        };
        assert!(policy.decide(&ctx).changes_model());
    }

    #[test]
    fn abort_on_many_failures() {
        let policy = DefaultSteeringPolicy::default();
        let ctx = SteeringContext {
            consecutive_failures: 5,
            ..Default::default()
        };
        assert!(policy.decide(&ctx).is_terminal());
    }
}
```
