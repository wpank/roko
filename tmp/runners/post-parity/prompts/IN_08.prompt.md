//! SkillSelector — routes tasks to models based on skill profiles.
//!
//! Each model has a skill profile (coding, research, reasoning, etc.).
//! Each task has skill requirements. The selector matches tasks to
//! models that have the strongest relevant skills.

use serde::{Deserialize, Serialize};
use std::collections::HashMap;

/// Skill dimensions for model and task profiles.
#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash, Serialize, Deserialize)]
pub enum Skill {
    /// Code generation and editing.
    Coding,
    /// Logical reasoning and problem solving.
    Reasoning,
    /// Research and information synthesis.
    Research,
    /// Documentation and writing.
    Writing,
    /// Multi-step planning.
    Planning,
    /// Tool use and function calling.
    ToolUse,
    /// Reviewing and critique.
    Review,
}

/// Skill profile for a model or task.
#[derive(Clone, Debug, Default, Serialize, Deserialize)]
pub struct SkillProfile {
    /// Skill scores (0.0 to 1.0).
    pub scores: HashMap<Skill, f64>,
}

impl SkillProfile {
    /// Create a profile with given skill scores.
    pub fn new(scores: Vec<(Skill, f64)>) -> Self {
        Self {
            scores: scores.into_iter().collect(),
        }
    }

    /// Score for a specific skill (0.0 if not set).
    pub fn score(&self, skill: Skill) -> f64 {
        self.scores.get(&skill).copied().unwrap_or(0.0)
    }

    /// Compute similarity to another profile.
    pub fn similarity(&self, other: &SkillProfile) -> f64 {
        let all_skills: Vec<Skill> = self.scores.keys()
            .chain(other.scores.keys())
            .copied()
            .collect::<std::collections::HashSet<_>>()
            .into_iter()
            .collect();

        if all_skills.is_empty() {
            return 0.0;
        }

        let dot: f64 = all_skills.iter()
            .map(|s| self.score(*s) * other.score(*s))
            .sum();

        let mag_a: f64 = all_skills.iter()
            .map(|s| self.score(*s).powi(2))
            .sum::<f64>()
            .sqrt();

        let mag_b: f64 = all_skills.iter()
            .map(|s| other.score(*s).powi(2))
            .sum::<f64>()
            .sqrt();

        if mag_a == 0.0 || mag_b == 0.0 {
            0.0
        } else {
            dot / (mag_a * mag_b)
        }
    }
}

/// Routes tasks to models based on skill matching.
pub struct ModelSkillRouter {
    /// Known model skill profiles.
    model_profiles: HashMap<String, SkillProfile>,
    /// Task category to skill requirements mapping.
    task_requirements: HashMap<String, SkillProfile>,
}

impl ModelSkillRouter {
    /// Create a router with default model profiles.
    pub fn new() -> Self {
        let mut model_profiles = HashMap::new();

        // Default profiles for common models
        model_profiles.insert("claude-opus-4-20250514".into(), SkillProfile::new(vec![
            (Skill::Coding, 0.95), (Skill::Reasoning, 0.95), (Skill::Research, 0.9),
            (Skill::Writing, 0.95), (Skill::Planning, 0.95), (Skill::ToolUse, 0.9),
            (Skill::Review, 0.95),
        ]));

        model_profiles.insert("claude-sonnet-4-20250514".into(), SkillProfile::new(vec![
            (Skill::Coding, 0.9), (Skill::Reasoning, 0.85), (Skill::Research, 0.8),
            (Skill::Writing, 0.85), (Skill::Planning, 0.8), (Skill::ToolUse, 0.85),
            (Skill::Review, 0.85),
        ]));

        model_profiles.insert("claude-haiku-3-5-20241022".into(), SkillProfile::new(vec![
            (Skill::Coding, 0.7), (Skill::Reasoning, 0.6), (Skill::Research, 0.5),
            (Skill::Writing, 0.7), (Skill::Planning, 0.5), (Skill::ToolUse, 0.6),
            (Skill::Review, 0.6),
        ]));

        let mut task_requirements = HashMap::new();
        task_requirements.insert("code".into(), SkillProfile::new(vec![
            (Skill::Coding, 0.9), (Skill::ToolUse, 0.7),
        ]));
        task_requirements.insert("research".into(), SkillProfile::new(vec![
            (Skill::Research, 0.8), (Skill::Reasoning, 0.7),
        ]));
        task_requirements.insert("docs".into(), SkillProfile::new(vec![
            (Skill::Writing, 0.7),
        ]));
        task_requirements.insert("review".into(), SkillProfile::new(vec![
            (Skill::Review, 0.8), (Skill::Reasoning, 0.7),
        ]));
        task_requirements.insert("plan".into(), SkillProfile::new(vec![
            (Skill::Planning, 0.8), (Skill::Reasoning, 0.8),
        ]));

        Self { model_profiles, task_requirements }
    }

    /// Register a model's skill profile.
    pub fn register_model(&mut self, slug: impl Into<String>, profile: SkillProfile) {
        self.model_profiles.insert(slug.into(), profile);
    }

    /// Select the best model for a task category.
    pub fn select(&self, task_category: &str, available_models: &[String]) -> Option<String> {
        let requirements = self.task_requirements.get(task_category)?;

        available_models
            .iter()
            .filter_map(|slug| {
                let profile = self.model_profiles.get(slug)?;
                Some((slug.clone(), profile.similarity(requirements)))
            })
            .max_by(|(_, a), (_, b)| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal))
            .map(|(slug, _)| slug)
    }

    /// Rank all available models by suitability for a task.
    pub fn rank(&self, task_category: &str, available_models: &[String]) -> Vec<(String, f64)> {
        let requirements = match self.task_requirements.get(task_category) {
            Some(r) => r,
            None => return available_models.iter().map(|m| (m.clone(), 0.5)).collect(),
        };

        let mut ranked: Vec<(String, f64)> = available_models
            .iter()
            .map(|slug| {
                let score = self.model_profiles.get(slug)
                    .map(|p| p.similarity(requirements))
                    .unwrap_or(0.5);
                (slug.clone(), score)
            })
            .collect();

        ranked.sort_by(|(_, a), (_, b)| b.partial_cmp(a).unwrap_or(std::cmp::Ordering::Equal));
        ranked
    }
}

impl Default for ModelSkillRouter {
    fn default() -> Self {
        Self::new()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn select_best_for_coding() {
        let router = ModelSkillRouter::new();
        let models = vec![
            "claude-opus-4-20250514".into(),
            "claude-sonnet-4-20250514".into(),
            "claude-haiku-3-5-20241022".into(),
        ];
        let best = router.select("code", &models);
        assert_eq!(best, Some("claude-opus-4-20250514".into()));
    }

    #[test]
    fn rank_for_docs() {
        let router = ModelSkillRouter::new();
        let models = vec![
            "claude-opus-4-20250514".into(),
            "claude-haiku-3-5-20241022".into(),
        ];
        let ranked = router.rank("docs", &models);
        assert_eq!(ranked[0].0, "claude-opus-4-20250514");
        assert!(ranked[0].1 > ranked[1].1);
    }

    #[test]
    fn unknown_category_returns_defaults() {
        let router = ModelSkillRouter::new();
        let models = vec!["claude-sonnet-4-20250514".into()];
        let ranked = router.rank("unknown", &models);
        assert_eq!(ranked[0].1, 0.5);
    }
}
