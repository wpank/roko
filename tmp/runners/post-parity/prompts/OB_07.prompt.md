# OB_07: Add GateRow widgets to TUI

## Task
Create a `GateRow` widget for the ratatui TUI that displays gate pipeline results as a horizontal row of pass/fail/skip indicators.

## Runner Context
Runner OB_B (Observability: Dashboard), batch 3 of 4. No dependencies.

## Problem
Gate results are logged but not visualized in the TUI. Operators need a quick at-a-glance view of which gates passed, failed, or were skipped in the current run.

## Current Code

### Step 1: Create `crates/roko-cli/src/tui/widgets/gate_row.rs`

```rust
//! GateRow widget — horizontal gate result indicators (OB_07).

use ratatui::prelude::*;
use ratatui::widgets::{Block, Borders};

/// Status of a single gate.
#[derive(Debug, Clone, Copy, PartialEq, Eq)]
pub enum GateStatus {
    /// Gate passed.
    Pass,
    /// Gate failed.
    Fail,
    /// Gate was skipped.
    Skip,
    /// Gate is currently running.
    Running,
    /// Gate has not been reached yet.
    Pending,
}

/// A single gate result for display.
#[derive(Debug, Clone)]
pub struct GateResult {
    pub name: String,
    pub rung: u8,
    pub status: GateStatus,
    pub duration_ms: Option<u64>,
    pub detail: Option<String>,
}

/// Widget that displays gate results as a horizontal row.
pub struct GateRow<'a> {
    gates: &'a [GateResult],
    title: &'a str,
}

impl<'a> GateRow<'a> {
    pub fn new(gates: &'a [GateResult]) -> Self {
        Self {
            gates,
            title: "Gates",
        }
    }

    pub fn title(mut self, title: &'a str) -> Self {
        self.title = title;
        self
    }
}

impl Widget for GateRow<'_> {
    fn render(self, area: Rect, buf: &mut Buffer) {
        let block = Block::default()
            .borders(Borders::ALL)
            .title(self.title);
        let inner = block.inner(area);
        Widget::render(block, area, buf);

        if self.gates.is_empty() || inner.width == 0 || inner.height == 0 {
            return;
        }

        let gate_width = (inner.width as usize / self.gates.len().max(1)).max(6);

        for (i, gate) in self.gates.iter().enumerate() {
            let x = inner.x + (i * gate_width) as u16;
            if x >= inner.x + inner.width {
                break;
            }

            let (symbol, style) = match gate.status {
                GateStatus::Pass => ("PASS", Style::default().fg(Color::Green)),
                GateStatus::Fail => ("FAIL", Style::default().fg(Color::Red).add_modifier(Modifier::BOLD)),
                GateStatus::Skip => ("SKIP", Style::default().fg(Color::DarkGray)),
                GateStatus::Running => (" .. ", Style::default().fg(Color::Yellow).add_modifier(Modifier::BOLD)),
                GateStatus::Pending => (" -- ", Style::default().fg(Color::DarkGray)),
            };

            // Gate name on first line
            let name = if gate.name.len() > gate_width - 1 {
                &gate.name[..gate_width - 1]
            } else {
                &gate.name
            };
            buf.set_string(x, inner.y, name, Style::default().fg(Color::White));

            // Status on second line
            if inner.height > 1 {
                buf.set_string(x, inner.y + 1, symbol, style);
            }

            // Duration on third line
            if inner.height > 2 {
                if let Some(ms) = gate.duration_ms {
                    let dur = if ms > 1000 {
                        format!("{:.1}s", ms as f64 / 1000.0)
                    } else {
                        format!("{ms}ms")
                    };
                    buf.set_string(x, inner.y + 2, &dur, Style::default().fg(Color::DarkGray));
                }
            }
        }
    }
}
```
