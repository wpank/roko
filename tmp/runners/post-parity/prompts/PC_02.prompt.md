### Step 1: Define `StreamingCallback` trait in `streaming.rs`

Add after the imports, before the `StreamingState` struct:

```rust
/// Callback for receiving streaming token deltas during LLM generation.
pub trait StreamingCallback: Send + 'static {
    /// Called when a new token delta arrives.
    fn on_token(&self, delta: &str);

    /// Called when streaming starts (phase transitions from Thinking to Streaming).
    fn on_start(&self, _model: &str) {}

    /// Called when streaming completes.
    fn on_complete(&self, _token_count: u64) {}

    /// Called when usage data arrives (tokens, cost).
    fn on_usage(&self, _input_tokens: u64, _output_tokens: u64, _cost_usd: f64) {}
}
```

### Step 2: Implement for `Arc<Mutex<StreamingState>>`

```rust
use std::sync::{Arc, Mutex};

impl StreamingCallback for Arc<Mutex<StreamingState>> {
    fn on_token(&self, delta: &str) {
        if let Ok(mut state) = self.lock() {
            state.append(delta);
        }
    }

    fn on_start(&self, model: &str) {
        if let Ok(mut state) = self.lock() {
            state.set_phase("Streaming");
            state.model = model.to_string();
        }
    }

    fn on_complete(&self, token_count: u64) {
        if let Ok(mut state) = self.lock() {
            state.set_phase("Done");
            state.token_count = token_count;
        }
    }

    fn on_usage(&self, input_tokens: u64, output_tokens: u64, cost_usd: f64) {
        if let Ok(mut state) = self.lock() {
            state.token_count = input_tokens + output_tokens;
            state.cost_usd = cost_usd;
        }
    }
}
```
