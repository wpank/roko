## Exact Changes

### Step 1: Update the `/effort` handler

Replace the match arm at chat_inline.rs:2250-2279:

```rust
_ if cmd.starts_with("/effort") => {
    let arg = cmd.strip_prefix("/effort").unwrap().trim();
    if arg.is_empty() {
        // Show current effort
        let current = &agent_session.effort;
        term.push_lines(&[styled::continuation(theme, "effort",
            &format!("current: {current}"),
            Some("options: low, medium, high, max"))])?;
    } else {
        let normalized = match arg {
            "low" | "medium" | "med" | "high" | "max" => {
                if arg == "med" { "medium" } else { arg }
            }
            _ => {
                term.push_lines(&[styled::continuation(theme, "effort",
                    &format!("unknown: {arg}"),
                    Some("options: low, medium, high, max"))])?;
                return Ok(());  // or continue
            }
        };
        agent_session.set_effort(normalized);
        term.push_lines(&[styled::continuation(theme, "effort",
            &format!("set to {normalized}"), None)])?;
    }
}
```
