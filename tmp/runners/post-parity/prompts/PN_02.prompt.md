## Task

## Exact Changes

### Step 1: Add cancel handler

```rust
pub async fn cancel_bench_run(
    State(state): State<Arc<AppState>>,
    Path(run_id): Path<String>,
) -> impl IntoResponse {
    // Find the running bench
    let bench_store = state.bench_store.lock().await;
    match bench_store.get(&run_id) {
        None => (StatusCode::NOT_FOUND, Json(serde_json::json!({
            "error": "bench run not found",
            "run_id": run_id,
        }))),
        Some(run) if run.is_completed() => (StatusCode::CONFLICT, Json(serde_json::json!({
            "error": "bench run already completed",
            "status": run.status,
        }))),
        Some(run) => {
            // Cancel the running bench
            run.cancel_token.cancel();
            drop(bench_store);

            // Update status
            let mut store = state.bench_store.lock().await;
            if let Some(run) = store.get_mut(&run_id) {
                run.status = "cancelled".to_string();
                run.completed_at = Some(chrono::Utc::now().timestamp_millis() as u64);
            }

            (StatusCode::OK, Json(serde_json::json!({
                "status": "cancelled",
                "run_id": run_id,
            })))
        }
    }
}
```
