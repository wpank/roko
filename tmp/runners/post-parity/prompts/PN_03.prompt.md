### Step 1: Add /api/knowledge/edges endpoint (or fix existing)

```rust
pub async fn get_knowledge_edges(
    State(state): State<Arc<AppState>>,
) -> impl IntoResponse {
    let neuro = match &state.neuro_store {
        Some(store) => store,
        None => return Json(serde_json::json!([])),
    };

    // Get all knowledge entries
    let entries = match neuro.list_all(100) {
        Ok(entries) => entries,
        Err(_) => return Json(serde_json::json!([])),
    };

    // Build edges from shared topics and episode references
    let mut edges: Vec<serde_json::Value> = Vec::new();
    for (i, a) in entries.iter().enumerate() {
        for b in entries.iter().skip(i + 1) {
            let shared_topics = a.topics.iter()
                .filter(|t| b.topics.contains(t))
                .count();
            let same_episode = a.source_episode == b.source_episode
                && a.source_episode.is_some();

            let frequency = shared_topics + if same_episode { 2 } else { 0 };
            if frequency > 0 {
                edges.push(serde_json::json!({
                    "source": a.id,
                    "target": b.id,
                    "frequency": frequency,
                }));
            }
        }
    }

    Json(serde_json::Value::Array(edges))
}
```

### Step 2: Register the route

```rust
.route("/knowledge/edges", get(get_knowledge_edges))
```
