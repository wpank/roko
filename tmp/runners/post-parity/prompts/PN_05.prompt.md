## Exact Changes

### Step 1: Add or fix the provider-outcomes handler

```rust
pub async fn get_provider_outcomes(
    State(state): State<Arc<AppState>>,
) -> impl IntoResponse {
    let tracker = &state.provider_health;
    let providers = tracker.all_providers().await;

    let outcomes: Vec<serde_json::Value> = providers.iter().map(|health| {
        let success_count = health.total_requests - health.total_failures;
        let success_rate = if health.total_requests > 0 {
            success_count as f64 / health.total_requests as f64
        } else {
            1.0
        };

        serde_json::json!({
            "name": health.provider_id,
            "success_count": success_count,
            "failure_count": health.total_failures,
            "total_requests": health.total_requests,
            "success_rate": success_rate,
            "circuit_state": format!("{:?}", health.state),
            "consecutive_failures": health.consecutive_failures,
        })
    }).collect();

    Json(serde_json::json!({ "providers": outcomes }))
}
```
