```rust
// BEFORE: bespoke one-liner
fn push_tool_outputs(tools: &[ToolOutput], lines: &mut Vec<Line>) {
    for tool in tools {
        lines.push(format!("  {} {}: {}", "⚙", tool.name, truncate(&tool.preview, 60)));
    }
}

// AFTER: use ToolCallBlock
fn push_tool_outputs(tools: &[ToolOutput], lines: &mut Vec<Line>, theme: &Theme) {
    for tool in tools {
        let block = ToolCallBlock {
            name: tool.name.clone(),
            input_summary: tool.input_summary(),  // type-specific summary
            input: tool.input.clone(),
            result: tool.result.clone(),
            duration_s: tool.duration_s,
            success: tool.success,
            expanded: false,  // collapsed by default in inline view
        };
        lines.extend(block.to_lines(theme));
    }
}
```
