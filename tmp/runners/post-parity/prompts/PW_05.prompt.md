### Step 1: Add c_factor handler

The C-factor is computed by `CFactorSummary` in `orchestrate.rs`. Extract the computation logic or read from `.roko/learn/efficiency.jsonl`:

```rust
async fn c_factor(
    State(state): State<Arc<AppState>>,
) -> Result<impl IntoResponse, ApiError> {
    let learn_root = state.workdir.join(".roko/learn");
    let episodes = read_episodes(&state.workdir)?;
    let episode_count = episodes.len();
    // Compute sub-metrics from episodes
    let gate_pass_rate = compute_gate_pass_rate(&episodes);
    let cost_efficiency = compute_cost_efficiency(&episodes);
    let speed = compute_speed_metric(&episodes);
    let reuse_rate = compute_reuse_rate(&episodes);
    let overall = (gate_pass_rate + cost_efficiency + speed + reuse_rate) / 4.0;

    Ok(Json(json!({
        "composite": { "overall": overall, "episode_count": episode_count },
        "sub_metrics": {
            "gate_pass_rate": gate_pass_rate,
            "cost_efficiency": cost_efficiency,
            "speed": speed,
            "reuse_rate": reuse_rate,
        }
    })))
}
```

### Step 2: Add trend handler

```rust
async fn c_factor_trend(
    State(state): State<Arc<AppState>>,
    Query(params): Query<TrendParams>,
) -> Result<impl IntoResponse, ApiError> {
    let window = parse_window(&params.window.unwrap_or_else(|| "24h".to_string()));
    // Bucket episodes by time window, compute c-factor per bucket
    let episodes = read_episodes(&state.workdir)?;
    let trend = bucket_and_compute(&episodes, window);
    Ok(Json(json!({ "trend": trend })))
}
```

### Step 3: Register routes

```rust
.route("/metrics/c_factor", get(c_factor))
.route("/c-factor/trend", get(c_factor_trend))
```
