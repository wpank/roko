# RF_01: Create performance benchmark harness

## Task
Create a benchmark harness using `criterion` to measure roko CLI runtime overhead: config loading, prompt assembly, dispatch setup.

## Runner Context
Runner RF, batch 1 of 11. No dependencies.

## Problem
There is no systematic way to measure performance regressions. Config loading, prompt assembly, and dispatch setup are on the critical path for every run. Benchmarks catch overhead creep before it becomes noticeable.

## Current Code

**Cargo.toml** -- `crates/roko-cli/Cargo.toml`:
Does not have a `[[bench]]` section or criterion dependency.

## Exact Changes

### Step 1: Add criterion dev-dependency

In `crates/roko-cli/Cargo.toml`, add:

```toml
[dev-dependencies]
criterion = { version = "0.5", features = ["html_reports"] }

[[bench]]
name = "runtime_overhead"
harness = false
```

### Step 2: Create `crates/roko-cli/benches/runtime_overhead.rs`

```rust
//! Benchmarks for roko CLI runtime overhead.

use criterion::{black_box, criterion_group, criterion_main, Criterion};
use std::path::PathBuf;

fn bench_config_load(c: &mut Criterion) {
    // Create a minimal config for benchmarking
    let dir = tempfile::tempdir().unwrap();
    let roko_dir = dir.path().join(".roko");
    std::fs::create_dir_all(&roko_dir).unwrap();

    // Write minimal roko.toml
    let config_path = dir.path().join("roko.toml");
    std::fs::write(&config_path, r#"
[agent]
model = "claude-sonnet-4-20250514"
effort = "standard"
"#).unwrap();

    c.bench_function("config_load", |b| {
        b.iter(|| {
            let config = roko_core::config::schema::RokoConfig::load_from_path(
                black_box(&config_path)
            );
            black_box(config)
        });
    });
}

fn bench_prompt_assembly(c: &mut Criterion) {
    use roko_compose::prompt_assembly_service::PromptAssemblyService;

    let service = PromptAssemblyService::new();

    c.bench_function("prompt_assembly_empty", |b| {
        b.iter(|| {
            let _ = black_box(&service);
            // Measure service creation overhead
            let svc = PromptAssemblyService::new();
            black_box(svc)
        });
    });
}

fn bench_token_counting(c: &mut Criterion) {
    use roko_compose::token_counter::count_tokens;

    let short_text = "Hello, world!";
    let medium_text = "fn main() { println!(\"hello\"); }\n".repeat(100);
    let long_text = "fn main() { println!(\"hello\"); }\n".repeat(1000);

    c.bench_function("token_count_short", |b| {
        b.iter(|| count_tokens(black_box(short_text)))
    });

    c.bench_function("token_count_medium", |b| {
        b.iter(|| count_tokens(black_box(&medium_text)))
    });

    c.bench_function("token_count_long", |b| {
        b.iter(|| count_tokens(black_box(&long_text)))
    });
}

criterion_group!(
    benches,
    bench_config_load,
    bench_prompt_assembly,
    bench_token_counting,
);
criterion_main!(benches);
```
