# RF_09: Create WarmDispatchPool with pre-spawn lifecycle

## Task
Create a `WarmDispatchPool` in `roko-runtime` that manages pre-spawned model call workers for fast dispatch.

## Runner Context
Runner RF, batch 9 of 11. No dependencies.

## Problem
The existing `WarmPool` in `roko-cli/src/dispatch/warm_pool.rs` is a typed LRU container for agent handles but does not pre-spawn real agents. A `WarmDispatchPool` in `roko-runtime` should manage pre-spawned workers at the model call level (below the agent abstraction) to reduce first-call latency.

### Step 1: Create `crates/roko-runtime/src/warm_dispatch_pool.rs`

```rust
//! WarmDispatchPool — pre-spawned workers for fast model call dispatch.
//!
//! Maintains a pool of ready-to-use workers keyed by model slug.
//! Workers are pre-allocated (but not pre-connected) so the dispatch
//! path can skip cold-start overhead.

use std::collections::HashMap;
use std::sync::{Arc, Mutex};
use std::time::{Duration, Instant};

/// Configuration for the warm dispatch pool.
#[derive(Clone, Debug)]
pub struct WarmPoolConfig {
    /// Maximum workers per model.
    pub max_per_model: usize,
    /// TTL for idle workers.
    pub idle_ttl: Duration,
    /// Models to pre-warm.
    pub pre_warm_models: Vec<String>,
}

impl Default for WarmPoolConfig {
    fn default() -> Self {
        Self {
            max_per_model: 2,
            idle_ttl: Duration::from_secs(300), // 5 min
            pre_warm_models: Vec::new(),
        }
    }
}

/// A single warm worker slot.
#[derive(Debug, Clone)]
pub struct WarmWorker {
    /// Stable identifier.
    pub id: String,
    /// Model this worker is configured for.
    pub model: String,
    /// When the worker was created.
    pub created_at: Instant,
    /// When the worker was last used.
    pub last_used: Instant,
    /// TTL for this worker.
    pub ttl: Duration,
    /// Whether the worker is currently in use.
    pub in_use: bool,
}

impl WarmWorker {
    /// Whether this worker has expired.
    pub fn is_expired(&self) -> bool {
        self.last_used.elapsed() > self.ttl
    }
}

/// Statistics for the warm pool.
#[derive(Debug, Clone, Default)]
pub struct WarmPoolStats {
    /// Total workers currently in the pool.
    pub total_workers: usize,
    /// Workers currently idle (available for dispatch).
    pub idle_workers: usize,
    /// Workers currently in use.
    pub active_workers: usize,
    /// Total workers evicted due to TTL.
    pub evicted: u64,
    /// Total cache hits (worker reused).
    pub hits: u64,
    /// Total cache misses (cold start required).
    pub misses: u64,
}

/// Pool of pre-spawned dispatch workers.
pub struct WarmDispatchPool {
    config: WarmPoolConfig,
    workers: Mutex<HashMap<String, Vec<WarmWorker>>>,
    stats: Mutex<WarmPoolStats>,
    next_id: std::sync::atomic::AtomicU64,
}

impl WarmDispatchPool {
    /// Create a new pool with the given configuration.
    pub fn new(config: WarmPoolConfig) -> Self {
        Self {
            config,
            workers: Mutex::new(HashMap::new()),
            stats: Mutex::new(WarmPoolStats::default()),
            next_id: std::sync::atomic::AtomicU64::new(0),
        }
    }

    /// Create a pool with default configuration.
    pub fn default_pool() -> Self {
        Self::new(WarmPoolConfig::default())
    }

    /// Try to take an idle worker for the given model.
    ///
    /// Returns `Some(worker)` if one is available, `None` if a cold start is needed.
    pub fn take(&self, model: &str) -> Option<WarmWorker> {
        let mut workers = self.workers.lock().unwrap_or_else(|e| e.into_inner());
        let mut stats = self.stats.lock().unwrap_or_else(|e| e.into_inner());

        if let Some(pool) = workers.get_mut(model) {
            // Evict expired workers
            pool.retain(|w| !w.is_expired());

            // Find first idle worker
            if let Some(pos) = pool.iter().position(|w| !w.in_use) {
                let mut worker = pool.remove(pos);
                worker.in_use = true;
                worker.last_used = Instant::now();
                stats.hits += 1;
                stats.active_workers += 1;
                stats.idle_workers = stats.idle_workers.saturating_sub(1);
                return Some(worker);
            }
        }

        stats.misses += 1;
        None
    }

    /// Return a worker to the pool after use.
    pub fn return_worker(&self, mut worker: WarmWorker) {
        worker.in_use = false;
        worker.last_used = Instant::now();

        let mut workers = self.workers.lock().unwrap_or_else(|e| e.into_inner());
        let mut stats = self.stats.lock().unwrap_or_else(|e| e.into_inner());

        let pool = workers.entry(worker.model.clone()).or_default();
        if pool.len() < self.config.max_per_model {
            pool.push(worker);
            stats.idle_workers += 1;
        }
        stats.active_workers = stats.active_workers.saturating_sub(1);
    }

    /// Create and insert a new warm worker for a model.
    pub fn warm_up(&self, model: &str) -> WarmWorker {
        let id = format!("warm-{}", self.next_id.fetch_add(1, std::sync::atomic::Ordering::Relaxed));
        let worker = WarmWorker {
            id: id.clone(),
            model: model.to_string(),
            created_at: Instant::now(),
            last_used: Instant::now(),
            ttl: self.config.idle_ttl,
            in_use: false,
        };

        let mut workers = self.workers.lock().unwrap_or_else(|e| e.into_inner());
        let mut stats = self.stats.lock().unwrap_or_else(|e| e.into_inner());

        let pool = workers.entry(model.to_string()).or_default();
        if pool.len() < self.config.max_per_model {
            pool.push(worker.clone());
            stats.total_workers += 1;
            stats.idle_workers += 1;
        }

        worker
    }

    /// Evict all expired workers across all models.
    pub fn evict_expired(&self) -> usize {
        let mut workers = self.workers.lock().unwrap_or_else(|e| e.into_inner());
        let mut stats = self.stats.lock().unwrap_or_else(|e| e.into_inner());
        let mut evicted = 0;

        for pool in workers.values_mut() {
            let before = pool.len();
            pool.retain(|w| !w.is_expired());
            evicted += before - pool.len();
        }

        stats.evicted += evicted as u64;
        stats.total_workers = stats.total_workers.saturating_sub(evicted);
        stats.idle_workers = stats.idle_workers.saturating_sub(evicted);

        evicted
    }

    /// Get current pool statistics.
    pub fn stats(&self) -> WarmPoolStats {
        self.stats.lock().unwrap_or_else(|e| e.into_inner()).clone()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn take_and_return() {
        let pool = WarmDispatchPool::default_pool();
        pool.warm_up("claude-sonnet");

        let worker = pool.take("claude-sonnet");
        assert!(worker.is_some());
        assert_eq!(pool.stats().hits, 1);

        pool.return_worker(worker.unwrap());
        assert_eq!(pool.stats().idle_workers, 1);
    }

    #[test]
    fn miss_on_empty() {
        let pool = WarmDispatchPool::default_pool();
        let worker = pool.take("claude-sonnet");
        assert!(worker.is_none());
        assert_eq!(pool.stats().misses, 1);
    }

    #[test]
    fn evict_expired() {
        let config = WarmPoolConfig {
            idle_ttl: Duration::from_millis(1),
            ..Default::default()
        };
        let pool = WarmDispatchPool::new(config);
        pool.warm_up("claude-sonnet");
        std::thread::sleep(Duration::from_millis(5));
        let evicted = pool.evict_expired();
        assert_eq!(evicted, 1);
    }
}
```
