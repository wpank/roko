# TV_01: Create unified test harness

## Task
Create a unified test harness module that provides common setup/teardown, temporary directory management, and mock configuration for integration tests across the roko-cli crate.

//! Unified test harness for roko-cli integration tests (TV_01).
//!
//! Provides:
//! - Temporary workspace with .roko/ directory
//! - Default roko.toml configuration
//! - Mock agent backends
//! - Common assertion helpers

use std::path::{Path, PathBuf};
use std::sync::Once;

use tempfile::TempDir;

static INIT: Once = Once::new();

/// Initialize test environment (logging, etc).
pub fn init() {
    INIT.call_once(|| {
        // Set up tracing for tests
        let _ = tracing_subscriber::fmt()
            .with_test_writer()
            .with_max_level(tracing::Level::DEBUG)
            .try_init();
    });
}

/// A test workspace with a temporary directory and roko configuration.
pub struct TestWorkspace {
    /// Temporary directory (cleaned up on drop).
    pub dir: TempDir,
    /// Path to the workspace root.
    pub root: PathBuf,
    /// Path to the .roko directory.
    pub roko_dir: PathBuf,
}

impl TestWorkspace {
    /// Create a new test workspace with default configuration.
    pub fn new() -> Self {
        init();
        let dir = TempDir::new().expect("create temp dir");
        let root = dir.path().to_path_buf();
        let roko_dir = root.join(".roko");

        // Create standard directory structure
        std::fs::create_dir_all(&roko_dir).expect("create .roko");
        std::fs::create_dir_all(roko_dir.join("state")).expect("create .roko/state");
        std::fs::create_dir_all(roko_dir.join("learn")).expect("create .roko/learn");
        std::fs::create_dir_all(roko_dir.join("prd")).expect("create .roko/prd");
        std::fs::create_dir_all(roko_dir.join("contracts")).expect("create .roko/contracts");

        // Write default roko.toml
        let config = default_config();
        std::fs::write(root.join("roko.toml"), config).expect("write roko.toml");

        // Initialize git repo
        let _ = std::process::Command::new("git")
            .args(["init", "--initial-branch=main"])
            .current_dir(&root)
            .output();
        let _ = std::process::Command::new("git")
            .args(["add", "."])
            .current_dir(&root)
            .output();
        let _ = std::process::Command::new("git")
            .args([
                "-c", "user.email=test@test.com",
                "-c", "user.name=test",
                "commit", "-m", "init",
            ])
            .current_dir(&root)
            .output();

        Self { dir, root, roko_dir }
    }

    /// Create a file in the workspace.
    pub fn write_file(&self, rel_path: &str, content: &str) {
        let path = self.root.join(rel_path);
        if let Some(parent) = path.parent() {
            std::fs::create_dir_all(parent).expect("create parent dirs");
        }
        std::fs::write(&path, content).expect("write file");
    }

    /// Read a file from the workspace.
    pub fn read_file(&self, rel_path: &str) -> String {
        std::fs::read_to_string(self.root.join(rel_path)).expect("read file")
    }

    /// Check if a file exists.
    pub fn file_exists(&self, rel_path: &str) -> bool {
        self.root.join(rel_path).exists()
    }

    /// Path to the workspace root.
    pub fn path(&self) -> &Path {
        &self.root
    }
}

fn default_config() -> String {
    r#"
[agent]
default_backend = "mock"
default_model = "mock-model"
effort = "standard"

[server]
bind = "127.0.0.1"
port = 0
"#
    .to_string()
}

/// Assert that a file contains a specific string.
pub fn assert_file_contains(workspace: &TestWorkspace, path: &str, expected: &str) {
    let content = workspace.read_file(path);
    assert!(
        content.contains(expected),
        "file {path} does not contain: {expected}\nactual content:\n{content}"
    );
}

/// Assert that a directory exists.
pub fn assert_dir_exists(workspace: &TestWorkspace, path: &str) {
    assert!(
        workspace.root.join(path).is_dir(),
        "directory does not exist: {path}"
    );
}
