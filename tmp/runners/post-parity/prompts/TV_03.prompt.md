# TV_03: Add gate pipeline integration test

## Task
Add an integration test for the gate pipeline that runs compile, clippy, and test gates against a real temporary Rust project and verifies the gate report.

## Runner Context
Runner TV_A (Testing: Infrastructure), batch 3 of 3. No dependencies.

## Problem
The gate pipeline (`GateService`) has unit tests for individual gates but no integration test that runs the full pipeline end-to-end with a real project. This test would catch issues with gate ordering, skipping, and report aggregation.

## Exact Changes

### Step 1: Create `crates/roko-gate/tests/pipeline_integration.rs`

```rust
//! Integration test for the gate pipeline (TV_03).
//!
//! Creates a temporary Rust project, runs the gate pipeline, and
//! verifies the gate report.

use std::path::Path;
use std::process::Command;

use tempfile::TempDir;

use roko_gate::gate_service::GateService;
use roko_gate::payload::BuildSystem;
use roko_core::foundation::{GateConfig, GateReport, GateVerdict};

/// Create a minimal Rust project in a temp directory.
fn create_test_project() -> TempDir {
    let dir = TempDir::new().expect("create temp dir");

    // cargo init
    let output = Command::new("cargo")
        .args(["init", "--name", "test_project"])
        .current_dir(dir.path())
        .output()
        .expect("cargo init");
    assert!(output.status.success(), "cargo init failed");

    // Write a simple lib.rs
    std::fs::write(
        dir.path().join("src/lib.rs"),
        r#"
/// Add two numbers.
pub fn add(a: i32, b: i32) -> i32 {
    a + b
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn test_add() {
        assert_eq!(add(2, 3), 5);
    }
}
"#,
    )
    .expect("write lib.rs");

    dir
}

/// Create a project with a compile error.
fn create_broken_project() -> TempDir {
    let dir = create_test_project();
    std::fs::write(
        dir.path().join("src/lib.rs"),
        "pub fn broken() -> i32 { \"not an int\" }",
    )
    .expect("write broken lib.rs");
    dir
}

#[tokio::test]
async fn test_full_pipeline_passes() {
    let project = create_test_project();
    let service = GateService::new();

    let config = GateConfig {
        gates: vec![
            "compile".to_string(),
            "test".to_string(),
        ],
        shell_gates: vec![],
    };

    let report = service
        .run_gates(&config.gates, project.path(), &config.shell_gates)
        .await;

    assert!(
        report.passed(),
        "expected all gates to pass, but got: {report:?}"
    );
}

#[tokio::test]
async fn test_compile_gate_fails_on_error() {
    let project = create_broken_project();
    let service = GateService::new();

    let config = GateConfig {
        gates: vec!["compile".to_string()],
        shell_gates: vec![],
    };

    let report = service
        .run_gates(&config.gates, project.path(), &config.shell_gates)
        .await;

    assert!(
        !report.passed(),
        "expected compile gate to fail, but it passed: {report:?}"
    );
}

#[tokio::test]
async fn test_gate_ordering() {
    let project = create_test_project();
    let service = GateService::new();

    let config = GateConfig {
        gates: vec![
            "compile".to_string(),
            "clippy".to_string(),
            "test".to_string(),
        ],
        shell_gates: vec![],
    };

    let report = service
        .run_gates(&config.gates, project.path(), &config.shell_gates)
        .await;

    // Verify gates ran in order: compile (rung 0) before clippy (rung 1) before test (rung 2)
    let gate_names: Vec<&str> = report
        .results()
        .iter()
        .map(|r| r.gate.as_str())
        .collect();

    // All three should be present (may be in any order if run in parallel,
    // but the report should list them)
    assert!(gate_names.contains(&"compile"));
    assert!(gate_names.contains(&"test"));
}
```
