# UX_01: Implement roko next command

## Task
Create the `roko next` command that inspects the workspace and suggests the next action the user should take.

## Runner Context
Runner UX, batch 1 of 5. No dependencies.

## Problem
Users open a roko workspace and don't know what to do next. The `roko next` command should inspect workspace state (plans, PRDs, agents, recent runs) and suggest the most useful next action.

## Current Code

**commands/mod.rs** -- `crates/roko-cli/src/commands/mod.rs`:
Lists all command modules. Does not include `next`.

**status command** -- `crates/roko-cli/src/commands/status.rs`:
Queries signals and reports counts. Reference for workspace inspection.

**main.rs** -- `crates/roko-cli/src/main.rs`:
Registers subcommands via clap.

### Step 1: Create `crates/roko-cli/src/commands/next.rs`

```rust
//! `roko next` — suggest the next action based on workspace state.

use anyhow::{Result, Context};
use std::path::Path;

/// Inspect workspace and suggest the next action.
pub fn run_next(workdir: &Path) -> Result<()> {
    let roko_dir = workdir.join(".roko");

    if !roko_dir.exists() {
        println!("No .roko/ directory found.");
        println!();
        println!("  roko init    Initialize this workspace");
        return Ok(());
    }

    let mut suggestions = Vec::new();

    // Check for unfinished plans
    let plans_dir = workdir.join("plans");
    if plans_dir.exists() {
        let has_plans = std::fs::read_dir(&plans_dir)
            .map(|d| d.count() > 0)
            .unwrap_or(false);

        if has_plans {
            // Check for executor snapshots (incomplete runs)
            let state_dir = roko_dir.join("state");
            let has_snapshots = state_dir.exists() && std::fs::read_dir(&state_dir)
                .map(|d| d.count() > 0)
                .unwrap_or(false);

            if has_snapshots {
                suggestions.push(Suggestion {
                    priority: 1,
                    action: "roko plan run plans/ --resume .roko/state/executor.json".into(),
                    reason: "Interrupted plan run detected — resume where you left off".into(),
                });
            } else {
                suggestions.push(Suggestion {
                    priority: 3,
                    action: "roko plan run plans/".into(),
                    reason: "Plans are ready to execute".into(),
                });
            }
        }
    }

    // Check for draft PRDs
    let prd_dir = roko_dir.join("prd");
    if prd_dir.exists() {
        let drafts = count_files_matching(&prd_dir, "draft");
        let published = count_files_matching(&prd_dir, "published");

        if drafts > 0 {
            suggestions.push(Suggestion {
                priority: 2,
                action: "roko prd list".into(),
                reason: format!("{drafts} draft PRD(s) need review/promotion"),
            });
        }

        if published > 0 {
            suggestions.push(Suggestion {
                priority: 2,
                action: "roko prd plan <slug>".into(),
                reason: format!("{published} published PRD(s) ready for plan generation"),
            });
        }
    }

    // Check episodes for recent failures
    let episodes_path = roko_dir.join("episodes.jsonl");
    if episodes_path.exists() {
        let recent_failures = count_recent_failures(&episodes_path);
        if recent_failures > 0 {
            suggestions.push(Suggestion {
                priority: 2,
                action: "roko learn episodes".into(),
                reason: format!("{recent_failures} recent task failure(s) — review and adjust"),
            });
        }
    }

    // Check learning state
    let learn_dir = roko_dir.join("learn");
    if !learn_dir.exists() || dir_is_empty(&learn_dir) {
        suggestions.push(Suggestion {
            priority: 4,
            action: "roko learn all".into(),
            reason: "No learning data yet — run tasks to build up learning state".into(),
        });
    }

    // Default suggestion
    if suggestions.is_empty() {
        suggestions.push(Suggestion {
            priority: 5,
            action: "roko prd idea \"<your idea>\"".into(),
            reason: "Workspace is clean — capture a new work item".into(),
        });
    }

    // Sort by priority and display
    suggestions.sort_by_key(|s| s.priority);

    println!("Suggested next actions:");
    println!();
    for (i, s) in suggestions.iter().enumerate() {
        let marker = if i == 0 { ">>>" } else { "   " };
        println!("{marker} {}", s.action);
        println!("    {}", s.reason);
        println!();
    }

    Ok(())
}

struct Suggestion {
    priority: u8,
    action: String,
    reason: String,
}

fn count_files_matching(dir: &Path, pattern: &str) -> usize {
    std::fs::read_dir(dir)
        .map(|entries| {
            entries
                .filter_map(|e| e.ok())
                .filter(|e| e.file_name().to_string_lossy().contains(pattern))
                .count()
        })
        .unwrap_or(0)
}

fn count_recent_failures(episodes_path: &Path) -> usize {
    let content = std::fs::read_to_string(episodes_path).unwrap_or_default();
    // Count lines containing "fail" in the last 20 lines
    content
        .lines()
        .rev()
        .take(20)
        .filter(|line| line.contains("\"fail\"") || line.contains("\"Fail\""))
        .count()
}

fn dir_is_empty(dir: &Path) -> bool {
    std::fs::read_dir(dir)
        .map(|mut d| d.next().is_none())
        .unwrap_or(true)
}
```
