## Task

## Problem
Raw context packs (from `roko pack create`) can be large and noisy. A synthesis pass uses an LLM to extract the most relevant information, summarize, and compress the pack to fit within a smaller token budget.

```rust
/// Synthesize a context pack through multi-pass compression.
///
/// Reads a `.pack` file, splits into chunks, sends each to the agent
/// for summarization, then combines summaries.
pub async fn synthesize_pack(
    pack_path: &Path,
    target_tokens: usize,
    model: &str,
    workdir: &Path,
) -> Result<PathBuf> {
    let content = std::fs::read_to_string(pack_path)
        .with_context(|| format!("reading pack: {}", pack_path.display()))?;

    let chunks = split_into_chunks(&content, 8000); // ~8K tokens per chunk
    println!("Synthesizing {} chunks (target: ~{target_tokens} tokens)...", chunks.len());

    let mut summaries = Vec::new();
    for (i, chunk) in chunks.iter().enumerate() {
        println!("  Pass {}/{}: summarizing...", i + 1, chunks.len());

        let prompt = format!(
            "Summarize the following context for a software development agent. \
             Extract key facts, APIs, patterns, and decisions. \
             Be concise but preserve technical accuracy.\n\n---\n{chunk}\n---"
        );

        // Use ModelCallService for the synthesis call
        let summary = call_model_for_synthesis(&prompt, model, workdir).await?;
        summaries.push(summary);
    }

    // Combine summaries
    let mut synthesized = String::new();
    synthesized.push_str(&format!("# Synthesized Context Pack\n\n"));
    synthesized.push_str(&format!("Source: {}\n", pack_path.display()));
    synthesized.push_str(&format!("Model: {model}\n"));
    synthesized.push_str(&format!("Sections: {}\n\n", summaries.len()));

    for (i, summary) in summaries.iter().enumerate() {
        synthesized.push_str(&format!("## Section {}\n\n", i + 1));
        synthesized.push_str(summary);
        synthesized.push_str("\n\n");
    }

    // If still over budget, do a final compression pass
    let char_budget = target_tokens * 4;
    if synthesized.len() > char_budget && summaries.len() > 1 {
        println!("  Final compression pass...");
        let prompt = format!(
            "Compress the following summaries into a single concise document \
             of at most {target_tokens} tokens. Preserve the most important \
             technical details.\n\n---\n{synthesized}\n---"
        );
        synthesized = call_model_for_synthesis(&prompt, model, workdir).await?;
    }

    // Write output
    let output_path = pack_path.with_extension("synthesized.pack");
    std::fs::write(&output_path, &synthesized)?;

    let approx_tokens = synthesized.len() / 4;
    println!("Synthesized pack: {}", output_path.display());
    println!("  Est tokens: ~{approx_tokens} (target: {target_tokens})");

    Ok(output_path)
}

fn split_into_chunks(content: &str, max_chars_per_chunk: usize) -> Vec<String> {
    let mut chunks = Vec::new();
    let mut current = String::new();

    for line in content.lines() {
        if current.len() + line.len() + 1 > max_chars_per_chunk && !current.is_empty() {
            chunks.push(std::mem::take(&mut current));
        }
        if !current.is_empty() {
            current.push('\n');
        }
        current.push_str(line);
    }

    if !current.is_empty() {
        chunks.push(current);
    }

    chunks
}

async fn call_model_for_synthesis(
    prompt: &str,
    model: &str,
    workdir: &Path,
) -> Result<String> {
    // Use the Claude CLI for synthesis (simplest integration)
    let output = tokio::process::Command::new("claude")
        .args(["--model", model, "--print", "--no-input", "-p", prompt])
        .current_dir(workdir)
        .output()
        .await
        .context("failed to run claude CLI for synthesis")?;

    if !output.status.success() {
        let stderr = String::from_utf8_lossy(&output.stderr);
        bail!("synthesis call failed: {stderr}");
    }

    let text = String::from_utf8_lossy(&output.stdout).to_string();
    Ok(text)
}
```
