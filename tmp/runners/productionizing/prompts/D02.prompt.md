# D02: Add `SemanticCache` (HDC fingerprint Hamming similarity) alongside `ResponseCache`

## Goal

Add a *semantic* response cache that finds near-duplicate prompts via HDC fingerprint Hamming similarity. Sits alongside the existing exact-match `ResponseCache`. Lookup order: exact → semantic → dispatch.

In-memory only. Bounded at 10k entries. 1h TTL. Skip prompts <100 tokens. Skip tool-calling responses.

## Step 1: extend `cache.rs`

Open `crates/roko-agent/src/cache.rs`. Append, near `ResponseCache`:

```rust
use serde::{Deserialize, Serialize};
use std::collections::VecDeque;
use std::sync::RwLock;
use std::time::{Duration, Instant};

/// Configuration for [`SemanticCache`].
#[derive(Debug, Clone, Copy, Serialize, Deserialize)]
pub struct SemanticConfig {
    pub enabled: bool,
    /// Hamming similarity threshold in `[0.0, 1.0]`. Hits require sim >= threshold.
    pub threshold: f64,
    pub max_entries: usize,
    pub ttl: Duration,
    /// Skip caching when prompt token count (estimated) is below this.
    pub min_prompt_tokens: u64,
}

impl Default for SemanticConfig {
    fn default() -> Self {
        Self {
            enabled: false, // opt-in via config
            threshold: 0.92,
            max_entries: 10_000,
            ttl: Duration::from_secs(3600),
            min_prompt_tokens: 100,
        }
    }
}

#[derive(Debug, Clone)]
struct SemanticEntry {
    fingerprint: Vec<u64>,
    response_bytes: Vec<u8>,
    inserted_at: Instant,
}

/// Memory-only semantic-similarity response cache.
///
/// Lookup is O(N) over `max_entries`. Entries are evicted by FIFO when the
/// cap is reached and lazily by TTL on lookup.
pub struct SemanticCache {
    cfg: SemanticConfig,
    entries: RwLock<VecDeque<SemanticEntry>>,
    hits: std::sync::atomic::AtomicU64,
    misses: std::sync::atomic::AtomicU64,
}

impl SemanticCache {
    pub fn new(cfg: SemanticConfig) -> Self {
        Self {
            cfg,
            entries: RwLock::new(VecDeque::new()),
            hits: 0.into(),
            misses: 0.into(),
        }
    }

    pub fn config(&self) -> &SemanticConfig {
        &self.cfg
    }

    /// Look up a response. Returns `Some((bytes, similarity))` on hit.
    pub fn lookup(&self, fingerprint: &[u64]) -> Option<(Vec<u8>, f64)> {
        if !self.cfg.enabled {
            return None;
        }
        let now = Instant::now();
        let entries = self.entries.read().ok()?;
        let mut best: Option<(usize, f64)> = None;
        for (idx, e) in entries.iter().enumerate() {
            if now.duration_since(e.inserted_at) > self.cfg.ttl {
                continue;
            }
            let sim = hamming_similarity(&e.fingerprint, fingerprint);
            if sim >= self.cfg.threshold {
                match best {
                    Some((_, prev)) if sim <= prev => {}
                    _ => best = Some((idx, sim)),
                }
            }
        }
        match best {
            Some((idx, sim)) => {
                self.hits.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
                Some((entries[idx].response_bytes.clone(), sim))
            }
            None => {
                self.misses.fetch_add(1, std::sync::atomic::Ordering::Relaxed);
                None
            }
        }
    }

    /// Insert a response.
    pub fn insert(&self, fingerprint: Vec<u64>, response_bytes: Vec<u8>) {
        if !self.cfg.enabled {
            return;
        }
        let mut entries = match self.entries.write() {
            Ok(g) => g,
            Err(p) => p.into_inner(),
        };
        if entries.len() >= self.cfg.max_entries {
            entries.pop_front();
        }
        entries.push_back(SemanticEntry {
            fingerprint,
            response_bytes,
            inserted_at: Instant::now(),
        });
    }

    pub fn stats(&self) -> (u64, u64) {
        (
            self.hits.load(std::sync::atomic::Ordering::Relaxed),
            self.misses.load(std::sync::atomic::Ordering::Relaxed),
        )
    }
}

fn hamming_similarity(a: &[u64], b: &[u64]) -> f64 {
    let n = a.len().min(b.len());
    if n == 0 {
        return 0.0;
    }
    let total_bits = (n as u64) * 64;
    let differing: u32 = a.iter().zip(b.iter()).take(n).map(|(x, y)| (x ^ y).count_ones()).sum();
    1.0 - (differing as f64 / total_bits as f64)
}

#[cfg(test)]
mod semantic_tests {
    use super::*;

    fn fp(seed: u64) -> Vec<u64> {
        (0..8).map(|i| seed.wrapping_mul(i + 1)).collect()
    }

    #[test]
    fn disabled_returns_none() {
        let mut cfg = SemanticConfig::default();
        cfg.enabled = false;
        let c = SemanticCache::new(cfg);
        c.insert(fp(1), b"resp".to_vec());
        assert!(c.lookup(&fp(1)).is_none());
    }

    #[test]
    fn identical_fingerprint_hits() {
        let mut cfg = SemanticConfig::default();
        cfg.enabled = true;
        let c = SemanticCache::new(cfg);
        c.insert(fp(42), b"resp".to_vec());
        let (bytes, sim) = c.lookup(&fp(42)).unwrap();
        assert_eq!(bytes, b"resp");
        assert!(sim >= 0.92);
    }

    #[test]
    fn evicts_oldest_on_overflow() {
        let mut cfg = SemanticConfig::default();
        cfg.enabled = true;
        cfg.max_entries = 2;
        let c = SemanticCache::new(cfg);
        c.insert(fp(1), b"a".to_vec());
        c.insert(fp(2), b"b".to_vec());
        c.insert(fp(3), b"c".to_vec());
        let entries = c.entries.read().unwrap();
        assert_eq!(entries.len(), 2);
    }
}
```
