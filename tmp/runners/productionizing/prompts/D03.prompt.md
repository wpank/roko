## Step 1: define `PlanCostSummary`

In `crates/roko-learn/src/costs_db.rs`:

```rust
#[derive(Debug, Clone, Default, serde::Serialize)]
pub struct PlanCostSummary {
    pub plan_id: String,
    pub total_cost_usd: f64,
    pub total_input_tokens: u64,
    pub total_output_tokens: u64,
    pub total_cached_tokens: u64,
    pub task_count: u64,
    pub successful_task_count: u64,
    pub passing_gate_count: u64,
    pub per_task_avg_usd: f64,
    pub per_passing_gate_usd: f64,
    /// Sum of diff line counts across gate artifacts, when available.
    pub total_sloc: Option<u64>,
    pub per_sloc_usd: Option<f64>,
}
```

Add the implementation:

```rust
impl CostsDb {
    pub fn plan_cost_summary(&self, plan_id: &str) -> PlanCostSummary {
        let records = self.query_by_plan(plan_id);
        let mut summary = PlanCostSummary {
            plan_id: plan_id.to_string(),
            ..Default::default()
        };
        let mut tasks_seen: std::collections::HashSet<&str> = std::collections::HashSet::new();
        let mut tasks_succeeded: std::collections::HashSet<&str> = std::collections::HashSet::new();
        for r in &records {
            summary.total_cost_usd     += r.cost_usd;
            summary.total_input_tokens += r.input_tokens.unwrap_or(0);
            summary.total_output_tokens+= r.output_tokens.unwrap_or(0);
            summary.total_cached_tokens+= r.cached_tokens.unwrap_or(0);
            if let Some(task) = r.task_id.as_deref() {
                tasks_seen.insert(task);
                if r.passed.unwrap_or(false) {
                    tasks_succeeded.insert(task);
                }
            }
            if r.gate_passed.unwrap_or(false) {
                summary.passing_gate_count += 1;
            }
        }
        summary.task_count = tasks_seen.len() as u64;
        summary.successful_task_count = tasks_succeeded.len() as u64;

        if summary.task_count > 0 {
            summary.per_task_avg_usd = summary.total_cost_usd / summary.task_count as f64;
        }
        if summary.passing_gate_count > 0 {
            summary.per_passing_gate_usd = summary.total_cost_usd / summary.passing_gate_count as f64;
        }
        // SLOC: sum diff_lines from gate artifacts when present.
        let total_sloc: u64 = records
            .iter()
            .filter_map(|r| r.gate_diff_lines)
            .sum();
        if total_sloc > 0 {
            summary.total_sloc = Some(total_sloc);
            summary.per_sloc_usd = Some(summary.total_cost_usd / total_sloc as f64);
        }
        summary
    }
}
```

Adjust field names to match the actual `CostRecord` struct. If a field doesn't exist (e.g. `gate_diff_lines`), drop the SLOC computation and leave the field as `None`. Don't invent fields.

```rust
pub struct CostsCmd {
    /// Show breakdown for one plan (overrides the global summary).

    /// Output as JSON instead of a human-readable table.
    #[arg(long)]
    pub json: bool,
}
```

Dispatch:

```rust
LearnCmd::Costs(cmd) => {
    let workdir = std::env::current_dir()?;
    let costs_db = roko_learn::costs_db::CostsDb::open_in_workdir(&workdir)?;
    if let Some(plan_id) = cmd.plan.as_deref() {
        let s = costs_db.plan_cost_summary(plan_id);
        if cmd.json {
            println!("{}", serde_json::to_string_pretty(&s)?);
        } else {
            print_plan_summary(&s);
        }
    } else {
        // existing global behavior — keep
    }
    Ok(())
}
```

`open_in_workdir` may not exist; copy whatever `CostsDb::open_default` / `CostsDb::at` constructor pattern the rest of the codebase uses (search `let costs = CostsDb::` in `roko-learn` and `roko-cli`).

`print_plan_summary` — a one-screen formatter. Suggested:

```rust
fn print_plan_summary(s: &roko_learn::costs_db::PlanCostSummary) {
    println!("Plan {}", s.plan_id);
    println!("  Tasks:           {} ({} successful)", s.task_count, s.successful_task_count);
    println!("  Total cost:      ${:.4}", s.total_cost_usd);
    println!("  Input tokens:    {}", s.total_input_tokens);
    println!("  Output tokens:   {}", s.total_output_tokens);
    println!("  Cached tokens:   {}", s.total_cached_tokens);
    println!("  Avg per task:    ${:.4}", s.per_task_avg_usd);
    if s.passing_gate_count > 0 {
        println!("  Per passing gate:${:.4}", s.per_passing_gate_usd);
    }
    if let Some(sloc) = s.total_sloc {
        println!("  SLOC:            {} ({}: ${:.6}/SLOC)", sloc, "diff lines", s.per_sloc_usd.unwrap_or(0.0));
    }
}
```
