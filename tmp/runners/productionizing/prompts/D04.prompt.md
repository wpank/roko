# D04: Add `BenchHistory` regression detection

## Goal

After every bench run, append the result to a JSONL history at `.roko/learn/bench-history.jsonl` and compare against the trailing-5-run average. Flag a regression when:

- pass_rate drops > 10%
- avg_cost_usd increases > 20%
- avg_latency_ms increases > 30%

Detection is post-run, advisory, log-only. No external alerts.

## Step 1: write `crates/roko-learn/src/bench_history.rs`

```rust
//! Append-only bench-history ledger with simple regression detection.
//!
//! Each line is a [`BenchRunRecord`]. Regression checks compare the most
//! recent run against the trailing-5-run average. Detection is advisory
//! and post-run; it does not interact with the bench scheduler.

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};
use std::path::{Path, PathBuf};
use tokio::fs::OpenOptions;
use tokio::io::{AsyncBufReadExt, AsyncWriteExt, BufReader};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct BenchRunRecord {
    pub run_id: String,
    pub suite_name: String,
    pub timestamp: DateTime<Utc>,
    pub task_count: u64,
    pub passed_count: u64,
    pub failed_count: u64,
    pub avg_cost_usd: f64,
    pub avg_latency_ms: u64,
}

impl BenchRunRecord {
    pub fn pass_rate(&self) -> f64 {
        let n = self.task_count;
        if n == 0 {
            0.0
        } else {
            self.passed_count as f64 / n as f64
        }
    }
}

#[derive(Debug, Clone)]
pub struct BenchHistory {
    path: PathBuf,
}

impl BenchHistory {
    pub fn new(path: impl Into<PathBuf>) -> Self {
        Self { path: path.into() }
    }

    pub async fn append(&self, record: &BenchRunRecord) -> std::io::Result<()> {
        if let Some(parent) = self.path.parent() {
            tokio::fs::create_dir_all(parent).await?;
        }
        let mut bytes = serde_json::to_vec(record).map_err(std::io::Error::other)?;
        bytes.push(b'\n');
        let mut file = OpenOptions::new()
            .create(true)
            .append(true)
            .open(&self.path)
            .await?;
        file.write_all(&bytes).await?;
        file.flush().await?;
        Ok(())
    }

    /// Read the trailing N records for the same suite.
    pub async fn trailing(&self, suite: &str, n: usize) -> std::io::Result<Vec<BenchRunRecord>> {
        let bytes = tokio::fs::read(&self.path).await.unwrap_or_default();
        let mut lines: Vec<BenchRunRecord> = bytes
            .split(|b| *b == b'\n')
            .filter_map(|line| serde_json::from_slice::<BenchRunRecord>(line).ok())
            .filter(|r| r.suite_name == suite)
            .collect();
        let take = lines.len().saturating_sub(n);
        lines.drain(..take);
        Ok(lines)
    }

    /// Detect regression of `current` against the trailing-N average for the
    /// same suite. Returns the report whether or not a regression is detected.
    pub async fn check(
        &self,
        current: &BenchRunRecord,
        trailing_n: usize,
    ) -> std::io::Result<RegressionReport> {
        let history = self.trailing(&current.suite_name, trailing_n).await?;
        if history.is_empty() {
            return Ok(RegressionReport::no_baseline(current));
        }
        let avg_pass_rate = history.iter().map(|r| r.pass_rate()).sum::<f64>() / history.len() as f64;
        let avg_cost_usd  = history.iter().map(|r| r.avg_cost_usd).sum::<f64>()  / history.len() as f64;
        let avg_latency   = history.iter().map(|r| r.avg_latency_ms as f64).sum::<f64>() / history.len() as f64;

        let pass_drop = avg_pass_rate - current.pass_rate();
        let cost_growth = if avg_cost_usd > 0.0 {
            (current.avg_cost_usd - avg_cost_usd) / avg_cost_usd
        } else {
            0.0
        };
        let lat_growth = if avg_latency > 0.0 {
            (current.avg_latency_ms as f64 - avg_latency) / avg_latency
        } else {
            0.0
        };

        Ok(RegressionReport {
            run_id: current.run_id.clone(),
            suite_name: current.suite_name.clone(),
            pass_rate_baseline: avg_pass_rate,
            pass_rate_current: current.pass_rate(),
            pass_rate_drop: pass_drop,
            cost_baseline_usd: avg_cost_usd,
            cost_current_usd: current.avg_cost_usd,
            cost_growth,
            latency_baseline_ms: avg_latency as u64,
            latency_current_ms: current.avg_latency_ms,
            latency_growth: lat_growth,
            regression_detected: pass_drop > 0.10 || cost_growth > 0.20 || lat_growth > 0.30,
        })
    }
}

#[derive(Debug, Clone, Serialize)]
pub struct RegressionReport {
    pub run_id: String,
    pub suite_name: String,
    pub pass_rate_baseline: f64,
    pub pass_rate_current: f64,
    pub pass_rate_drop: f64,
    pub cost_baseline_usd: f64,
    pub cost_current_usd: f64,
    pub cost_growth: f64,
    pub latency_baseline_ms: u64,
    pub latency_current_ms: u64,
    pub latency_growth: f64,
    pub regression_detected: bool,
}

impl RegressionReport {
    fn no_baseline(current: &BenchRunRecord) -> Self {
        Self {
            run_id: current.run_id.clone(),
            suite_name: current.suite_name.clone(),
            pass_rate_baseline: 0.0,
            pass_rate_current: current.pass_rate(),
            pass_rate_drop: 0.0,
            cost_baseline_usd: 0.0,
            cost_current_usd: current.avg_cost_usd,
            cost_growth: 0.0,
            latency_baseline_ms: 0,
            latency_current_ms: current.avg_latency_ms,
            latency_growth: 0.0,
            regression_detected: false,
        }
    }
}

#[cfg(test)]
mod tests {
    use super::*;
    use tempfile::tempdir;

    fn rec(id: &str, suite: &str, pass: u64, n: u64, cost: f64, lat: u64) -> BenchRunRecord {
        BenchRunRecord {
            run_id: id.into(),
            suite_name: suite.into(),
            timestamp: Utc::now(),
            task_count: n,
            passed_count: pass,
            failed_count: n - pass,
            avg_cost_usd: cost,
            avg_latency_ms: lat,
        }
    }

    #[tokio::test]
    async fn detects_pass_rate_regression() {
        let dir = tempdir().unwrap();
        let h = BenchHistory::new(dir.path().join("hist.jsonl"));
        for i in 0..5 {
            h.append(&rec(&format!("r{i}"), "s1", 90, 100, 0.10, 1000)).await.unwrap();
        }
        let cur = rec("r5", "s1", 70, 100, 0.10, 1000); // 70% vs 90% trailing
        let report = h.check(&cur, 5).await.unwrap();
        assert!(report.regression_detected);
        assert!(report.pass_rate_drop > 0.10);
    }

    #[tokio::test]
    async fn no_regression_with_no_baseline() {
        let dir = tempdir().unwrap();
        let h = BenchHistory::new(dir.path().join("hist.jsonl"));
        let cur = rec("first", "s1", 10, 10, 0.05, 500);
        let report = h.check(&cur, 5).await.unwrap();
        assert!(!report.regression_detected);
        assert_eq!(report.pass_rate_baseline, 0.0);
    }
}
```

Open `crates/roko-serve/src/bench.rs`. Find the function that finalizes a bench run (search `BenchRunCompleted`, `finalize`, `summarize`). Aggregate the record:

```rust
let summary = roko_learn::bench_history::BenchRunRecord {
    run_id: run.id.clone(),
    suite_name: run.suite_name.clone(),
    timestamp: chrono::Utc::now(),
    task_count: run.task_count,
    passed_count: run.passed_count,
    failed_count: run.failed_count,
    avg_cost_usd: run.total_cost_usd / (run.task_count.max(1) as f64),
    avg_latency_ms: run.avg_latency_ms,
};

let history = roko_learn::bench_history::BenchHistory::new(
    state.workdir.join(".roko").join("learn").join("bench-history.jsonl"),
);
if let Err(err) = history.append(&summary).await {
    tracing::warn!(error = %err, "bench-history append failed");
}
match history.check(&summary, 5).await {
    Ok(report) if report.regression_detected => {
        tracing::warn!(
            run_id = %report.run_id,
            suite = %report.suite_name,
            pass_drop = report.pass_rate_drop,
            cost_growth = report.cost_growth,
            latency_growth = report.latency_growth,
            "bench regression detected"
        );
    }
    Ok(_) => {}
    Err(err) => tracing::warn!(error = %err, "bench-history check failed"),
}
```
