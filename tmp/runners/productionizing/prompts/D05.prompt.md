Find the insert method. After insert:



## Step 3: expose `stats()` method

Add:

```rust
#[derive(Debug, Clone, Copy, serde::Serialize)]
pub struct ResponseCacheStats {
    pub hits: u64,
    pub misses: u64,
    pub evictions: u64,
    pub total_entries: u64,
    pub estimated_savings_cents: u64,
    /// Hit rate in `[0.0, 1.0]`. `0.0` when there have been no lookups.
    pub hit_rate: f64,
}

impl ResponseCache {
    pub fn stats(&self) -> ResponseCacheStats {
        let hits = self.hits.load(Ordering::Relaxed);
        let misses = self.misses.load(Ordering::Relaxed);
        let total = hits + misses;
        ResponseCacheStats {
            hits,
            misses,
            evictions: self.evictions.load(Ordering::Relaxed),
            total_entries: self.total_entries.load(Ordering::Relaxed),
            estimated_savings_cents: self.estimated_savings_cents.load(Ordering::Relaxed),
            hit_rate: if total > 0 { hits as f64 / total as f64 } else { 0.0 },
        }
    }
}
```

Re-export from `lib.rs` if `ResponseCacheStats` needs cross-crate use:

```rust
pub use cache::{ResponseCache, ResponseCacheStats};
```
