# D07: Compliance export — JSON / CSV audit trail

## Step 1: write `compliance_export.rs`

```rust
//! Structured compliance export.
//!
//! Produces an audit report for a date range, drawing from the existing
//! JSONL ledgers under `.roko/`. Output is JSON or CSV.
//!
//! IMPORTANT: this module deliberately strips raw prompts, raw model
//! responses, and any free-text fields that may contain sensitive data.
//! Only structured metadata (model, cost, tokens, gate verdict, timestamps,
//! provenance source) appears in the output.

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};
use std::path::{Path, PathBuf};

#[derive(Debug, Clone, Serialize)]
pub struct AuditReport {
    pub generated_at: DateTime<Utc>,
    pub from: DateTime<Utc>,
    pub to: DateTime<Utc>,
    pub agent_invocations: Vec<AgentInvocationRow>,
    pub gate_results: Vec<GateRow>,
    pub routing_decisions: Vec<RoutingRow>,
    pub config_changes: Vec<ConfigChangeRow>,
    pub cost_summary: CostSummaryRow,
}

#[derive(Debug, Clone, Serialize)]
pub struct AgentInvocationRow {
    pub timestamp: DateTime<Utc>,
    pub plan_id: Option<String>,
    pub task_id: Option<String>,
    pub model: String,
    pub provider: String,
    pub input_tokens: u64,
    pub output_tokens: u64,
    pub cost_usd: f64,
    pub passed: Option<bool>,
}

#[derive(Debug, Clone, Serialize)]
pub struct GateRow {
    pub timestamp: DateTime<Utc>,
    pub plan_id: Option<String>,
    pub task_id: Option<String>,
    pub gate_name: String,
    pub passed: bool,
    /// Gate "evidence kind" — e.g. `cargo_test`, `cargo_clippy` — NEVER raw output.
    pub evidence_kind: Option<String>,
}

#[derive(Debug, Clone, Serialize)]
pub struct RoutingRow {
    pub timestamp: DateTime<Utc>,
    pub plan_id: Option<String>,
    pub task_id: Option<String>,
    pub selected_model: String,
    pub stage: String, // "static" | "confidence" | "ucb"
    pub justification: Option<String>,
}

#[derive(Debug, Clone, Serialize)]
pub struct ConfigChangeRow {
    pub timestamp: DateTime<Utc>,
    pub field_path: String,
    pub source: String, // "user" | "default" | "env" | "cli" | "hot-reload"
    pub before_hash: Option<String>,
    pub after_hash: Option<String>,
}

#[derive(Debug, Clone, Default, Serialize)]
pub struct CostSummaryRow {
    pub total_usd: f64,
    pub by_provider: std::collections::BTreeMap<String, f64>,
    pub by_model: std::collections::BTreeMap<String, f64>,
}

#[derive(Debug, Clone, Copy)]
pub enum ExportFormat {
    Json,
    Csv,
}

#[derive(Debug, Clone)]
pub struct ExportSources {
    pub workdir: PathBuf,
}

impl ExportSources {
    pub fn at(workdir: impl Into<PathBuf>) -> Self {
        Self { workdir: workdir.into() }
    }

    fn episodes_path(&self) -> PathBuf {
        self.workdir.join(".roko").join("episodes.jsonl")
    }
    fn costs_path(&self) -> PathBuf {
        self.workdir.join(".roko").join("learn").join("costs.jsonl")
    }
    fn routing_log_path(&self) -> PathBuf {
        self.workdir.join(".roko").join("learn").join("routing-log.jsonl")
    }
    fn provenance_path(&self) -> PathBuf {
        self.workdir.join(".roko").join("config-provenance.jsonl")
    }
}

pub async fn export_audit_trail(
    sources: &ExportSources,
    from: DateTime<Utc>,
    to: DateTime<Utc>,
) -> std::io::Result<AuditReport> {
    let agent_invocations = read_filter_jsonl::<AgentInvocationRow>(&sources.episodes_path(), from, to, |r| r.timestamp).await?;
    let gate_results     = read_filter_jsonl::<GateRow>(&sources.episodes_path(), from, to, |r| r.timestamp).await?;
    let routing_decisions= read_filter_jsonl::<RoutingRow>(&sources.routing_log_path(), from, to, |r| r.timestamp).await?;
    let config_changes   = read_filter_jsonl::<ConfigChangeRow>(&sources.provenance_path(), from, to, |r| r.timestamp).await?;

    let mut cost = CostSummaryRow::default();
    for r in &agent_invocations {
        cost.total_usd += r.cost_usd;
        *cost.by_model.entry(r.model.clone()).or_default() += r.cost_usd;
        *cost.by_provider.entry(r.provider.clone()).or_default() += r.cost_usd;
    }

    Ok(AuditReport {
        generated_at: Utc::now(),
        from,
        to,
        agent_invocations,
        gate_results,
        routing_decisions,
        config_changes,
        cost_summary: cost,
    })
}

async fn read_filter_jsonl<T>(
    path: &Path,
    from: DateTime<Utc>,
    to: DateTime<Utc>,
    ts_of: fn(&T) -> DateTime<Utc>,
) -> std::io::Result<Vec<T>>
where
    T: for<'de> Deserialize<'de>,
{
    let bytes = match tokio::fs::read(path).await {
        Ok(b) => b,
        Err(err) if err.kind() == std::io::ErrorKind::NotFound => return Ok(Vec::new()),
        Err(err) => return Err(err),
    };
    let mut out = Vec::new();
    for line in bytes.split(|b| *b == b'\n') {
        if line.is_empty() {
            continue;
        }
        if let Ok(record) = serde_json::from_slice::<T>(line) {
            let ts = ts_of(&record);
            if ts >= from && ts <= to {
                out.push(record);
            }
        }
    }
    Ok(out)
}

pub fn write_report(report: &AuditReport, format: ExportFormat, mut writer: impl std::io::Write) -> std::io::Result<()> {
    match format {
        ExportFormat::Json => {
            serde_json::to_writer_pretty(&mut writer, report).map_err(std::io::Error::other)
        }
        ExportFormat::Csv => write_csv(report, writer),
    }
}

fn write_csv(report: &AuditReport, mut w: impl std::io::Write) -> std::io::Result<()> {
    // Single denormalized stream — agent invocations primary, others appended.
    writeln!(w, "section,timestamp,plan_id,task_id,model,provider,input_tokens,output_tokens,cost_usd,passed,gate,evidence_kind,routing_stage,routing_justification,config_field,config_source")?;
    for r in &report.agent_invocations {
        writeln!(
            w,
            "agent_invocation,{},{},{},{},{},{},{},{},{},,,,,,,",
            r.timestamp.to_rfc3339(),
            r.plan_id.as_deref().unwrap_or(""),
            r.task_id.as_deref().unwrap_or(""),
            r.model, r.provider, r.input_tokens, r.output_tokens, r.cost_usd,
            r.passed.map(|b| b.to_string()).unwrap_or_default(),
        )?;
    }
    for r in &report.gate_results {
        writeln!(
            w,
            "gate,{},{},{},,,,,,{},{},{},,,,",
            r.timestamp.to_rfc3339(),
            r.plan_id.as_deref().unwrap_or(""),
            r.task_id.as_deref().unwrap_or(""),
            r.passed,
            r.gate_name,
            r.evidence_kind.as_deref().unwrap_or(""),
        )?;
    }
    for r in &report.routing_decisions {
        writeln!(
            w,
            "routing,{},{},{},{},,,,,,,,{},{},,",
            r.timestamp.to_rfc3339(),
            r.plan_id.as_deref().unwrap_or(""),
            r.task_id.as_deref().unwrap_or(""),
            r.selected_model,
            r.stage,
            r.justification.as_deref().unwrap_or(""),
        )?;
    }
    for r in &report.config_changes {
        writeln!(
            w,
            "config_change,{},,,,,,,,,,,,{},{}",
            r.timestamp.to_rfc3339(),
            r.field_path,
            r.source,
        )?;
    }
    Ok(())
}
```

The CSV is intentionally denormalized into a single stream with a `section` column — easy to load into spreadsheets, harder to misinterpret. Don't try to produce a relational schema.

## Step 3: CLI

In `crates/roko-cli/src/commands/learn.rs`:

```rust
LearnCmd::Export(cmd) => {
    let workdir = std::env::current_dir()?;
    let sources = roko_learn::compliance_export::ExportSources::at(&workdir);
    let from = parse_dt_or_default(cmd.from.as_deref(), default_from);
    let to   = parse_dt_or_default(cmd.to.as_deref(), default_to);
    let report = roko_learn::compliance_export::export_audit_trail(&sources, from, to).await?;
    let fmt = match cmd.format.as_deref().unwrap_or("json") {
        "csv" => roko_learn::compliance_export::ExportFormat::Csv,
        _ => roko_learn::compliance_export::ExportFormat::Json,
    };
    roko_learn::compliance_export::write_report(&report, fmt, std::io::stdout())?;
    Ok(())
}
```

`ExportCmd`:

```rust
#[derive(clap::Args, Debug)]
pub struct ExportCmd {
    #[arg(long)]
    pub from: Option<String>, // RFC 3339 / ISO 8601
    #[arg(long)]
    pub to: Option<String>,
    #[arg(long, default_value = "json")]
    pub format: Option<String>, // "json" | "csv"
}
```

`parse_dt_or_default` is a small helper (or use `chrono::DateTime::parse_from_rfc3339`). Defaults: `from` = beginning of current month, `to` = `Utc::now()`.
