## Step 1: write `competitive_bench.rs`

```rust
//! Static competitive baselines for cost / token / latency comparison.
//!
//! Numbers are manually maintained from public sources. Update when sources
//! publish new figures. Each baseline carries a citation and a date.
//!
//! IMPORTANT: this is an INTERNAL measurement system. It is NOT a marketing
//! tool. If our numbers are worse than a baseline, surface that honestly.

use serde::{Deserialize, Serialize};

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct CompetitorBaseline {
    pub name: String,
    pub avg_cost_per_task_usd: f64,
    pub avg_tokens_per_task: u64,
    pub avg_latency_ms: u64,
    pub source: String,
    /// Date the source data was captured (`YYYY-MM-DD`).
    pub date: String,
    /// Free-form note (e.g. "estimated from published case study X").
    pub notes: Option<String>,
}

#[derive(Debug, Clone, Serialize)]
pub struct ComparisonRow {
    pub competitor: String,
    pub cost_ratio: f64,        // their_cost / ours; >1 means we're cheaper
    pub token_ratio: f64,
    pub latency_ratio: f64,
    pub source: String,
    pub date: String,
}

#[derive(Debug, Clone, Serialize)]
pub struct ComparisonReport {
    pub our_avg_cost_per_task_usd: f64,
    pub our_avg_tokens_per_task: u64,
    pub our_avg_latency_ms: u64,
    pub rows: Vec<ComparisonRow>,
}

/// Returns the canonical baseline set. Update this list when new numbers come in.
pub fn baselines() -> Vec<CompetitorBaseline> {
    vec![
        // LangGraph — typical agent loop without semantic caching.
        CompetitorBaseline {
            name: "LangGraph".into(),
            avg_cost_per_task_usd: 0.45,
            avg_tokens_per_task: 18_000,
            avg_latency_ms: 25_000,
            source: "https://blog.langchain.dev/langgraph-cloud-pricing/".into(),
            date: "2026-03-15".into(),
            notes: Some("estimated for a 5-node graph with default model gpt-4-turbo".into()),
        },
        // CrewAI — published case studies.
        CompetitorBaseline {
            name: "CrewAI".into(),
            avg_cost_per_task_usd: 0.38,
            avg_tokens_per_task: 14_000,
            avg_latency_ms: 22_000,
            source: "https://docs.crewai.com/benchmarks (case study)".into(),
            date: "2026-02-10".into(),
            notes: None,
        },
        // Raw API — direct LLM call without any agent framework.
        CompetitorBaseline {
            name: "Raw API (no framework)".into(),
            avg_cost_per_task_usd: 0.28,
            avg_tokens_per_task: 12_000,
            avg_latency_ms: 18_000,
            source: "Anthropic claude-sonnet-4-6 published pricing × estimated tokens".into(),
            date: "2026-04-01".into(),
            notes: Some("baseline for 'how cheap can it possibly be without retries or routing'".into()),
        },
    ]
}

pub fn compare(
    our_avg_cost_per_task_usd: f64,
    our_avg_tokens_per_task: u64,
    our_avg_latency_ms: u64,
) -> ComparisonReport {
    let rows = baselines()
        .into_iter()
        .map(|b| ComparisonRow {
            competitor: b.name,
            cost_ratio: ratio(b.avg_cost_per_task_usd, our_avg_cost_per_task_usd),
            token_ratio: ratio(b.avg_tokens_per_task as f64, our_avg_tokens_per_task as f64),
            latency_ratio: ratio(b.avg_latency_ms as f64, our_avg_latency_ms as f64),
            source: b.source,
            date: b.date,
        })
        .collect();

    ComparisonReport {
        our_avg_cost_per_task_usd,
        our_avg_tokens_per_task,
        our_avg_latency_ms,
        rows,
    }
}

fn ratio(numer: f64, denom: f64) -> f64 {
    if denom <= 0.0 {
        f64::NAN
    } else {
        numer / denom
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn baselines_have_citations() {
        for b in baselines() {
            assert!(!b.source.is_empty(), "baseline {} missing source", b.name);
            assert!(!b.date.is_empty(), "baseline {} missing date", b.name);
        }
    }

    #[test]
    fn compare_returns_one_row_per_baseline() {
        let report = compare(0.10, 5_000, 8_000);
        assert_eq!(report.rows.len(), baselines().len());
    }

    #[test]
    fn cost_ratio_greater_than_one_means_we_are_cheaper() {
        let report = compare(0.10, 5_000, 8_000);
        // baselines all have higher costs than 0.10
        for row in &report.rows {
            assert!(row.cost_ratio > 1.0, "expected {} cost_ratio > 1, got {}", row.competitor, row.cost_ratio);
        }
    }
}
```

## Step 3: extend `roko learn costs` output

Open `crates/roko-cli/src/commands/learn.rs`. The `CostsCmd` from D03 already exists. Add a flag:

```rust
#[derive(clap::Args, Debug)]
pub struct CostsCmd {
    #[arg(long)]
    pub plan: Option<String>,
    #[arg(long)]
    pub json: bool,
    /// Show comparison against published competitor baselines.
    #[arg(long)]
    pub vs_competitors: bool,
}
```

In the dispatcher, when `cmd.vs_competitors` is set:

```rust
if cmd.vs_competitors {
    let report = roko_learn::competitive_bench::compare(
        summary.per_task_avg_usd,
        avg_tokens_per_task(&summary),
        avg_latency_ms_from_costs_db(&costs_db),
    );
    if cmd.json {
        println!("{}", serde_json::to_string_pretty(&report)?);
    } else {
        print_comparison(&report);
    }
}
```

`avg_tokens_per_task` and `avg_latency_ms_from_costs_db` are small helpers — derive from existing `PlanCostSummary` (D03) and `CostsDb`. If those values aren't readily available, pass 0 and let the ratio logic produce NaN, which `print_comparison` will skip.

`print_comparison`:

```rust
fn print_comparison(r: &roko_learn::competitive_bench::ComparisonReport) {
    println!("Roko vs competitors (this plan / aggregate):");
    println!("  Our avg cost/task:    ${:.4}", r.our_avg_cost_per_task_usd);
    println!("  Our avg tokens/task:  {}",     r.our_avg_tokens_per_task);
    println!("  Our avg latency:      {}ms",   r.our_avg_latency_ms);
    println!();
    for row in &r.rows {
        println!("  {} (source: {}, {})", row.competitor, row.source, row.date);
        if !row.cost_ratio.is_nan() {
            let label = if row.cost_ratio > 1.0 { "cheaper" } else { "more expensive" };
            println!("    cost: {:.2}x {}", row.cost_ratio, label);
        }
        if !row.token_ratio.is_nan() {
            println!("    tokens: {:.2}x", row.token_ratio);
        }
        if !row.latency_ratio.is_nan() {
            println!("    latency: {:.2}x", row.latency_ratio);
        }
    }
}
```
