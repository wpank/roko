## Step 1: write `crates/roko-learn/src/adas.rs`

```rust
//! ADAS — Autocatalytic Design of AI Systems (LEARN-08).
//!
//! Observes per-(role, task_category) pass rates and proposes difficulty-model
//! deltas to feed into [`crate::curriculum::CurriculumScheduler`].
//!
//! ADAS is observation-only by default: it records proposals to the
//! `.roko/learn/adas.json` ledger; the curriculum scheduler may consult them
//! when running in [`CurriculumMode::Adas`](crate::curriculum::CurriculumMode::Adas).
//!
//! State persists via the same JSON pattern as `cascade_router::save`.

use serde::{Deserialize, Serialize};
use std::collections::BTreeMap;
use std::path::Path;

/// Adjustment proposal for a single (role, category) cell.
#[derive(Debug, Clone, Serialize, Deserialize, Default)]
pub struct DifficultyDelta {
    /// Move difficulty up (positive) or down (negative) in [-1.0, 1.0].
    pub delta: f64,
    /// Confidence in [0.0, 1.0] derived from observation count.
    pub confidence: f64,
    /// Number of observations supporting this delta.
    pub support: u64,
}

/// Per-cell observation aggregates.
#[derive(Debug, Clone, Default, Serialize, Deserialize)]
pub struct CellStats {
    pub pass_count: u64,
    pub fail_count: u64,
    pub last_difficulty: f64,
}

impl CellStats {
    pub fn pass_rate(&self) -> f64 {
        let n = self.pass_count + self.fail_count;
        if n == 0 {
            0.5
        } else {
            self.pass_count as f64 / n as f64
        }
    }
    pub fn observations(&self) -> u64 {
        self.pass_count + self.fail_count
    }
}

/// ADAS optimizer state.
#[derive(Debug, Clone, Default, Serialize, Deserialize)]
pub struct AdasOptimizer {
    cells: BTreeMap<String, CellStats>,
    /// Proposed deltas, keyed by `format!("{role}::{category}")`.
    proposals: BTreeMap<String, DifficultyDelta>,
}

impl AdasOptimizer {
    pub fn new() -> Self {
        Self::default()
    }

    fn key(role: &str, category: &str) -> String {
        format!("{role}::{category}")
    }

    /// Record a single observation. Call this after each task verdict.
    pub fn observe(&mut self, role: &str, category: &str, difficulty: f64, passed: bool) {
        let cell = self
            .cells
            .entry(Self::key(role, category))
            .or_default();
        if passed {
            cell.pass_count += 1;
        } else {
            cell.fail_count += 1;
        }
        cell.last_difficulty = difficulty;
    }

    /// Recompute proposals. Call periodically (e.g., per plan completion).
    ///
    /// Heuristic: if pass rate > 0.8 and we have ≥5 obs, push difficulty up;
    /// if pass rate < 0.4 and we have ≥5 obs, push it down. Δ scales with
    /// distance from 0.6 (the "sweet spot").
    pub fn recompute(&mut self) {
        self.proposals.clear();
        for (key, cell) in &self.cells {
            let n = cell.observations();
            if n < 5 {
                continue;
            }
            let rate = cell.pass_rate();
            let delta = if rate > 0.8 {
                ((rate - 0.6) * 0.5).clamp(0.0, 1.0)
            } else if rate < 0.4 {
                ((rate - 0.6) * 0.5).clamp(-1.0, 0.0)
            } else {
                continue;
            };
            // Confidence grows with sqrt(n), saturating at 100 obs.
            let confidence = ((n as f64).sqrt() / 10.0).min(1.0);
            self.proposals.insert(
                key.clone(),
                DifficultyDelta {
                    delta,
                    confidence,
                    support: n,
                },
            );
        }
    }

    /// Read-only access to proposals for downstream consumers (curriculum, dashboards).
    pub fn proposals(&self) -> &BTreeMap<String, DifficultyDelta> {
        &self.proposals
    }

    /// Suggested difficulty adjustment for a single cell, or `None` when no proposal.
    pub fn proposal_for(&self, role: &str, category: &str) -> Option<&DifficultyDelta> {
        self.proposals.get(&Self::key(role, category))
    }

    // ── Persistence (JSON, atomic via roko_fs::atomic). ─────────────────────

    pub fn save(&self, path: &Path) -> std::io::Result<()> {
        let bytes = serde_json::to_vec_pretty(self)
            .map_err(std::io::Error::other)?;
        roko_fs::atomic::write_atomic(path, &bytes)
    }

    pub fn load_or_default(path: &Path) -> Self {
        std::fs::read(path)
            .ok()
            .and_then(|b| serde_json::from_slice(&b).ok())
            .unwrap_or_default()
    }
}

#[cfg(test)]
mod tests {
    use super::*;

    #[test]
    fn proposes_up_after_high_pass_rate() {
        let mut a = AdasOptimizer::new();
        for _ in 0..6 {
            a.observe("implementer", "code", 0.5, true);
        }
        a.recompute();
        let p = a.proposal_for("implementer", "code").unwrap();
        assert!(p.delta > 0.0);
        assert!(p.support >= 5);
    }

    #[test]
    fn proposes_down_after_low_pass_rate() {
        let mut a = AdasOptimizer::new();
        for _ in 0..6 {
            a.observe("implementer", "code", 0.5, false);
        }
        a.recompute();
        let p = a.proposal_for("implementer", "code").unwrap();
        assert!(p.delta < 0.0);
    }

    #[test]
    fn no_proposal_with_few_observations() {
        let mut a = AdasOptimizer::new();
        a.observe("implementer", "code", 0.5, true);
        a.recompute();
        assert!(a.proposal_for("implementer", "code").is_none());
    }
}
```

```rust
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum CurriculumMode {
    EasyFirst,
    HardFirst,
    Interleaved,
    Adaptive,
    /// ADAS-augmented Adaptive: when an [`AdasOptimizer`](crate::adas::AdasOptimizer)
    /// proposal exists for the (role, category), bias next task selection by it.
    /// Falls back to Adaptive when no proposal is available.
    Adas,
}
```

Add a field on `CurriculumScheduler` to hold the optional optimizer:

```rust
pub struct CurriculumScheduler {
    mode: CurriculumMode,
    // ... existing fields ...
    adas: Option<crate::adas::AdasOptimizer>,
}
```

Add a constructor and accessors:

```rust
impl CurriculumScheduler {
    pub fn with_adas(mut self, opt: crate::adas::AdasOptimizer) -> Self {
        self.adas = Some(opt);
        self
    }

    pub fn observe_outcome(&mut self, role: &str, category: &str, difficulty: f64, passed: bool) {
        if let Some(ref mut a) = self.adas {
            a.observe(role, category, difficulty, passed);
        }
    }

    /// Recompute ADAS proposals (call periodically).
    pub fn refresh_adas(&mut self) {
        if let Some(ref mut a) = self.adas {
            a.recompute();
        }
    }
}
```
