## Step 1: write `crates/roko-learn/src/research_pipeline.rs`

```rust
//! Research pipeline — Paper → Claim → Trial → Ledger.
//!
//! A claim is an assertion derived from a paper or playbook. A trial is a
//! task (or plan) that tests the claim. The ledger records (claim, trial,
//! verdict, evidence_uris) per attempt.
//!
//! This module owns the data shapes and JSONL persistence; trial *execution*
//! is delegated to the existing plan runner. We do not invoke LLMs from here.

use chrono::{DateTime, Utc};
use serde::{Deserialize, Serialize};
use std::path::{Path, PathBuf};
use tokio::fs::OpenOptions;
use tokio::io::AsyncWriteExt;

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Claim {
    /// Stable id, e.g. SHA-256 of the claim text or a uuid.
    pub id: String,
    pub source: ClaimSource,
    pub statement: String,
    pub created_at: DateTime<Utc>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(rename_all = "snake_case", tag = "kind")]
pub enum ClaimSource {
    Paper { citation: String, doi: Option<String> },
    Playbook { playbook_id: String },
    Manual { author: String },
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct Trial {
    pub id: String,
    pub claim_id: String,
    /// Path to a plan file or task spec the runner can execute.
    pub plan_path: PathBuf,
    pub created_at: DateTime<Utc>,
}

#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct TrialResult {
    pub trial_id: String,
    pub claim_id: String,
    pub verdict: TrialVerdict,
    pub evidence_uris: Vec<String>,
    pub completed_at: DateTime<Utc>,
}

#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
#[serde(rename_all = "snake_case")]
pub enum TrialVerdict {
    Supports,
    Refutes,
    Inconclusive,
    Failed,
}

/// Append-only JSONL ledger.
///
/// Format: one `LedgerEntry` per line. Readers `BufRead::lines` and
/// `serde_json::from_str` each line — same idiom as `costs_log.rs`.
#[derive(Debug, Clone, Serialize, Deserialize)]
#[serde(tag = "type", rename_all = "snake_case")]
pub enum LedgerEntry {
    Claim(Claim),
    Trial(Trial),
    Result(TrialResult),
}

#[derive(Debug, Clone)]
pub struct ResearchLedger {
    path: PathBuf,
}

impl ResearchLedger {
    pub fn new(path: impl Into<PathBuf>) -> Self {
        Self { path: path.into() }
    }

    pub async fn append(&self, entry: &LedgerEntry) -> std::io::Result<()> {
        if let Some(parent) = self.path.parent() {
            tokio::fs::create_dir_all(parent).await?;
        }
        let mut bytes = serde_json::to_vec(entry)
            .map_err(std::io::Error::other)?;
        bytes.push(b'\n');

        let mut file = OpenOptions::new()
            .create(true)
            .append(true)
            .open(&self.path)
            .await?;
        file.write_all(&bytes).await?;
        file.flush().await?;
        Ok(())
    }

    pub fn path(&self) -> &Path {
        &self.path
    }
}

/// Generate a deterministic claim id from a statement (BLAKE3 of the text).
pub fn claim_id_for(statement: &str) -> String {
    let hash = blake3::hash(statement.as_bytes());
    format!("claim_{}", hash.to_hex())
}

/// Generate a deterministic trial id from (claim_id, plan_path, timestamp_ms).
pub fn trial_id_for(claim_id: &str, plan_path: &Path, ts_ms: i64) -> String {
    let mut hasher = blake3::Hasher::new();
    hasher.update(claim_id.as_bytes());
    hasher.update(plan_path.as_os_str().to_string_lossy().as_bytes());
    hasher.update(&ts_ms.to_le_bytes());
    let hash = hasher.finalize();
    format!("trial_{}", hash.to_hex())
}

#[cfg(test)]
mod tests {
    use super::*;
    use tempfile::tempdir;

    #[tokio::test]
    async fn append_roundtrip() {
        let dir = tempdir().unwrap();
        let path = dir.path().join("research-ledger.jsonl");
        let ledger = ResearchLedger::new(&path);

        let claim = Claim {
            id: claim_id_for("smaller models are cheaper"),
            source: ClaimSource::Manual { author: "tester".into() },
            statement: "smaller models are cheaper".into(),
            created_at: Utc::now(),
        };

        ledger.append(&LedgerEntry::Claim(claim.clone())).await.unwrap();

        let contents = tokio::fs::read_to_string(&path).await.unwrap();
        let line = contents.lines().next().unwrap();
        let parsed: LedgerEntry = serde_json::from_str(line).unwrap();
        match parsed {
            LedgerEntry::Claim(c) => assert_eq!(c.id, claim.id),
            _ => panic!("expected Claim"),
        }
    }
}
```

## Step 3: CLI subcommand

Open `crates/roko-cli/src/commands/learn.rs`. Find the existing `Learn` clap subcommand enum. Add a `Research` arm:

```rust
#[derive(clap::Subcommand, Debug)]
pub enum LearnCmd {
    // ... existing variants ...

    /// Research-claim ledger and trial dispatch.
    Research(ResearchCmd),
}

#[derive(clap::Args, Debug)]
pub struct ResearchCmd {
    #[command(subcommand)]
    pub op: ResearchOp,
}

#[derive(clap::Subcommand, Debug)]
pub enum ResearchOp {
    /// Record a claim and (optionally) queue a trial.
    Trial {
        /// Plain-text claim statement.
        #[arg(long)]
        claim: String,

        /// Path to a plan file that exercises the claim.
        #[arg(long)]
        plan: std::path::PathBuf,

        /// Citation if the claim came from a paper.
        #[arg(long)]
        citation: Option<String>,
    },
    /// Print the latest N ledger entries.
    Tail {
        #[arg(long, default_value_t = 20)]
        n: usize,
    },
}
```

In the dispatcher (search for `match cmd { LearnCmd::... }`), add:

```rust
LearnCmd::Research(rc) => {
    use roko_learn::research_pipeline::{
        Claim, ClaimSource, LedgerEntry, ResearchLedger, Trial,
        claim_id_for, trial_id_for,
    };
    let workdir = std::env::current_dir()?;
    let ledger_path = workdir.join(".roko").join("learn").join("research-ledger.jsonl");
    let ledger = ResearchLedger::new(&ledger_path);

    match rc.op {
        ResearchOp::Trial { claim, plan, citation } => {
            let now = chrono::Utc::now();
            let cid = claim_id_for(&claim);
            let claim_entry = Claim {
                id: cid.clone(),
                source: match citation {
                    Some(c) => ClaimSource::Paper { citation: c, doi: None },
                    None => ClaimSource::Manual { author: whoami::username() },
                },
                statement: claim,
                created_at: now,
            };
            ledger.append(&LedgerEntry::Claim(claim_entry)).await?;

            let trial = Trial {
                id: trial_id_for(&cid, &plan, now.timestamp_millis()),
                claim_id: cid,
                plan_path: plan,
                created_at: now,
            };
            ledger.append(&LedgerEntry::Trial(trial.clone())).await?;
            println!("recorded trial {}", trial.id);
            println!("execute it with: roko plan run {}", trial.plan_path.display());
            Ok(())
        }
        ResearchOp::Tail { n } => {
            let bytes = tokio::fs::read(&ledger_path).await.unwrap_or_default();
            let lines: Vec<&str> = std::str::from_utf8(&bytes)
                .unwrap_or("")
                .lines()
                .collect();
            let start = lines.len().saturating_sub(n);
            for line in &lines[start..] {
                println!("{line}");
            }
            Ok(())
        }
    }
}
```

If `whoami` isn't a workspace dep, replace with `std::env::var("USER").unwrap_or_else(|_| "unknown".into())`.

If the existing dispatcher doesn't return `anyhow::Result<()>`, match its real signature.

If `roko-learn` is not already a dep of `roko-cli` (it almost certainly is), no change needed.

## Step 4: trial execution is delegated

We do NOT invoke the plan runner inline. The `Trial { plan_path }` is enough — the user runs `roko plan run <path>` themselves, and a future batch can auto-dispatch. For now F02 just records.

After plan completion, the runner can append a `LedgerEntry::Result` to the same ledger path. That post-plan hook is out-of-scope for this batch (it'd require touching `orchestrate.rs`).
