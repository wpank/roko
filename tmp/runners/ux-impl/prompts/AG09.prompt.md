    assert_same_shape(&actual, &golden);

/// Recursively assert that `actual` has the same key set and value
/// types as `golden`. Values may differ.
fn assert_same_shape(actual: &Value, golden: &Value) {
    match (actual, golden) {
        (Value::Object(a), Value::Object(g)) => {
            for (k, gv) in g {
                let av = a
                    .get(k)
                    .unwrap_or_else(|| panic!("missing key {k} in {a:?}"));
                assert_same_shape(av, gv);
            }
            for k in a.keys() {
                assert!(g.contains_key(k), "unexpected key {k} in actual");
            }
        }
        (Value::Array(a), Value::Array(g)) if !g.is_empty() => {
            // Compare first element shape; arrays may have differing length.
            let template = &g[0];
            for av in a {
                assert_same_shape(av, template);
            }
        }
        (a, g) => {
            assert_eq!(
                std::mem::discriminant(a),
                std::mem::discriminant(g),
                "shape mismatch: actual={:?} golden={:?}",
                a,
                g
            );
        }
    }
}

  later one with explicit anvil setup.
