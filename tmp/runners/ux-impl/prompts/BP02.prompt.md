# BP02: Claude tests via parity_kit (golden reference)

## Goal

Rewrite the existing Claude tests to use `parity_kit` so the kit's
shape is proven by the reference backend. After this batch, the kit's
correctness is locked.

## Changes Required

1. Create `crates/roko-agent/tests/claude_parity.rs`:

```rust
mod _helpers;
use _helpers::parity_kit::*;

use roko_agent::claude_agent::ClaudeAgent;

fn build_backend(mock_url: String) -> ClaudeAgent {
    ClaudeAgent::for_test(mock_url)   // helper on ClaudeAgent if it exists
}

#[tokio::test]
async fn happy_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/v1/messages")
        .with_status(200)
        .with_body(r#"{"id":"m1","content":[{"type":"text","text":"hello back"}]}"#)
        .create_async().await;

    let backend = build_backend(server.url());
    assert_happy_path(&backend, "hello", "hello back").await;
}

#[tokio::test]
async fn streaming() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/v1/messages")
        .match_query(mockito::Matcher::UrlEncoded("stream".into(), "true".into()))
        .with_status(200)
        .with_header("content-type", "text/event-stream")
        .with_body("event: content_block_delta\ndata: {\"delta\":{\"text\":\"hello\"}}\n\nevent: content_block_delta\ndata: {\"delta\":{\"text\":\" back\"}}\n\nevent: message_stop\ndata: {}\n\n")
        .create_async().await;

    let backend = build_backend(server.url());
    assert_streaming(&backend, "hello", "hello back").await;
}

#[tokio::test]
async fn tool_call() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/v1/messages")
        .with_status(200)
        .with_body(r#"{"id":"m2","content":[{"type":"tool_use","name":"read_file","input":{"path":"x"}}]}"#)
        .create_async().await;

    let backend = build_backend(server.url());
    assert_tool_call(&backend, "read_file").await;
}

#[tokio::test]
async fn error_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/v1/messages")
        .with_status(503)
        .with_body(r#"{"error":"upstream busy"}"#)
        .create_async().await;

    let backend = build_backend(server.url());
    assert_error_path(&backend, 503).await;
}

#[tokio::test]
async fn session_continuation() {
    let mut server = mockito::Server::new_async().await;
    let _m1 = server.mock("POST", "/v1/messages")
        .with_status(200)
        .with_body(r#"{"id":"m1","content":[{"type":"text","text":"first"}]}"#)
        .expect(1)
        .create_async().await;

    // Second call must include the previous turn's content.
    let _m2 = server.mock("POST", "/v1/messages")
        .match_body(mockito::Matcher::PartialJsonString(r#"{"messages":[{"role":"assistant","content":"first"}]}"#.into()))
        .with_status(200)
        .with_body(r#"{"id":"m2","content":[{"type":"text","text":"second"}]}"#)
        .expect(1)
        .create_async().await;

    let backend = build_backend(server.url());
    assert_session_continuation(&backend, "first?", "second?").await;
}
```

2. If `ClaudeAgent::for_test(url)` doesn't exist, add it:

3. Move (or remove) older Claude tests that now duplicate the parity
