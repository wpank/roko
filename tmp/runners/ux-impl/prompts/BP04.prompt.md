   ```rust
   async fn send_turn_streaming(
       &self,
       request: ModelCallRequest,
   ) -> Result<mpsc::Receiver<String>, DispatchError> {
       let (tx, rx) = mpsc::channel(32);
       let url = format!("{}/v1/chat/completions", self.base_url);
       let body = self.build_streaming_body(request)?;
       let mut response = self.http
           .post(url)
           .bearer_auth(&self.api_key)
           .json(&body)
           .send()
           .await?
           .error_for_status()?;
       tokio::spawn(async move {
           while let Some(chunk) = response.chunk().await.ok().flatten() {
               for line in parse_sse_chunk(&chunk) {
                   if let Some(text) = extract_delta(&line) {
                       if tx.send(text).await.is_err() { break; }
                   }
               }
           }
       });
       Ok(rx)
   }
   ```

```rust
mod _helpers;
use _helpers::parity_kit::*;
use roko_agent::cursor_agent::CursorAgent;

fn build_backend(url: String) -> CursorAgent {
    CursorAgent::for_test(url)   // add this helper if absent
}

#[tokio::test]
async fn happy_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/v1/chat/completions")
        .with_status(200)
        .with_body(r#"{"choices":[{"message":{"content":"hello back"}}]}"#)
        .create_async().await;
    let backend = build_backend(server.url());
    assert_happy_path(&backend, "hello", "hello back").await;
}

#[tokio::test]
async fn streaming() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/v1/chat/completions")
        .with_status(200)
        .with_header("content-type", "text/event-stream")
        .with_body("data: {\"choices\":[{\"delta\":{\"content\":\"hello\"}}]}\n\ndata: {\"choices\":[{\"delta\":{\"content\":\" back\"}}]}\n\ndata: [DONE]\n\n")
        .create_async().await;
    let backend = build_backend(server.url());
    assert_streaming(&backend, "hi", "hello back").await;
}

#[tokio::test]
async fn error_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/v1/chat/completions")
        .with_status(429)
        .with_body(r#"{"error":"rate limited"}"#)
        .create_async().await;
    let backend = build_backend(server.url());
    assert_error_path(&backend, 429).await;
}

#[tokio::test]
#[ignore = "Cursor doesn't expose tool-call frames; skip in parity kit"]

#[tokio::test]
async fn session_continuation() {
    // similar to claude; verify second request includes prior assistant message
    let mut server = mockito::Server::new_async().await;
    let _m1 = server.mock("POST", "/v1/chat/completions")

        .with_body(r#"{"choices":[{"message":{"content":"second"}}]}"#)
        .expect(1)
        .create_async().await;
    let backend = build_backend(server.url());
    assert_session_continuation(&backend, "first?", "second?").await;
```
