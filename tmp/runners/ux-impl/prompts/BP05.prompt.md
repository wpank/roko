# BP05: Gemini parity tests

## Goal

`crates/roko-agent/tests/gemini_parity.rs` covers the canonical 5
scenarios for `GeminiAgent` (`crates/roko-agent/src/gemini/native.rs`).

## Changes Required

1. Add `GeminiAgent::for_test(url)` if absent (mirrors `ClaudeAgent::for_test`).

2. Create `crates/roko-agent/tests/gemini_parity.rs`:

```rust
mod _helpers;
use _helpers::parity_kit::*;
use roko_agent::gemini::native::GeminiAgent;

fn build_backend(url: String) -> GeminiAgent {
    GeminiAgent::for_test(url)
}

#[tokio::test]
async fn happy_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", mockito::Matcher::Regex(r"^/v1beta/models/.*:generateContent$".into()))
        .with_status(200)
        .with_body(r#"{"candidates":[{"content":{"parts":[{"text":"hello back"}]}}]}"#)
        .create_async().await;
    assert_happy_path(&build_backend(server.url()), "hi", "hello back").await;
}

#[tokio::test]
async fn streaming() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", mockito::Matcher::Regex(r"^/v1beta/models/.*:streamGenerateContent$".into()))
        .with_status(200)
        .with_header("content-type", "application/json")
        .with_body(r#"[{"candidates":[{"content":{"parts":[{"text":"hello"}]}}]},{"candidates":[{"content":{"parts":[{"text":" back"}]}}]}]"#)
        .create_async().await;
    assert_streaming(&build_backend(server.url()), "hi", "hello back").await;
}

#[tokio::test]
async fn tool_call() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", mockito::Matcher::Regex(r"^/v1beta/models/.*:generateContent$".into()))
        .with_status(200)
        .with_body(r#"{"candidates":[{"content":{"parts":[{"functionCall":{"name":"read_file","args":{"path":"x"}}}]}}]}"#)
        .create_async().await;
    assert_tool_call(&build_backend(server.url()), "read_file").await;
}

#[tokio::test]
async fn error_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", mockito::Matcher::Regex(r"^/v1beta/models/.*$".into()))
        .with_status(500)
        .with_body(r#"{"error":{"code":500}}"#)
        .create_async().await;
    assert_error_path(&build_backend(server.url()), 500).await;
}

#[tokio::test]
async fn session_continuation() {
    let mut server = mockito::Server::new_async().await;
    let _m1 = server.mock("POST", mockito::Matcher::Regex(r"^/v1beta/models/.*:generateContent$".into()))
        .with_status(200)
        .with_body(r#"{"candidates":[{"content":{"parts":[{"text":"first"}]}}]}"#)
        .expect(1)
        .create_async().await;
    let _m2 = server.mock("POST", mockito::Matcher::Regex(r"^/v1beta/models/.*:generateContent$".into()))
        .match_body(mockito::Matcher::PartialJsonString(r#"{"contents":[{"role":"model","parts":[{"text":"first"}]}]}"#.into()))
        .with_status(200)
        .with_body(r#"{"candidates":[{"content":{"parts":[{"text":"second"}]}}]}"#)
        .expect(1)
        .create_async().await;
    assert_session_continuation(&build_backend(server.url()), "first?", "second?").await;
}
```
