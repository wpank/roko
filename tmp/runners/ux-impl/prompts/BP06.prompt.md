# BP06: Perplexity parity tests

## Goal

`crates/roko-agent/tests/perplexity_parity.rs` covers the canonical 5
scenarios for the Perplexity backend.

## Notes

- Perplexity has chat (`crates/roko-agent/src/perplexity/chat.rs`),
  tool_loop, and search variants. Test against the **chat** entry
  point — it's the LlmBackend implementer.
- Perplexity exposes search rather than function/tool calls; the
  `tool_call` scenario `#[ignore]`s with reason
  `"Perplexity uses search-cited responses, not tool-call frames"`.

## Changes Required

1. Add `PerplexityChatAgent::for_test(url)` if absent.

2. Create `crates/roko-agent/tests/perplexity_parity.rs`:

```rust
mod _helpers;
use _helpers::parity_kit::*;
use roko_agent::perplexity::chat::PerplexityChatAgent;

fn build_backend(url: String) -> PerplexityChatAgent {
    PerplexityChatAgent::for_test(url)
}

#[tokio::test]
async fn happy_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/chat/completions")
        .with_status(200)
        .with_body(r#"{"choices":[{"message":{"content":"hello back"}}]}"#)
        .create_async().await;
    assert_happy_path(&build_backend(server.url()), "hi", "hello back").await;
}

#[tokio::test]
async fn streaming() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/chat/completions")
        .with_status(200)
        .with_header("content-type", "text/event-stream")
        .with_body("data: {\"choices\":[{\"delta\":{\"content\":\"hello\"}}]}\n\ndata: {\"choices\":[{\"delta\":{\"content\":\" back\"}}]}\n\ndata: [DONE]\n\n")
        .create_async().await;
    assert_streaming(&build_backend(server.url()), "hi", "hello back").await;
}

#[tokio::test]
#[ignore = "Perplexity uses search-cited responses, not tool-call frames"]
async fn tool_call() {}

#[tokio::test]
async fn error_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/chat/completions")
        .with_status(502).with_body(r#"{"error":"upstream"}"#)
        .create_async().await;
    assert_error_path(&build_backend(server.url()), 502).await;
}

#[tokio::test]
async fn session_continuation() {
    let mut server = mockito::Server::new_async().await;
    let _m1 = server.mock("POST", "/chat/completions")
        .with_status(200)
        .with_body(r#"{"choices":[{"message":{"content":"first"}}]}"#)
        .expect(1).create_async().await;
    let _m2 = server.mock("POST", "/chat/completions")
        .match_body(mockito::Matcher::PartialJsonString(r#"{"messages":[{"role":"assistant","content":"first"}]}"#.into()))
        .with_status(200)
        .with_body(r#"{"choices":[{"message":{"content":"second"}}]}"#)
        .expect(1).create_async().await;
    assert_session_continuation(&build_backend(server.url()), "first?", "second?").await;
}
```
