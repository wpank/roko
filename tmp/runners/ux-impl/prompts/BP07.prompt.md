# BP07: Ollama parity tests

## Goal

`crates/roko-agent/tests/ollama_parity.rs` covers the canonical 5
scenarios for `OllamaAgent`.

## Notes

- Ollama runs as a local daemon (`http://127.0.0.1:11434` by default).
- Tool calling is supported in recent Ollama (model-dependent); test
  against a model that supports it (e.g. `llama3.1:70b`-class models).
- Streaming is the default; non-streaming requires `stream: false`.

## Changes Required

1. Add `OllamaAgent::for_test(url)` if absent.

2. Create `crates/roko-agent/tests/ollama_parity.rs`:

```rust
mod _helpers;
use _helpers::parity_kit::*;
use roko_agent::ollama_agent::OllamaAgent;

fn build_backend(url: String) -> OllamaAgent {
    OllamaAgent::for_test(url)
}

#[tokio::test]
async fn happy_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/api/chat")
        .with_status(200)
        .with_body(r#"{"message":{"role":"assistant","content":"hello back"},"done":true}"#)
        .create_async().await;
    assert_happy_path(&build_backend(server.url()), "hi", "hello back").await;
}

#[tokio::test]
async fn streaming() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/api/chat")
        .with_status(200)
        // Ollama streaming is line-delimited JSON, not SSE
        .with_body("{\"message\":{\"role\":\"assistant\",\"content\":\"hello\"},\"done\":false}\n{\"message\":{\"role\":\"assistant\",\"content\":\" back\"},\"done\":false}\n{\"message\":{\"role\":\"assistant\",\"content\":\"\"},\"done\":true}\n")
        .create_async().await;
    assert_streaming(&build_backend(server.url()), "hi", "hello back").await;
}

#[tokio::test]
async fn tool_call() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/api/chat")
        .with_status(200)
        .with_body(r#"{"message":{"role":"assistant","content":"","tool_calls":[{"function":{"name":"read_file","arguments":{"path":"x"}}}]},"done":true}"#)
        .create_async().await;
    assert_tool_call(&build_backend(server.url()), "read_file").await;
}

#[tokio::test]
async fn error_path() {
    let mut server = mockito::Server::new_async().await;
    let _m = server.mock("POST", "/api/chat")
        .with_status(503).with_body(r#"{"error":"model loading"}"#)
        .create_async().await;
    assert_error_path(&build_backend(server.url()), 503).await;
}

#[tokio::test]
async fn session_continuation() {
    let mut server = mockito::Server::new_async().await;
    let _m1 = server.mock("POST", "/api/chat")
        .with_status(200)
        .with_body(r#"{"message":{"role":"assistant","content":"first"},"done":true}"#)
        .expect(1).create_async().await;
    let _m2 = server.mock("POST", "/api/chat")
        .match_body(mockito::Matcher::PartialJsonString(r#"{"messages":[{"role":"assistant","content":"first"}]}"#.into()))
        .with_status(200)
        .with_body(r#"{"message":{"role":"assistant","content":"second"},"done":true}"#)
        .expect(1).create_async().await;
    assert_session_continuation(&build_backend(server.url()), "first?", "second?").await;
}
```
