#[derive(Debug, serde::Serialize)]
pub struct BootstrapResult {
    pub passport_id: String,
    pub name: String,
    pub address: String,
    pub mask: u64,
    pub tx_hash: String,
}

/// Register all 5 demo passports against the configured registry.
/// Returns the list of bootstrap results in the same order as
/// `demo_passports()`.
///
/// # Errors
///
/// Returns an error if any registration tx fails to submit. Partial
/// success is allowed (the operator can re-run individual ones).
pub async fn run(
    registry_address: &str,
    rpc_url: &str,
    output_path: &std::path::Path,
) -> Result<Vec<BootstrapResult>> {
    let mut results = Vec::new();
    for p in demo_passports() {
        let card_uri = build_card_uri(&p)?;
        let wallet = build_wallet(rpc_url, &p.private_key).await?;
        let passport_id = next_passport_id(rpc_url, registry_address).await?;
        // Step 1: register passport
        let tx = TxRequest {
            to: Some(registry_address.into()),
            data: register_calldata()?,
            ..TxRequest::default()
        };
        let _ = wallet.sign_and_submit(tx).await
            .context(format!("register passport for {}", p.name))?;
        // Step 2: updateAgentCard with capabilityMask
        let tx = TxRequest {
            to: Some(registry_address.into()),
            data: update_agent_card_calldata(&passport_id, &card_uri, p.mask)?,
            ..TxRequest::default()
        };
        let hash = wallet.sign_and_submit(tx).await
            .context(format!("updateAgentCard for {}", p.name))?;
        results.push(BootstrapResult {
            passport_id,
            name: p.name.clone(),
            address: p.address.clone(),
            mask: p.mask,
            tx_hash: hash.as_str().to_string(),
        });
    }
    std::fs::write(output_path, serde_json::to_vec_pretty(&results)?)
        .context("write bootstrap result file")?;
    Ok(results)
}

fn build_card_uri(p: &DemoPassport) -> Result<String> {
    let card = json!({
        "name": p.name,
        "capabilities": p.capabilities,
        "endpoints": { "rest": format!("http://localhost:{}", 9100 + (p.mask as u16 % 100)) },
        "domain_tags": ["roko"],
        "version": "1.0",
    });
    Ok(format!(
        "data:application/json;base64,{}",
        base64::engine::general_purpose::STANDARD_NO_PAD.encode(serde_json::to_vec(&card)?)
    ))
}

- Do not generate fresh keys per run. Use the hardhat default keys

  the bootstrap is broken; the operator should run on a fresh fork.
- Do not make the bootstrap non-async. The chain calls are async.
