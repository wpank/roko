// crates/roko-cli/tests/smoke_learning_loop.rs

use std::sync::Arc;
use std::time::Duration;

use serde_json::Value;
use tempfile::TempDir;

#[tokio::test]
async fn learning_loop_smoke() {
    let tmp = TempDir::new().unwrap();
    let workdir = tmp.path();

    // 1) Initialise an empty .roko under the tempdir.
    std::env::set_current_dir(workdir).unwrap();
    let init = std::process::Command::new(env!("CARGO_BIN_EXE_roko"))
        .arg("init").arg("--workdir").arg(workdir)
        .output().unwrap();
    assert!(init.status.success(), "roko init failed: {init:?}");

    // 2) Drive a single one-shot run with a MockDispatcher.
    //    The MockDispatcher must produce an Episode + a gate verdict.
    //    Use the existing roko_std::mock::MockBackend.
    //    If the smoke needs a binary entry, use the roko CLI with an
    //    env knob pointing it at MockDispatcher; otherwise call into
    //    the library API directly.

    // Simplification: drive the library API.
    let dispatcher = build_smoke_dispatcher();   // helper using mock backend
    let outcome = roko_cli::run_for_smoke(workdir, dispatcher, "fix the bug")
        .await
        .expect("smoke run completes");
    assert!(outcome.episode_count >= 1, "no episode produced");

    // 3) Episode JSONL has at least one entry.
    let episodes = std::fs::read_to_string(workdir.join(".roko/episodes.jsonl"))
        .expect(".roko/episodes.jsonl exists");
    assert!(episodes.lines().any(|l| !l.is_empty()), "episodes empty");
    let first: Value = serde_json::from_str(episodes.lines().next().unwrap())
        .expect("episode parses");
    assert!(first["task"].is_string() || first["prompt"].is_string());

    // 4) Cascade router updated.
    let cascade = workdir.join(".roko/learn/cascade-router.json");
    if cascade.exists() {
        let raw = std::fs::read_to_string(&cascade).unwrap();
        let parsed: Value = serde_json::from_str(&raw).expect("cascade parses");
        assert!(parsed["arms"].is_array() || parsed["winner"].is_string(),
            "cascade router state shape unexpected");
    } else {
        // Cascade requires multiple backends; if absent, log but don't fail.
        eprintln!("note: cascade-router.json not produced by this smoke");
    }

    // 5) At least one gate verdict landed in the substrate.
    let verdicts_dir = workdir.join(".roko/substrate/Kind=GateVerdict");
    if verdicts_dir.exists() {
        let count = std::fs::read_dir(&verdicts_dir).unwrap().count();
        assert!(count >= 1, "no gate verdicts in substrate");
    }
}
