//! Incremental tailer for files that are a top-level JSON array.
//!
//! Scans newly-appended bytes for closing-`}\\n  {`-style record
//! boundaries and parses each complete record into `T`. Tolerates
//! files that grow by record-append (writer keeps the trailing `]`
//! shifted forward). Does not handle pretty-printed nested arrays
//! that contain `}` mid-record without quoting — fine for our use
//! case (RuntimeEvent is a flat enum-tagged object).

use std::path::PathBuf;

use super::jsonl_cursor::JsonlCursor;

pub struct JsonArrayTailer<T> {
    cursor: JsonlCursor,           // reused for byte-tracking
    items: Vec<T>,
    pending: Vec<u8>,              // partial record buffer
    pub parse_errors: usize,
}

impl<T: serde::de::DeserializeOwned> JsonArrayTailer<T> {
    pub fn new(path: impl Into<PathBuf>) -> Self {
        Self {
            cursor: JsonlCursor::new(path),
            items: Vec::new(),
            pending: Vec::new(),
            parse_errors: 0,
        }
    }

    pub fn tick(&mut self) -> std::io::Result<usize> {
        let bytes = self.cursor.read_new_bytes()?;   // (extend JsonlCursor with this method)
        if bytes.is_empty() { return Ok(0); }
        self.pending.extend_from_slice(&bytes);

        let mut added = 0;
        loop {
            // Find a complete record: starts with `{`, ends with `}` at depth 0.
            let Some((start, end)) = next_record_span(&self.pending) else { break };
            let slice = &self.pending[start..=end];
            match serde_json::from_slice::<T>(slice) {
                Ok(item) => {
                    self.items.push(item);
                    added += 1;
                }
                Err(_) => self.parse_errors += 1,
            }
            self.pending.drain(0..=end);
        }
        Ok(added)
    }

    pub fn items(&self) -> &[T] { &self.items }
    pub fn len(&self) -> usize { self.items.len() }
}

fn next_record_span(buf: &[u8]) -> Option<(usize, usize)> {
    let start = buf.iter().position(|&b| b == b'{')?;
    let mut depth = 0i32;
    let mut in_str = false;
    let mut esc = false;
    for (i, &b) in buf[start..].iter().enumerate() {
        if esc { esc = false; continue; }
        match b {
            b'\\' if in_str => { esc = true; }
            b'"' => { in_str = !in_str; }
            b'{' if !in_str => { depth += 1; }
            b'}' if !in_str => {
                depth -= 1;
                if depth == 0 { return Some((start, start + i)); }
            }
            _ => {}
        }
    }
    None
}

mod tests {

    #[derive(Debug, Deserialize)]

        let p = dir.path().join("events.json");
        fs::write(&p, r#"[ {"kind":"a"}, {"kind":"b"}, {"kind":"c"} ]"#).unwrap();
        let mut t = JsonArrayTailer::<Ev>::new(&p);
        let n = t.tick().unwrap();
        assert_eq!(n, 3);
        assert_eq!(t.items()[2].kind, "c");
    }
}

## Do NOT

- Do not switch to Option A (JSONL conversion) in this batch. That's
  a separate breaking change for `event_bus.rs`.
- Do not use a regex-based record splitter. The depth-tracking parser
  above handles strings + nesting correctly.
- Do not assume the writer holds a lock during append. Tolerate
  partially-written records (they show up in `pending` and complete
  on the next tick).
