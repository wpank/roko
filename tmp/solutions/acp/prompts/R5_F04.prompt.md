#[tokio::test]
async fn acp_dispatch_produces_telemetry() {
    // Setup: temp dir with .roko/ structure
    let tmp = tempdir().unwrap();
    let episodes_path = tmp.path().join(".roko/episodes.jsonl");
    let router_path = tmp.path().join(".roko/learn/cascade-router.json");

    // Create ACP session with mock dispatcher
    let session = create_test_session(tmp.path());

    // Dispatch a prompt (uses mock provider that returns fixed response)
    let result = session.dispatch_prompt("Fix the bug in main.rs").await;
    assert!(result.is_ok());

    // Verify episode was logged
    let episodes = std::fs::read_to_string(&episodes_path).unwrap();
    assert!(!episodes.is_empty(), "episodes.jsonl should have content");
    let episode: serde_json::Value = serde_json::from_str(episodes.lines().last().unwrap()).unwrap();
    assert_eq!(episode["entry_point"], "acp");
    assert!(episode["turns"].as_array().unwrap().len() >= 2); // user + assistant

    // Verify router observation was recorded
    let router_data = std::fs::read_to_string(&router_path).unwrap();
    let router: serde_json::Value = serde_json::from_str(&router_data).unwrap();
    assert!(router["observations"].as_array().unwrap().len() > 0);

    // Verify cost is not zero (when mock provides usage)
    assert!(result.unwrap().total_tokens.unwrap_or(0) > 0);
}
