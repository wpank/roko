### Part 2: Inline Phase Badges

4. **In `crates/roko-acp/src/runner.rs`**: At each phase transition, emit an `AgentMessageChunk` with a styled badge:

```rust
fn phase_badge(phase: &PipelinePhase, iteration: u32) -> String {
    let icon = match phase {
        PipelinePhase::Strategizing => "тКХ",
        PipelinePhase::Implementing => "тЩж",
        PipelinePhase::Gating => "тКХ",
        PipelinePhase::AutoFixing => "ЁЯФз",
        PipelinePhase::Reviewing => "ЁЯФН",
        PipelinePhase::Committing => "тЬУ",
        PipelinePhase::Complete => "тЬЕ",
        PipelinePhase::Halted { .. } => "тЫФ",
        PipelinePhase::Cancelled => "тКШ",
        PipelinePhase::Pending => return String::new(),
    };

    let name = match phase {
        PipelinePhase::Strategizing => "Strategizing",
        PipelinePhase::Implementing => "Implementing",
        PipelinePhase::Gating => "Gating",
        PipelinePhase::AutoFixing => "Auto-fixing",
        PipelinePhase::Reviewing => "Reviewing",
        PipelinePhase::Committing => "Committing",
        _ => return String::new(),
    };

    if iteration > 1 {
        format!("\n\n{icon} **{name} iter {iteration}**\n\n")
    } else {
        format!("\n\n{icon} **{name}**\n\n")
    }
}
```

5. Emit the badge as an `AgentMessageChunk` before each phase starts:
```rust
// In run_workflow_pipeline(), after each step():
let badge = phase_badge(&run.pipeline.phase, run.pipeline.iteration());
if !badge.is_empty() {
    event_sender.send(CognitiveEvent::TokenChunk(badge)).await?;
}
```
