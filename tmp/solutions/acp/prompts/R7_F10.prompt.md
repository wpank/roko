## Problem

Roko can do better: analyze the failure, trace it back through the agent's actions, cross-reference with similar past failures, classify the error type, and present a structured diagnosis. This is roko's forensic replay + episode matching + post-gate reflection working together.

```rust
async fn analyze_gate_failure(
    gate_name: &str,
    error_output: &str,
    workdir: &Path,
    event_sender: &mpsc::Sender<CognitiveEvent>,
) -> GateAutopsy {
    let mut autopsy = GateAutopsy {
        error_type: classify_error(error_output),
        root_cause: extract_root_cause(error_output),
        causal_chain: vec![],
        similar_past_failure: None,
        confidence: 0.0,
    };

    // 1. Build causal chain from git diff
    if let Ok(diff_output) = Command::new("git")
        .args(["diff", "--stat"])
        .current_dir(workdir)
        .output()
        .await
    {
        let diff = String::from_utf8_lossy(&diff_output.stdout);
        autopsy.causal_chain.push(format!("Files changed: {}", diff.trim()));
    }

    // 2. Parse error to identify the specific failure point
    match &autopsy.error_type {
        ErrorType::CompileError { file, line, message } => {
            autopsy.causal_chain.push(format!("Compile error at {}:{}", file, line));
            autopsy.causal_chain.push(format!("Error: {}", message));
        }
        ErrorType::TestFailure { test_name, expected, actual } => {
            autopsy.causal_chain.push(format!("Test {} failed", test_name));
            if let (Some(exp), Some(act)) = (expected, actual) {
                autopsy.causal_chain.push(format!("Expected: {}", exp));
                autopsy.causal_chain.push(format!("Actual: {}", act));
            }
        }
        ErrorType::ClippyWarning { lint, file } => {
            autopsy.causal_chain.push(format!("Clippy lint {} in {}", lint, file));
        }
        ErrorType::Unknown => {
            autopsy.causal_chain.push("Could not classify error".into());
        }
    }

    // 3. Cross-reference with similar past failures
    if let Ok(store) = EpisodeStore::open(workdir) {
        if let Ok(similar) = store.similar_failures(error_output, 1) {
            if let Some(past) = similar.first() {
                autopsy.similar_past_failure = Some(PastFailure {
                    episode_id: past.id.clone(),
                    summary: past.summary.clone(),
                    resolution: past.resolution.clone(),
                });
            }
        }
    }

    // 4. Determine confidence (higher if we've seen this before)
    autopsy.confidence = if autopsy.similar_past_failure.is_some() { 0.89 } else { 0.65 };

    autopsy
}

struct GateAutopsy {
    error_type: ErrorType,
    root_cause: String,
    causal_chain: Vec<String>,
    similar_past_failure: Option<PastFailure>,
    confidence: f64,
}

enum ErrorType {
    CompileError { file: String, line: u32, message: String },
    TestFailure { test_name: String, expected: Option<String>, actual: Option<String> },
    ClippyWarning { lint: String, file: String },
    Unknown,
}

struct PastFailure {
    episode_id: String,
    summary: String,
    resolution: String,
}
```

2. **Classify errors** by parsing output:

```rust
fn classify_error(output: &str) -> ErrorType {
    if output.contains("error[E") {
        // Parse rustc error: file:line:col: error[EXXXX]: message
        ErrorType::CompileError { ... }
    } else if output.contains("FAILED") || output.contains("test result: FAILED") {
        // Parse test failure
        ErrorType::TestFailure { ... }
    } else if output.contains("warning:") && output.contains("clippy") {
        ErrorType::ClippyWarning { ... }
    } else {
        ErrorType::Unknown
    }
}
```

3. **Emit autopsy card** between gate failure and auto-fix:
```rust
// After gate fails, before spawning auto-fixer:
let autopsy = analyze_gate_failure(gate_name, &error_output, workdir, &event_sender).await;

let mut card_text = format!("Root cause: {}\n\nCausal chain:\n", autopsy.root_cause);
for (i, step) in autopsy.causal_chain.iter().enumerate() {
    card_text.push_str(&format!("  {}. {}\n", i + 1, step));
}

if let Some(past) = &autopsy.similar_past_failure {
    card_text.push_str(&format!(
        "\nSimilar past failure: Episode #{}\n  {}\n  Resolution: {}\n",
        past.episode_id, past.summary, past.resolution
    ));
}

card_text.push_str(&format!("\nConfidence: {:.2} ({})",
    autopsy.confidence,
    autopsy.error_type.description()
));

event_sender.send(CognitiveEvent::ToolCallStart {
    tool_call_id: "gate-autopsy".into(),
    title: "🔬 Gate autopsy — forensic analysis".into(),
    kind: ToolCallKind::Other,
}).await?;

event_sender.send(CognitiveEvent::ToolCallComplete {
    tool_call_id: "gate-autopsy".into(),
    status: ToolCallStatus::Completed,
    content: vec![ContentBlock::Text { text: card_text }],
}).await?;
```

4. **Feed autopsy into auto-fixer prompt**: The auto-fixer should receive the classified error and past resolution, not just raw output:
```rust
let fix_prompt = if let Some(past) = &autopsy.similar_past_failure {

        past.resolution,
        error_output
    )
} else {
```
